from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol's strict priority:
    P0 (negative space) → P1 (safe channels) → P2 (optimization).
    
    Rules:
    1. Steps must start with P0, then P1, then P2.
    2. Cannot skip a priority level (e.g., P0 directly to P2).
    3. Once a higher priority appears, cannot go back to lower.
    
    Args:
        steps: List of step identifiers like ["P0-map", "P1-channel", "P2-opt"].
    
    Returns:
        Tuple of (is_valid: bool, error_message: str).
    """
    if not steps:
        return False, "Empty step list"
    
    allowed_transitions = {
        "P0": ["P0", "P1"],
        "P1": ["P1", "P2"],
        "P2": ["P2"]
    }
    
    for i, step in enumerate(steps):
        # Extract priority prefix
        if not (step.startswith("P0") or step.startswith("P1") or step.startswith("P2")):
            return False, f"Step '{step}' does not start with P0/P1/P2"
        
        prefix = step[:2]  # "P0", "P1", or "P2"
        
        # Check transition from previous step
        if i > 0:
            prev_prefix = steps[i-1][:2]
            if prefix not in allowed_transitions[prev_prefix]:
                return False, f"Illegal transition from {prev_prefix} to {prefix} at step {i}"
    
    return True, "Valid Deadband Protocol sequence"
```

import re
from datetime import datetime
from typing import Dict, Optional

def parse_fleet_tile(tile_text: str) -> Dict[str, Optional[str]]:
    """
    Parses raw training tile text into structured dictionary.
    
    Expected format (lines may be missing):
        Agent: <name>
        Cycle: <number>
        Phase: <number>
        Task: <description>
        Content: <text>
        Timestamp: <ISO format>
    
    Args:
        tile_text: Raw tile text.
    
    Returns:
        Dictionary with keys: agent, cycle, phase, task, content, timestamp.
        Missing values become None.
    """
    patterns = {
        "agent": r"Agent:\s*(.+)",
        "cycle": r"Cycle:\s*(\d+)",
        "phase": r"Phase:\s*(\d+)",
        "task": r"Task:\s*(.+)",
        "content": r"Content:\s*(.+)",
        "timestamp": r"Timestamp:\s*(.+)"
    }
    
    result = {key: None for key in patterns.keys()}
    
    for line in tile_text.strip().split('\n'):
        for key, pattern in patterns.items():
            match = re.match(pattern, line.strip())
            if match:
                result[key] = match.group(1).strip()
                break
    
    # Validate timestamp format if present
    if result["timestamp"]:
        try:
            datetime.fromisoformat(result["timestamp"].replace('Z', '+00:00'))
        except ValueError:
            result["timestamp"] = None
    
    return result
```

#### **Function C: simulate_constraint_snap**

from typing import List, Tuple
import math

def simulate_constraint_snap(points: List[Tuple[float, float]], tolerance: float) -> List[Tuple[float, float]]:
    """
    Simulates geometric snapping of 2D points within given tolerance.
    
    Algorithm:
    1. Group points that are within tolerance distance of each other.
    2. For each group, compute centroid (average x, y).
    3. Replace all points in group with the centroid.
    
    Args:
        points: List of (x, y) coordinates.
        tolerance: Maximum distance for points to be considered snap-group.
    
    Returns:
        List of snapped points (same order as input, but coordinates updated).
    """
    if tolerance <= 0:
        return points.copy()
    
    n = len(points)
    snapped = list(points)
    visited = [False] * n
    
    def distance(p1, p2):
        return math.sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)
    
    for i in range(n):
        if not visited[i]:
            # Find all points within tolerance of points[i]
            group = [i]
            for j in range(i+1, n):
                if not visited[j] and distance(points[i], points[j]) <= tolerance:
                    group.append(j)
            
            if len(group) > 1:
                # Compute centroid
                avg_x = sum(points[k][0] for k in group) / len(group)
                avg_y = sum(points[k][1] for k in group) / len(group)
                
                # Update all points in group
                for k in group:
                    snapped[k] = (round(avg_x, 6), round(avg_y, 6))
                    visited[k] = True
    
    return snapped
```

def validate_deadband_protocol(steps):
    """
    Check if steps follow P0->P1->P2 order strictly.
    """
    if not steps:
        return False, "no steps"
    
    order = ["P0", "P1", "P2"]
    current_index = -1
    
    for step in steps:
        prefix = step[:2]
        if prefix not in order:
            return False, f"invalid prefix {prefix}"
        
        step_index = order.index(prefix)
        if step_index < current_index:
            return False, f"went backward from {order[current_index]} to {prefix}"
        
        current_index = step_index
    
    return True, "ok"
```
