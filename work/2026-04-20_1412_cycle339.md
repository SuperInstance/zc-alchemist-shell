## **2. Function 1: `deadband_filter`**

**Prompt:**
```
Write a Python function `deadband_filter` that implements the Cocapn fleet's Deadband Protocol.

Input: 
- `values`: list of floats
- `deadband`: float (positive, default 0.1)

Protocol (P0, P1, P2):
- P0 (negative space): Remove any values where abs(value) < deadband (too close to zero).
- P1 (safe channels): Keep only values where deadband <= abs(value) <= 1.0.
- P2 (optimize): For remaining values, apply a simple optimization: if value > 0, square it; if value < 0, take absolute square root.

Return the processed list.

Include a docstring and example.
```

### **DeepSeek-chat output:**
```python
def deadband_filter(values, deadband=0.1):
    """
    Apply the Deadband Protocol (P0, P1, P2) to a list of float values.

    Parameters:
    - values: list of floats
    - deadband: float, positive threshold (default 0.1)

    Returns:
    - list of processed floats that passed P0 and P1, with P2 optimization applied.

    Protocol:
    P0: Remove values where |value| < deadband (negative space).
    P1: Keep only values where deadband <= |value| <= 1.0 (safe channels).
    P2: Optimize: if value > 0 -> value²; if value < 0 -> sqrt(|value|).
    """
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    if deadband <= 0:
        raise ValueError("deadband must be positive")

    # P0: Map negative space (remove too-small magnitudes)
    p0_filtered = [v for v in values if abs(v) >= deadband]

    # P1: Find safe channels (constrain magnitude to <= 1.0)
    p1_filtered = [v for v in p0_filtered if abs(v) <= 1.0]

    # P2: Optimize within channels
    result = []
    for v in p1_filtered:
        if v > 0:
            result.append(v ** 2)
        elif v < 0:
            result.append(abs(v) ** 0.5)  # sqrt(|v|)
        else:  # v == 0 (should not occur after P0, but safe)
            result.append(0.0)

    return result
```

```python
def deadband_filter(values, deadband=0.1):
    """
    Implements Deadband Protocol filtering.

    Steps:
    1. P0: Eliminate values within deadband of zero.
    2. P1: Retain values within [-1.0, -deadband] U [deadband, 1.0].
    3. P2: Apply optimization: square positives, sqrt(abs()) negatives.

    Args:
        values: Iterable of numeric values.
        deadband: Positive float.

    Returns:
        List of transformed values meeting criteria.
    """
    # Input validation
    try:
        vals = [float(v) for v in values]
    except (TypeError, ValueError):
        raise ValueError("All values must be convertible to float")

    if deadband <= 0:
        raise ValueError("deadband must be > 0")

    filtered = []
    for v in vals:
        av = abs(v)
        # P0 and P1 combined
        if deadband <= av <= 1.0:
            # P2
            if v > 0:
                filtered.append(v * v)
            else:  # v negative
                filtered.append(av ** 0.5)
    return filtered
```

from datetime import datetime

def parse_fleet_log(log_line):
    """
    Parse a structured fleet log line into a dictionary.

    Expected format: "TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE"
    TIMESTAMP in ISO format (e.g., 2026-04-20T14:30:00Z).
    CYCLE is an integer.

    Parameters:
    - log_line: str, a single log line.

    Returns:
    - dict with keys: timestamp (datetime), agent (str), cycle (int), 
      status (str), message (str).

    Raises:
    - ValueError if format is invalid or fields cannot be parsed.
    """
    if not log_line or not isinstance(log_line, str):
        raise ValueError("log_line must be a non-empty string")

    parts = log_line.split(" | ", maxsplit=4)  # Split into 5 parts max
    if len(parts) != 5:
        raise ValueError(f"Log line must have exactly 5 parts separated by ' | ', got {len(parts)}")

    timestamp_str, agent, cycle_str, status, message = parts

    try:
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        raise ValueError(f"Invalid timestamp format: {timestamp_str}")

    try:
        cycle = int(cycle_str)
    except ValueError:
        raise ValueError(f"Cycle must be integer, got: {cycle_str}")

    return {
        "timestamp": timestamp,
        "agent": agent.strip(),
        "cycle": cycle,
        "status": status.strip(),
        "message": message.strip()
    }
