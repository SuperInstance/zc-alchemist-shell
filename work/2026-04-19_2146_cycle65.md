def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply the Deadband Protocol to a list of float values.
    
    The Deadband Protocol (P0/P1/P2) filters out values that fall within a 
    symmetric deadband around zero, representing 'negative space' to avoid.
    
    Args:
        values: List of input float values.
        deadband: Absolute threshold around zero. Values in [-deadband, deadband]
                  are filtered out (P0 avoidance).
    
    Returns:
        List of values outside the deadband (safe channels for P1 exploration).
    
    Edge cases:
        - Empty input returns empty list.
        - Non‑float values raise TypeError.
    """
    if not isinstance(values, list):
        raise TypeError("Input must be a list")
    if len(values) == 0:
        return []
    
    filtered = []
    for v in values:
        if not isinstance(v, (int, float)):
            raise TypeError(f"Value {v} is not a float")
        if abs(v) > deadband:
            filtered.append(v)
    return filtered

from typing import List

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Implements the Deadband Protocol: removes values within +/- deadband of zero.
    
    P0: Avoid values inside deadband (negative space).
    P1: Keep values outside deadband (safe channels).
    
    Args:
        values: A list of floating‑point numbers.
        deadband: The width of the deadband region (default 0.1).
    
    Returns:
        A list containing only values whose absolute value > deadband.
    
    Raises:
        ValueError: If deadband is negative.
        TypeError: If any element is not a float.
    """
    if deadband < 0:
        raise ValueError("deadband must be non‑negative")
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    
    result = []
    for v in values:
        if not isinstance(v, (int, float)):
            raise TypeError(f"Element {v} is not a numeric type")
        if abs(v) > deadband:
            result.append(float(v))
    return result

import re
from typing import Dict, Optional

def parse_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a fleet service log line into structured fields.
    
    Expected format:
        [TIMESTAMP] SERVICE:LEVEL Message key=value key=value ...
    
    Example:
        [2026‑04‑19T21:43:00Z] holodeck‑rust:INFO actors=5 rooms=2
    
    Returns:
        Dictionary with keys: timestamp, service, level, message, and extracted
        key‑value pairs. Returns None if line does not match the pattern.
    
    Edge cases:
        - Empty or whitespace‑only lines return None.
        - Missing key‑value pairs are omitted.
    """
    if not line or line.isspace():
        return None
    
    pattern = r'^\[([^\]]+)\]\s+([^:]+):([^\s]+)\s+(.*)$'
    match = re.match(pattern, line.strip())
    if not match:
        return None
    
    timestamp, service, level, rest = match.groups()
    result = {
        'timestamp': timestamp,
        'service': service,
        'level': level,
        'message': rest
    }
    
    # Extract key=value pairs
    kv_pattern = r'(\w+)=([^\s]+)'
    for key, val in re.findall(kv_pattern, rest):
        result[key] = val
    
    return result
```

import re
from typing import Dict, Any, Optional

def parse_log_line(log_line: str) -> Optional[Dict[str, Any]]:
    """
    Parses a standardized fleet log line.
    
    Format: [ISO_TIMESTAMP] SERVICE_NAME:LOG_LEVEL message key1=val1 key2=val2 ...
    
    Args:
        log_line: A string containing one log entry.
    
    Returns:
        A dictionary with keys: timestamp, service, level, msg, and any extracted
        key‑value pairs. Returns None if the line does not match the expected format.
    
    Edge cases:
        - Lines without a valid timestamp return None.
        - Key‑value pairs with missing values are skipped.
    """
    if not log_line or not log_line.startswith('['):
        return None
    
    # Main regex
    regex = r'^\[([^\]]+)\]\s+([^:]+):([^\s]+)\s+(.*)$'
    m = re.match(regex, log_line.strip())
    if not m:
        return None
    
    timestamp, service, level, remainder = m.groups()
    parsed = {
        'timestamp': timestamp,
        'service': service,
        'level': level,
        'msg': remainder
    }
    
    # Find key=value patterns
    kv_re = r'(\w+)=([^\s]+)'
    for k, v in re.findall(kv_re, remainder):
        parsed[k] = v
    
    return parsed
```
