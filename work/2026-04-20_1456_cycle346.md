# Function 1: deadband_filter
def deadband_filter(decisions, p0_threshold=0.1, p1_margin=0.2):
    """
    Apply Deadband Protocol to a list of decision scores.
    
    P0: Reject decisions below p0_threshold (negative space)
    P1: Accept decisions within p1_margin of max remaining score (safe channels)
    P2: Return the highest scoring decision from P1 set (optimize)
    
    Args:
        decisions: List of float scores [0.0-1.0]
        p0_threshold: Minimum score to pass P0 (default 0.1)
        p1_margin: Range for safe channel selection (default 0.2)
    
    Returns:
        Tuple of (selected_index, selected_score, filtered_indices)
        Returns (-1, 0.0, []) if no decisions pass P0
    """
    if not decisions:
        return (-1, 0.0, [])
    
    # P0: Filter out negative space
    p0_passing = [(i, score) for i, score in enumerate(decisions) 
                  if score >= p0_threshold]
    
    if not p0_passing:
        return (-1, 0.0, [])
    
    # Find max score among P0 survivors
    max_score = max(score for _, score in p0_passing)
    
    # P1: Find safe channels (within margin of max)
    p1_candidates = [(i, score) for i, score in p0_passing 
                     if score >= max_score - p1_margin]
    
    # P2: Optimize (select highest from P1)
    best_idx, best_score = max(p1_candidates, key=lambda x: x[1])
    
    filtered_indices = [i for i, _ in p1_candidates]
    
    return (best_idx, best_score, filtered_indices)

def parse_fleet_log(log_line):
    """
    Parse a structured fleet log line.
    
    Format: "TIMESTAMP | AGENT | CYCLE | ACTION | STATUS | DETAILS"
    Example: "2026-04-20T14:39:00Z | Alchemist | 345 | tile_submit | accepted | size=342"
    
    Args:
        log_line: String in fleet log format
    
    Returns:
        Dictionary with keys: timestamp, agent, cycle, action, status, details
        Returns empty dict if parsing fails
    """
    try:
        parts = [part.strip() for part in log_line.split('|')]
        if len(parts) != 6:
            return {}
        
        return {
            'timestamp': parts[0],
            'agent': parts[1],
            'cycle': int(parts[2]) if parts[2].isdigit() else parts[2],
            'action': parts[3],
            'status': parts[4],
            'details': parts[5]
        }
    except Exception:
        return {}

def validate_training_tile(tile, schema=None):
    """
    Validate a training tile against schema.
    
    Required tile fields: id, agent, cycle, content, timestamp
    Optional: tags, metrics, parent_tile
    
    Args:
        tile: Dictionary containing tile data
        schema: Optional custom schema dict, defaults to basic fleet schema
    
    Returns:
        Tuple of (is_valid, errors)
        errors is list of string error messages
    """
    if schema is None:
        schema = {
            'required': ['id', 'agent', 'cycle', 'content', 'timestamp'],
            'optional': ['tags', 'metrics', 'parent_tile'],
            'types': {
                'id': str,
                'agent': str,
                'cycle': int,
                'content': str,
                'timestamp': str,
                'tags': list,
                'metrics': dict,
                'parent_tile': str
            }
        }
    
    errors = []
    
    # Check required fields
    for field in schema['required']:
        if field not in tile:
            errors.append(f"Missing required field: {field}")
        elif field in schema['types']:
            expected_type = schema['types'][field]
            if not isinstance(tile[field], expected_type):
                errors.append(f"Field {field} should be {expected_type.__name__}, got {type(tile[field]).__name__}")
    
    # Check for unknown fields
    all_fields = set(schema['required'] + schema['optional'])
    for field in tile.keys():
        if field not in all_fields:
            errors.append(f"Unknown field: {field}")
    
    # Special validation for cycle
    if 'cycle' in tile and isinstance(tile['cycle'], int):
        if tile['cycle'] < 0:
            errors.append("Cycle must be non-negative")
    
    return (len(errors) == 0, errors)
