### **DeepSeek-chat Output:**
```python
def deadband_filter(values, lower_bound, upper_bound, p2_factor=0.9):
    """
    Apply the Deadband Protocol (P0/P1/P2) to a list of values.
    
    P0 (negative space): values <= lower_bound → 0.0
    P1 (safe channel): lower_bound < value < upper_bound → unchanged
    P2 (optimization): value >= upper_bound → value * p2_factor
    
    Parameters
    ----------
    values : list of float
        Input data points.
    lower_bound : float
        Threshold for P0 rejection.
    upper_bound : float
        Threshold for P2 optimization.
    p2_factor : float, optional
        Scaling factor for P2 zone (default 0.9).
    
    Returns
    -------
    list of float
        Filtered values.
    
    Raises
    ------
    ValueError
        If lower_bound >= upper_bound.
    """
    if lower_bound >= upper_bound:
        raise ValueError("lower_bound must be less than upper_bound")
    
    result = []
    for v in values:
        if v <= lower_bound:
            # P0: Map negative space to zero
            result.append(0.0)
        elif v < upper_bound:
            # P1: Safe channel, pass through
            result.append(v)
        else:
            # P2: Optimize by scaling
            result.append(v * p2_factor)
    return result
```

```python
import re

def parse_fleet_context(md_text):
    """
    Parse fleet context markdown into structured dictionary.
    
    Sections are identified by '## ' headers. Content continues until next
    '## ' header or end of document.
    
    Parameters
    ----------
    md_text : str
        Raw markdown text from FLEET-CONTEXT.md
    
    Returns
    -------
    dict
        Dictionary with section titles (without '## ') as keys and
        section content (stripped) as values.
    """
    # Pattern to match ## headers and capture title
    section_pattern = r'^##\s+(.+?)$'
    
    lines = md_text.split('\n')
    sections = {}
    current_section = None
    current_content = []
    
    for line in lines:
        header_match = re.match(section_pattern, line.strip())
        if header_match:
            # Save previous section
            if current_section is not None:
                sections[current_section] = '\n'.join(current_content).strip()
            # Start new section
            current_section = header_match.group(1)
            current_content = []
        else:
            if current_section is not None:
                current_content.append(line)
    
    # Don't forget the last section
    if current_section is not None:
        sections[current_section] = '\n'.join(current_content).strip()
    
    return sections
```
