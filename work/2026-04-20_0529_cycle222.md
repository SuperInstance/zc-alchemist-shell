from typing import List

def deadband_filter(values: List[float], epsilon: float) -> List[float]:
    """
    Apply a deadband filter to a list of float values.
    
    Args:
        values: List of float values to filter.
        epsilon: Deadband threshold. Values with |v| < epsilon become 0.0.
    
    Returns:
        List of filtered float values.
    """
    if epsilon < 0:
        raise ValueError("epsilon must be non-negative")
    
    result = []
    for v in values:
        if abs(v) < epsilon:
            result.append(0.0)
        else:
            result.append(v)
    return result
```

import re
from typing import Optional, Dict

def parse_fleet_log(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a fleet log line.
    
    Expected format: "YYYY-MM-DD HH:MM:SS | AGENT | LEVEL | message"
    
    Args:
        line: A string log line.
    
    Returns:
        Dictionary with keys: timestamp, agent, level, message.
        Returns None if line doesn't match expected format.
    """
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \| (\w+) \| (\w+) \| (.+)$'
    match = re.match(pattern, line.strip())
    if not match:
        return None
    
    return {
        'timestamp': match.group(1),
        'agent': match.group(2),
        'level': match.group(3),
        'message': match.group(4)
    }
```

Write a Python function `constraint_snap(point, segments)` that given a point (x, y) and a list of line segments [(x1, y1, x2, y2), ...] finds the closest point on any segment. Return (snapped_x, snapped_y, segment_index). Use Euclidean distance. If segments empty, return original point and -1.

**Response:**
```python
from typing import Tuple, List
import math

def point_to_segment_distance(px: float, py: float, x1: float, y1: float, x2: float, y2: float) -> Tuple[float, float, float]:
    """
    Compute closest point on segment (x1,y1)-(x2,y2) to point (px,py).
    Returns (closest_x, closest_y, distance).
    """
    dx = x2 - x1
    dy = y2 - y1
    if dx == 0 and dy == 0:
        # Segment is a point
        return x1, y1, math.hypot(px - x1, py - y1)
    
    t = ((px - x1) * dx + (py - y1) * dy) / (dx * dx + dy * dy)
    t = max(0.0, min(1.0, t))  # Clamp to segment
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    distance = math.hypot(px - closest_x, py - closest_y)
    return closest_x, closest_y, distance

def constraint_snap(point: Tuple[float, float], segments: List[Tuple[float, float, float, float]]) -> Tuple[float, float, int]:
    """
    Snap a point to the closest line segment.
    
    Args:
        point: (x, y) tuple.
        segments: List of (x1, y1, x2, y2) line segments.
    
    Returns:
        (snapped_x, snapped_y, segment_index)
        If segments empty, returns original point and segment_index = -1.
    """
    if not segments:
        return point[0], point[1], -1
    
    best_distance = float('inf')
    best_point = (point[0], point[1])
    best_index = -1
    
    for i, (x1, y1, x2, y2) in enumerate(segments):
        closest_x, closest_y, dist = point_to_segment_distance(point[0], point[1], x1, y1, x2, y2)
        if dist < best_distance:
            best_distance = dist
            best_point = (closest_x, closest_y)
            best_index = i
    
    return best_point[0], best_point[1], best_index
```
