**DeepSeek-chat Output:**

from typing import List, Dict

def normalize_and_aggregate(data: List[Dict], key: str) -> Dict[str, float]:
    """
    Normalizes a numeric 'score' field within each group defined by `key`
    and returns the average normalized score per group.

    Args:
        data: List of dictionaries containing at least `key` and 'score' keys.
        key: The dictionary key to group by.

    Returns:
        Dictionary mapping key values to their mean normalized score.

    Raises:
        ValueError: If data is empty or missing required keys.
    """
    if not data:
        return {}

    # Validate required keys exist
    for item in data:
        if key not in item or 'score' not in item:
            raise ValueError(f"Item missing required key: {item}")

    # Group data
    groups: Dict[str, List[float]] = {}
    for item in data:
        groups.setdefault(item[key], []).append(item['score'])

    # Normalize and aggregate per group
    result: Dict[str, float] = {}
    for group_key, scores in groups.items():
        if not scores:
            result[group_key] = 0.0
            continue
        min_score = min(scores)
        max_score = max(scores)
        if max_score == min_score:
            # All scores identical, set normalized to 0.5 or keep as 0
            normalized = [0.5 for _ in scores]
        else:
            normalized = [(s - min_score) / (max_score - min_score) for s in scores]
        result[group_key] = sum(normalized) / len(normalized)

    return result
```

from typing import List, Dict

def normalize_and_aggregate(data: List[Dict], key: str) -> Dict[str, float]:
    """
    Groups data by `key`, normalizes the 'score' values within each group
    to the range [0, 1], and computes the mean normalized score for each group.
    """
    if not data:
        return {}

    # Group scores by the provided key
    groups = {}
    for entry in data:
        k = entry.get(key)
        score = entry.get('score')
        if k is None or score is None:
            continue  # Skip malformed entries
        groups.setdefault(k, []).append(score)

    # Process each group
    result = {}
    for k, scores in groups.items():
        if not scores:
            result[k] = 0.0
            continue
        lo, hi = min(scores), max(scores)
        if hi == lo:
            # Avoid division by zero; all values are identical
            norm_scores = [0.5] * len(scores)
        else:
            norm_scores = [(s - lo) / (hi - lo) for s in scores]
        result[k] = sum(norm_scores) / len(norm_scores)

    return result
```

*   **Clarity & Style:** DeepSeek includes a more detailed docstring with `Raises:` section and uses explicit type hints (`Dict[str, List[float]]`). Groq's docstring is adequate but shorter. Both are readable.
*   **Robustness:** DeepSeek handles the `max_score == min_score` edge case by setting normalized values to 0.5, which is a reasonable default. Groq does the same. DeepSeek's explicit validation could prevent silent bugs later.
*   **Fleet Relevance:** Both use only standard library. DeepSeek's stricter error handling may align better with fleet systems where data integrity is P0.

### **Function 2: `find_safe_channels`**

**DeepSeek-chat Output:**
```python
from typing import Set, Tuple, List
from collections import deque

def find_safe_channels(obstacles: Set[Tuple[int, int]], grid_size: int) -> List[List[Tuple[int, int]]]:
    """
    Find all non-intersecting safe channels from (0,0) to (grid_size-1, grid_size-1)
    using BFS, avoiding obstacles.

    A 'channel' is a path where no cell is used by another path.
    This implementation finds one path at a time and marks its cells as used.

    Args:
        obstacles: Set of (x, y) coordinates that are blocked.
        grid_size: Size of the N x N grid.

    Returns:
        List of paths, where each path is a list of (x, y) coordinates.
    """
    target = (grid_size - 1, grid_size - 1)
    channels = []
    used_cells = set(obstacles)  # Start with obstacles as permanently used

    while True:
        # BFS for a single path
        queue = deque()
        start = (0, 0)
        if start in used_cells:
            break  # Start is blocked, no more channels possible
        queue.append((start, [start]))
        visited = {start}
        found_path = None

        while queue and found_path is None:
            (x, y), path = queue.popleft()

            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:  # 4-directional
                nx, ny = x + dx, y + dy
                if 0 <= nx < grid_size and 0 <= ny < grid_size:
                    next_cell = (nx, ny)
                    if next_cell in used_cells or next_cell in visited:
                        continue
                    if next_cell == target:
                        found_path = path + [next_cell]
                        break
                    visited.add(next_cell)
                    queue.append((next_cell, path + [next_cell]))

        if not found_path:
            break  # No more paths exist
        channels.append(found_path)
        # Add all cells in this path to used_cells to prevent intersection
        used_cells.update(found_path)

    return channels
```
