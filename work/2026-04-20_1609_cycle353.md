```python
from typing import List

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply deadband filtering to a list of values.
    
    Args:
        values: List of float values
        deadband: Threshold around zero (positive)
        
    Returns:
        List where values within [-deadband, deadband] are set to 0.0
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for v in values:
        if -deadband <= v <= deadband:
            result.append(0.0)
        else:
            result.append(v)
    return result
```

#### **Function 2: constraint_snap**
```python
from typing import Tuple

def constraint_snap(px: float, py: float, x1: float, y1: float, x2: float, y2: float) -> Tuple[float, float]:
    """
    Snap point (px, py) to the nearest point on line segment (x1,y1)-(x2,y2).
    
    Args:
        px, py: Point coordinates
        x1, y1, x2, y2: Line segment endpoints
        
    Returns:
        (x, y) of snapped point
    """
    # Vector from start to end
    dx = x2 - x1
    dy = y2 - y1
    
    # If segment is a point, return that point
    if dx == 0 and dy == 0:
        return (x1, y1)
    
    # Vector from start to point
    vx = px - x1
    vy = py - y1
    
    # Projection parameter t = dot(v, d) / dot(d, d)
    t = (vx * dx + vy * dy) / (dx * dx + dy * dy)
    
    # Clamp t to [0, 1] to stay on segment
    t_clamped = max(0.0, min(1.0, t))
    
    # Calculate snapped point
    x_snap = x1 + t_clamped * dx
    y_snap = y1 + t_clamped * dy
    
    return (x_snap, y_snap)
```

```python
from typing import Tuple, List, Dict, Any
from datetime import datetime

def tile_validate(tile: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile structure.
    
    Args:
        tile: Dictionary representing a tile
        
    Returns:
        (is_valid, list_of_error_messages)
    """
    errors = []
    required_keys = ["id", "timestamp", "agent", "content", "room"]
    
    # Check required keys
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    # If missing keys, skip further checks
    if errors:
        return (False, errors)
    
    # Validate id is string
    if not isinstance(tile["id"], str):
        errors.append("'id' must be a string")
    
    # Validate timestamp format
    try:
        datetime.fromisoformat(tile["timestamp"])
    except (ValueError, TypeError):
        errors.append("'timestamp' must be ISO format string")
    
    # Validate agent is string
    if not isinstance(tile["agent"], str):
        errors.append("'agent' must be a string")
    
    # Validate content is string
    if not isinstance(tile["content"], str):
        errors.append("'content' must be a string")
    
    # Validate room is string
    if not isinstance(tile["room"], str):
        errors.append("'room' must be a string")
    
    return (len(errors) == 0, errors)
```

```python
    """

    Example:
    >>> deadband_filter([-0.1, 0.05, 0.2, -0.3], 0.1)

    result = []
    for value in values:
        if abs(value) <= deadband:
            result.append(0.0)
        else:
            result.append(value)
    return result
```
