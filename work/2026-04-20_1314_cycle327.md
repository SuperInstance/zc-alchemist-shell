def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol to filter values.
    
    Args:
        values: List of float values
        deadband: Absolute threshold around zero (default 0.1)
    
    Returns:
        List of values outside the deadband region [-deadband, deadband]
    
    Example:
        >>> deadband_filter([0.05, -0.03, 0.15, -0.2, 0.0])
        [0.15, -0.2]
    """
    if not values:
        return []
    
    filtered = []
    for v in values:
        if abs(v) > deadband:
            filtered.append(v)
    return filtered
```

from datetime import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse structured fleet log line.
    
    Format: "YYYY-MM-DD HH:MM:SS | AGENT | CYCLE | STATUS | MESSAGE"
    
    Returns:
        Dictionary with parsed components or None if malformed
    
    Example:
        >>> parse_fleet_log("2026-04-20 13:07:00 | Alchemist | 327 | ACTIVE | Testing")
        {'timestamp': datetime(2026, 4, 20, 13, 7, 0), 
         'agent': 'Alchemist', 
         'cycle': 327, 
         'status': 'ACTIVE', 
         'message': 'Testing'}
    """
    try:
        parts = log_line.split(" | ")
        if len(parts) != 5:
            return None
        
        timestamp = datetime.strptime(parts[0], "%Y-%m-%d %H:%M:%S")
        agent = parts[1].strip()
        cycle = int(parts[2].strip())
        status = parts[3].strip()
        message = parts[4].strip()
        
        return {
            "timestamp": timestamp,
            "agent": agent,
            "cycle": cycle,
            "status": status,
            "message": message
        }
    except (ValueError, IndexError):
        return None
```

#### **Function 3: constraint_snap**
```python
import math

def constraint_snap(point: tuple[float, float], 
                    grid_size: float = 1.0,
                    tolerance: float = 0.25) -> tuple[float, float]:
    """
    Snap 2D point to nearest grid point if within tolerance.
    
    Args:
        point: (x, y) coordinates
        grid_size: Spacing between grid points
        tolerance: Maximum distance for snapping
    
    Returns:
        Original point or snapped grid point
    
    Example:
        >>> constraint_snap((0.6, 0.6), grid_size=1.0, tolerance=0.25)
        (1.0, 1.0)  # Distance to (1,1) is ~0.566 > 0.25, so no snap
        >>> constraint_snap((0.12, 0.12), grid_size=1.0, tolerance=0.25)
        (0.0, 0.0)  # Distance to (0,0) is ~0.17 <= 0.25, so snap
    """
    x, y = point
    
    # Find nearest grid point
    grid_x = round(x / grid_size) * grid_size
    grid_y = round(y / grid_size) * grid_size
    
    # Calculate distance
    distance = math.sqrt((x - grid_x)**2 + (y - grid_y)**2)
    
    # Snap if within tolerance
    if distance <= tolerance:
        return (grid_x, grid_y)
    else:
        return (x, y)
```

def deadband_filter(values, deadband=0.1):
    """
    Implements the Deadband Protocol filtering.
    
    Parameters:
    values (list): List of floating point numbers
    deadband (float): Threshold around zero (default 0.1)
    
    Returns:
    list: Values outside the deadband range [-deadband, deadband]

    >>> deadband_filter([0.05, -0.03, 0.15, -0.2, 0.0])
    [0.15, -0.2]

    for val in values:
        if val < -deadband or val > deadband:
