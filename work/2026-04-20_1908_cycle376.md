```
1. `deadband_filter(signal: list[float], p0_threshold: float, p1_width: float) -> list[float]`:

2. `parse_fleet_log(log_line: str) -> dict`:

   `phase` must be 1, 2, 3, or 4.
   Return a tuple: (is_valid: bool, list_of_error_messages: list[str]).

Write only the Python code. No explanations.
```

### **A. DeepSeek-chat Output**
```python
import re
from datetime import datetime

def deadband_filter(signal: list[float], p0_threshold: float, p1_width: float) -> list[float]:
    """
    Apply Deadband Protocol (P0/P1/P2) to a signal.
    P0: |value| <= p0_threshold -> 0.0
    P1: p0_threshold < |value| <= p0_threshold + p1_width -> original value
    P2: |value| > p0_threshold + p1_width -> value * 0.9
    """
    result = []
    boundary = p0_threshold + p1_width
    
    for val in signal:
        abs_val = abs(val)
        if abs_val <= p0_threshold:
            result.append(0.0)
        elif abs_val <= boundary:
            result.append(val)
        else:  # P2
            result.append(val * 0.9)
    return result

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse fleet agent log line.
    Format: "YYYY-MM-DD HH:MM:SS | AGENT_NAME | STATUS | MESSAGE"
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \| ([^|]+) \| ([^|]+) \| (.+)'
    match = re.match(pattern, log_line)
    if not match:
        raise ValueError(f"Invalid log format: {log_line}")
    
    timestamp_str, agent, status, message = match.groups()
    timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
    
    return {
        'timestamp': timestamp,
        'agent': agent.strip(),
        'status': status.strip(),
        'message': message.strip()
    }

def validate_training_tile(tile_data: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile dictionary.
    Required keys: id (str), cycle (int), agent_role (str), content (str), phase (int)
    Phase must be in {1, 2, 3, 4}
    """
    required_keys = ['id', 'cycle', 'agent_role', 'content', 'phase']
    errors = []
    
    # Check required keys
    for key in required_keys:
        if key not in tile_data:
            errors.append(f"Missing required key: {key}")
    
    # If missing keys, skip further validation
    if errors:
        return (False, errors)
    
    # Validate types and constraints
    if not isinstance(tile_data['id'], str):
        errors.append("'id' must be a string")
    
    if not isinstance(tile_data['cycle'], int):
        errors.append("'cycle' must be an integer")
    elif tile_data['cycle'] < 0:
        errors.append("'cycle' must be non-negative")
    
    if not isinstance(tile_data['agent_role'], str):
        errors.append("'agent_role' must be a string")
    
    if not isinstance(tile_data['content'], str):
        errors.append("'content' must be a string")
    
    if not isinstance(tile_data['phase'], int):
        errors.append("'phase' must be an integer")
    elif tile_data['phase'] not in {1, 2, 3, 4}:
        errors.append("'phase' must be 1, 2, 3, or 4")
    
    return (len(errors) == 0, errors)
```

import re
from datetime import datetime

def deadband_filter(signal: list[float], p0_threshold: float, p1_width: float) -> list[float]:
    filtered = []
    p1_upper = p0_threshold + p1_width
    for x in signal:
        abs_x = abs(x)
        if abs_x <= p0_threshold:
            filtered.append(0.0)
        elif abs_x <= p1_upper:
            filtered.append(x)
        else:
            filtered.append(x * 0.9)
    return filtered

def parse_fleet_log(log_line: str) -> dict:
    # Expected format: "2024-04-20 12:34:56 | AGENT_NAME | STATUS | MESSAGE"
    parts = log_line.split(" | ", 3)
    if len(parts) != 4:
        raise ValueError("Log line must have exactly four parts separated by ' | '")
    date_str, agent, status, message = parts
    try:
        timestamp = datetime.strptime(date_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        raise ValueError("Invalid date format in log line")
    return {
        "timestamp": timestamp,
        "agent": agent.strip(),
        "status": status.strip(),
        "message": message.strip()
    }

        errors.append("cycle must be an integer")
