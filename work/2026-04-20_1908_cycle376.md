# Function 1: Deadband Protocol Validator
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of Deadband Protocol steps.
    Rules: Must start with P0, proceed to P1, then P2.
    No skipping (e.g., P0 -> P2 is invalid). No regressions (e.g., P1 -> P0).
    """
    expected = ["P0", "P1", "P2"]
    seen = set()
    
    for i, step in enumerate(steps):
        if step not in expected:
            return False, f"Invalid step '{step}' at index {i}. Must be P0, P1, or P2."
        
        # Check for regression
        if i > 0:
            prev_idx = expected.index(steps[i-1])
            curr_idx = expected.index(step)
            if curr_idx < prev_idx:
                return False, f"Regression from {steps[i-1]} to {step} at index {i}."
        
        seen.add(step)
    
    # Check for skips
    for required in expected:
        if required not in seen:
            # If we missed a required step, check if it was skipped
            if required == "P0" and ("P1" in seen or "P2" in seen):
                return False, "P0 is required but missing. Never skip to P1/P2."
            if required == "P1" and "P0" not in seen:
                return False, "P1 requires P0 first."
            if required == "P2" and ("P0" not in seen or "P1" not in seen):
                return False, "P2 requires P0 and P1 first."
    
    return True, ""

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parses a raw fleet tile text into structured fields.
    Expected format lines: 'Key: value'. Handles missing keys.
    """
    result = {
        "agent": "",
        "cycle": 0,
        "phase": 0,
        "task": "",
        "content": ""
    }
    
    lines = raw_text.strip().split('\n')
    content_lines = []
    
    for line in lines:
        if ': ' in line:
            key, value = line.split(': ', 1)
            key_lower = key.strip().lower()
            if key_lower == "agent":
                result["agent"] = value.strip()
            elif key_lower == "cycle":
                try:
                    result["cycle"] = int(value.strip())
                except ValueError:
                    result["cycle"] = 0
            elif key_lower == "phase":
                try:
                    result["phase"] = int(value.strip())
                except ValueError:
                    result["phase"] = 0
            elif key_lower == "task":
                result["task"] = value.strip()
        else:
            content_lines.append(line.strip())
    
    result["content"] = '\n'.join(content_lines).strip()
    return result

def calculate_safe_channel(boundaries: list[tuple[float, float]], position: float) -> tuple[float, float]:
    """
    Finds the nearest safe channel (interval) that doesn't intersect any boundary.
    Boundaries are (min, max) unsafe ranges (P0 negative space).
    Returns (channel_min, channel_max) representing the safe channel.
    """
    if not boundaries:
        return (position - 1.0, position + 1.0)  # Default safe channel
    
    # Sort boundaries
    sorted_bounds = sorted(boundaries, key=lambda x: x[0])
    
    # Find gaps between boundaries
    gaps = []
    current_max = float('-inf')
    
    for b_min, b_max in sorted_bounds:
        if b_min > current_max:
            # Found a gap
            gaps.append((current_max, b_min))
        current_max = max(current_max, b_max)
    
    # Add gap after last boundary
    gaps.append((current_max, float('inf')))
    
    # Filter valid gaps (must have finite positive width)
    valid_gaps = [(g_min, g_max) for g_min, g_max in gaps if g_max > g_min and g_max != float('inf')]
    
    if not valid_gaps:
        # No finite safe channel found
        return (position, position)  # Zero-width fallback
    
    # Find nearest gap to position
    nearest_gap = min(valid_gaps, key=lambda gap: min(abs(position - gap[0]), abs(position - gap[1])))
    
    # Return a safe channel within the gap (centered on position if possible)
    channel_width = 0.1  # Default width
    channel_min = max(nearest_gap[0], position - channel_width/2)
    channel_max = min(nearest_gap[1], position + channel_width/2)
    
    # Adjust if channel is too narrow
    if channel_max - channel_min < 0.01:
        channel_min = nearest_gap[0]
        channel_max = nearest_gap[1]
    
    return (channel_min, channel_max)
