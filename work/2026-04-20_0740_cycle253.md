from datetime import datetime
import re
from typing import Dict, Optional

def parse_log_line(log_line: str) -> Dict[str, str]:
    """
    Parse a structured log line into its components.

    Expected format: "[TIMESTAMP] [LEVEL] Message"
    Example: "[2023-10-05 14:30:00] [INFO] Service started"

    Args:
        log_line: A string containing the log entry.

    Returns:
        Dictionary with keys 'timestamp', 'level', 'message'.

    Raises:
        ValueError: If the log line does not match the expected format.
    """
    pattern = r'^\[([^\]]+)\] \[([^\]]+)\] (.+)$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        raise ValueError(f"Log line does not match expected format: {log_line}")
    
    timestamp, level, message = match.groups()
    
    # Optional: validate timestamp format
    try:
        datetime.strptime(timestamp, '%Y-%m-%d %H:%M:%S')
    except ValueError:
        # If timestamp format is invalid, still return it but warn in doc
        pass
    
    return {
        'timestamp': timestamp,
        'level': level.upper(),
        'message': message.strip()

**DeepSeek-chat output:**

from typing import Tuple, Dict, Optional

def deadband_filter(
    value: float,
    p0_range: Tuple[float, float],
    p1_range: Tuple[float, float]
) -> Dict[str, Optional[str, float]]:
    """
    Apply deadband protocol filtering to a sensor reading.

    Protocol:
    - P0_REJECT: Value outside P0 range (unsafe)
    - P1_ACCEPT: Value within P0 range but outside P1 range (safe channel)
    - P2_OPTIMIZE: Value within P1 range (optimization zone)

    Args:
        value: Sensor reading.
        p0_range: Tuple (min, max) defining absolute safe boundaries.
        p1_range: Tuple (min, max) defining optimization zone within P0.

    Returns:
        Dictionary with:
        - 'status': One of 'P0_REJECT', 'P1_ACCEPT', 'P2_OPTIMIZE'
        - 'filtered_value': Original value if P1_ACCEPT or P2_OPTIMIZE, None if P0_REJECT

    Raises:
        ValueError: If ranges are invalid (P1 not within P0).
    """
    p0_min, p0_max = p0_range
    p1_min, p1_max = p1_range
    
    # Validate ranges
    if not (p0_min <= p1_min <= p1_max <= p0_max):
        raise ValueError("P1 range must be within P0 range")
    
    # P0 check
    if value < p0_min or value > p0_max:
        return {'status': 'P0_REJECT', 'filtered_value': None}
    
    # P1/P2 check
    if p1_min <= value <= p1_max:
        return {'status': 'P2_OPTIMIZE', 'filtered_value': value}
    else:
        return {'status': 'P1_ACCEPT', 'filtered_value': value}
