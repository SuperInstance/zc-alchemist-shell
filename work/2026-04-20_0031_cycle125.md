#### **DeepSeek-chat output:**
```python
def deadband_filter(values, lower, upper):
    """
    Apply a deadband filter to input values.
    
    If a value is within [lower, upper], it passes unchanged.
    If outside, it snaps to the nearest bound.
    
    Args:
        values: Single numeric value or list of numeric values.
        lower: Lower bound (inclusive).
        upper: Upper bound (inclusive).
    
    Returns:
        Filtered value(s) of same type as input.
    
    Raises:
        ValueError: If lower > upper.
        TypeError: If inputs are non-numeric.
    """
    if lower > upper:
        raise ValueError(f"lower ({lower}) cannot be greater than upper ({upper})")
    
    # Helper for single value
    def clamp(x):
        if not isinstance(x, (int, float)):
            raise TypeError(f"Expected numeric, got {type(x).__name__}")
        if x < lower:
            return lower
        if x > upper:
            return upper
        return x
    
    if isinstance(values, list):
        return [clamp(v) for v in values]
    else:
        return clamp(values)

# Example
if __name__ == "__main__":
    # Single value
    print(deadband_filter(5, 0, 10))   # 5
    print(deadband_filter(-2, 0, 10))  # 0
    print(deadband_filter(15, 0, 10))  # 10
    
    # List
    print(deadband_filter([1, 5, 12, -3, 7], 0, 10))  # [1, 5, 10, 0, 7]
```

```python
def deadband_filter(values, lower, upper):
    """
    Deadband filter: values within [lower, upper] pass through,
    values outside snap to nearest bound.
    
    Parameters:
        values: int, float, or list of int/float
        lower: numeric lower bound
        upper: numeric upper bound
    
    Returns:
        Same type as input with deadband applied.
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    # Single value processing
    if isinstance(values, (int, float)):
        if values < lower:
            return lower
        elif values > upper:
            return upper
        else:
            return values
    
    # List processing
    elif isinstance(values, list):
        result = []
        for v in values:
            if v < lower:
                result.append(lower)
            elif v > upper:
                result.append(upper)
            else:
                result.append(v)
        return result
    else:
        raise TypeError("values must be numeric or list of numeric")

# Example

print(deadband_filter([-1, 5, 11], 0, 10))  # [0, 5, 10]
```

- **Error handling:** DeepSeek checks numeric type inside `clamp()`; Groq checks at top. Both raise `ValueError` for `lower > upper`.
- **Edge cases:** DeepSeek explicitly mentions empty list (works due to list comprehension). Groq's list loop also works.
- **Style:** DeepSeek's docstring more detailed. Groq's is concise.
- **Spec adherence:** Both meet requirements. DeepSeek returns same type more robustly via `isinstance(values, list)` check.

#### **DeepSeek-chat output:**

```python
import re
from datetime import datetime

def parse_fleet_log_line(line):
    """
    Parse a fleet log line into structured components.
    
    Expected format: [TIMESTAMP] [AGENT] [LEVEL] message
    TIMESTAMP in ISO-like format, e.g., 2026-04-20T00:30:00Z
    
    Args:
        line (str): Log line string.
    
    Returns:
        dict: {'timestamp': datetime, 'agent': str, 'level': str, 'message': str}
    
    Raises:
        ValueError: If line does not match expected format.
    """
    pattern = r'^\[([^\]]+)\]\s+\[([^\]]+)\]\s+\[([^\]]+)\]\s+(.+)$'
    match = re.match(pattern, line.strip())
    if not match:
        raise ValueError(f"Line does not match fleet log format: {line}")
    
    timestamp_str, agent, level, message = match.groups()
    
    try:
        # Parse ISO format with Zulu time
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%dT%H:%M:%SZ")
    except ValueError:
        raise ValueError(f"Invalid timestamp format: {timestamp_str}")
    
    return {
        'timestamp': timestamp,
        'agent': agent.strip(),
        'level': level.strip(),
        'message': message.strip()
    }

# Example
if __name__ == "__main__":
    line = "[2026-04-20T00:30:00Z] [Alchemist] [INFO] Task started"
    parsed = parse_fleet_log_line(line)
    print(parsed)
    # {'timestamp': datetime(2026, 4, 20, 0, 30), 'agent': 'Alchemist', 'level': 'INFO', 'message': 'Task started'}
```
