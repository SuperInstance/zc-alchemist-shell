from typing import List, Dict, Any

def validate_deadband_protocol(actions: List[str]) -> Dict[str, Any]:
    """
    Validates that actions follow Deadband Protocol ordering: P0 → P1 → P2.
    
    Args:
        actions: List of action strings, e.g. ["P0: check", "P2: optimize", "P1: find"]
    
    Returns:
        Dict with keys:
        - valid: bool
        - violations: list of violation descriptions
        - corrected_order: list of actions sorted by phase (P0, P1, P2)
    """
    # Phase detection regex
    import re
    phase_pattern = re.compile(r'^(P[0-2]):\s*(.+)$')
    
    # Extract phases and actions
    parsed = []
    for i, action in enumerate(actions):
        match = phase_pattern.match(action.strip())
        if match:
            phase, desc = match.groups()
            parsed.append({
                'index': i,
                'phase': phase,
                'description': desc.strip(),
                'original': action
            })
        else:
            parsed.append({
                'index': i,
                'phase': None,
                'description': action.strip(),
                'original': action
            })
    
    # Check for missing phases
    phases_found = {p['phase'] for p in parsed if p['phase'] is not None}
    expected_phases = {'P0', 'P1', 'P2'}
    missing_phases = expected_phases - phases_found
    
    # Check ordering violations
    violations = []
    phase_sequence = []
    
    for i, item in enumerate(parsed):
        if item['phase'] is not None:
            phase_sequence.append((i, item['phase']))
    
    # Check for P2 before P1, P1 before P0, etc.
    for j in range(1, len(phase_sequence)):
        prev_phase = phase_sequence[j-1][1]
        curr_phase = phase_sequence[j][1]
        
        if prev_phase == 'P2' and curr_phase in ['P0', 'P1']:
            violations.append(f"P2 found at index {phase_sequence[j-1][0]} before {curr_phase} at index {phase_sequence[j][0]}")
        elif prev_phase == 'P1' and curr_phase == 'P0':
            violations.append(f"P1 found at index {phase_sequence[j-1][0]} before P0 at index {phase_sequence[j][0]}")
    
    # Generate corrected order (sort by phase priority)
    phased_actions = [p for p in parsed if p['phase'] is not None]
    phase_order = {'P0': 0, 'P1': 1, 'P2': 2}
    phased_actions.sort(key=lambda x: phase_order[x['phase']])
    corrected_order = [p['original'] for p in phased_actions]
    
    # Add non-phased actions to end
    non_phased = [p['original'] for p in parsed if p['phase'] is None]
    corrected_order.extend(non_phased)
    
    # Add missing phase violations
    for missing in missing_phases:
        violations.append(f"Missing phase: {missing}")
    
    return {
        'valid': len(violations) == 0,
        'violations': violations,
        'corrected_order': corrected_order
    }

import re
from typing import List, Dict

def extract_training_tiles(log_text: str) -> List[Dict[str, str]]:
    """
    Extract training tiles from agent interaction logs.
    
    Args:
        log_text: Multi-line log text containing agent cycles and tile content.
    
    Returns:
        List of dicts with keys: agent, cycle, tile_content, timestamp
    """
    tiles = []
    
    # Pattern to match cycle headers
    cycle_pattern = re.compile(r'Cycle\s+(\d+)\s*—\s*(\w+)', re.IGNORECASE)
    
    # Split by possible delimiters
    lines = log_text.split('\n')
    i = 0
    
    while i < len(lines):
        line = lines[i].strip()
        cycle_match = cycle_pattern.match(line)
        
        if cycle_match:
            cycle_num = cycle_match.group(1)
            agent_name = cycle_match.group(2)
            
            # Look for tile content (until next cycle or empty section)
            tile_lines = []
            i += 1
            
            while i < len(lines) and not cycle_pattern.match(lines[i].strip()):
                if lines[i].strip() == '---' and tile_lines:
                    # Tile separator
                    break
                tile_lines.append(lines[i].rstrip())
                i += 1
            
            tile_content = '\n'.join(tile_lines).strip()
            
            # Extract timestamp if present in tile content
            timestamp = None
            ts_match = re.search(r'\d{4}-\d{2}-\d{2}\s+\d{2}:\d{2}', tile_content)
            if ts_match:
                timestamp = ts_match.group(0)
            
            if tile_content:  # Only add if we have content
                tiles.append({
                    'agent': agent_name,
                    'cycle': cycle_num,
                    'tile_content': tile_content,
                    'timestamp': timestamp or 'unknown'
                })
        else:
            i += 1
    
    return tiles
