def deadband_filter(values: List[float], p0_threshold: float, 
                    p1_min: float, p1_max: float) -> Optional[float]:
    """
    Apply the Deadband Protocol (P0/P1/P2) to filter and select an optimal value.
    
    P0: Remove values where abs(value) > p0_threshold (negative space/rocks).
    P1: Keep values within the safe channel [p1_min, p1_max].
    P2: From survivors, select value closest to channel midpoint.
    
    Args:
        values: List of input values.
        p0_threshold: Maximum absolute value for P0 survival.
        p1_min: Lower bound of safe channel.
        p1_max: Upper bound of safe channel.
    
    Returns:
        Optimal value or None if no values survive P0/P1.
    """
    # P0: Eliminate rocks
    p0_safe = [v for v in values if abs(v) <= p0_threshold]
    if not p0_safe:
        return None
    
    # P1: Keep within safe channel
    p1_safe = [v for v in p0_safe if p1_min <= v <= p1_max]
    if not p1_safe:
        return None
    
    # P2: Optimize - find closest to midpoint
    midpoint = (p1_min + p1_max) / 2
    optimal = min(p1_safe, key=lambda x: abs(x - midpoint))
    return optimal
