## **3. DeepSeek-chat Implementation**

```python
# Generated by DeepSeek-chat (self)

import re
from datetime import datetime
from math import sqrt, copysign

def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply deadband filtering to a list of values.
    
    Deadband protocol: If absolute value < deadband, return 0.
    Otherwise, return value minus deadband * sign(value).
    
    Args:
        values: List of float values
        deadband: Deadband threshold (default 0.1)
    
    Returns:
        Filtered list of values
    """
    filtered = []
    for val in values:
        if abs(val) < deadband:
            filtered.append(0.0)
        else:
            # Subtract deadband magnitude in direction of value
            filtered.append(val - copysign(deadband, val))
    return filtered


def parse_fleet_log(log_text: str) -> dict:
    """
    Parse fleet log format into structured data.
    
    Log format example:
    [2026-04-19 23:20:45] AGENT=Alchemist CYCLE=97 STATUS=ACTIVE TILES=4
    
    Returns dict with keys:
    - timestamp: datetime object
    - agent: str
    - cycle: int
    - status: str
    - tiles: int
    - message: str (optional)
    """
    # Extract timestamp
    timestamp_match = re.search(r'\[(.*?)\]', log_text)
    if not timestamp_match:
        raise ValueError("No timestamp found in log")
    
    timestamp_str = timestamp_match.group(1)
    timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
    
    # Extract key-value pairs
    result = {'timestamp': timestamp}
    
    # Parse AGENT=CYCLE=STATUS=TILES pattern
    kv_pattern = r'(\w+)=([^\s\]]+)'
    matches = re.findall(kv_pattern, log_text)
    
    for key, value in matches:
        key = key.lower()
        if key == 'agent':
            result['agent'] = value
        elif key == 'cycle':
            result['cycle'] = int(value)
        elif key == 'status':
            result['status'] = value
        elif key == 'tiles':
            result['tiles'] = int(value)
        elif key == 'message':
            result['message'] = value
    
    # Check for required fields
    required = ['agent', 'cycle', 'status', 'tiles']
    for req in required:
        if req not in result:
            raise ValueError(f"Missing required field: {req}")
    
    return result


def constraint_snap(point: tuple[float, float], 
                   constraints: list[tuple[tuple[float, float], tuple[float, float]]],
                   tolerance: float = 0.01) -> tuple[float, float]:
    """
    Snap a point to the nearest constraint line within tolerance.
    
    Args:
        point: (x, y) coordinates
        constraints: List of line segments as ((x1, y1), (x2, y2))
        tolerance: Maximum distance to snap (default 0.01)
    
    Returns:
        Snapped point coordinates
    """
    px, py = point
    best_point = point
    min_distance = float('inf')
    
    for (x1, y1), (x2, y2) in constraints:
        # Line vector
        dx = x2 - x1
        dy = y2 - y1
        line_length_sq = dx*dx + dy*dy
        
        if line_length_sq == 0:
            # Line is a point
            distance = sqrt((px - x1)**2 + (py - y1)**2)
            if distance < min_distance and distance <= tolerance:
                min_distance = distance
                best_point = (x1, y1)
            continue
        
        # Project point onto line
        t = ((px - x1) * dx + (py - y1) * dy) / line_length_sq
        t = max(0, min(1, t))  # Clamp to segment
        
        # Closest point on segment
        closest_x = x1 + t * dx
        closest_y = y1 + t * dy
        
        distance = sqrt((px - closest_x)**2 + (py - closest_y)**2)
        
        if distance < min_distance and distance <= tolerance:
            min_distance = distance
            best_point = (closest_x, closest_y)
    
    return best_point
```
