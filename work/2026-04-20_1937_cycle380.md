**Output:**
```python
from typing import List, Tuple

def deadband_filter(values: List[float], 
                    danger_threshold: float = 0.0,
                    safe_min: float = 0.3,
                    safe_max: float = 0.7) -> List[float]:
    """
    Apply the Deadband Protocol (P0/P1/P2) to a list of numeric values.
    
    P0 (negative space): Remove values below danger_threshold (unsafe).
    P1 (safe channels): Keep only values within [safe_min, safe_max].
    P2 (optimization): Apply simple normalization to remaining values.
    
    Args:
        values: Input list of float values.
        danger_threshold: Values below this are considered dangerous (P0).
        safe_min: Lower bound of safe channel (P1).
        safe_max: Upper bound of safe channel (P1).
    
    Returns:
        Filtered and optimized list of values.
    """
    # P0: Remove dangerous values
    safe_values = [v for v in values if v >= danger_threshold]
    
    # P1: Keep only values within safe channel
    channel_values = [v for v in safe_values if safe_min <= v <= safe_max]
    
    # P2: Optimize (normalize to 0-1 range within channel)
    if not channel_values:
        return []
    
    min_val = min(channel_values)
    max_val = max(channel_values)
    if max_val == min_val:
        return [0.5] * len(channel_values)  # Default midpoint
    
    optimized = [(v - min_val) / (max_val - min_val) for v in channel_values]
    return optimized
```

import json
from typing import Dict, Any, Tuple, List
from datetime import datetime

def tile_validator(tile_data: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile JSON structure against fleet schema.
    
    Required fields:
    - agent_id: str, non-empty
    - cycle: int, positive
    - content: str, non-empty
    - room_id: str, format 'room-XXX'
    - timestamp: str, ISO 8601 format
    
    Args:
        tile_data: Dictionary containing tile data.
    
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    """
    errors = []
    required_fields = ['agent_id', 'cycle', 'content', 'room_id', 'timestamp']
    
    # Check required fields exist
    for field in required_fields:
        if field not in tile_data:
            errors.append(f"Missing required field: {field}")
    
    if errors:
        return False, errors
    
    # Validate agent_id
    if not isinstance(tile_data['agent_id'], str) or not tile_data['agent_id'].strip():
        errors.append("agent_id must be non-empty string")
    
    # Validate cycle
    if not isinstance(tile_data['cycle'], int) or tile_data['cycle'] <= 0:
        errors.append("cycle must be positive integer")
    
    # Validate content
    if not isinstance(tile_data['content'], str) or not tile_data['content'].strip():
        errors.append("content must be non-empty string")
    
    # Validate room_id format
    if not isinstance(tile_data['room_id'], str) or not tile_data['room_id'].startswith('room-'):
        errors.append("room_id must start with 'room-' prefix")
    
    # Validate timestamp
    try:
        datetime.fromisoformat(tile_data['timestamp'].replace('Z', '+00:00'))
    except (ValueError, TypeError):
        errors.append("timestamp must be valid ISO 8601 format")
    
    return len(errors) == 0, errors
```

from typing import Dict, Any
from datetime import datetime, timedelta

def repo_health_score(repo_data: Dict[str, Any]) -> float:
    """
    Calculate health score (0-100) for a git repository.
    
    Factors considered:
    1. Commit activity last 30 days (weight: 40%)
    2. Issue resolution ratio (weight: 30%)
    3. Days since last commit (weight: 30%)
    
    Args:
        repo_data: Dictionary containing:
            - 'commits': list of commit dates (ISO strings)
            - 'open_issues': int
            - 'closed_issues': int
            - 'last_commit_date': str (ISO)
    
    Returns:
        Health score between 0 and 100.
    """
    # Default weights
    commit_weight = 0.4
    issue_weight = 0.3
    recency_weight = 0.3
    
    total_score = 0.0
    
    # 1. Commit activity score (0-100)
    thirty_days_ago = datetime.now() - timedelta(days=30)
    recent_commits = 0
    
    if 'commits' in repo_data and repo_data['commits']:
        for commit_date in repo_data['commits']:
            try:
                commit_dt = datetime.fromisoformat(commit_date.replace('Z', '+00:00'))
                if commit_dt >= thirty_days_ago:
                    recent_commits += 1
            except (ValueError, TypeError):
                continue
    
    # Normalize: 10+ commits in 30 days = 100 points
    commit_score = min(recent_commits * 10, 100)
    total_score += commit_score * commit_weight
    
    # 2. Issue resolution score (0-100)
    open_issues = repo_data.get('open_issues', 0)
    closed_issues = repo_data.get('closed_issues', 0)
    
    if open_issues + closed_issues > 0:
        resolution_ratio = closed_issues / (open_issues + closed_issues)
        issue_score = resolution_ratio * 100
    else:
        issue_score = 50  # Neutral score for no issues
    
    total_score += issue_score * issue_weight
    
    # 3. Recency score (0-100)
    if 'last_commit_date' in repo_data:
        try:
            last_commit = datetime.fromisoformat(
                repo_data['last_commit_date'].replace('Z', '+00:00')
            )
            days_since = (datetime.now() - last_commit).days
            # 0 days = 100, 30+ days = 0
            recency_score = max(0, 100 - (days_since * 3.33))
        except (ValueError, TypeError):
            recency_score = 0
    else:
        recency_score = 0
    
    total_score += recency_score * recency_weight
    
    return round(total_score, 2)
```
