```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of agent steps against the Deadband Protocol.
    The protocol requires strict priority: P0 (avoid) -> P1 (safe channels) -> P2 (optimize).
    Each step must be prefixed with P0:, P1:, or P2:.
    Sequence must follow priority order, though not all levels may be present.
    
    Args:
        steps: List of step strings like ["P0: check rocks", "P1: find channel"]
    
    Returns:
        Tuple of (is_valid, error_message). Error message empty if valid.
    """
    if not isinstance(steps, list):
        return False, "Input must be a list"
    
    allowed_prefixes = {"P0:", "P1:", "P2:"}
    last_seen_priority = -1  # -1 for none, 0 for P0, 1 for P1, 2 for P2
    
    for i, step in enumerate(steps):
        if not isinstance(step, str):
            return False, f"Step {i} must be a string"
        
        step_stripped = step.strip()
        if not any(step_stripped.startswith(prefix) for prefix in allowed_prefixes):
            return False, f"Step {i} must start with P0:, P1:, or P2:"
        
        # Determine priority index
        if step_stripped.startswith("P0:"):
            current_priority = 0
        elif step_stripped.startswith("P1:"):
            current_priority = 1
        else:  # P2:
            current_priority = 2
        
        # Check priority order
        if current_priority < last_seen_priority:
            return False, f"Protocol violation at step {i}: P{current_priority} after P{last_seen_priority}. Priority must not decrease."
        
        last_seen_priority = current_priority
    
    return True, ""
```

```python
def validate_deadband_protocol(steps):
    """
    Validate steps against Deadband Protocol.
    P0 (avoid negatives) must come before P1 (find safe channels),
    which must come before P2 (optimize within channels).
    Steps must be labeled with P0:, P1:, or P2:.
    """
    if not steps:
        return True, "No steps to validate"
    
    priorities = []
    for idx, step in enumerate(steps):
        if not step.startswith(("P0:", "P1:", "P2:")):
            return False, f"Step {idx} missing priority prefix: {step}"
        if step.startswith("P0:"):
            priorities.append(0)
        elif step.startswith("P1:"):
            priorities.append(1)
        elif step.startswith("P2:"):
            priorities.append(2)
    
    # Check non-decreasing order
    for i in range(1, len(priorities)):
        if priorities[i] < priorities[i - 1]:
            return False, f"Priority regression at step {i}: P{priorities[i]} after P{priorities[i-1]}"
    
    return True, ""
```

*   **Style:** DeepSeek's use of a `last_seen_priority` variable avoids building a separate list. Groq's approach is straightforward.

```python
import re
from datetime import datetime

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a raw fleet training tile text into a structured dictionary.
    Expected format (from cycle files):
        # Cycle X — AgentName
        **Phase Y: ...**
        **Task:** ...
        ---
        Content...
    
    Returns dict with keys: agent, cycle, phase, task, content, timestamp.
    Missing fields will be None.
    """
    result = {
        "agent": None,
        "cycle": None,
        "phase": None,
        "task": None,
        "content": None,
        "timestamp": datetime.utcnow().isoformat()  # default to now
    }
    
    if not tile_text or not isinstance(tile_text, str):
        return result
    
    lines = tile_text.strip().split('\n')
    
    # Parse header line: "# Cycle 379 — Alchemist"
    if lines and lines[0].startswith('# Cycle'):
        header_match = re.match(r'# Cycle\s+(\d+)\s*—\s*(\w+)', lines[0])
        if header_match:
            result["cycle"] = int(header_match.group(1))
            result["agent"] = header_match.group(2)
    
    # Look for Phase and Task lines (could be anywhere before content separator)
    for line in lines:
        if line.startswith('**Phase'):
            # Extract phase number: "**Phase 4: Build & Test**" -> 4
            phase_match = re.search(r'Phase\s+(\d+)', line)
            if phase_match:
                result["phase"] = int(phase_match.group(1))
        elif line.startswith('**Task:'):
            # Extract task description after "**Task:** "
            task_text = line.replace('**Task:**', '').strip()
            if task_text:
                result["task"] = task_text
    
    # Find content separator "---" and extract everything after it
    try:
        sep_index = lines.index('---')
        content_lines = lines[sep_index + 1:]
        result["content"] = '\n'.join(content_lines).strip()
    except ValueError:
        # No separator, use all lines after header as content
        result["content"] = '\n'.join(lines[1:] if len(lines) > 1 else []).strip()
    
    return result
```
