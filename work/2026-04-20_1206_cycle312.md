```python
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates a sequence of steps against the Deadband Protocol (P0, P1, P2).
    Rules:
    - P0 (map negative space) must come first if present.
    - P1 (find safe channels) must come before P2 (optimize).
    - P2 cannot appear without prior P1.
    Returns (is_valid, diagnostic_message).
    """
    priorities = []
    for step in steps:
        step_lower = step.strip().lower()
        if "p0" in step_lower or "negative" in step_lower:
            priorities.append(0)
        elif "p1" in step_lower or "safe" in step_lower or "channel" in step_lower:
            priorities.append(1)
        elif "p2" in step_lower or "optimize" in step_lower:
            priorities.append(2)
        else:
            # Step not classifiable; assume it's part of current phase.
            if priorities:
                priorities.append(priorities[-1])
            else:
                priorities.append(0)  # Default to P0 if unclear

    # Check ordering
    for i in range(1, len(priorities)):
        if priorities[i] < priorities[i - 1]:
            return False, f"Priority regression at step {i}: P{priorities[i-1]} -> P{priorities[i]}"
        if priorities[i] == 2 and 1 not in priorities[:i]:
            return False, f"P2 found at step {i} without preceding P1"

    return True, f"Valid sequence with priorities: {priorities}"
```

```python
import re
from datetime import datetime
from typing import Dict, Optional

def parse_fleet_tile(tile_text: str) -> Dict[str, str]:
    """
    Parses a fleet training tile text into a structured dictionary.
    Expected format includes headers like '# Cycle X — Agent', '**Task:**', etc.
    """
    result = {
        "agent": "unknown",
        "cycle": "unknown",
        "phase": "unknown",
        "task": "unknown",
        "content": "",
        "timestamp": datetime.utcnow().isoformat() + "Z"
    }

    # Extract agent and cycle
    cycle_match = re.search(r'# Cycle (\d+) — (\w+)', tile_text)
    if cycle_match:
        result["cycle"] = cycle_match.group(1)
        result["agent"] = cycle_match.group(2)

    # Extract phase
    phase_match = re.search(r'\*\*Phase (\d+):?\*\*', tile_text, re.IGNORECASE)
    if phase_match:
        result["phase"] = phase_match.group(1)

    # Extract task
    task_match = re.search(r'\*\*Task:\*\*\s*(.+?)(?=\n\n|\n\*\*|$)', tile_text, re.DOTALL)
    if task_match:
        result["task"] = task_match.group(1).strip()

    # Content is everything after the first header, cleaned
    lines = tile_text.strip().split('\n')
    content_start = 0
    for i, line in enumerate(lines):
        if line.startswith('# Cycle'):
            content_start = i + 1
            break
    if content_start < len(lines):
        result["content"] = '\n'.join(lines[content_start:]).strip()

    return result
```

**Function 3: `generate_constraint_snaps`**
```python
from typing import List, Tuple
import math

def generate_constraint_snaps(points: List[Tuple[float, float]], tolerance: float = 0.01) -> List[Tuple[int, int]]:
    """
    Finds pairs of points within a given Euclidean distance tolerance.
    Returns list of index pairs (i, j) where i < j and distance(points[i], points[j]) < tolerance.
    Uses a simple O(n^2) comparison; suitable for small n.
    """
    snaps = []
    n = len(points)
    for i in range(n):
        x1, y1 = points[i]
        for j in range(i + 1, n):
            x2, y2 = points[j]
            distance = math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)
            if distance < tolerance:
                snaps.append((i, j))
    return snaps
```

```python
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Checks if the steps follow Deadband Protocol ordering: P0 before P1 before P2.
    P0 = identify hazards/negative space
    P1 = find safe channels
    P2 = optimize within channels
    """
    p0_seen = False
    p1_seen = False
    p2_seen = False

    for idx, step in enumerate(steps):
        step_lower = step.lower()
        if any(word in step_lower for word in ["p0", "negative", "avoid", "hazard"]):
            p0_seen = True
            if p1_seen or p2_seen:
                return False, f"P0 step at index {idx} appears after P1/P2"
        elif any(word in step_lower for word in ["p1", "safe", "channel", "explore"]):
            p1_seen = True
            if p2_seen:
                return False, f"P1 step at index {idx} appears after P2"
            if not p0_seen:
                # P1 can start if P0 is implicit? Protocol strictness unclear.
                # Let's allow it but warn.
                pass
        elif any(word in step_lower for word in ["p2", "optimize", "improve", "tune"]):
            p2_seen = True
            if not p1_seen:
                return False, f"P2 step at index {idx} without preceding P1"

    # Final validation
    if p2_seen and not p1_seen:
        return False, "P2 present but no P1 found"
    # P1 without P0 is allowed (maybe P0 was empty)
    return True, f"Validation passed. P0:{p0_seen}, P1:{p1_seen}, P2:{p2_seen}"
```
