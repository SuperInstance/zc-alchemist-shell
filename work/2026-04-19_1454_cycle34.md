### **Function 1: deadband_filter**
```python
def deadband_filter(data, lower, upper):
    """
    Implements Deadband Protocol P0/P1/P2 logic on numeric array.
    
    P0: Values below lower bound -> map to None (negative space)
    P1: Values within deadband [lower, upper] -> map to 0 (safe channel)
    P2: Values above upper bound -> map to (value - upper) (optimize)
    
    Args:
        data: List[float] or np.ndarray
        lower: float, lower deadband threshold
        upper: float, upper deadband threshold
    
    Returns:
        List[float or None] filtered according to protocol
    """
    import numpy as np
    
    if not isinstance(data, (list, np.ndarray)):
        raise TypeError("data must be list or numpy array")
    
    if lower >= upper:
        raise ValueError("lower must be < upper")
    
    result = []
    for val in data:
        # P0: Map negative space
        if val < lower:
            result.append(None)
        # P1: Find safe channels
        elif lower <= val <= upper:
            result.append(0.0)
        # P2: Optimize within channels
        else:  # val > upper
            result.append(val - upper)
    
    return result
```

### **Function 2: parse_fleet_context**
```python
import re
from typing import Dict, List, Optional

def parse_fleet_context(markdown_text: str) -> Dict:
    """
    Parses fleet-knowledge markdown into structured data.
    
    Extracts:
    - Section headers and content
    - Code blocks with language
    - Key-value pairs from lists
    - Agent assignments from tables
    
    Returns:
        Dict with 'sections', 'code_blocks', 'agents', 'metadata'
    """
    if not markdown_text or not isinstance(markdown_text, str):
        return {"error": "Invalid input"}
    
    lines = markdown_text.split('\n')
    
    result = {
        "sections": [],
        "code_blocks": [],
        "agents": [],
        "metadata": {}
    }
    
    current_section = None
    in_code_block = False
    code_buffer = []
    code_lang = ""
    
    for line in lines:
        # Detect section headers
        if line.startswith('## '):
            if current_section:
                result["sections"].append(current_section)
            current_section = {"title": line[3:], "content": []}
        
        # Detect code blocks
        elif line.startswith('```'):
            if not in_code_block:
                in_code_block = True
                code_lang = line[3:] or "text"
                code_buffer = []
            else:
                in_code_block = False
                result["code_blocks"].append({
                    "language": code_lang,
                    "code": "\n".join(code_buffer)
                })
        elif in_code_block:
            code_buffer.append(line)
        
        # Detect agent assignments (pattern: **Agent**: description)
        elif re.match(r'^\*\*(\w+)\*\*: (.+)$', line):
            match = re.match(r'^\*\*(\w+)\*\*: (.+)$', line)
            result["agents"].append({
                "role": match.group(1),
                "task": match.group(2)
            })
        
        # Collect section content
        elif current_section and line.strip():
            current_section["content"].append(line)
    
    if current_section:
        result["sections"].append(current_section)
    
    return result
```

from typing import Dict, List, Any

def validate_training_tile(tile_json: Dict[str, Any]) -> Dict[str, Any]:
    """
    Validates a training tile against fleet submission schema.
    
    Required fields:
    - agent_id: str, must match pattern "zeroclaw-[a-z0-9]+"
    - cycle: int, 1-100
    - phase: int, 1-4
    - content: str, non-empty
    - timestamp: ISO 8601 format
    
    Optional but recommended:
    - task_id: str
    - metrics: dict
    - tags: list[str]
    
    Returns:
        Dict with 'valid': bool, 'errors': list[str], 'warnings': list[str]
    """
    errors = []
    warnings = []
    
    # Check required fields
    required = ['agent_id', 'cycle', 'phase', 'content', 'timestamp']
    for field in required:
        if field not in tile_json:
            errors.append(f"Missing required field: {field}")
    
    if errors:
        return {"valid": False, "errors": errors, "warnings": warnings}
    
    # Validate agent_id pattern
    if not re.match(r'^zeroclaw-[a-z0-9]{6,12}$', tile_json['agent_id']):
        errors.append("agent_id must match pattern: zeroclaw-[a-z0-9]{6,12}")
    
    # Validate cycle range
    if not isinstance(tile_json['cycle'], int) or not 1 <= tile_json['cycle'] <= 100:
        errors.append("cycle must be integer between 1 and 100")
    
    # Validate phase range
    if not isinstance(tile_json['phase'], int) or not 1 <= tile_json['phase'] <= 4:
        errors.append("phase must be integer between 1 and 4")
    
    # Validate content
    if not isinstance(tile_json['content'], str) or len(tile_json['content'].strip()) < 10:
        errors.append("content must be non-empty string with at least 10 characters")
    
    # Validate timestamp
    try:
        datetime.fromisoformat(tile_json['timestamp'].replace('Z', '+00:00'))
    except (ValueError, AttributeError):
        errors.append("timestamp must be valid ISO 8601 format")
    
    # Check for recommended fields
    recommended = ['task_id', 'metrics', 'tags']
    for field in recommended:
        if field not in tile_json:
            warnings.append(f"Missing recommended field: {field}")
    
    # Validate metrics if present
    if 'metrics' in tile_json and not isinstance(tile_json['metrics'], dict):
        warnings.append("metrics should be a dictionary")
    
    return {
        "valid": len(errors) == 0,
        "errors": errors,
        "warnings": warnings
    }

def deadband_filter(data, lower, upper):
    """
    Apply deadband filtering to data.
    
    Parameters:
    data: array-like of numbers
    lower: lower threshold
    upper: upper threshold
    
    Returns:
    filtered data according to deadband rules
    """
    result = []
    
    for x in data:
        if x < lower:
            # Below deadband - ignore
            result.append(None)
        elif x > upper:
            # Above deadband - pass through with offset
            result.append(x - upper)
        else:
            # Within deadband - zero
            result.append(0)
    
    return result
```
