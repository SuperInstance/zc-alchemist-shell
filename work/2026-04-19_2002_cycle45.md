### **Function 1: `deadband_filter`**

**DeepSeek-chat output:**
```python
from typing import List, Optional
import statistics

def deadband_filter(values: List[float], threshold: float) -> List[float]:
    """
    Implements Deadband Protocol filtering.
    
    P0: Avoid values within deadband of moving median (negative space).
    P1: Find safe channels -> keep values outside deadband.
    P2: Optimize -> return filtered list.
    
    Moving median is computed over a window of 3 (current, previous, next).
    Edge cases use available values.
    
    Args:
        values: List of numeric values.
        threshold: Deadband threshold (absolute).
        
    Returns:
        Filtered list where only values outside deadband around moving median are kept.
    """
    if not values:
        return []
    
    filtered = []
    n = len(values)
    
    for i in range(n):
        # Determine window indices
        start = max(0, i - 1)
        end = min(n, i + 2)  # i+2 because slice end is exclusive
        window = values[start:end]
        
        median_val = statistics.median(window)
        
        # Check if value is outside deadband
        if abs(values[i] - median_val) > threshold:
            filtered.append(values[i])
    
    return filtered
```

```python
from typing import List

def deadband_filter(values: List[float], threshold: float) -> List[float]:
    """
    Filter values based on deadband protocol.
    
    For each value, compute the median of a window around it (size 3).
    If the absolute difference between the value and its window median exceeds
    the threshold, keep the value.
    
    Args:
        values: A list of floats.
        threshold: The deadband threshold.
        
    Returns:
        A list of floats that are outside the deadband.
    """
    if len(values) == 0:
        return []
    
    result = []
    n = len(values)
    
    for i in range(n):
        # Define window: previous, current, next (if available)
        window_start = max(0, i - 1)
        window_end = min(n, i + 2)
        window = values[window_start:window_end]
        
        median_val = statistics.median(window)
        
        if abs(values[i] - median_val) > threshold:
            result.append(values[i])
            
    return result


if __name__ == "__main__":
    # Example usage
    sample_data = [1.0, 2.0, 1.5, 3.0, 1.4, 2.9, 1.6]
    deadband_threshold = 0.3
    filtered = deadband_filter(sample_data, deadband_threshold)
    print(f"Sample data: {sample_data}")
    print(f"Filtered data (threshold={deadband_threshold}): {filtered}")
```

```python
from typing import Dict, List, Any
import re

def parse_fleet_context(markdown_text: str) -> Dict[str, Any]:
    """
    Parse a fleet context markdown string into structured data.
    
    Extracts:
    - Sections (## headings)
    - Key technologies (bulleted items under "Key Technologies")
    - Rules (bulleted items under "Rules")
    
    Args:
        markdown_text: String containing markdown similar to FLEET-CONTEXT.md.
        
    Returns:
        Dictionary with keys: 'sections', 'technologies', 'rules'.
    """
    lines = markdown_text.split('\n')
    
    sections = []
    technologies = []
    rules = []
    
    current_section = None
    in_technologies = False
    in_rules = False
    
    for line in lines:
        # Detect section headers
        if line.startswith('## '):
            current_section = line[3:].strip()
            sections.append(current_section)
            
            # Check if this is the key technologies or rules section
            if 'key technology' in current_section.lower():
                in_technologies = True
                in_rules = False
            elif 'rules' in current_section.lower():
                in_rules = True
                in_technologies = False
            else:
                in_technologies = False
                in_rules = False
        
        # Collect bullet items in relevant sections
        elif line.strip().startswith('- '):
            item = line.strip()[2:].strip()
            if in_technologies:
                technologies.append(item)
            elif in_rules:
                rules.append(item)
    
    return {
        'sections': sections,
        'technologies': technologies,
        'rules': rules
    }
```
