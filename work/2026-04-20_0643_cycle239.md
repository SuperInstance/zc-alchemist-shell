#### **DeepSeek-chat Output (Simulated Self):**

from typing import List, Optional

def deadband_filter(values: List[float], lower_bound: float, upper_bound: float) -> Optional[float]:
    """
    Applies the Deadband Protocol (P0-P2) to a list of values.

    P0 (Negative Space): Filter out values not within the absolute bounds.
    P1 (Safe Channel): Keep values within the central 60% of the safe range.
    P2 (Optimize): Return the average of values in the P1 channel.

    Args:
        values: List of numeric values to filter.
        lower_bound: Absolute lower bound for P0.
        upper_bound: Absolute upper bound for P0.

    Returns:
        Average of values in the P1 channel, or None if no values survive P0 or P1.
    """
    # P0: Map negative space - exclude values outside absolute bounds
    p0_values = [v for v in values if lower_bound <= v <= upper_bound]
    if not p0_values:
        return None

    # Define P1 safe channel (middle 60% of the allowed range)
    range_width = upper_bound - lower_bound
    channel_lower = lower_bound + 0.2 * range_width
    channel_upper = upper_bound - 0.2 * range_width

    # P1: Find safe channel
    p1_values = [v for v in p0_values if channel_lower <= v <= channel_upper]
    if not p1_values:
        return None

    # P2: Optimize within channel
    return sum(p1_values) / len(p1_values)
```

Example: '2026-04-20T06:38:00Z | Alchemist | 239 | 4 | Starting model benchmark'. Return a dict with keys: timestamp, agent, cycle (int), phase (int), message. Handle malformed lines gracefully."

#### **DeepSeek-chat Output:**
```python
import re
from typing import Dict, Any

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parses a standardized fleet log line into its components.

    Expected format: 'TIMESTAMP | AGENT | CYCLE | PHASE | MESSAGE'
    Example: '2026-04-20T06:38:00Z | Alchemist | 239 | 4 | Starting model benchmark'

    Args:
        log_line: A single line from a fleet log.

    Returns:
        Dictionary with keys: 'timestamp' (str), 'agent' (str), 'cycle' (int),
        'phase' (int), 'message' (str). Returns a dict with 'error' key if parsing fails.
    """
    pattern = r'^(.+?) \| (.+?) \| (\d+) \| (\d+) \| (.+)$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        return {"error": f"Malformed log line: {log_line[:50]}..."}
    
    timestamp, agent, cycle_str, phase_str, message = match.groups()
    
    try:
        cycle = int(cycle_str)
        phase = int(phase_str)
    except ValueError:
        return {"error": f"Non-numeric cycle/phase in: {log_line[:50]}..."}
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "phase": phase,
        "message": message.strip()
    }
```

def parse_fleet_log(log_line):
    """
    Parse fleet log line to extract components.
    
    Format: TIMESTAMP | AGENT | CYCLE | PHASE | MESSAGE
    """
    parts = log_line.split('|')
    
    if len(parts) != 5:
        return {'error': 'Invalid format', 'input': log_line}
    
    timestamp = parts[0].strip()
    agent = parts[1].strip()
    
    try:
        cycle = int(parts[2].strip())
        phase = int(parts[3].strip())
    except ValueError:
        return {'error': 'Cycle or phase not numeric', 'input': log_line}
    
    message = parts[4].strip()
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': cycle,
        'phase': phase,
        'message': message
    }
