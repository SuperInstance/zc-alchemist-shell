from typing import List, Tuple

def deadband_filter(values: List[float], 
                    deadband_threshold: float = 0.1) -> List[float]:
    """
    Apply the Deadband Protocol (P0/P1/P2) to a list of float values.
    
    P0: Remove values that are NaN or infinite (negative space).
    P1: Keep values within safe channel [-deadband_threshold, deadband_threshold] as zero.
    P2: Optimize by rounding filtered values to 3 decimal places.
    
    Args:
        values: List of input float values.
        deadband_threshold: Absolute threshold for deadband (default 0.1).
    
    Returns:
        Filtered list of floats with deadband applied.
    """
    if not isinstance(values, list):
        raise TypeError("Input must be a list of floats.")
    
    filtered = []
    for v in values:
        # P0: Map negative space
        if not isinstance(v, (int, float)):
            continue  # skip non-numeric
        if v in (float('inf'), float('-inf')) or v != v:  # NaN check
            continue
        
        # P1: Find safe channel
        if -deadband_threshold <= v <= deadband_threshold:
            filtered.append(0.0)
        else:
            filtered.append(v)
    
    # P2: Optimize within channel
    filtered = [round(v, 3) for v in filtered]
    return filtered
```

def deadband_filter(values: list[float], threshold: float = 0.1) -> list[float]:
    """
    Implements the Deadband Protocol (P0/P1/P2) for signal filtering.
    
    P0: Remove invalid inputs (NaN, Inf).
    P1: Values within [-threshold, threshold] are clamped to zero.
    P2: Output is rounded to 3 decimal places.
    """
    import math
    
    if not values:
        return []
    
    result = []
    for val in values:
        # P0: negative space
        if not isinstance(val, (int, float)):
            continue
        if math.isnan(val) or math.isinf(val):
            continue
        
        # P1: safe channel
        if -threshold <= val <= threshold:
            result.append(0.0)
        else:
            result.append(val)
    
    # P2: optimize
    return [round(x, 3) for x in result]

import json
from typing import Dict, Any, List, Optional

def validate_fleet_message(message_json: str, 
                          schema_version: str = "v1") -> Dict[str, Any]:
    """
    Validate a fleet message JSON string against expected schema.
    
    Expected schema (v1):
    {
        "sender": str,           # e.g., "Oracle1", "JetsonClaw1"
        "recipient": str,        # fleet agent name
        "timestamp": str,        # ISO 8601
        "message_type": str,     # "command", "data", "log", "tile"
        "content": Dict[str, Any],
        "priority": int,         # 0=P0, 1=P1, 2=P2
        "cycle": int             # agent cycle number
    }
    
    Args:
        message_json: JSON string to validate.
        schema_version: Schema version to check against.
    
    Returns:
        Dict with keys:
        - "valid": bool
        - "errors": list of error strings (empty if valid)
        - "parsed": parsed dict (if valid)
    
    Raises:
        ValueError: If schema_version not supported.
    """
    if schema_version != "v1":
        raise ValueError(f"Unsupported schema version: {schema_version}")
    
    result = {"valid": False, "errors": [], "parsed": None}
    
    # Parse JSON
    try:
        data = json.loads(message_json)
    except json.JSONDecodeError as e:
        result["errors"].append(f"JSON decode error: {e}")
        return result
    
    # Required fields
    required = ["sender", "recipient", "timestamp", "message_type", "content", "priority", "cycle"]
    for field in required:
        if field not in data:
            result["errors"].append(f"Missing required field: {field}")
    
    # Type checks
    if "sender" in data and not isinstance(data["sender"], str):
        result["errors"].append("sender must be string")
    if "priority" in data and data["priority"] not in (0, 1, 2):
        result["errors"].append("priority must be 0, 1, or 2")
    if "cycle" in data and not isinstance(data["cycle"], int):
        result["errors"].append("cycle must be integer")
    
    # Message type enum
    valid_types = ["command", "data", "log", "tile"]
    if "message_type" in data and data["message_type"] not in valid_types:
        result["errors"].append(f"message_type must be one of {valid_types}")
    
    # Timestamp format (basic ISO check)
    if "timestamp" in data and isinstance(data["timestamp"], str):
        if "T" not in data["timestamp"]:
            result["errors"].append("timestamp should be ISO 8601 format")
    
    result["valid"] = len(result["errors"]) == 0
    if result["valid"]:
        result["parsed"] = data
    
    return result
```

import json

def validate_fleet_message(message: str) -> dict:
    """
    Validates a fleet message JSON.
    
    Required fields: sender, recipient, timestamp, msg_type, content, priority, cycle.
    Priority must be 0,1,2 (P0,P1,P2).
    Timestamp must contain 'T' (ISO 8601).
    msg_type must be in ['command','data','log','tile'].
    """
    try:
        data = json.loads(message)
    except Exception as e:
        return {"valid": False, "error": f"JSON parse failed: {e}"}
    
    errors = []
    
    required = ["sender", "recipient", "timestamp", "msg_type", "content", "priority", "cycle"]
    for r in required:
        if r not in data:
            errors.append(f"Missing {r}")
    
    if "priority" in data and data["priority"] not in [0,1,2]:
        errors.append("Priority must be 0,1,2")
    
    if "msg_type" in data and data["msg_type"] not in ["command","data","log","tile"]:
        errors.append("Invalid msg_type")
    
    if "timestamp" in data and isinstance(data["timestamp"], str):
        if 'T' not in data['timestamp']:
            errors.append("Timestamp not ISO 8601")
    
    if errors:
        return {"valid": False, "errors": errors}
    else:
        return {"valid": True, "data": data}
