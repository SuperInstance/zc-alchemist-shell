def deadband_filter(values: List[float], p0_thresh: float, p1_thresh: float, p2_thresh: float) -> List[str]:
    """

    Phases:
    - P0 (negative space): value <= p0_thresh
    - P1 (safe channel): p0_thresh < value <= p1_thresh
    - P2 (optimization zone): p1_thresh < value <= p2_thresh
    - OUT_OF_BOUNDS: value > p2_thresh

    Args:
        values: A list of float values.
        p0_thresh: The P0 threshold.
        p1_thresh: The P1 threshold.
        p2_thresh: The P2 threshold.

    Returns:
        A list of strings indicating the phase for each value.
    """
    # Validate thresholds
    if p0_thresh >= p1_thresh or p1_thresh >= p2_thresh:
        raise ValueError("Thresholds must satisfy p0 < p1 < p2")

    phases = []
    for val in values:
        if val <= p0_thresh:
            phases.append("P0")
        elif val <= p1_thresh:
            phases.append("P1")
        elif val <= p2_thresh:
            phases.append("P2")
        else:
            phases.append("OUT_OF_BOUNDS")
    return phases

from datetime import datetime
from typing import Optional, Dict, Any

def parse_fleet_log(log_line: str) -> Optional[Dict[str, Any]]:
    """
    Parse a standardized fleet log line.

    Expected format: 'YYYY-MM-DD HH:MM:SS | AGENT_NAME | CYCLE_NUM | STATUS | MESSAGE'

    Args:
        log_line: A single log line string.

    Returns:
        Dictionary with parsed fields, or None if line is malformed.
    """
    # Basic format check: must have exactly 5 pipe-separated parts
    parts = [part.strip() for part in log_line.split('|')]
    if len(parts) != 5:
        return None

    date_str, agent, cycle_str, status, message = parts

    try:
        # Parse timestamp
        timestamp = datetime.strptime(date_str, '%Y-%m-%d %H:%M:%S')
        # Parse cycle number
        cycle = int(cycle_str)
    except ValueError:
        return None

    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': cycle,
        'status': status,
        'message': message
    }
