**Function 3 spec:**
```

Each constraint is (a, b, c) representing line a*x + b*y + c = 0.
If the point's orthogonal distance to any constraint line is <= epsilon, snap the point
to that line (project onto line). If multiple constraints within epsilon, snap to the
nearest. Return the snapped (or original) coordinates.
```

```python
import re
from datetime import datetime
from typing import List, Tuple, Dict, Any
import math

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol to suppress small fluctuations.
    P0: No division by zero, no index errors on empty list.
    P1: Build safe channel by carrying forward values within deadband.
    P2: O(n) time, single pass.
    """
    if not values:
        return []
    
    filtered = [values[0]]
    for v in values[1:]:
        if abs(v - filtered[-1]) < deadband:
            filtered.append(filtered[-1])  # stay in channel
        else:
            filtered.append(v)
    return filtered

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a semi‑structured fleet log line into structured dict.
    P0: Handle missing fields, extra spaces, malformed brackets.
    P1: Use regex with named groups for robustness.
    P2: Return sensible defaults (None) for missing data.
    """
    pattern = r'(?P<timestamp>\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\s*\[(?P<level>\w+)\]\s*(.*)'
    match = re.match(pattern, log_line)
    if not match:
        return {}
    
    result = {
        'timestamp': datetime.strptime(match.group('timestamp'), '%Y-%m-%d %H:%M:%S'),
        'level': match.group('level')
    }
    
    # Parse key=value pairs
    rest = match.group(3)
    pairs = re.findall(r'(\w+)=(\S+)', rest)
    for key, val in pairs:
        if key in ('cycle', 'phase', 'tiles'):
            try:
                result[key] = int(val)
            except ValueError:
                result[key] = None
        else:
            result[key] = val
    
    # Ensure expected keys exist
    for field in ('agent', 'cycle', 'phase', 'tiles'):
        if field not in result:
            result[field] = None
    
    return result

def constraint_snap(point: Tuple[float, float],
                    constraints: List[Tuple[float, float, float]],
                    epsilon: float = 1e-3) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest constraint line within epsilon.
    P0: Avoid math domain errors, handle empty constraints.
    P1: Compute orthogonal distances safely.
    P2: Choose nearest constraint if multiple within epsilon.
    """
    x, y = point
    if not constraints:
        return (x, y)
    
    best_dist = float('inf')
    best_proj = (x, y)
    
    for a, b, c in constraints:
        norm = math.hypot(a, b)
        if norm == 0:
            continue  # invalid constraint, skip (P0)
        dist = abs(a * x + b * y + c) / norm
        if dist <= epsilon:
            # Project onto line
            t = -(a * x + b * y + c) / (a*a + b*b)
            proj_x = x + a * t
            proj_y = y + b * t
            if dist < best_dist:
                best_dist = dist
                best_proj = (proj_x, proj_y)
    
    return best_proj
```
