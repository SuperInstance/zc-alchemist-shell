from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol's strict P0->P1->P2 priority.

    Args:
        steps: List of step strings, e.g., ["P0", "P1", "P2"].

    Returns:
        Tuple of (is_valid: bool, error_message: str). Empty error string if valid.
    """
    expected_sequence = ["P0", "P1", "P2"]
    seen = set()

    for i, step in enumerate(steps):
        step_upper = step.upper()
        if step_upper not in expected_sequence:
            return False, f"Invalid step '{step}'. Must be one of {expected_sequence}."

        # Check for out-of-order or duplicate
        if step_upper in seen:
            return False, f"Duplicate step '{step_upper}' found."
        if i > 0:
            prev_step = steps[i-1].upper()
            if expected_sequence.index(step_upper) < expected_sequence.index(prev_step):
                return False, f"Step '{step_upper}' appears before its prerequisite '{prev_step}'."

        seen.add(step_upper)

    # Ensure all required steps are present (could be relaxed if protocol allows subsets?)
    # For strict completeness:
    if len(seen) < len(expected_sequence):
        missing = [s for s in expected_sequence if s not in seen]
        return False, f"Missing required step(s): {missing}"

    return True, ""
```

**Function 2: `parse_fleet_tile`**

import re
from typing import Dict, Union

def parse_fleet_tile(raw_text: str) -> Dict[str, Union[str, int, bool]]:
    """
    Parses raw text of a potential fleet training tile.

    Expected header format:
        # Cycle <number> — <AgentName>
        **Phase <number>: <Phase Name>**
        **Task:** <task description>

    Args:
        raw_text: The full tile text.

    Returns:
        Dictionary with keys: "agent", "cycle", "content", "is_valid_format".
    """
    result = {
        "agent": "",
        "cycle": -1,
        "content": raw_text.strip(),
        "is_valid_format": False
    }

    # Regex for the header
    header_pattern = r"^# Cycle\s+(\d+)\s*—\s*([^\n]+)\n\*\*Phase\s+\d+:.*?\*\*\n\*\*Task:\*\*\s*(.+?)(?:\n|$)"
    match = re.search(header_pattern, raw_text, re.MULTILINE | re.DOTALL)

    if match:
        result["cycle"] = int(match.group(1))
        result["agent"] = match.group(2).strip()
        # Optional: store task separately if needed
        result["is_valid_format"] = True
    else:
        # Fallback: try to find agent and cycle with simpler patterns
        cycle_match = re.search(r"Cycle\s+(\d+)", raw_text)
        agent_match = re.search(r"—\s*([^\n]+)", raw_text)
        if cycle_match:
            result["cycle"] = int(cycle_match.group(1))
        if agent_match:
            result["agent"] = agent_match.group(1).strip()
        # Consider it a valid format if we found at least a cycle and agent
        if result["cycle"] != -1 and result["agent"]:
            result["is_valid_format"] = True

    return result
```

**Function 3: `simulate_channel_optimization`**

from typing import List, Tuple

def simulate_channel_optimization(resources: List[float],
                                  constraints: List[Tuple[float, float]]) -> List[float]:
    """
    P2 Optimization: Allocates resources within safe channel constraints.

    Args:
        resources: Initial resource allocation values.
        constraints: List of (min, max) tuples for each channel.
                    Must be same length as resources.

    Returns:
        Optimized allocation where each resource is clamped to [min, max].
        If lengths mismatch, returns empty list.
    """
    if len(resources) != len(constraints):
        # P0: Don't proceed with invalid input
        return []

    optimized = []
    for res, (chan_min, chan_max) in zip(resources, constraints):
        # Clamp to channel bounds
        clamped = max(chan_min, min(res, chan_max))
        optimized.append(clamped)

    return optimized
```

def validate_deadband_protocol(steps):
    """
    Check if steps follow P0, P1, P2 order.
    """
    order = ["P0", "P1", "P2"]
    for i, step in enumerate(steps):
        if step not in order:
            return False, f"Bad step {step}"
        if i > 0:
            if order.index(step) < order.index(steps[i-1]):
                return False, f"Wrong order"
    return True, ""

def parse_fleet_tile(raw_text):
    """
    Parse fleet tile text.
    """
    lines = raw_text.split('\n')
    agent = None
    cycle = None
    for line in lines:
        if line.startswith('# Cycle'):
            parts = line.split('—')
            if len(parts) > 1:
                agent = parts[1].strip()
                cycle_part = parts[0].split()[2]
                cycle = int(cycle_part)
    content = raw_text
    is_valid = agent is not None and cycle is not None
    return {"agent": agent, "cycle": cycle, "content": content, "is_valid_format": is_valid}
```

**Function 3: `simulate_channel_optimization`**
```python
def simulate_channel_optimization(resources, constraints):
    """
    Optimize resources within constraints.
    """
    result = []
    for r, (low, high) in zip(resources, constraints):
        if r < low:
            result.append(low)
        elif r > high:
            result.append(high)
        else:
            result.append(r)
    return result
```
