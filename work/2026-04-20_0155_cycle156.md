### **Function 1: `deadband_filter`**

**Fleet Context:** "Deadband Protocol: P0 map negative space (what NOT to do), P1 find safe channels, P2 optimize within channels. Applied to sensor streams to filter noise while preserving meaningful transitions."

#### **DeepSeek-chat output:**
```python
from typing import List, Optional

def deadband_filter(
    readings: List[float],
    deadband_width: float = 0.1,
    initial_state: Optional[float] = None
) -> List[float]:
    """
    Apply deadband filtering to a sensor reading stream.
    
    Implements the fleet's Deadband Protocol:
    - P0: Ignore changes smaller than deadband_width (negative space)
    - P1: Track when reading exits deadband (safe channel)
    - P2: Output optimized filtered stream
    
    Args:
        readings: List of sensor readings
        deadband_width: Minimum change required to update output
        initial_state: Starting value (defaults to first reading)
    
    Returns:
        Filtered list where small oscillations are removed
    
    Example:
        >>> deadband_filter([1.0, 1.05, 1.12, 1.08], 0.1)
        [1.0, 1.0, 1.12, 1.12]
    """
    if not readings:
        return []
    
    filtered = []
    current = initial_state if initial_state is not None else readings[0]
    
    for reading in readings:
        if abs(reading - current) >= deadband_width:
            current = reading
        filtered.append(current)
    
    return filtered
```

```python
    Implements deadband filtering per Cocapn fleet protocol.

    initial_value: optional starting state

    return output
```

```python
from typing import Tuple, List
import math

Point = Tuple[float, float]
Constraint = Tuple[str, List[float]]  # e.g., ("line", [a, b, c])

def constraint_snap(
    point: Point,
    constraints: List[Constraint],
    max_iterations: int = 10
) -> Point:
    """
    Snap a 2D point to the nearest valid position given geometric constraints.
    
    Simplified version for fleet training. Supports:
    - "line": ax + by + c = 0 (snap to orthogonal projection)
    - "circle": [cx, cy, radius] (snap to nearest point on circumference)
    
    Args:
        point: (x, y) coordinate to snap
        constraints: List of constraint definitions
        max_iterations: For future iterative solving (not used in simple version)
    
    Returns:
        Snapped (x, y) coordinate
    
    Example:
        >>> constraint_snap((1, 1), [("line", [1, 0, -2])])
        (2.0, 1.0)  # snaps to x = 2 line
    """
    if not constraints:
        return point
    
    best_point = point
    min_distance = float('inf')
    
    for const_type, params in constraints:
        if const_type == "line":
            a, b, c = params
            # Orthogonal projection onto line
            denom = a*a + b*b
            if abs(denom) < 1e-10:
                continue
            x, y = point
            t = -(a*x + b*y + c) / denom
            snapped = (x + a*t, y + b*t)
            
        elif const_type == "circle":
            cx, cy, radius = params
            # Vector from center to point
            dx, dy = point[0] - cx, point[1] - cy
            dist = math.hypot(dx, dy)
            if dist < 1e-10:
                snapped = (cx + radius, cy)
            else:
                scale = radius / dist
                snapped = (cx + dx * scale, cy + dy * scale)
        
        else:
            continue
        
        # Distance from original point
        dist = math.hypot(snapped[0] - point[0], snapped[1] - point[1])
        if dist < min_distance:
            min_distance = dist
            best_point = snapped
    
    return best_point
```
