```python
from typing import List, Dict

def validate_deadband_protocol(actions: List[str]) -> Dict[str, bool]:
    """
    Validates a sequence of actions against the Deadband Protocol priorities.
    
    Args:
        actions: List of action strings with prefixes 'P0_', 'P1_', 'P2_'.
    
    Returns:
        Dict with three boolean flags:
        - p0_violation: True if any P0 action appears after a non-P0 action.
        - p1_missing: True if no P1 action is present in the list.
        - p2_without_p1: True if any P2 action appears before the first P1 action.
    """
    p0_seen = False
    p1_seen = False
    p2_without_p1_flag = False
    p0_violation_flag = False
    
    for action in actions:
        if action.startswith('P0_'):
            p0_seen = True
            # If we already saw a non-P0 action earlier, violation
            if any(not a.startswith('P0_') for a in actions[:actions.index(action)]):
                p0_violation_flag = True
        elif action.startswith('P1_'):
            p1_seen = True
        elif action.startswith('P2_'):
            if not p1_seen:
                p2_without_p1_flag = True
    
    return {
        "p0_violation": p0_violation_flag,
        "p1_missing": not p1_seen,
        "p2_without_p1": p2_without_p1_flag
    }
```

**Notes:** Logic flaw: `actions.index(action)` re-finds first occurrence, not current index. Could mislabel violations. Edge case handling is present but buggy.

### **Function 2: `parse_fleet_tile`**
```python
from typing import Optional, Dict
import datetime

def parse_fleet_tile(tile_text: str) -> Optional[Dict]:
    """
    Parses a fleet training tile string into structured fields.
    
    Expected format: "AGENT=CYCLE=PHASE=TASK=CONTENT=TIMESTAMP"
    Returns None if format is invalid or missing required fields.
    """
    if not tile_text or not isinstance(tile_text, str):
        return None
    
    parts = tile_text.split('=', maxsplit=5)
    if len(parts) != 6:
        return None
    
    agent, cycle_str, phase_str, task, content, timestamp_str = parts
    
    try:
        cycle = int(cycle_str)
        phase = int(phase_str)
        # Validate timestamp format (assuming ISO format)
        timestamp = datetime.datetime.fromisoformat(timestamp_str)
    except (ValueError, TypeError):
        return None
    
    return {
        "agent": agent,
        "cycle": cycle,
        "phase": phase,
        "task": task,
        "content": content,
        "timestamp": timestamp
    }
```

**Notes:** Clean, uses `maxsplit=5` correctly. Handles type conversions and malformed input. Good.

### **Function 3: `calculate_similarity_score`**
```python
from typing import Dict, Set

def jaccard_similarity(set1: Set, set2: Set) -> float:
    """Helper: Jaccard similarity between two sets."""
    if not set1 and not set2:
        return 1.0
    intersection = len(set1.intersection(set2))
    union = len(set1.union(set2))
    return intersection / union if union > 0 else 0.0

def calculate_similarity_score(tile1: Dict, tile2: Dict, weights: Dict[str, float]) -> float:
    """
    Calculates weighted Jaccard similarity between two parsed tile dicts.
    
    Args:
        tile1, tile2: Parsed tile dictionaries.
        weights: Dict mapping field names to weight values (sum optional).
    
    Returns:
        Weighted similarity score between 0.0 and 1.0.
    """
    total_score = 0.0
    total_weight = sum(weights.values())
    
    for field, weight in weights.items():
        if field not in tile1 or field not in tile2:
            continue
        
        val1 = tile1[field]
        val2 = tile2[field]
        
        if field in ["agent", "cycle", "phase"]:
            # Exact match for categorical fields
            similarity = 1.0 if val1 == val2 else 0.0
        else:
            # Text field: tokenize into sets (simple split)
            set1 = set(str(val1).lower().split())
            set2 = set(str(val2).lower().split())
            similarity = jaccard_similarity(set1, set2)
        
        total_score += similarity * weight
    
    return total_score / total_weight if total_weight > 0 else 0.0
```

**Notes:** Solid. Handles categorical vs text fields, normalization, missing fields. Good practices.

from typing import List, Dict

def validate_deadband_protocol(actions: List[str]) -> Dict[str, bool]:
    """
    Check Deadband Protocol compliance.
    P0: Must come first if present.
    P1: Must exist before any P2.
    P2: Cannot appear before P1.
    """
    has_p0 = False
    has_p1 = False
    p0_violation = False
    p2_without_p1 = False
    
    for i, action in enumerate(actions):
        if action.startswith('P0_'):
            has_p0 = True
            # Check if any non-P0 before this P0
            for prev in actions[:i]:
                if not prev.startswith('P0_'):
                    p0_violation = True
                    break
        elif action.startswith('P1_'):
            has_p1 = True
        elif action.startswith('P2_'):
            if not has_p1:
                p2_without_p1 = True
    
    return {
        "p0_violation": p0_violation,
        "p1_missing": not has_p1,
        "p2_without_p1": p2_without_p1
    }
