"""
Fleet utility functions for Alchemist model comparison.
"""

from typing import Dict, Any, Optional, Tuple
import re
from datetime import datetime
import math

def deadband_filter(data: float, lower: float, upper: float) -> float:
    """
    Apply a deadband filter to a single data point.
    
    Args:
        data: Input value
        lower: Lower deadband threshold
        upper: Upper deadband threshold
        
    Returns:
        Filtered value where:
        - If lower <= data <= upper: returns 0.0
        - If data < lower: returns data - lower
        - If data > upper: returns data - upper
        
    Example:
        >>> deadband_filter(5.0, 2.0, 8.0)
        0.0
        >>> deadband_filter(1.0, 2.0, 8.0)
        -1.0
    """
    if lower <= data <= upper:
        return 0.0
    elif data < lower:
        return data - lower
    else:  # data > upper
        return data - upper


def parse_fleet_log_line(line: str) -> Optional[Dict[str, Any]]:
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format:
        [timestamp] [LEVEL] key1=value1 key2=value2 ...
    
    Args:
        line: Log line string
        
    Returns:
        Dictionary with parsed fields, or None if format invalid.
        
    Example:
        >>> parse_fleet_log_line('[2026-04-20T05:21:00Z] [INFO] agent=Alchemist cycle=220')
        {
            'timestamp': datetime.datetime(2026, 4, 20, 5, 21, 0),
            'level': 'INFO',
            'agent': 'Alchemist',
            'cycle': 220
        }
    """
    pattern = r'\[([^\]]+)\] \[([^\]]+)\] (.*)'
    match = re.match(pattern, line.strip())
    if not match:
        return None
    
    timestamp_str, level, kv_pairs = match.groups()
    
    try:
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        timestamp = None
    
    result = {'timestamp': timestamp, 'level': level}
    
    # Parse key=value pairs
    for pair in kv_pairs.strip().split():
        if '=' in pair:
            key, value = pair.split('=', 1)
            # Try to convert numeric values
            try:
                if '.' in value:
                    value = float(value)
                else:
                    value = int(value)
            except ValueError:
                pass  # Keep as string
            result[key] = value
    
    return result


def constraint_snap(
    point: Tuple[float, float],
    line_start: Tuple[float, float],
    line_end: Tuple[float, float],
    tolerance: float
) -> Optional[Tuple[float, float]]:
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Args:
        point: (x, y) coordinate to snap
        line_start: (x, y) start of line segment
        line_end: (x, y) end of line segment
        tolerance: Maximum distance for snapping
        
    Returns:
        (x, y) snapped coordinates if within tolerance, None otherwise.
        
    Based on vector projection geometry.
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector from line_start to line_end
    dx, dy = x2 - x1, y2 - y1
    line_length_sq = dx*dx + dy*dy
    
    # If line segment is degenerate (zero length), snap to start if within tolerance
    if line_length_sq < 1e-12:
        dist = math.hypot(px - x1, py - y1)
        return (x1, y1) if dist <= tolerance else None
    
    # Project point onto line vector
    # Vector from line_start to point
    vx, vy = px - x1, py - y1
    t = (vx*dx + vy*dy) / line_length_sq
    
    # Clamp t to [0, 1] to stay on segment
    t = max(0.0, min(1.0, t))
    
    # Nearest point on segment
    nearest_x = x1 + t * dx
    nearest_y = y1 + t * dy
    
    # Check distance
    dist = math.hypot(px - nearest_x, py - nearest_y)
    
    return (nearest_x, nearest_y) if dist <= tolerance else None
