def deadband_filter(readings, p0_threshold=5.0, p1_margin=2.0):
    """
    Apply Deadband Protocol (P0/P1/P2) to a list of numeric readings.
    
    P0: Readings below p0_threshold are considered 'negative space' -> None
    P1: Readings within [p0_threshold, p0_threshold + p1_margin] are 'safe channels' -> keep as-is
    P2: Readings above p0_threshold + p1_margin are 'optimization zone' -> scale down by 0.8
    
    Args:
        readings: List of float readings
        p0_threshold: Minimum safe value (P0 boundary)
        p1_margin: Width of P1 safe channel
        
    Returns:
        List of processed values (None for P0)
    """
    processed = []
    p1_upper = p0_threshold + p1_margin
    
    for val in readings:
        if val < p0_threshold:
            processed.append(None)  # P0: negative space
        elif val <= p1_upper:
            processed.append(val)   # P1: safe channel
        else:
            processed.append(val * 0.8)  # P2: optimization zone
            
    return processed
```

def deadband_filter(values, deadband_low=5.0, deadband_high=7.0):
    """
    Implements a deadband filter with three zones:
    - Zone 0 (P0): values < deadband_low → discard (return None)
    - Zone 1 (P1): deadband_low <= values <= deadband_high → pass through
    - Zone 2 (P2): values > deadband_high → scale by 0.8
    
    Parameters:
        values: list of floats
        deadband_low: lower threshold for P0
        deadband_high: upper threshold for P1 (P2 starts above this)
    
    Returns:
        list of filtered values (with Nones for discarded)
    """
    result = []
    for v in values:
        if v < deadband_low:
            result.append(None)
        elif v <= deadband_high:
            result.append(v)
        else:
            result.append(v * 0.8)
    return result

import re
from datetime import datetime
from typing import List, Dict, Any

def parse_fleet_log(log_text: str) -> List[Dict[str, Any]]:
    """
    Parse Cocapn fleet log entries into structured events.
    
    Expected format: 
    [YYYY-MM-DD HH:MM:SS] [AGENT] [LEVEL] message (key=value, ...)
    
    Example:
    [2026-04-20 12:34:56] [Oracle1] [INFO] heartbeat (cycle=324, status=ACTIVE)
    
    Returns:
        List of dictionaries with keys: timestamp, agent, level, message, params
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] \[(\w+)\] \[(\w+)\] (.+?)(?: \((.+)\))?$'
    events = []
    
    for line in log_text.strip().split('\n'):
        match = re.match(pattern, line)
        if not match:
            continue
            
        timestamp_str, agent, level, message, params_str = match.groups()
        timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
        
        # Parse key=value parameters
        params = {}
        if params_str:
            for pair in params_str.split(', '):
                if '=' in pair:
                    key, value = pair.split('=', 1)
                    # Try to convert numeric values
                    try:
                        params[key] = int(value)
                    except ValueError:
                        try:
                            params[key] = float(value)
                        except ValueError:
                            params[key] = value
        
        events.append({
            'timestamp': timestamp,
            'agent': agent,
            'level': level,
            'message': message.strip(),
            'params': params
        })
    
    return events
```

def parse_fleet_log(log_content):
    """
    Extracts structured events from Cocapn fleet logs.
    
    Log format: [timestamp] [agent] [level] message (key1=val1, key2=val2)
    
    Returns a list of dicts with fields:
        time: datetime object
        agent: str
        level: str (INFO, WARN, ERROR, etc.)
        msg: str
        data: dict of key-value pairs from parentheses
    """
    entries = []
    # Regex to capture: [timestamp] [agent] [level] message (params)
    regex = r'\[([^\]]+)\] \[([^\]]+)\] \[([^\]]+)\] ([^(]+)(?: \((.+)\))?'
    
    for line in log_content.splitlines():
        line = line.strip()
        if not line:
            continue
        m = re.match(regex, line)
        if m:
            ts_str, agent, level, msg, param_str = m.groups()
            # Parse timestamp
            try:
                ts = datetime.strptime(ts_str, "%Y-%m-%d %H:%M:%S")
            except ValueError:
                ts = None
            # Parse parameters
            params = {}
            if param_str:
                for item in param_str.split(', '):
                    if '=' in item:
                        k, v = item.split('=', 1)
                        # Try to convert to int/float if possible
                        try:
                            v = int(v)
                        except ValueError:
                            try:
                                v = float(v)
                            except ValueError:
                                pass
                        params[k] = v
            entries.append({
                'time': ts,
                'agent': agent,
                'level': level,
                'msg': msg.strip(),
                'data': params
            })
    return entries

def constraint_snap(point, tolerance=0.1, axes=None):
    """
    Snap a 2D point to nearest axis within tolerance.
    
    Args:
        point: Tuple (x, y)
        tolerance: Maximum distance to snap
        axes: List of axes to consider, default is ['x', 'y']
    
    Returns:
        Tuple (snapped_x, snapped_y, snapped_axis)
        snapped_axis is None if no snap occurred
    """
    if axes is None:
        axes = ['x', 'y']
    
    x, y = point
    snapped_x, snapped_y = x, y
    snapped_axis = None
    
    # Check x-axis snap (y = 0)
    if 'x' in axes and abs(y) <= tolerance:
        snapped_y = 0.0
        snapped_axis = 'x'
    
    # Check y-axis snap (x = 0)
    if 'y' in axes and abs(x) <= tolerance:
        snapped_x = 0.0
        snapped_axis = 'y' if snapped_axis is None else 'both'
    
    return (snapped_x, snapped_y, snapped_axis)
