import re
from typing import Any

def deadband_filter(signal: list[float], p0_thresh: float, p1_thresh: float) -> list[str]:
    """
    Apply Deadband Protocol (P0/P1/P2) to a 1D signal.
    
    Args:
        signal: List of float values.
        p0_thresh: Threshold for P0 (danger zone).
        p1_thresh: Threshold for P1 (safe channel). Assumes p0_thresh > p1_thresh.
    
    Returns:
        List of protocol phase strings for each input value.
    """
    result = []
    for val in signal:
        abs_val = abs(val)
        if abs_val >= p0_thresh:
            result.append("P0")
        elif abs_val >= p1_thresh:
            result.append("P1")
        else:
            result.append("P2")
    return result

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a structured fleet log line into its components.
    
    Expected format: [TIMESTAMP] AGENT@CYCLE: "MESSAGE" (STATUS)
    Components may be missing or malformed.
    
    Args:
        log_line: A single line of fleet log.
    
    Returns:
        Dictionary with parsed components or an error dict.
    """
    # Regex to capture components (allowing for missing parts)
    pattern = r'\[([^\]]+)\]\s+([^@]+)@(\d+):\s*"([^"]*)"\s*\(([^)]*)\)'
    match = re.match(pattern, log_line)
    
    if match:
        timestamp, agent, cycle, message, status = match.groups()
        return {
            "timestamp": timestamp,
            "agent": agent.strip(),
            "cycle": cycle,
            "message": message,
            "status": status.strip()
        }
    else:
        # Fallback: try to extract what we can
        parts = {}
        # Timestamp
        ts_match = re.search(r'\[([^\]]+)\]', log_line)
        if ts_match:
            parts["timestamp"] = ts_match.group(1)
        # Agent and cycle
        ac_match = re.search(r'([^@]+)@(\d+)', log_line)
        if ac_match:
            parts["agent"] = ac_match.group(1).strip()
            parts["cycle"] = ac_match.group(2)
        # Message
        msg_match = re.search(r'"([^"]*)"', log_line)
        if msg_match:
            parts["message"] = msg_match.group(1)
        # Status
        stat_match = re.search(r'\(([^)]*)\)', log_line)
        if stat_match:
            parts["status"] = stat_match.group(1).strip()
        
        if len(parts) >= 4:  # Have all essential fields?
            # Ensure all keys exist
            for key in ["timestamp", "agent", "cycle", "message", "status"]:
                if key not in parts:
                    parts[key] = ""
            return parts
        else:
            return {"error": f"Failed to parse log line: {log_line}"}

def validate_training_tile(tile_data: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile dictionary against required schema.
    
    Required keys: agent_id, cycle, content, room_id (non-empty strings).
    Optional key: metadata (must be dict if present).
    
    Args:
        tile_data: Dictionary to validate.
    
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    """
    errors = []
    required_keys = ["agent_id", "cycle", "content", "room_id"]
    
    # Check required keys exist and are non-empty strings
    for key in required_keys:
        if key not in tile_data:
            errors.append(f"Missing required key: {key}")
        else:
            if not isinstance(tile_data[key], str):
                errors.append(f"Key '{key}' must be a string")
            elif not tile_data[key].strip():
                errors.append(f"Key '{key}' cannot be empty or whitespace")
    
    # Check metadata if present
    if "metadata" in tile_data:
        if not isinstance(tile_data["metadata"], dict):
            errors.append("Key 'metadata' must be a dictionary")
    
    # Check for extra keys? (Not required by spec, could be omitted)
    
    is_valid = len(errors) == 0
    return (is_valid, errors)
