def validate_deadband_protocol(priority_sequence):
    """Validate strict P0->P1->P2 ordering with no skips."""
    allowed = {"P0", "P1", "P2"}
    current_phase = 0  # 0: P0, 1: P1, 2: P2
    phase_map = {"P0": 0, "P1": 1, "P2": 2}
    
    for priority in priority_sequence:
        if priority not in allowed:
            return False
        phase = phase_map[priority]
        # Cannot go backwards in phase
        if phase < current_phase:
            return False
        # If phase jumps more than 1, it's a skip (e.g., P0 to P2)
        if phase > current_phase + 1:
            return False
        current_phase = phase
    return True

def parse_fleet_tile(raw_tile):
    """Parse a fleet training tile string into structured data."""
    # Default result structure
    result = {
        "agent_id": None,
        "cycle": None,
        "content_hash": None,
        "is_valid_format": False
    }
    
    try:
        parsed = {}
        for item in raw_tile.strip().split(';'):
            if '=' in item:
                key, value = item.split('=', 1)
                parsed[key.strip().upper()] = value.strip()
        
        # Check for required fields
        required = {"AGENT", "CYCLE", "CONTENT", "HASH"}
        if required.issubset(parsed.keys()):
            result["agent_id"] = parsed["AGENT"]
            result["cycle"] = int(parsed["CYCLE"])
            result["content_hash"] = parsed["HASH"]
            result["is_valid_format"] = True
            
    except (ValueError, KeyError, AttributeError):
        # If parsing fails, return default (is_valid_format=False)
        pass
    
    return result

def simulate_channel_optimization(obstacles, start, end):
    """
    Simplified P1/P2 path simulation.
    P1: Find safe points (Manhattan distance > 2 from any obstacle).
    P2: If start and end are safe, return straight-line path.
    """
    def manhattan_dist(p1, p2):
        return abs(p1[0] - p2[0]) + abs(p1[1] - p2[1])
    
    # P1: Check if points are in safe channels
    start_safe = all(manhattan_dist(start, obs) > 2 for obs in obstacles)
    end_safe = all(manhattan_dist(end, obs) > 2 for obs in obstacles)
    
    # P2: Optimize within channel (straight line if both safe)
    if start_safe and end_safe:
        return [start, end]
    
    # No safe channel found
    return []
