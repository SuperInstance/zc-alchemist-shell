import numpy as np

def deadband_filter(data, lower, upper):
    """
    Apply a deadband filter to input data.
    
    Parameters:
    data : list or numpy.ndarray
        Input data values.
    lower : float
        Lower bound of deadband.
    upper : float
        Upper bound of deadband.
        
    Returns:
    filtered : numpy.ndarray
        Data with values in [lower, upper] set to zero.
        
    Raises:
    ValueError: if lower > upper.
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
        
    # Convert to numpy array for vectorized operations
    arr = np.asarray(data, dtype=np.float64)
    
    # Create mask for values inside deadband
    inside_mask = (arr >= lower) & (arr <= upper)
    
    # Apply filter: inside -> 0, outside -> original value
    filtered = np.where(inside_mask, 0.0, arr)
    
    return filtered
```

import re

def parse_fleet_log_line(line):
    """
    Parse a fleet log line in format: [TIMESTAMP] SERVICE: LEVEL - message
    
    Parameters:
    line : str
        Log line string.
        
    Returns:
    dict : Parsed components with keys: timestamp, service, level, message.
        
    Raises:
    ValueError: if line does not match expected format.
    """
    pattern = r'^\[([^\]]+)\]\s+([^:]+):\s+([^\s-]+)\s+-\s+(.+)$'
    match = re.match(pattern, line.strip())
    
    if not match:
        raise ValueError(f"Line does not match fleet log format: {line}")
    
    timestamp, service, level, message = match.groups()
    
    return {
        'timestamp': timestamp.strip(),
        'service': service.strip(),
        'level': level.strip(),
        'message': message.strip()
    }
```

import numpy as np

def constraint_snap(point, constraints):
    """
    Snap a 2D point to the nearest constraint line.
    
    Parameters:
    point : tuple or list of two floats
        (x, y) coordinates.
    constraints : list of tuples
        Each tuple is (A, B, C) for line equation Ax + By + C = 0.
        
    Returns:
    snapped_point : numpy.ndarray
        (x, y) coordinates of nearest point on nearest constraint line.
        
    Raises:
    ValueError: if constraints list is empty.
    """
    if not constraints:
        raise ValueError("constraints list cannot be empty")
    
    x, y = point
    best_distance = float('inf')
    best_point = None
    
    for A, B, C in constraints:
        # Distance from point to line
        denom = np.sqrt(A**2 + B**2)
        if denom == 0:
            continue  # Invalid line, skip
        
        distance = abs(A*x + B*y + C) / denom
        
        # Find projected point on line
        # Using perpendicular projection formula
        t = -(A*x + B*y + C) / (A**2 + B**2)
        proj_x = x + A * t
        proj_y = y + B * t
        
        if distance < best_distance:
            best_distance = distance
            best_point = np.array([proj_x, proj_y])
    
    return best_point
```

import numpy as np

def constraint_snap(point, constraints):
    """
    Snap point to nearest constraint line.
    
    Args:
        point: tuple (x, y)
        constraints: list of (A, B, C) for line Ax + By + C = 0
        
    Returns:
        snapped point (x, y)
    """
    x, y = point
    min_dist = float('inf')
    snapped = None
    
    for A, B, C in constraints:
        # Avoid division by zero for degenerate lines
        if A == 0 and B == 0:
            continue
            
        # Distance from point to line
        dist = abs(A*x + B*y + C) / np.sqrt(A**2 + B**2)
        
        # Project point onto line
        denom = A**2 + B**2
        proj_x = x - A*(A*x + B*y + C) / denom
        proj_y = y - B*(A*x + B*y + C) / denom
        
        if dist < min_dist:
            min_dist = dist
            snapped = (proj_x, proj_y)
    
    if snapped is None:
        raise ValueError("No valid constraints provided")
        
    return snapped
```
