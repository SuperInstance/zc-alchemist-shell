```python
def deadband_filter(data: list[float], lower: float, upper: float) -> list[float]:
    """
    Apply a deadband filter to a list of data points.
    Values within the inclusive range [lower, upper] are set to 0.0.
    Values outside the range pass through unchanged.

    Args:
        data: List of floating-point values.
        lower: Lower bound of deadband.
        upper: Upper bound of deadband.

    Returns:
        Filtered list where values in deadband are zeroed.

    Raises:
        ValueError: If lower > upper.
    """
    if lower > upper:
        raise ValueError(f"Lower bound ({lower}) cannot be greater than upper bound ({upper})")

    result = []
    for value in data:
        if lower <= value <= upper:
            result.append(0.0)
        else:
            result.append(value)
    return result
```

```python
def parse_fleet_log_line(line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line.
    Expected format: '[timestamp] [LEVEL] key1=value1 key2=value2 ...'

    Args:
        line: A log line string.

    Returns:
        Dictionary with parsed fields including 'timestamp', 'level', and key-value pairs.
        Returns empty dict for unparsable lines.
    """
    pattern = r'^\[([^\]]+)\] \[([^\]]+)\] (.*)$'
    match = re.match(pattern, line.strip())
    if not match:
        return {}

    timestamp_str, level, kv_string = match.groups()
    try:
        # Try ISO format parsing
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        timestamp = timestamp_str  # Fallback to string

    result = {'timestamp': timestamp, 'level': level}

    # Parse key=value pairs
    for part in kv_string.split():
        if '=' in part:
            key, value = part.split('=', 1)
            # Try to convert numeric values
            try:
                if '.' in value:
                    value = float(value)
                else:
                    value = int(value)
            except ValueError:
                pass  # Keep as string
            result[key] = value

    return result
```

```python
import re
from datetime import datetime
from typing import Dict, Any

def parse_fleet_log_line(line: str) -> Dict[str, Any]:
    """
    Parses a fleet log line of format:
    [YYYY-MM-DDTHH:MM:SSZ] [LEVEL] key1=value1 key2=value2 ...

    Returns a dictionary with keys: timestamp (datetime), level (str), and extracted key-value pairs.
    If line doesn't match expected format, returns empty dict.
    """
    line = line.strip()
    # Regex to capture timestamp, level, and the rest
    match = re.match(r'^\[([^\]]+)\] \[([^\]]+)\] (.+)$', line)
    if not match:
        return {}

    ts_str, level, kv_str = match.groups()

    # Parse timestamp
    try:
        # Remove Z and parse as ISO format
        if ts_str.endswith('Z'):
            ts_str = ts_str[:-1] + '+00:00'
        timestamp = datetime.fromisoformat(ts_str)
    except Exception:
        timestamp = ts_str  # fallback to string if parsing fails

    result = {
        'timestamp': timestamp,
        'level': level
    }

    # Parse key=value pairs
    for token in kv_str.split():
        if '=' in token:
            k, v = token.split('=', 1)
            # Try to convert to int or float if possible
            try:
                if '.' in v:
                    v = float(v)
                else:
                    v = int(v)
            except ValueError:
                pass  # remain as string
            result[k] = v

    return result
```

**Analysis:**
*   **Correctness:** Both implement the core logic correctly. DeepSeek's timestamp handling (`replace('Z', '+00:00')`) is cleaner than Llama's string manipulation.
*   **Robustness:** Both handle malformed lines by returning `{}`. Both attempt type conversion for values.
*   **Clarity:** DeepSeek's regex pattern is slightly more readable (`[^\]]+` vs `[^\]]+` is the same). Llama's docstring is more specific about the expected timestamp format.
*   **Edge:** DeepSeek's fallback for timestamp parsing keeps the original string, which is good for debugging.
*   **Verdict:** Slight edge to **DeepSeek-chat** for more elegant timestamp handling and a cleaner fallback strategy.
