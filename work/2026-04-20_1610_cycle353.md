```python
def deadband_filter(data, lower, upper):
    """
    Apply a deadband filter to input data.
    
    Parameters:
    data (float or list): Input value or list of values
    lower (float): Lower deadband threshold
    upper (float): Upper deadband threshold
    
    Returns:
    float or list: Filtered output where values within [lower, upper] 
                   are mapped to 0, others pass through linearly.
    
    Example:
    >>> deadband_filter([1, 2, 3, 4, 5], 2, 4)
    [1, 0, 0, 0, 5]
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    # Handle single value
    if isinstance(data, (int, float)):
        if lower <= data <= upper:
            return 0.0
        return float(data)
    
    # Handle list
    result = []
    for value in data:
        if lower <= value <= upper:
            result.append(0.0)
        else:
            result.append(float(value))
    return result
```

### **Function 2: parse_fleet_log**
```python
import re
from datetime import datetime

def parse_fleet_log(log_text):
    """
    Parse a structured fleet log entry.
    
    Expected format: "[TIMESTAMP] AGENT:CYCLE - MESSAGE"
    Example: "[2026-04-20 16:30:42] Alchemist:353 - Testing deadband filter"
    
    Parameters:
    log_text (str): Log entry string
    
    Returns:
    dict: Parsed components with keys: timestamp, agent, cycle, message
          Returns None if format doesn't match
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+):(\d+) - (.+)'
    match = re.match(pattern, log_text)
    
    if not match:
        return None
    
    timestamp_str, agent, cycle_str, message = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
        cycle = int(cycle_str)
    except ValueError:
        return None
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "message": message.strip()
    }
```

### **Function 3: constraint_snap**

```python
import math

def constraint_snap(point, constraints):
    """
    Snap a 2D point to the nearest constraint line.
    
    Parameters:
    point (tuple): (x, y) coordinates
    constraints (list): List of line constraints as (A, B, C) for Ax + By + C = 0
    
    Returns:
    tuple: (snapped_x, snapped_y, constraint_index) or original point if no constraints
    
    Example:
    >>> constraint_snap((1, 1), [(1, 0, -2), (0, 1, -3)])
    (2.0, 1.0, 0)  # Snaps to vertical line x=2
    """
    if not constraints:
        return (*point, -1)
    
    x, y = point
    min_distance = float('inf')
    best_snap = point
    best_idx = -1
    
    for idx, (A, B, C) in enumerate(constraints):
        # Distance from point to line |Ax + By + C| / sqrt(A² + B²)
        denominator = math.sqrt(A*A + B*B)
        if denominator == 0:
            continue  # Invalid constraint
        
        distance = abs(A*x + B*y + C) / denominator
        
        # Calculate snapped point (orthogonal projection)
        if A == 0 and B == 0:
            continue  # Invalid
        
        # Projection formula
        t = -(A*x + B*y + C) / (A*A + B*B)
        snapped_x = x + A * t
        snapped_y = y + B * t
        
        if distance < min_distance:
            min_distance = distance
            best_snap = (snapped_x, snapped_y)
            best_idx = idx
    
    return (*best_snap, best_idx)
```

```python
def deadband_filter(data, lower, upper):
    """
    Implements a deadband filter. If data is within [lower, upper], output is zero.
    Otherwise, output equals input.
    
    Args:
        data: scalar or list of values
        lower: lower bound of deadband
        upper: upper bound of deadband
    
    Returns:
        filtered data
    """
    if lower > upper:
        raise ValueError("lower must be less than or equal to upper")
    
    if isinstance(data, list):
        return [0.0 if lower <= v <= upper else float(v) for v in data]
    else:
        return 0.0 if lower <= data <= upper else float(data)
```

```python
import re
from datetime import datetime

def parse_fleet_log(log_text):
    """
    Parses a fleet log line.
    
    Expected format: [YYYY-MM-DD HH:MM:SS] AGENT:CYCLE - MESSAGE
    
    Returns dict with keys: timestamp (datetime), agent (str), cycle (int), message (str).
    Returns None if parsing fails.
    """
    pattern = r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+):(\d+) - (.+)$'
    m = re.match(pattern, log_text)
    if not m:
        return None
    
    ts_str, agent, cycle_str, msg = m.groups()
    try:
        ts = datetime.strptime(ts_str, '%Y-%m-%d %H:%M:%S')
        cycle = int(cycle_str)
    except Exception:
        return None
    
    return {
        'timestamp': ts,
        'agent': agent,
        'cycle': cycle,
        'message': msg
    }
```
