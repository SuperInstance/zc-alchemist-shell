```python
from typing import List, Tuple

def deadband_filter(readings: List[float], deadband: float) -> List[float]:
    """
    Apply the Deadband Protocol (P0: avoid, P1: find safe channels, P2: optimize).
    Filters a list of sensor readings, returning only values that fall within
    'safe channels' defined as being outside a +/- deadband zone around zero.
    This models P1: finding where you CAN be.

    Args:
        readings: List of raw sensor readings.
        deadband: Absolute threshold. Readings with absolute value <= deadband are filtered out.

    Returns:
        Filtered list of readings where abs(value) > deadband.
    """
    if deadband < 0:
        raise ValueError("Deadband must be non-negative.")
    # P0: Map negative space - filter out values within the deadband.
    filtered = [val for val in readings if abs(val) > deadband]
    return filtered
```

```python
def deadband_filter(signal: list[float], threshold: float) -> list[float]:
    """
    Implements a deadband filter on a time-series signal.
    All values whose absolute magnitude is less than or equal to the threshold
    are set to zero, representing the 'dead' zone. Other values are passed through.

    Parameters:
        signal (list[float]): Input list of signal values.
        threshold (float): Deadband threshold (non-negative).

    Returns:
        list[float]: Filtered signal.
    """
    if threshold < 0:
        raise ValueError("Threshold cannot be negative")

    output = []
    for x in signal:
        if abs(x) <= threshold:
            output.append(0.0)  # Represent deadband region explicitly
        else:
            output.append(x)
    return output

# Example
```

```python
import re
from typing import Optional, Dict

def parse_fleet_log(log_line: str) -> Optional[Dict[str, str]]:
    """
    Parse a standardized fleet log line.
    Expected format: '[YYYY-MM-DD HH:MM:SS] [AGENT] MESSAGE'

    Args:
        log_line: A single line from the fleet log.

    Returns:
        Dictionary with keys 'timestamp', 'agent', 'message' if parse succeeds,
        otherwise None.
    """
    pattern = r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] \[(\w+)\] (.+)$'
    match = re.match(pattern, log_line.strip())
    if match:
        return {
            'timestamp': match.group(1),
            'agent': match.group(2),
            'message': match.group(3)
        }
    return None

# Example
```

```python
import datetime

def parse_fleet_log(entry: str) -> dict:
    """
    Parses a fleet log entry string into its components.

    The log format is: TIMESTAMP | AGENT | LEVEL | MESSAGE
    Example: "2026-04-20T05:54:00Z | Alchemist | INFO | Cycle 229 started"

    Returns a dictionary with keys: time, agent, level, text.
    If parsing fails, returns an empty dict.
    """
    parts = entry.split(" | ", maxsplit=3)
    if len(parts) != 4:
        return {}

    try:
        dt = datetime.datetime.fromisoformat(parts[0].replace('Z', '+00:00'))
    except ValueError:
        dt = parts[0]

    return {
        "timestamp": dt,
        "agent": parts[1],
        "level": parts[2],
        "message": parts[3]
    }
```

```python
from typing import Dict, Any, List, Tuple

def validate_training_tile(tile_data: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile dictionary against fleet requirements.
    Required fields: 'id' (str), 'agent' (str), 'cycle' (int), 'content' (str),
    'phase' (int, 1-4), 'timestamp' (str ISO format).

    Args:
        tile_data: Dictionary containing tile data.

    Returns:
        Tuple of (is_valid: bool, errors: list of error messages).
    """
    required = {
        'id': str,
        'agent': str,
        'cycle': int,
        'content': str,
        'phase': int,
        'timestamp': str
    }
    errors = []

    for field, expected_type in required.items():
        if field not in tile_data:
            errors.append(f"Missing required field: '{field}'")
        elif not isinstance(tile_data[field], expected_type):
            errors.append(f"Field '{field}' must be of type {expected_type.__name__}")

    # Additional semantic checks
    if 'phase' in tile_data and isinstance(tile_data['phase'], int):
        if not 1 <= tile_data['phase'] <= 4:
            errors.append("Field 'phase' must be between 1 and 4 (inclusive).")

    if 'timestamp' in tile_data and isinstance(tile_data['timestamp'], str):
        # Simple ISO format check (basic)
        if len(tile_data['timestamp']) < 10 or 'T' not in tile_data['timestamp']:
            errors.append("Field 'timestamp' should be in ISO 8601 format (e.g., '2026-04-20T05:54:00Z').")

    is_valid = len(errors) == 0
    return (is_valid, errors)
```
