def deadband_filter(values: list[float], deadband: tuple[float, float], p0_threshold: float) -> list[str]:
    """
    Applies the Deadband Protocol (P0, P1, P2) to a list of sensor readings.

    Args:
        values: List of floating-point sensor readings.
        deadband: A tuple (lower, upper) defining the 'safe channel' (P1 region).

    Returns:
        A list of strings, same length as `values`, with classifications:
        - "P0": |value| > p0_threshold (Negative Space / ROCK)
        - "P1": lower <= value <= upper (Safe Channel)
        - "P2": otherwise (Optimization Zone)
    """
    lower, upper = deadband
    results = []
    for v in values:
        if abs(v) > p0_threshold:
            results.append("P0")
        elif lower <= v <= upper:
            results.append("P1")
        else:
            results.append("P2")

import re
from datetime import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parses a single line from a fleet agent's state log.
    Expected format: 'YYYY-MM-DD HH:MM:SS | AGENT_NAME | STATUS | CYCLE: N | MSG: free text'

    Args:
        log_line: A string in the fleet log format.

    Returns:
        A dictionary with keys: 'timestamp', 'agent', 'status', 'cycle', 'message'.
        Returns None if the line cannot be parsed.
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \| (\w+) \| (\w+) \| CYCLE: (\d+) \| MSG: (.+)'
    match = re.match(pattern, log_line.strip())
    if not match:
        return None
    ts_str, agent, status, cycle_str, message = match.groups()
    try:
        timestamp = datetime.strptime(ts_str, '%Y-%m-%d %H:%M:%S')
        cycle = int(cycle_str)
    except ValueError:
        return None
    return {
        'timestamp': timestamp,
        'agent': agent,
        'status': status,
        'cycle': cycle,
        'message': message.strip()
    }
