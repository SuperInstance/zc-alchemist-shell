/// Parse a structured log line into a `LogEntry`.
///
/// Expected format: "YYYY-MM-DD HH:MM:SS LEVEL [MODULE] MESSAGE"
/// Returns `Err` if the line does not match the expected format.
#[derive(Debug, PartialEq)]
pub struct LogEntry {
    pub timestamp: String, // For simplicity; could be chrono::DateTime
    pub level: String,
    pub module: String,
    pub message: String,
}

pub fn parse_log_line(line: &str) -> Result<LogEntry, String> {
    let parts: Vec<&str> = line.splitn(5, ' ').collect();
    if parts.len() != 5 {
        return Err(format!("Invalid log line format: {}", line));
    }

    let timestamp = format!("{} {}", parts[0], parts[1]);
    let level = parts[2].to_string();
    
    // Extract module from brackets
    let module_part = parts[3];
    if !module_part.starts_with('[') || !module_part.ends_with(']') {
        return Err(format!("Module not in brackets: {}", module_part));
    }
    let module = module_part[1..module_part.len() - 1].to_string();
    
    let message = parts[4].to_string();

    Ok(LogEntry {
        timestamp,
        level,
        module,
        message,
    })
}

#[cfg(test)]

use chrono::{DateTime, NaiveDateTime};
use thiserror::Error;

#[derive(Debug, PartialEq)]
pub struct LogEntry {
    pub timestamp: DateTime<chrono::Utc>,
    pub level: LogLevel,
    pub module: String,
    pub message: String,
}

#[derive(Debug, PartialEq)]
pub enum LogLevel {
    Error,
    Warn,
    Info,
    Debug,
    Trace,
}

impl std::str::FromStr for LogLevel {
    type Err = ParseError;

    fn from_str(s: &str) -> Result<Self, Self::Err> {
        match s.to_uppercase().as_str() {
            "ERROR" => Ok(LogLevel::Error),
            "WARN" => Ok(LogLevel::Warn),
            "INFO" => Ok(LogLevel::Info),
            "DEBUG" => Ok(LogLevel::Debug),
            "TRACE" => Ok(LogLevel::Trace),
            _ => Err(ParseError::InvalidLevel(s.to_string())),
        }
    }
}

#[derive(Error, Debug)]
pub enum ParseError {
    #[error("Invalid timestamp format: {0}")]
    InvalidTimestamp(String),
    #[error("Invalid log level: {0}")]
    InvalidLevel(String),
    #[error("Missing module brackets")]
    MissingModuleBrackets,
    #[error("Malformed log line: {0}")]
    MalformedLine(String),
}

pub fn parse_log_line(line: &str) -> Result<LogEntry, ParseError> {
    let mut parts = line.splitn(5, ' ');
    let date = parts.next().ok_or_else(|| ParseError::MalformedLine(line.to_string()))?;
    let time = parts.next().ok_or_else(|| ParseError::MalformedLine(line.to_string()))?;
    let level_str = parts.next().ok_or_else(|| ParseError::MalformedLine(line.to_string()))?;
    let module_with_brackets = parts.next().ok_or_else(|| ParseError::MalformedLine(line.to_string()))?;
    let message = parts.next().ok_or_else(|| ParseError::MalformedLine(line.to_string()))?;

    let timestamp_str = format!("{} {}", date, time);
    let naive = NaiveDateTime::parse_from_str(&timestamp_str, "%Y-%m-%d %H:%M:%S")
        .map_err(|e| ParseError::InvalidTimestamp(e.to_string()))?;
    let timestamp = DateTime::from_naive_utc_and_offset(naive, chrono::Utc);

    let level = level_str.parse()?;

    if !module_with_brackets.starts_with('[') || !module_with_brackets.ends_with(']') {
        return Err(ParseError::MissingModuleBrackets);
    }
    let module = module_with_brackets[1..module_with_brackets.len() - 1].to_string();

    Ok(LogEntry {
        timestamp,
        level,
        module,
        message: message.to_string(),
    })
}

#[cfg(test)]
mod tests {
    use super::*;
    use chrono::TimeZone;

    #[test]
    fn test_parse_log_line() {
        let line = "2026-04-19 12:52:34 INFO [network] Connection established";
        let entry = parse_log_line(line).unwrap();
        assert_eq!(entry.timestamp, chrono::Utc.with_ymd_and_hms(2026, 4, 19, 12, 52, 34).unwrap());
        assert_eq!(entry.level, LogLevel::Info);
        assert_eq!(entry.module, "network");
        assert_eq!(entry.message, "Connection established");
    }
}

**Analysis for Function 1:**
*   **DeepSeek:** Simpler, no external dependencies. Returns a basic struct with `String` fields. Error handling is a simple `String`. Good for a quick utility. Test is minimal but correct.
*   **Groq:** More production-ready. Uses `chrono` for proper datetime parsing, an enum for `LogLevel`, and a custom `ParseError` enum with `thiserror`. More robust and idiomatic, but adds dependencies. Test uses chrono's API correctly.
*   **Verdict:** Groq's output is higher quality for a library. DeepSeek's is acceptable for a simple script.
