import json
import re
from datetime import datetime
from math import isinf, isnan

def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol to a list of float values.
    
    P0: Remove values that are NaN or infinite (negative space).
    P1: Keep values within [-deadband, +deadband] as zero (safe channel).
    P2: For values outside deadband, subtract deadband*sign(value) (optimize).
    
    Returns filtered list with same length, None for removed values.
    """
    result = []
    for val in values:
        # P0: Map negative space
        if isnan(val) or isinf(val):
            result.append(None)
            continue
            
        # P1: Find safe channel
        if -deadband <= val <= deadband:
            result.append(0.0)
        # P2: Optimize within channels
        else:
            if val > 0:
                result.append(val - deadband)
            else:
                result.append(val + deadband)
    return result

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a fleet log line with format:
    'YYYY-MM-DD HH:MM:SS | AGENT | CYCLE | STATUS | MESSAGE'
    
    Returns dict with keys: timestamp, agent, cycle, status, message.
    Handles malformed lines gracefully.
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \| (\w+) \| (\d+) \| (\w+) \| (.+)'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        # Return minimal structure for malformed lines
        return {
            'timestamp': None,
            'agent': 'UNKNOWN',
            'cycle': 0,
            'status': 'ERROR',
            'message': log_line
        }
    
    timestamp_str, agent, cycle_str, status, message = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
        cycle = int(cycle_str)
    except (ValueError, TypeError):
        timestamp = None
        cycle = 0
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': cycle,
        'status': status,
        'message': message.strip()
    }

def tile_validator(tile_json: dict, schema_version: str = "v1") -> tuple[bool, list[str]]:
    """
    Validate a training tile JSON against fleet schema.
    
    Required fields: id, agent, cycle, phase, content, timestamp.
    Content must be non-empty string.
    Returns (is_valid, list_of_errors).
    """
    errors = []
    required_fields = ['id', 'agent', 'cycle', 'phase', 'content', 'timestamp']
    
    # Check required fields
    for field in required_fields:
        if field not in tile_json:
            errors.append(f"Missing required field: {field}")
    
    # Validate field types and constraints
    if 'id' in tile_json and not isinstance(tile_json['id'], str):
        errors.append("Field 'id' must be string")
    
    if 'agent' in tile_json and not isinstance(tile_json['agent'], str):
        errors.append("Field 'agent' must be string")
    
    if 'cycle' in tile_json:
        if not isinstance(tile_json['cycle'], int):
            errors.append("Field 'cycle' must be integer")
        elif tile_json['cycle'] < 0:
            errors.append("Field 'cycle' must be non-negative")
    
    if 'phase' in tile_json:
        if not isinstance(tile_json['phase'], int):
            errors.append("Field 'phase' must be integer")
        elif tile_json['phase'] not in [1, 2, 3, 4]:
            errors.append("Field 'phase' must be 1, 2, 3, or 4")
    
    if 'content' in tile_json:
        if not isinstance(tile_json['content'], str):
            errors.append("Field 'content' must be string")
        elif len(tile_json['content'].strip()) == 0:
            errors.append("Field 'content' cannot be empty")
    
    if 'timestamp' in tile_json:
        if not isinstance(tile_json['timestamp'], str):
            errors.append("Field 'timestamp' must be string")
        else:
            try:
                datetime.fromisoformat(tile_json['timestamp'].replace('Z', '+00:00'))
            except ValueError:
                errors.append("Field 'timestamp' must be ISO 8601 format")
    
    return (len(errors) == 0, errors)

from datetime import datetime
import math

def deadband_filter(values, deadband=0.1):
    """
    Apply Deadband Protocol to a list of float values.
    P0: Remove NaN/inf values.
    P1: Values within [-deadband, deadband] become 0.
    P2: Outside values get deadband subtracted with sign.
    Returns list with filtered values (None for removed).
    """
    filtered = []
    for v in values:
        # P0: negative space
        if math.isnan(v) or math.isinf(v):
            filtered.append(None)
            continue
        
        # P1: safe channel
        if abs(v) <= deadband:
            filtered.append(0.0)
        # P2: optimize
        else:
            sign = 1 if v > 0 else -1
            filtered.append(v - sign * deadband)
    return filtered

    Parse fleet log: 'YYYY-MM-DD HH:MM:SS | AGENT | CYCLE | STATUS | MESSAGE'

    """

    parts = log_line.split(' | ', maxsplit=4)

            'timestamp': None,
            'agent': 'UNKNOWN',
