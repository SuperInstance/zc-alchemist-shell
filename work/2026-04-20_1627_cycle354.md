```python
    """
    Apply Deadband Protocol P0/P1/P2 logic to a list of sensor readings.
```

```python
def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol P0/P1/P2 logic to a list of sensor readings.
    
    P0: If value is NaN or infinite, return 0.0 (avoid catastrophic).
    P1: If value within +/- deadband of zero, return 0.0 (suppress noise).
    P2: Otherwise, return value unchanged (optimize within safe channel).
    
    Args:
        values: List of float readings.
        deadband: Absolute threshold for noise suppression.
    
    Returns:
        Filtered list of floats.
    """
    import math
    filtered = []
    for v in values:
        # P0: Handle non-finite values
        if not math.isfinite(v):
            filtered.append(0.0)
            continue
        # P1: Within deadband?
        if abs(v) <= deadband:
            filtered.append(0.0)
        else:
            # P2: Safe channel
            filtered.append(v)
    return filtered
```

```python
def parse_fleet_log(line: str) -> dict[str, str | int | float]:
    """
    Parse a structured fleet log line.
    
    Format: "2026-04-20T16:30:00Z [AGENT] [LEVEL] message key=value key2=value2"
    
    Returns dict with keys: timestamp, agent, level, message, and all key=value pairs.
    Numeric values are converted to int or float if possible.
    """
    import re
    
    # Split timestamp, agent, level, rest
    parts = line.split(' ', 3)
    if len(parts) < 4:
        raise ValueError(f"Invalid log line: {line}")
    
    timestamp = parts[0]
    agent_match = re.search(r'\[([^\]]+)\]', parts[1])
    level_match = re.search(r'\[([^\]]+)\]', parts[2])
    if not agent_match or not level_match:
        raise ValueError(f"Could not parse agent/level: {line}")
    
    agent = agent_match.group(1)
    level = level_match.group(1)
    remainder = parts[3]
    
    # Split message from key=value pairs
    kv_parts = remainder.split(' ')
    message_parts = []
    kv_dict = {}
    
    for p in kv_parts:
        if '=' in p:
            k, v = p.split('=', 1)
            # Try to convert to int or float
            try:
                if '.' in v:
                    v = float(v)
                else:
                    v = int(v)
            except ValueError:
                pass  # keep as string
            kv_dict[k] = v
        else:
            message_parts.append(p)
    
    result = {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': ' '.join(message_parts)
    }
    result.update(kv_dict)
    return result
```

```python
    """
    Parses fleet log lines into structured dictionaries.
    
    Expected format: TIMESTAMP [AGENT] [LEVEL] message key1=val1 key2=val2
    """
    import re
    
    pattern = r'^(\S+)\s+\[(\w+)\]\s+\[(\w+)\]\s+(.+?)(?:\s+(\S+=\S+.*))?$'
    match = re.match(pattern, line)
    if not match:
        return {}
    
    timestamp, agent, level, message, kv_string = match.groups()
    
    result = {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': message.strip()
    }
    
    if kv_string:
        for pair in kv_string.split():
            if '=' in pair:
                key, val = pair.split('=', 1)
                # Try numeric conversion
                try:
                    val = float(val) if '.' in val else int(val)
                except:
                    pass
                result[key] = val
    
    return result
```
