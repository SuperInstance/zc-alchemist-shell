from datetime import datetime
from typing import Optional

def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply the Deadband Protocol to filter values.
    
    P0: Map negative space - remove values within deadband of zero.
    P1: Find safe channels - keep values outside deadband.
    P2: Return filtered list.
    
    Args:
        values: List of float values.
        deadband: Absolute threshold around zero (default 0.1).
        
    Returns:
        List of values safely outside the deadband.
    """
    if not values:
        return []
    
    filtered = []
    for val in values:
        if abs(val) > deadband:
            filtered.append(val)
    return filtered


def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a standardized fleet log line.
    
    Expected format: "YYYY-MM-DD HH:MM:SS [LEVEL] key1=value1 key2=value2"
    
    Args:
        log_line: A single log line string.
        
    Returns:
        Dictionary with parsed fields: timestamp, level, agent, cycle, phase.
        Missing fields are set to None.
    """
    # Initialize with defaults
    result = {
        "timestamp": None,
        "level": None,
        "agent": None,
        "cycle": None,
        "phase": None
    }
    
    if not log_line or not log_line.strip():
        return result
    
    parts = log_line.strip().split()
    if len(parts) < 3:
        return result
    
    # Parse timestamp
    try:
        timestamp_str = f"{parts[0]} {parts[1]}"
        result["timestamp"] = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except (ValueError, IndexError):
        result["timestamp"] = None
    
    # Parse level
    if len(parts) >= 3 and parts[2].startswith("[") and parts[2].endswith("]"):
        result["level"] = parts[2][1:-1]
    
    # Parse key=value pairs
    for part in parts[3:]:
        if "=" in part:
            key, value = part.split("=", 1)
            if key in result:
                # Convert cycle and phase to int if possible
                if key in ["cycle", "phase"]:
                    try:
                        result[key] = int(value)
                    except ValueError:
                        result[key] = None
                else:
                    result[key] = value
    
    return result


def validate_tile(tile: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile for fleet submission.
    
    Required keys: agent, cycle, content, timestamp.
    
    Args:
        tile: Dictionary representing a training tile.
        
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    """
    errors = []
    required_keys = ["agent", "cycle", "content", "timestamp"]
    
    # Check required keys
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    # Validate agent
    if "agent" in tile:
        if not isinstance(tile["agent"], str) or not tile["agent"].strip():
            errors.append("Agent must be a non-empty string")
    
    # Validate cycle
    if "cycle" in tile:
        if not isinstance(tile["cycle"], int) or tile["cycle"] <= 0:
            errors.append("Cycle must be a positive integer")
    
    # Validate content
    if "content" in tile:
        if not isinstance(tile["content"], str) or not tile["content"].strip():
            errors.append("Content must be a non-empty string")
    
    # Validate timestamp
    if "timestamp" in tile:
        if not isinstance(tile["timestamp"], (str, datetime)):
            errors.append("Timestamp must be string or datetime")
        elif isinstance(tile["timestamp"], str):
            try:
                datetime.fromisoformat(tile["timestamp"].replace('Z', '+00:00'))
            except ValueError:
                errors.append("Timestamp string must be ISO format")
    
    return len(errors) == 0, errors
