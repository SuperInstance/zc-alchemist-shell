# Function 1: Deadband Protocol Validator
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.
    Protocol: Must start with P0, then P1, then P2. No skipping allowed.
    Steps can repeat within a phase (e.g., [P0, P0, P1, P2]) but cannot regress.
    
    Args:
        steps: List of step strings like ["P0", "P1", "P2"]
    
    Returns:
        Tuple of (is_valid: bool, error_message: str)
    """
    expected_sequence = ["P0", "P1", "P2"]
    current_phase_index = 0  # Start expecting P0
    
    for i, step in enumerate(steps):
        step_upper = step.upper()
        
        # Check if step is a valid phase
        if step_upper not in expected_sequence:
            return False, f"Invalid step '{step}' at position {i}. Must be P0, P1, or P2."
        
        # Find index of this step in expected sequence
        step_index = expected_sequence.index(step_upper)
        
        # Check for phase regression
        if step_index < current_phase_index:
            return False, f"Protocol regression at position {i}: '{step}' after '{expected_sequence[current_phase_index]}'."
        
        # Advance phase expectation if we moved forward
        if step_index > current_phase_index:
            # Ensure we didn't skip a phase
            if step_index - current_phase_index > 1:
                return False, f"Phase skip at position {i}: '{step}' jumps from '{expected_sequence[current_phase_index]}'."
            current_phase_index = step_index
    
    # Check if we reached P2 (not required by spec, but good practice)
    if current_phase_index < 2 and steps:
        return False, f"Protocol incomplete. Last phase was '{expected_sequence[current_phase_index]}', never reached P2."
    
    return True, ""

def parse_fleet_tile(content: str) -> dict:
    """
    Parses a fleet training tile markdown content.
    Extracts: agent_type, cycle, phase, task, output.
    
    Args:
        content: Raw markdown text of a tile
    
    Returns:
        Dictionary with extracted fields (empty strings if not found)
    """
    result = {
        "agent_type": "",
        "cycle": "",
        "phase": "",
        "task": "",
        "output": ""
    }
    
    lines = content.split('\n')
    in_output_section = False
    output_lines = []
    
    for line in lines:
        # Parse headers
        if line.startswith('**Agent Type:**'):
            result["agent_type"] = line.split(':', 1)[1].strip()
        elif line.startswith('**Cycle:**'):
            result["cycle"] = line.split(':', 1)[1].strip()
        elif line.startswith('**Phase:**'):
            result["phase"] = line.split(':', 1)[1].strip()
        elif line.startswith('**Task:**'):
            result["task"] = line.split(':', 1)[1].strip()
        
        # Detect output section (simplistic)
        elif line.strip() == '---' or line.strip().startswith('# Output'):
            in_output_section = True
            continue
        elif in_output_section and line.strip() and not line.startswith('#'):
            output_lines.append(line)
    
    result["output"] = '\n'.join(output_lines).strip()
    return result

# Function 3: Safe Channel Calculator
def calculate_safe_channel(boundaries: list[tuple[float, float]], position: float) -> float:
    """
    Finds the nearest safe position outside all unsafe boundaries.
    
    Args:
        boundaries: List of (lower, upper) unsafe ranges
        position: Current position
    
    Returns:
        Nearest safe position (could be the original if already safe)
    """
    def is_safe(pos: float) -> bool:
        """Check if a position is outside all unsafe boundaries."""
        for lower, upper in boundaries:
            if lower <= pos <= upper:
                return False
        return True
    
    # If already safe, return position
    if is_safe(position):
        return position
    
    # Generate candidate safe positions
    candidates = []
    
    # Add boundary edges
    for lower, upper in boundaries:
        candidates.extend([lower - 1e-9, upper + 1e-9])  # Just outside boundaries
    
    # Add position itself (though we know it's unsafe)
    candidates.append(position)
    
    # Add some default positions if boundaries exist
    if boundaries:
        min_boundary = min(lower for lower, _ in boundaries)
        max_boundary = max(upper for _, upper in boundaries)
        candidates.extend([min_boundary - 1.0, max_boundary + 1.0])
    
    # Filter to safe candidates and find closest
    safe_candidates = [c for c in candidates if is_safe(c)]
    
    if not safe_candidates:
        # Fallback: return position (though unsafe)
        return position
    
    return min(safe_candidates, key=lambda c: abs(c - position))
