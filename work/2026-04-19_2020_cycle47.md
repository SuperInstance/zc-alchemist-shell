```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates that a list of steps follows the Deadband Protocol's strict order:
    P0 (negative space) -> P1 (safe channels) -> P2 (optimization).
    Returns (is_valid, error_message).
    """
    valid_phases = {"P0", "P1", "P2"}
    phase_order = {"P0": 0, "P1": 1, "P2": 2}
    last_index = -1

    for i, step in enumerate(steps):
        if step not in valid_phases:
            return False, f"Invalid step '{step}' at position {i}. Must be P0, P1, or P2."
        current_index = phase_order[step]
        if current_index < last_index:
            return False, f"Protocol violation at position {i}: {step} appears after a higher phase."
        last_index = current_index

    # Check for missing phases (optional but good for completeness)
    present_phases = {step for step in steps}
    for phase in ["P0", "P1", "P2"]:
        if phase not in present_phases:
            return False, f"Missing phase {phase}."

    return True, "Protocol valid."

if __name__ == "__main__":
    print(validate_deadband_protocol(["P0", "P1", "P2"]))  # (True, "Protocol valid.")
    print(validate_deadband_protocol(["P0", "P2", "P1"]))  # (False, "Protocol violation...")
    print(validate_deadband_protocol(["P0", "P1", "P3"]))  # (False, "Invalid step 'P3'...")
    print(validate_deadband_protocol(["P1", "P2"]))        # (False, "Missing phase P0.")
```

```python
import re

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet training tile text into a structured dictionary.
    Expected header format: "YYYY-MM-DD_HHMM_cycleX.md" followed by content.
    """
    default_result = {
        "cycle": None,
        "phase": None,
        "task": "",
        "agent": "",
        "content": tile_text.strip()
    }

    lines = tile_text.strip().split('\n')
    if not lines:
        return default_result

    # Extract cycle from first line filename pattern
    first_line = lines[0]
    cycle_match = re.search(r'cycle(\d+)', first_line)
    if cycle_match:
        default_result["cycle"] = int(cycle_match.group(1))

    # Scan for Phase and Task patterns in first 10 lines
    for line in lines[:10]:
        phase_match = re.search(r'\*\*Phase\s*(\d+)\*\*', line)
        if phase_match and default_result["phase"] is None:
            default_result["phase"] = int(phase_match.group(1))

        task_match = re.search(r'\*\*Task:\*\*\s*(.+)', line)
        if task_match and not default_result["task"]:
            default_result["task"] = task_match.group(1).strip()

        agent_match = re.search(r'# Cycle \d+ — (\w+)', line)
        if agent_match and not default_result["agent"]:
            default_result["agent"] = agent_match.group(1)

    return default_result

if __name__ == "__main__":
    sample_tile = """2026-04-19_2002_cycle45.md
# Cycle 45 — Alchemist
**Phase 4**
**Task:** Compare models for code generation.
This is the content body."""
    print(parse_fleet_tile(sample_tile))
    print(parse_fleet_tile("Just some text without a header."))
```

```python
import math

def generate_constraint_snaps(points: list[tuple[float, float]], tolerance: float = 0.01) -> list[tuple[int, int]]:
    """
    Finds pairs of points within a given Euclidean distance tolerance.
    Returns list of (index_a, index_b) where index_a < index_b.
    """
    snaps = []
    n = len(points)
    for i in range(n):
        x1, y1 = points[i]
        for j in range(i + 1, n):  # Avoid self and duplicates
            x2, y2 = points[j]
            distance = math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)
            if distance <= tolerance:
                snaps.append((i, j))
    return snaps

if __name__ == "__main__":
    points = [(0.0, 0.0), (0.0, 0.009), (1.0, 1.0), (1.005, 1.0)]
    print(generate_constraint_snaps(points, 0.01))  # [(0, 1), (2, 3)]
    print(generate_constraint_snaps([], 0.1))       # []
    print(generate_constraint_snaps([(0, 0)], 0.1)) # []
```
