    """
    Parses a structured log line of format 'YYYY-MM-DD HH:MM:SS LEVEL [MODULE] Message'
    into a dictionary with keys: timestamp, level, module, message.

    Args:
        log_line: A string containing a single log line.

    Returns:
        A dictionary with parsed components or None if the line doesn't match the expected format.
    """

    pattern = r'^(\d{4}-\d{2}-\d{2})\s+(\d{2}:\d{2}:\d{2})\s+(\w+)\s+\[([^\]]+)\]\s+(.+)$'

        timestamp = dt.isoformat()
    except ValueError:
        # If datetime parsing fails, keep the raw strings
        timestamp = f"{date_str} {time_str}"

        "level": level,
        "module": module,
