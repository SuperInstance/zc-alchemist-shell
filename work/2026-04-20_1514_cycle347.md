from typing import List, Dict, Any, Optional, Tuple

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply deadband filtering to a list of values.
    
    Args:
        values: List of float values to filter.
        deadband: Threshold around zero. Values within [-deadband, deadband] become 0.0.
    
    Returns:
        Filtered list where values within deadband of zero are set to 0.0.
    
    Example:
        >>> deadband_filter([0.05, 0.2, -0.07, 1.5], 0.1)
        [0.0, 0.2, 0.0, 1.5]
    """
    return [0.0 if -deadband <= v <= deadband else v for v in values]


def parse_fleet_log(log_line: str) -> Optional[Dict[str, Any]]:
    """
    Parse a fleet log line into structured data.
    
    Expected format: [TIMESTAMP] AGENT@CYCLE: "MESSAGE" (METADATA)
    
    Args:
        log_line: A single line from fleet logs.
    
    Returns:
        Dictionary with keys: timestamp, agent, cycle, message, metadata.
        Returns None if line doesn't match format.
    
    Example:
        >>> parse_fleet_log('[2026-04-20T14:30:00Z] Alchemist@347: "Testing tile" (score=0.8)')
        {'timestamp': '2026-04-20T14:30:00Z', 'agent': 'Alchemist', 'cycle': 347,
         'message': 'Testing tile', 'metadata': 'score=0.8'}
    """
    import re
    
    pattern = r'\[([^\]]+)\]\s+([^@]+)@(\d+):\s+"([^"]+)"\s+\(([^)]+)\)'
    match = re.match(pattern, log_line.strip())
    if not match:
        return None
    
    timestamp, agent, cycle_str, message, metadata = match.groups()
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': int(cycle_str),
        'message': message,
        'metadata': metadata
    }


def validate_tile_json(tile_data: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile JSON against fleet schema.
    
    Required keys: id (str), agent (str), cycle (int), content (str)
    Optional keys: tags (list of str), score (float between 0 and 1)
    
    Args:
        tile_data: Dictionary containing tile data.
    
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    
    Example:
        >>> data = {'id': 'tile_001', 'agent': 'Alchemist', 'cycle': 347, 'content': 'Test'}
        >>> validate_tile_json(data)
        (True, [])
    """
    errors = []
    
    # Required fields
    required = {'id': str, 'agent': str, 'cycle': int, 'content': str}
    for field, expected_type in required.items():
        if field not in tile_data:
            errors.append(f"Missing required field: {field}")
        elif not isinstance(tile_data[field], expected_type):
            errors.append(f"Field '{field}' must be {expected_type.__name__}")
    
    # Optional fields validation
    if 'tags' in tile_data:
        if not isinstance(tile_data['tags'], list):
            errors.append("Field 'tags' must be a list")
        elif not all(isinstance(tag, str) for tag in tile_data['tags']):
            errors.append("All tags must be strings")
    
    if 'score' in tile_data:
        score = tile_data['score']
        if not isinstance(score, (int, float)):
            errors.append("Field 'score' must be numeric")
        elif not (0 <= score <= 1):
            errors.append("Field 'score' must be between 0 and 1")
    
    return (len(errors) == 0, errors)
