```python
def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply deadband filtering to a list of values.
    
    Deadband protocol: If absolute value is less than deadband, return 0.
    Otherwise, return the original value.
    
    Args:
        values: List of float values
        deadband: Threshold for filtering (default 0.1)
        
    Returns:
        Filtered list of floats
    """
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for val in values:
        if abs(val) < deadband:
            result.append(0.0)
        else:
            result.append(val)
    
    return result
```

**Function 2: parse_fleet_log**
```python
import re
from datetime import datetime

def parse_fleet_log(log_text: str) -> dict:
    """
    Parse a fleet log entry into structured data.
    
    Log format: "[TIMESTAMP] AGENT_NAME: MESSAGE (STATUS)"
    Example: "[2026-04-20 15:30:00] Oracle1: Tile submitted (ACCEPTED)"
    
    Returns dict with keys: timestamp, agent, message, status
    """
    pattern = r'\[(.*?)\] (\w+): (.*?) \((.*?)\)'
    match = re.match(pattern, log_text.strip())
    
    if not match:
        raise ValueError(f"Invalid log format: {log_text}")
    
    timestamp_str, agent, message, status = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        timestamp = timestamp_str  # Keep as string if parsing fails
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "message": message,
        "status": status
    }
```

```python
def validate_tile_json(tile_data: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile JSON structure.
    
    Required fields: tile_id, agent, cycle, content, timestamp
    Optional fields: tags, metadata
    
    Returns: (is_valid, list_of_errors)
    """
    required_fields = ["tile_id", "agent", "cycle", "content", "timestamp"]
    errors = []
    
    # Check required fields
    for field in required_fields:
        if field not in tile_data:
            errors.append(f"Missing required field: {field}")
        elif tile_data[field] is None or tile_data[field] == "":
            errors.append(f"Field {field} cannot be empty")
    
    # Type checking
    if "tile_id" in tile_data and not isinstance(tile_data["tile_id"], str):
        errors.append("tile_id must be a string")
    
    if "cycle" in tile_data and not isinstance(tile_data["cycle"], int):
        errors.append("cycle must be an integer")
    
    if "content" in tile_data and not isinstance(tile_data["content"], str):
        errors.append("content must be a string")
    
    # Optional field validation
    if "tags" in tile_data and not isinstance(tile_data["tags"], list):
        errors.append("tags must be a list if present")
    
    return (len(errors) == 0, errors)
```
