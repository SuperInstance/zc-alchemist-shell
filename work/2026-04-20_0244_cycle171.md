```python
def deadband_filter(data, lower, upper):
    """
    Apply Deadband Protocol filtering to a numeric series.
    
    P0: Values outside [lower, upper] are considered 'rocks' and set to None.
    P1: Values inside the band are kept as-is (safe channel).
    P2: No optimization applied in this basic filter.
    
    Args:
        data (list of float): Input series.
        lower (float): Lower bound of deadband.
        upper (float): Upper bound of deadband.
    
    Returns:
        list: Filtered series with out-of-band values replaced by None.
    """
    if not isinstance(data, list):
        raise TypeError("data must be a list")
    if lower >= upper:
        raise ValueError("lower must be less than upper")
    
    filtered = []
    for value in data:
        # P0: Identify rocks (negative space)
        if value < lower or value > upper:
            filtered.append(None)  # Mark as unsafe
        # P1: Safe channel
        else:
            filtered.append(value)
    return filtered
```

```python
import re

def parse_fleet_context(markdown_text):
    """
    Parse a fleet-context Markdown document into a structured dictionary.
    
    Expected sections: Who We Are, Fleet Doctrine, Key Technologies,
    Architecture, Rules.
    
    Args:
        markdown_text (str): Raw Markdown content.
    
    Returns:
        dict: Keys are section headers, values are list of lines.
    """
    if not markdown_text or not isinstance(markdown_text, str):
        return {}
    
    # Normalize line endings and split into lines
    lines = markdown_text.replace('\r\n', '\n').split('\n')
    
    sections = {}
    current_section = None
    current_content = []
    
    for line in lines:
        # Detect section headers (## Header)
        match = re.match(r'^##\s+(.+)$', line.strip())
        if match:
            # Save previous section
            if current_section:
                sections[current_section] = current_content
            # Start new section
            current_section = match.group(1).strip()
            current_content = []
        elif current_section:
            # Skip empty lines at start of section
            if line.strip() or current_content:
                current_content.append(line.rstrip())
    
    # Add last section
    if current_section:
        sections[current_section] = current_content
    
    return sections
```

```python
def tile_scorer(tile_text, rubric):
    """
    Score a training tile against a rubric.
    
    Rubric format: {
        'completeness': {'weight': 0.4, 'max': 10},
        'clarity': {'weight': 0.3, 'max': 10},
        'usefulness': {'weight': 0.3, 'max': 10}
    }
    
    Args:
        tile_text (str): The tile content to score.
        rubric (dict): Scoring rubric with weights and max scores.
    
    Returns:
        tuple: (total_score, breakdown_dict)
    """
    if not tile_text or not rubric:
        return 0.0, {}
    
    # Validate rubric structure
    required_keys = {'weight', 'max'}
    for key, params in rubric.items():
        if not all(k in params for k in required_keys):
            raise ValueError(f"Rubric item '{key}' missing required keys")
        if not 0 <= params['weight'] <= 1:
            raise ValueError(f"Weight for '{key}' must be between 0 and 1")
    
    # Simple scoring heuristics (in practice, would use more sophisticated NLP)
    breakdown = {}
    total_weight = 0
    
    for criterion, params in rubric.items():
        weight = params['weight']
        max_score = params['max']
        
        # Placeholder scoring logic
        if criterion == 'completeness':
            # Count non-empty lines as proxy for completeness
            lines = [l for l in tile_text.split('\n') if l.strip()]
            score = min(len(lines) / 20 * max_score, max_score)  # 20 lines = full score
        elif criterion == 'clarity':
            # Favor shorter sentences and bullet points
            sentences = tile_text.replace('!', '.').replace('?', '.').split('.')
            avg_len = sum(len(s.split()) for s in sentences) / max(len(sentences), 1)
            score = max_score if avg_len < 15 else max_score * 0.5
        elif criterion == 'usefulness':
            # Check for actionable keywords
            keywords = ['implement', 'test', 'fix', 'document', 'analyze']
            found = sum(1 for kw in keywords if kw in tile_text.lower())
            score = min(found / 3 * max_score, max_score)
        else:
            score = 0
        
        breakdown[criterion] = round(score, 2)
        total_weight += weight
    
    # Normalize weights to 1.0
    if total_weight == 0:
        total_score = 0.0
    else:
        weighted_sum = sum(breakdown[c] * rubric[c]['weight'] for c in breakdown)
        total_score = round(weighted_sum / total_weight, 2)
    
    return total_score, breakdown
```

```python
def deadband_filter(data, lower, upper):
    """
    Implements deadband filtering on a list of numeric values.
    
    Values within [lower, upper] are retained (safe channel).
    Values outside are replaced with None (negative space).
    
    Args:
        data (list): List of numeric values.
        lower (float): Lower bound.
        upper (float): Upper bound.
    
    Returns:
        list: List with same length as data, where values outside
              deadband are None.
    """
    # Input validation
    if not isinstance(data, list):
        raise TypeError("Input data must be a list")
    if lower >= upper:
        raise ValueError("Lower bound must be less than upper bound")
    
    result = []
    for val in data:
        # Check if value is numeric
        if not isinstance(val, (int, float)):
            result.append(None)  # Non-numeric treated as rock
            continue
        
        # Deadband logic
        if lower <= val <= upper:
            result.append(val)   # In safe channel
        else:
            result.append(None)  # In negative space
    
    return result
```
