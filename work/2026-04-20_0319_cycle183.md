def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol P0/P1/P2 logic to a list of values.
    
    P0: If value <= deadband, return 0 (negative space)
    P1: If deadband < value <= 2*deadband, return value (safe channel)
    P2: If value > 2*deadband, return value * 0.8 (optimized)
    
    Args:
        values: List of float values

        else:
            result.append(val * 0.8)  # P2: optimized

#### **Function 2: `parse_fleet_log`**
```python
import re
from datetime import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a fleet log line with format:
    "TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE"
    
    Example: "2026-04-20T03:18:00Z | Alchemist | 183 | ACTIVE | Testing models"
    
    Returns dict with keys: timestamp, agent, cycle, status, message.
    If format invalid, returns {'error': 'Invalid format', 'raw': log_line}
    """
    if not isinstance(log_line, str):
        return {'error': 'Input must be string', 'raw': log_line}
    
    pattern = r'^(.+?) \| (.+?) \| (\d+) \| (.+?) \| (.+)$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        return {'error': 'Invalid format', 'raw': log_line}
    
    timestamp_str, agent, cycle_str, status, message = match.groups()
    
    # Try to parse timestamp
    try:
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        timestamp = timestamp_str
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': int(cycle_str),
        'status': status,
        'message': message
    }
```

#### **Function 3: `constraint_snap`**
```python
import math

def constraint_snap(point: tuple[float, float], 
                   constraints: list[tuple[float, float, float]]) -> tuple[float, float]:
    """
    Snap a 2D point to the nearest geometric constraint.
    
    Each constraint is (A, B, C) representing line Ax + By + C = 0
    Returns the point on the nearest constraint line.
    
    Args:
        point: (x, y) tuple
        constraints: List of (A, B, C) line coefficients
    
    Returns:
        (x_snapped, y_snapped) tuple
    """
    if not constraints:
        return point
    
    x, y = point
    min_distance = float('inf')
    snapped_point = point
    
    for A, B, C in constraints:
        # Distance from point to line
        denominator = math.sqrt(A**2 + B**2)
        if denominator == 0:
            continue  # Invalid constraint
        
        distance = abs(A*x + B*y + C) / denominator
        
        if distance < min_distance:
            min_distance = distance
            # Find closest point on line
            if B != 0:
                # Solve for y when x is given
                y_snap = (-A*x - C) / B
                snapped_point = (x, y_snap)
            else:
                # Vertical line
                x_snap = -C / A if A != 0 else x
                snapped_point = (x_snap, y)
    
    return snapped_point
```
