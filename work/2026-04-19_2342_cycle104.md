    Parse a fleet state log line into structured data.

    Returns:
        Dict with keys: timestamp, agent, cycle, status, tiles

    parts = log_line.split(" | ")
    result = {}
    
    # Parse timestamp
    result["timestamp"] = datetime.datetime.strptime(parts[0], "%Y-%m-%d %H:%M:%S")
    
    # Parse key-value pairs
    for part in parts[1:]:
        if "=" in part:
            key, value = part.split("=", 1)
            key = key.lower()
            
            # Type conversion
            if key == "cycle" or key == "tiles":
                result[key] = int(value)
            else:
                result[key] = value
    
    return result

    Validate a training tile against fleet schema.
    
    Required fields: agent, cycle, task, output, timestamp
    Optional fields: metrics, artifacts, references
    
    Returns:
        (is_valid, list_of_errors)
    """
    required_fields = ["agent", "cycle", "task", "output", "timestamp"]
    errors = []
    
    # Check required fields
    for field in required_fields:
        if field not in tile_json:
            errors.append(f"Missing required field: {field}")

            errors.append(f"Empty value for required field: {field}")

        errors.append("Field 'cycle' must be integer")

        try:
            datetime.datetime.fromisoformat(tile_json["timestamp"].replace('Z', '+00:00'))
        except (ValueError, TypeError):
            errors.append("Field 'timestamp' must be valid ISO format")
    
    return (len(errors) == 0, errors)

# deadband_filter
def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol to a list of sensor readings.
    
    P0: Remove values that are within ±deadband of zero (negative space).
    P1: Return only values outside deadband (safe channels).
    
    Args:
        values: List of float readings
        deadband: Absolute threshold around zero
    
    Returns:
        Filtered list where abs(v) > deadband
    """
    if deadband < 0:
        raise ValueError("Deadband must be non-negative")
    
    filtered_values = []
    for val in values:
        if abs(val) > deadband:
            filtered_values.append(val)
    
    return filtered_values

# parse_fleet_state
def parse_fleet_state(log_line: str) -> dict:
    """
    Parse a fleet state log line into structured data.
    
    Format: "2026-04-19 23:45:01 | AGENT=Alchemist | CYCLE=104 | STATUS=ACTIVE | TILES=3"
    
    Returns:
        Dict with keys: timestamp, agent, cycle, status, tiles
    """
    import re
    from datetime import datetime
    
    # Split by pipe and strip whitespace
    segments = [seg.strip() for seg in log_line.split('|')]
    
    result = {}
    
    # First segment is timestamp
    try:
        result['timestamp'] = datetime.strptime(segments[0], '%Y-%m-%d %H:%M:%S')
    except ValueError:
        result['timestamp'] = segments[0]  # Keep as string if parsing fails
    
    # Parse key-value pairs
    for seg in segments[1:]:
        match = re.match(r'(\w+)=(.+)', seg)
        if match:
            key, value = match.groups()
            key_lower = key.lower()
            
            # Convert numeric fields
            if key_lower in ['cycle', 'tiles']:
                try:
                    result[key_lower] = int(value)
                except ValueError:
                    result[key_lower] = value
            else:
                result[key_lower] = value
    
    return result

# tile_validator
def tile_validator(tile_json: dict, schema_version: str = "v1") -> tuple[bool, list[str]]:
    """
    Validate a training tile against fleet schema.
    
    Required fields: agent, cycle, task, output, timestamp
    Optional fields: metrics, artifacts, references
    
    Returns:
        (is_valid, list_of_errors)
    """
    errors = []
    
    # Schema version check
    if schema_version != "v1":
        errors.append(f"Unsupported schema version: {schema_version}")
    
    # Required fields
    required = ['agent', 'cycle', 'task', 'output', 'timestamp']
    for field in required:
        if field not in tile_json:
            errors.append(f"Missing required field: {field}")
        elif not tile_json[field]:  # Check for empty/None
            errors.append(f"Field '{field}' is empty")
    
    # Data type validation
    if 'cycle' in tile_json and not isinstance(tile_json['cycle'], int):
        errors.append("Field 'cycle' must be an integer")
