from typing import List, Dict, Any
import re
from datetime import datetime

def validate_deadband_protocol(steps: List[str]) -> Dict[str, Any]:
    """
    Validate a sequence of deadband protocol steps.
    
    Args:
        steps: List of step strings, e.g. ["P0", "P1", "P2"]
    
    Returns:
        Dictionary with:
        - valid: bool indicating if sequence is valid
        - errors: list of error messages if invalid
        - phase_counts: dict counting occurrences of each phase
    """
    valid_phases = {"P0", "P1", "P2"}
    phase_counts = {phase: 0 for phase in valid_phases}
    errors = []
    
    # Track expected order
    expected_order = ["P0", "P1", "P2"]
    last_index = -1
    
    for i, step in enumerate(steps):
        # Check if valid phase
        if step not in valid_phases:
            errors.append(f"Step {i}: Invalid phase '{step}'")
            continue
        
        phase_counts[step] += 1
        
        # Check order
        current_index = expected_order.index(step)
        if current_index < last_index:
            errors.append(f"Step {i}: Phase '{step}' appears out of order")
        last_index = current_index
    
    # Check for missing phases (optional, but good for completeness)
    for phase in valid_phases:
        if phase_counts[phase] == 0:
            errors.append(f"Missing phase '{phase}'")
    
    # Check for skipping
    found_phases = [step for step in steps if step in valid_phases]
    for i in range(len(found_phases) - 1):
        curr_idx = expected_order.index(found_phases[i])
        next_idx = expected_order.index(found_phases[i + 1])
        if next_idx > curr_idx + 1:
            errors.append(f"Skipping from '{found_phases[i]}' to '{found_phases[i + 1]}'")
    
    return {
        "valid": len(errors) == 0,
        "errors": errors,
        "phase_counts": phase_counts
    }

def extract_tile_metadata(log_text: str) -> List[Dict[str, Any]]:
    """
    Extract tile metadata from PLATO room server log.
    
    Args:
        log_text: Multi-line log text
    
    Returns:
        List of dictionaries with agent, cycle, timestamp, status
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2})\] (\w+)@cycle(\d+) tile (accepted|rejected)'
    tiles = []
    
    for line in log_text.strip().split('\n'):
        match = re.search(pattern, line)
        if match:
            timestamp_str, agent, cycle_str, status = match.groups()
            try:
                timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M")
                cycle = int(cycle_str)
                tiles.append({
                    "agent": agent,
                    "cycle": cycle,
                    "timestamp": timestamp,
                    "status": status
                })
            except ValueError:
                # Skip malformed entries
                continue
    
    return tiles

def simulate_fleet_energy(agents: List[Dict[str, Any]], steps: int) -> List[float]:
    """
    Simulate fleet energy consumption over time.
    
    Args:
        agents: List of agent dicts with keys:
                - name: str
                - power_draw_watts: float
                - uptime_ratio: float (0.0 to 1.0)
        steps: Number of time steps to simulate
    
    Returns:
        List of total energy consumption in kWh for each step
    """
    if not agents:
        return [0.0] * steps
    
    # Validate agents
    validated_agents = []
    for agent in agents:
        required_keys = {"name", "power_draw_watts", "uptime_ratio"}
        if not all(key in agent for key in required_keys):
            raise ValueError(f"Agent {agent.get('name', 'unknown')} missing required keys")
        
        power = float(agent["power_draw_watts"])
        uptime = float(agent["uptime_ratio"])
        
        if power < 0:
            raise ValueError(f"Agent {agent['name']}: power_draw_watts must be non-negative")
        if uptime < 0 or uptime > 1:
            raise ValueError(f"Agent {agent['name']}: uptime_ratio must be between 0 and 1")
        
        validated_agents.append({
            "name": agent["name"],
            "power_draw_watts": power,
            "uptime_ratio": uptime
        })
    
    # Calculate total power per step (in watts)
    total_power_watts = sum(
        agent["power_draw_watts"] * agent["uptime_ratio"]
        for agent in validated_agents
    )
    
    # Convert to kWh per step (assuming 1 step = 1 hour for simplicity)
    # In real simulation, you'd have a time_delta parameter
    energy_kwh_per_step = total_power_watts / 1000.0
    
    return [energy_kwh_per_step] * steps
