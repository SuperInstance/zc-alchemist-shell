from typing import List

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply deadband filtering to a list of values.
    
    Args:
        values: List of float values.
        deadband: Threshold around zero. Values within [-deadband, deadband] become 0.0.
    
    Returns:
        Filtered list where values within deadband are set to 0.0.
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for val in values:
        if -deadband <= val <= deadband:
            result.append(0.0)
        else:
            result.append(val)
    return result
```

import re
from datetime import datetime
from typing import Dict, Any

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a fleet log line into structured components.
    
    Expected format: 'YYYY-MM-DD HH:MM:SS [LEVEL] agent@cycle - message'
    
    Args:
        log_line: A single log line string.
    
    Returns:
        Dictionary with keys: timestamp, level, agent, cycle, message.
    
    Raises:
        ValueError: If log_line doesn't match expected pattern.
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\] (\w+)@(\d+) - (.+)'
    match = re.match(pattern, log_line)
    
    if not match:
        raise ValueError(f"Log line does not match expected format: {log_line}")
    
    timestamp_str, level, agent, cycle_str, message = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
        cycle = int(cycle_str)
    except ValueError as e:
        raise ValueError(f"Invalid timestamp or cycle in log line: {e}")
    
    return {
        "timestamp": timestamp,
        "level": level,
        "agent": agent,
        "cycle": cycle,
        "message": message.strip()
    }
```

import uuid
from datetime import datetime
from typing import Dict, Any

def validate_training_tile(tile: Dict[str, Any]) -> bool:
    """
    Validate a training tile against fleet standards.
    
    Required keys: 'id', 'agent', 'cycle', 'content', 'timestamp'
    
    Args:
        tile: Dictionary representing a training tile.
    
    Returns:
        True if tile is valid, False otherwise.
    """
    required_keys = {'id', 'agent', 'cycle', 'content', 'timestamp'}
    if not all(key in tile for key in required_keys):
        return False
    
    # Validate id as UUID
    try:
        uuid.UUID(tile['id'])
    except (ValueError, TypeError):
        return False
    
    # Validate agent (non-empty string)
    if not isinstance(tile['agent'], str) or not tile['agent'].strip():
        return False
    
    # Validate cycle (positive int)
    if not isinstance(tile['cycle'], int) or tile['cycle'] <= 0:
        return False
    
    # Validate content (non-empty string)
    if not isinstance(tile['content'], str) or not tile['content'].strip():
        return False
    
    # Validate timestamp (ISO 8601 format)
    try:
        datetime.fromisoformat(tile['timestamp'].replace('Z', '+00:00'))
    except (ValueError, TypeError):
        return False
    
    return True

import re

from typing import Dict

def parse_fleet_log(log_line: str) -> Dict[str, str | int | datetime]:
    """
    Parse a fleet log line.
    
    Format: 'YYYY-MM-DD HH:MM:SS [LEVEL] agent@cycle - message'
    
    Args:
        log_line: The log line to parse.
    
    Returns:
        A dictionary with parsed components.
    
    Raises:
        ValueError: If the log line doesn't match the expected format.
    """
    pattern = re.compile(
        r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[([A-Z]+)\] ([a-zA-Z0-9_-]+)@(\d+) - (.*)$'
    )
    
    match = pattern.match(log_line.strip())
    if not match:
        raise ValueError(f"Invalid log line format: {log_line}")
    
    timestamp_str, level, agent, cycle_str, message = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
        cycle = int(cycle_str)
    except ValueError as e:
        raise ValueError(f"Failed to parse timestamp or cycle: {e}")
    
    return {
        "timestamp": timestamp,
        "level": level,
        "agent": agent,
        "cycle": cycle,
        "message": message
    }
