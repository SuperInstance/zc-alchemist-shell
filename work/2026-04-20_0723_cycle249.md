from typing import List

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply a deadband filter to a list of float values.
    
    The deadband filter sets values within [-deadband, deadband] to zero,
    preserving values outside this range.
    
    Args:
        values: A list of float values.
        deadband: The deadband threshold (default 0.1).
        
    Returns:
        A list of filtered float values.
        
    Example:
        >>> deadband_filter([0.05, 1.2, -0.08, 3.0])
        [0.0, 1.2, 0.0, 3.0]
    """
    if not values:
        return []
    
    # P0: Input validation
    if deadband < 0:
        raise ValueError("Deadband must be non-negative")
    
    filtered = []
    for val in values:
        # Check for valid float
        if not isinstance(val, (int, float)):
            raise TypeError(f"Expected numeric value, got {type(val)}")
        
        # Convert to float for consistency
        val_float = float(val)
        
        # Apply deadband
        if abs(val_float) <= deadband:
            filtered.append(0.0)
        else:
            filtered.append(val_float)
    
    return filtered
```

- **Correctness:** Both implement core deadband logic. DeepSeek includes explicit NaN/inf filtering (P0), while Groq raises TypeError on non-numeric.
- **Fleet context:** DeepSeek explicitly references P0/P1/P2 in docstring; Groq does not.
- **Edge cases:** DeepSeek handles NaN/inf silently (skips them); Groq raises exceptions. DeepSeek's type checking is redundant (repeated). Groq adds deadband < 0 validation.
- **Style:** Both have docstrings and type hints. DeepSeek's example includes deadband param explicitly; Groq's uses default.

import subprocess
from typing import Dict, Any, List
from datetime import datetime

def git_repo_analyzer(repo_path: str) -> Dict[str, Any]:
    """
    Analyze a local git repository for fleet training tile metrics.
    
    P0: Avoid errors - handle missing git, invalid repo.
    P1: Find safe channels - use subprocess with timeouts.
    P2: Optimize - extract key metrics for plato-torch room training.
    
    Args:
        repo_path: Path to git repository.
    
    Returns:
        Dictionary with analysis metrics.
    
    Example:
        >>> git_repo_analyzer("/home/fleet/plato-torch")
        {'branch': 'main', 'commits': 245, 'contributors': 8, ...}
    """
    result = {
        "repo_path": repo_path,
        "valid": False,
        "branch": None,
        "commit_count": 0,
        "contributors": [],
        "recent_commits": [],
        "error": None
    }
    
    # P0: Check if path exists and is git repo
    import os
    if not os.path.exists(repo_path):
        result["error"] = "Path does not exist"
        return result
    
    git_dir = os.path.join(repo_path, ".git")
    if not os.path.exists(git_dir):
        result["error"] = "Not a git repository"
        return result
    
    def run_git(cmd: List[str]) -> str:
        """Safe git command runner with timeout."""
        try:
            proc = subprocess.run(
                ["git"] + cmd,
                cwd=repo_path,
                capture_output=True,
                text=True,
                timeout=5.0
            )
            if proc.returncode != 0:
                return ""
            return proc.stdout.strip()
        except (subprocess.TimeoutExpired, FileNotFoundError):
            return ""
    
    # P1: Collect safe metrics
    branch = run_git(["branch", "--show-current"])
    if branch:
        result["branch"] = branch
        result["valid"] = True
    
    # Commit count
    commit_count = run_git(["rev-list", "--count", "HEAD"])
    if commit_count:
        result["commit_count"] = int(commit_count)
    
    # Contributors (unique authors)
    contributors_raw = run_git(["shortlog", "-s", "-n", "HEAD"])
    if contributors_raw:
        contributors = []
        for line in contributors_raw.split("\n"):
            if line:
                parts = line.strip().split("\t")
                if len(parts) == 2:
                    contributors.append(parts[1].strip())
        result["contributors"] = contributors[:10]  # Top 10
    
    # Recent commits (last 5)
    recent = run_git(["log", "--oneline", "-5"])
    if recent:
        result["recent_commits"] = recent.split("\n")
    
    return result
```
