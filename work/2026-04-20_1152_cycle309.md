def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.

    The protocol requires strict order: P0 -> P1 -> P2, with no skipping
    and no repeats. Returns (True, "") if valid, (False, error_msg) otherwise.

    Args:
        steps: List of step labels (e.g., ["P0", "P1", "P2"])

    Returns:
        Tuple of (is_valid, error_message)
    """
    expected = ["P0", "P1", "P2"]
    seen = set()

    for i, step in enumerate(steps):
        if step not in expected:
            return (False, f"Error: Invalid step '{step}' at position {i}. Must be P0, P1, or P2.")
        if step in seen:
            return (False, f"Error: Step '{step}' repeated at position {i}. No repeats allowed.")
        if i > 0:
            prev_idx = expected.index(steps[i-1])
            curr_idx = expected.index(step)
            if curr_idx != prev_idx + 1:
                return (False, f"Error: Illegal transition from '{steps[i-1]}' to '{step}' at position {i}. Must follow order P0->P1->P2.")
        seen.add(step)

        return (False, f"Error: Incomplete protocol. Missing steps: {set(expected) - seen}")

    return (True, "")


def parse_fleet_tile(content: str) -> dict[str, Optional[str]]:
    """
    Parses a fleet training tile string for key metadata.

    Extracts agent, cycle, phase, task, and output from a formatted
    text block. Uses simple string matching.

    Args:
        content: Raw tile content as a string.

    Returns:
        Dictionary with keys 'agent', 'cycle', 'phase', 'task', 'output'.
        Values are extracted strings or None if not found.
    """
    lines = content.split('\n')
    result = {
        'agent': None,
        'cycle': None,
        'phase': None,
        'task': None,
        'output': None
    }

    for i, line in enumerate(lines):
        line_stripped = line.strip()
        # Extract agent (from first line pattern "# Cycle X — Agent")
        if i == 0 and line_stripped.startswith('# Cycle'):
            parts = line_stripped.split('—')
            if len(parts) > 1:
                result['agent'] = parts[1].strip()

        # Extract cycle number
        if line_stripped.startswith('# Cycle'):
            import re
            match = re.search(r'Cycle (\d+)', line_stripped)
            if match:
                result['cycle'] = match.group(1)

        # Extract phase
        if line_stripped.startswith('**Phase'):
            # Remove markdown bold markers
            phase = line_stripped.replace('**', '').replace('Phase', '').strip()
            result['phase'] = phase

        # Extract task
        if line_stripped.startswith('**Task:**'):
            task = line_stripped.replace('**Task:**', '').strip()
            result['task'] = task

        # Extract output
        if line_stripped.startswith('**Output:**'):
            output = line_stripped.replace('**Output:**', '').strip()
            # If output spans multiple lines, join until next header or empty line
            output_lines = [output]
            for next_line in lines[i+1:]:
                if next_line.strip() == '' or next_line.strip().startswith('**'):
                    break
                output_lines.append(next_line.strip())
            result['output'] = ' '.join(output_lines)

    return result
