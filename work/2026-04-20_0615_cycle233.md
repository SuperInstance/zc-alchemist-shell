    Apply Deadband Protocol (P0/P1/P2) to a list of numeric values.
    
    P0: Map negative space (values <= 0)
    P1: Find safe channels (values > 0 and <= 50)
    P2: Optimize within channels (values > 50)
    
    Args:
        values: List of float values to categorize.
    
    Returns:
        Dict with keys 'p0_negative', 'p1_safe', 'p2_optimized' containing
        filtered lists.
    
    Example:
        >>> deadband_filter([-5.0, 10.0, 75.0, 0.0, 30.0, 100.0])
        {'p0_negative': [-5.0, 0.0], 'p1_safe': [10.0, 30.0], 'p2_optimized': [75.0, 100.0]}
    """
    if not isinstance(values, list):
        raise TypeError("Input must be a list")
    
    p0_negative = []
    p1_safe = []

        if not isinstance(val, (int, float)):
            continue  # Skip non-numeric values
        
        if val <= 0:
            p0_negative.append(val)
        elif val <= 50:
            p1_safe.append(val)
        else:
            p2_optimized.append(val)

        'p1_safe': p1_safe,

    }

def parse_fleet_log(log_line: str) -> Optional[Dict[str, str]]:
    """
    Parse a fleet log line into structured dictionary.
    
    Expected format: 'TIMESTAMP | AGENT | CYCLE | ACTION | STATUS'
    
    Args:
        log_line: A single log line string.
    
    Returns:
        Dictionary with keys: timestamp, agent, cycle, action, status
        Returns None for malformed lines.
    
    Example:
        >>> parse_fleet_log('2026-04-20 06:11 | Alchemist | 233 | TEST | PASS')
        {'timestamp': '2026-04-20 06:11', 'agent': 'Alchemist', 
         'cycle': '233', 'action': 'TEST', 'status': 'PASS'}
    """
    if not log_line or not isinstance(log_line, str):
        return None
    
    # Split by pipe with optional whitespace
    parts = [part.strip() for part in log_line.split('|')]
    
    if len(parts) != 5:
        # Try regex for more flexible parsing
        pattern = r'(.+?)\s*\|\s*(.+?)\s*\|\s*(.+?)\s*\|\s*(.+?)\s*\|\s*(.+)'
        match = re.match(pattern, log_line)
        if not match:
            return None
        parts = [match.group(i).strip() for i in range(1, 6)]
    
    return {
        'timestamp': parts[0],
        'agent': parts[1],
        'cycle': parts[2],
        'action': parts[3],
        'status': parts[4]
    }
