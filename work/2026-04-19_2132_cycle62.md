def deadband_filter(data, lower, upper):
    """
    Implements the Deadband Protocol for numeric data filtering.
    
    P0 (negative space): Reject invalid inputs.
    P1 (safe channels): Identify values within the deadband range [lower, upper].
    P2 (optimize): Return filtered list with only in-range values.
    
    Args:
        data (list): List of numeric values (int/float).
        lower (float): Lower bound of deadband (inclusive).
        upper (float): Upper bound of deadband (inclusive).
    
    Returns:
        list: Values within the deadband range.
    
    Raises:
        TypeError: If data is not a list or bounds are non-numeric.
        ValueError: If lower > upper.
    """
    # P0: Map negative space - validate inputs
    if not isinstance(data, list):
        raise TypeError("data must be a list")
    if not all(isinstance(x, (int, float)) for x in data):
        raise TypeError("All elements in data must be numeric")
    if not isinstance(lower, (int, float)) or not isinstance(upper, (int, float)):
        raise TypeError("lower and upper must be numeric")
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    # P1: Find safe channels - identify values within deadband
    safe_values = [x for x in data if lower <= x <= upper]
    
    # P2: Optimize within channels - return filtered list
    return safe_values
```

import re
from pathlib import Path

def parse_fleet_context(filepath):
    """
    Parse a fleet-knowledge markdown file into structured data.
    
    Args:
        filepath (str/Path): Path to the markdown file.
    
    Returns:
        dict: Structured data with keys:
            - title: Document title (first # header)
            - sections: List of section headers
            - agents: List of mentioned agents
            - technologies: List of technologies
            - rules: List of rules
    
    Raises:
        FileNotFoundError: If file doesn't exist.
        ValueError: If file is empty or malformed.
    """
    path = Path(filepath)
    if not path.exists():
        raise FileNotFoundError(f"File not found: {filepath}")
    
    content = path.read_text(encoding='utf-8')
    if not content.strip():
        raise ValueError("File is empty")
    
    result = {
        'title': '',
        'sections': [],
        'agents': [],
        'technologies': [],
        'rules': []
    }
    
    # Extract title (first # header)
    title_match = re.search(r'^#\s+(.+)$', content, re.MULTILINE)
    if title_match:
        result['title'] = title_match.group(1).strip()
    
    # Extract section headers (## or ###)
    section_headers = re.findall(r'^#{2,3}\s+(.+)$', content, re.MULTILINE)
    result['sections'] = [h.strip() for h in section_headers]
    
    # Simple pattern matching for agents (capitalized words ending in numbers)
    agent_pattern = r'\b([A-Z][a-z]+[0-9]+)\b'
    result['agents'] = list(set(re.findall(agent_pattern, content)))
    
    # Technologies (bold text in markdown)
    tech_pattern = r'\*\*([^*]+)\*\*'
    technologies = re.findall(tech_pattern, content)
    result['technologies'] = [t.strip() for t in technologies if len(t) > 3]
    
    # Rules (lines starting with dash after "Rules" section)
    lines = content.split('\n')
    in_rules_section = False
    for line in lines:
        if line.strip().startswith('## Rules'):
            in_rules_section = True
            continue
        if in_rules_section and line.strip().startswith('##'):
            break
        if in_rules_section and line.strip().startswith('-'):
            result['rules'].append(line.strip()[1:].strip())
    
    return result
```
