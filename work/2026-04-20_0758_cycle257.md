```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of steps against the Deadband Protocol (P0->P1->P2).
    P0: Map negative space. P1: Find safe channels. P2: Optimize.
    Strict priority: P0 must be present, P1 must follow P0, P2 must be last.
    No skipping allowed.

    Args:
        steps: List of step identifiers (e.g., ["P0", "P1", "P2"]).

    Returns:
        Tuple of (is_valid: bool, error_message: str). Error message is empty if valid.
    """
    if not steps:
        return False, "Empty step list. Protocol requires at least P0."

    # Check for required P0
    if "P0" not in steps:
        return False, "Missing P0 (map negative space). Cannot start protocol."

    p0_index = steps.index("P0")
    p1_index = steps.index("P1") if "P1" in steps else -1
    p2_index = steps.index("P2") if "P2" in steps else -1

    # P1 must exist and come after P0
    if p1_index == -1:
        return False, "Missing P1 (find safe channels). Protocol incomplete."
    if p1_index <= p0_index:
        return False, f"P1 at index {p1_index} must come after P0 at index {p0_index}."

    # P2 must exist and be the last step
    if p2_index == -1:
        return False, "Missing P2 (optimize). Protocol incomplete."
    if p2_index != len(steps) - 1:
        return False, f"P2 must be the final step. Found at index {p2_index}, last index is {len(steps)-1}."

    # Check for no skipping: order must be P0, P1, P2 with no other major steps between?
    # Protocol implies only these three steps in sequence. Reject any other "P" steps.
    allowed = {"P0", "P1", "P2"}
    for step in steps:
        if step not in allowed:
            return False, f"Invalid step '{step}'. Only P0, P1, P2 are allowed."

    # Verify sequence
    expected_order = ["P0", "P1", "P2"]
    filtered = [s for s in steps if s in allowed]
    if filtered != expected_order:
        return False, f"Step sequence {filtered} does not match required order {expected_order}. Skipping detected."

    return True, ""
```

```python
import re

def parse_fleet_tile(markdown_text: str) -> dict:
    """
    Parses a fleet training tile markdown text into a structured dictionary.
    Expected format mirrors our cycle files (e.g., starts with '# Cycle X').

    Args:
        markdown_text: Raw markdown string of a tile.

    Returns:
        Dictionary with keys: 'cycle', 'phase', 'task', 'content'.
        Values are extracted strings. Missing sections will have empty string values.
    """
    result = {
        "cycle": "",
        "phase": "",
        "task": "",
        "content": ""
    }

    lines = markdown_text.split('\n')
    content_lines = []

    for line in lines:
        # Extract Cycle number (e.g., '# Cycle 257')
        cycle_match = re.match(r'^# Cycle\s+(\d+)', line)
        if cycle_match and not result["cycle"]:
            result["cycle"] = cycle_match.group(1)
            continue

        # Extract Phase (e.g., '**Phase:** 4' or '**Phase 4:**')
        phase_match = re.match(r'^\*\*Phase[:\s]*(\d+)\*\*', line)
        if phase_match and not result["phase"]:
            result["phase"] = phase_match.group(1)
            continue

        # Extract Task (e.g., '**Task:** Compare models...')
        task_match = re.match(r'^\*\*Task[:\s]*(.+?)\*\*', line)
        if task_match and not result["task"]:
            result["task"] = task_match.group(1).strip()
            continue

        # Collect remaining lines as content, skip headers and bold markers
        if line.strip() and not line.startswith('#') and not re.match(r'^\*\*.+\*\*', line):
            content_lines.append(line.strip())

    result["content"] = '\n'.join(content_lines)
    return result
```

```python
from typing import List

def simulate_room_training(initial_score: float, tiles: List[float], decay: float = 0.95) -> List[float]:
    """
    Simulates a PLATO room's training score over a series of tile submissions.
    Implements a simple exponential moving average: new_score = (old_score * decay) + (tile * (1 - decay)).

    Args:
        initial_score: Starting performance score (0.0-1.0).
        tiles: List of tile quality scores (0.0-1.0).
        decay: Decay factor for previous score (default 0.95). Higher = memory persists longer.

    Returns:
        List of scores after processing each tile, length len(tiles).
    """
    if not 0 <= initial_score <= 1:
        raise ValueError(f"initial_score {initial_score} must be between 0.0 and 1.0")
    if not 0 < decay < 1:
        raise ValueError(f"decay {decay} must be between 0.0 and 1.0 (exclusive)")

    scores = []
    current = initial_score

    for tile in tiles:
        if not 0 <= tile <= 1:
            raise ValueError(f"Tile score {tile} must be between 0.0 and 1.0")
        current = (current * decay) + (tile * (1 - decay))
        scores.append(round(current, 4))  # Round for readability

    return scores
```
