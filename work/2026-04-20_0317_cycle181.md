def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a Deadband Protocol sequence.
    Returns (is_valid, message).
    """
    if not steps:
        return False, "Protocol must contain at least one step."
    
    allowed = {"P0", "P1", "P2"}
    transitions = {
        "P0": {"P0", "P1", "P2"},
        "P1": {"P1", "P2"},
        "P2": {"P2"}
    }
    
    if steps[0] != "P0":
        return False, "Protocol must start with P0."
    
    for i, step in enumerate(steps):
        if step not in allowed:
            return False, f"Invalid step '{step}' at position {i}."
        if i > 0:
            prev = steps[i-1]
            if step not in transitions[prev]:
                return False, f"Invalid transition from {prev} to {step} at position {i}."
    
    return True, "Valid protocol"


def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet tile text into a dictionary.
    Returns None if parsing fails.
    """
    try:
        lines = tile_text.strip().split('\n')
        result = {}
        for line in lines:
            line = line.strip()
            if line.startswith('-'):
                line = line[1:].strip()
                if ':' in line:
                    key, value = line.split(':', 1)
                    key = key.strip()
                    value = value.strip()
                    if key == 'score':
                        result[key] = int(value)
                    else:
                        result[key] = value
        required = {'model', 'score', 'task'}
        if not all(req in result for req in required):
            return None
        return result
    except Exception:
        return None


def simulate_room_training(initial_score: float, tiles: list[dict]) -> float:
    """
    Simulates room training with exponential moving average.
    Each tile updates: score = score * 0.9 + tile_score * 0.1
    """
    current = initial_score
    for tile in tiles:
        current = current * 0.9 + tile.get("score", 0) * 0.1
    return current
