#### **Function 1: deadband_filter**

```python
def deadband_filter(values: list[float], lower: float, upper: float) -> list[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0 (negative space): Values within [lower, upper] are set to 0.0 (avoid noise).
    P1 (safe channels): Values outside deadband pass through unchanged.
    P2 (optimization): Not applied here—this is a P0/P1 filter only.
    
    Args:
        values: Input list of float values.
        lower: Deadband lower bound.
        upper: Deadband upper bound.
    
    Returns:
        Filtered list where values inside deadband are zeroed.
    
    Raises:
        ValueError: If lower > upper.
    """
    if lower > upper:
        raise ValueError(f"Invalid deadband: lower ({lower}) > upper ({upper})")
    
    filtered = []
    for v in values:
        if lower <= v <= upper:
            filtered.append(0.0)  # P0: suppress noise
        else:
            filtered.append(v)    # P1: pass through safe channel
    return filtered
```

#### **Function 2: parse_fleet_context**

```python
import re
from typing import Dict, List, Optional

def parse_fleet_context(markdown_text: str) -> Dict[str, Optional[List[str]]]:
    """
    Parse a fleet-knowledge Markdown file into structured sections.
    
    Extracts:
    - Who We Are
    - Fleet Doctrine
    - Key Technologies
    - Architecture
    - Rules
    
    Returns a dict with section names as keys and list of lines as values.
    Sections not found will have value None.
    """
    sections = {
        "Who We Are": None,
        "Fleet Doctrine": None,
        "Key Technologies": None,
        "Architecture": None,
        "Rules": None
    }
    
    current_section = None
    lines = markdown_text.split('\n')
    
    for line in lines:
        # Detect section headers (## Heading)
        match = re.match(r'^##\s+(.+)$', line.strip())
        if match:
            header = match.group(1)
            if header in sections:
                current_section = header
                sections[current_section] = []
            else:
                current_section = None
        elif current_section and line.strip():
            # Skip sub-headers (###) and empty lines
            if not line.strip().startswith('###'):
                sections[current_section].append(line.strip())
    
    return sections
```

```python
import json

from typing import Dict, Any

def validate_training_tile(tile_json: Dict[str, Any]) -> tuple[bool, list[str]]:
    """
    Validate a training tile against fleet schema.
    
    Required fields:
    - agent_id: str
    - cycle: int
    - phase: int
    - content: str (non-empty)
    - timestamp: ISO 8601 format
    
    Returns (is_valid, list_of_errors).
    """
    errors = []
    
    # Required fields
    required = ['agent_id', 'cycle', 'phase', 'content', 'timestamp']
    for field in required:
        if field not in tile_json:
            errors.append(f"Missing required field: {field}")
    
    if errors:
        return False, errors
    
    # Type checks
    if not isinstance(tile_json['agent_id'], str):
        errors.append("agent_id must be string")
    if not isinstance(tile_json['cycle'], int) or tile_json['cycle'] < 0:
        errors.append("cycle must be non-negative integer")
    if not isinstance(tile_json['phase'], int) or tile_json['phase'] not in [1, 2, 3, 4]:
        errors.append("phase must be 1, 2, 3, or 4")
    if not isinstance(tile_json['content'], str) or len(tile_json['content'].strip()) == 0:
        errors.append("content must be non-empty string")
    
    # Timestamp format
    try:
        datetime.fromisoformat(tile_json['timestamp'].replace('Z', '+00:00'))
    except (ValueError, TypeError):
        errors.append("timestamp must be ISO 8601 format")
    
    return len(errors) == 0, errors
```

```python
from typing import List

def deadband_filter(values: List[float], lower: float, upper: float) -> List[float]:
    """
    Implements the Deadband Protocol: zero values within [lower, upper] (P0),
    pass others unchanged (P1).
    
    Args:
        values: List of floating-point numbers.
        lower: Lower bound of deadband (inclusive).
        upper: Upper bound of deadband (inclusive).
    
    Returns:
        List with same length as values, where values in deadband are 0.0.
    
    Raises:
        ValueError: If lower > upper.
    """
    if lower > upper:
        raise ValueError(f"Deadband bounds invalid: {lower} > {upper}")
    
    result = []
    for v in values:
        if v >= lower and v <= upper:
            result.append(0.0)  # P0: eliminate noise
        else:
            result.append(v)    # P1: preserve signal
    return result
```
