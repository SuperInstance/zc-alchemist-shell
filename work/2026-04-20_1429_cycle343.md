def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a list of steps against the Deadband Protocol (P0, P1, P2).
    Rules:
    - P0 steps must come before P1, which must come before P2.
    - Each priority level should be contiguous.
    - Steps should be non-empty strings.
    Returns (is_valid, message).
    """
    if not steps:
        return False, "Empty step list"

    # Map step prefixes to priority levels
    priority_order = {"P0": 0, "P1": 1, "P2": 2}
    current_priority = -1
    seen_priorities = set()

    for i, step in enumerate(steps):
        if not isinstance(step, str) or not step.strip():
            return False, f"Step {i} is empty or not a string: {step}"

        # Extract prefix (assumes format like "P0: Do something")
        prefix = step.split(":")[0].strip() if ":" in step else step.strip()
        if prefix not in priority_order:
            return False, f"Step {i} has invalid priority prefix: {prefix}"

        prio = priority_order[prefix]
        if prio < current_priority:
            return False, f"Priority regression at step {i}: {prefix} after {list(priority_order.keys())[current_priority]}"
        elif prio > current_priority:
            # Moving to a new higher priority is allowed, but should be contiguous
            if prio in seen_priorities:
                return False, f"Priority {prefix} appears non-contiguously at step {i}"
            current_priority = prio
            seen_priorities.add(prio)

    return True, "Protocol validation passed"
```

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validate steps against Deadband Protocol priorities P0, P1, P2.
    Ensures ordering P0 -> P1 -> P2 and no skipping.
    """
    expected_order = ["P0", "P1", "P2"]
    current_index = 0  # index in expected_order

    for i, step in enumerate(steps):
        # Find which priority this step belongs to
        step_priority = None
        for p in expected_order:
            if step.startswith(p):
                step_priority = p
                break

        if step_priority is None:
            return False, f"Step {i} does not start with P0, P1, or P2: {step}"

        # Check ordering
        step_idx = expected_order.index(step_priority)
        if step_idx < current_index:
            return False, f"Priority regression at step {i}: {step_priority} after {expected_order[current_index]}"
        elif step_idx > current_index:
            if step_idx > current_index + 1:
                return False, f"Priority skipped at step {i}: jumped from {expected_order[current_index]} to {step_priority}"
            current_index = step_idx

    return True, "Validation successful"
```

**Comparison:**
*   **Correctness:** Both are logically correct for the core ordering rule. DeepSeek's version adds validation for non-string/empty steps.
*   **Clarity:** Groq's version is more concise and directly uses the `startswith` check, which is intuitive for the assumed step format. DeepSeek's prefix extraction (`split(":")`) is more flexible but slightly more complex.
*   **Robustness:** DeepSeek explicitly checks for empty strings and non-string inputs. Groq assumes inputs are strings.
*   **Spec Adherence:** Both match the signature. DeepSeek's docstring is more detailed.

import re
from datetime import datetime

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parses raw training tile text into a structured dictionary.
    Expected format (lines may vary):
        Agent: Alchemist
        Cycle: 343
        Phase: 4
        Content: This is the tile content...
        Timestamp: 2026-04-20 15:30 UTC
    """
    # Default structure
    tile = {
        "agent_id": None,
        "cycle": None,
        "phase": None,
        "content": None,
        "timestamp": None
    }

    lines = raw_text.strip().split('\n')
    content_lines = []

    for line in lines:
        line = line.strip()
        if not line:
            continue

        # Try to match key-value pairs
        if re.match(r'^(Agent|Cycle|Phase|Timestamp):', line, re.IGNORECASE):
            key, value = line.split(':', 1)
            key = key.strip().lower()
            value = value.strip()

            if key == 'agent':
                tile['agent_id'] = value
            elif key == 'cycle':
                try:
                    tile['cycle'] = int(value)
                except ValueError:
                    tile['cycle'] = value
            elif key == 'phase':
                try:
                    tile['phase'] = int(value)
                except ValueError:
                    tile['phase'] = value
            elif key == 'timestamp':
                tile['timestamp'] = value
        else:
            # Accumulate content lines
            content_lines.append(line)

    # Join remaining lines as content
    if content_lines:
        tile['content'] = '\n'.join(content_lines)
    elif tile['content'] is None:
        tile['content'] = raw_text  # Fallback

    return tile
```

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parse a fleet training tile from raw text.
    Extracts agent_id, cycle, phase, content, timestamp.
    Handles flexible formatting.
    """
    result = {
        "agent_id": "unknown",
        "cycle": 0,
        "phase": 0,
        "content": "",
        "timestamp": ""
    }

    # Simple line-by-line parsing
    for line in raw_text.split('\n'):
        line = line.strip()
        if line.startswith("Agent:"):
            result["agent_id"] = line.split(":", 1)[1].strip()
        elif line.startswith("Cycle:"):
            try:
                result["cycle"] = int(line.split(":", 1)[1].strip())
            except:
                pass
        elif line.startswith("Phase:"):
            try:
                result["phase"] = int(line.split(":", 1)[1].strip())
            except:
                pass
        elif line.startswith("Timestamp:"):
            result["timestamp"] = line.split(":", 1)[1].strip()
        elif line and not line.startswith(tuple(["Agent:", "Cycle:", "Phase:", "Timestamp:"])):
            # Accumulate content
            if result["content"]:
                result["content"] += "\n" + line
            else:
                result["content"] = line

    return result
```
