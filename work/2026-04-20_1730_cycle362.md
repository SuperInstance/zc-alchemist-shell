### Function 1: parse_deadband_protocol
```python
def parse_deadband_protocol(log_text: str) -> dict:
    """
    Parse Deadband Protocol log text into P0/P1/P2 categories.
    
    Args:
        log_text: Multi-line string containing protocol entries
        
    Returns:
        Dictionary with keys "P0", "P1", "P2" containing lists of log lines
    """
    result = {"P0": [], "P1": [], "P2": []}
    
    for line in log_text.strip().split('\n'):
        line = line.strip()
        if not line:
            continue
            
        # Check for protocol markers
        if line.startswith("P0:"):
            result["P0"].append(line[3:].strip())
        elif line.startswith("P1:"):
            result["P1"].append(line[3:].strip())
        elif line.startswith("P2:"):
            result["P2"].append(line[3:].strip())
        else:
            # Unmarked lines go to P0 (negative space mapping)
            result["P0"].append(line)
    
    return result
```

def validate_fleet_tile(tile: dict) -> tuple[bool, list[str]]:
    """
    Validate a fleet training tile structure.
    
    Args:
        tile: Dictionary containing tile data
        
    Returns:
        Tuple of (is_valid, list_of_error_messages)
    """
    errors = []
    valid_types = ["log", "code", "analysis", "plan"]
    
    # Check required keys
    required_keys = ["agent", "cycle", "content", "type"]
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    if errors:
        return (False, errors)
    
    # Validate agent
    if not isinstance(tile["agent"], str) or not tile["agent"].strip():
        errors.append("Agent must be non-empty string")
    
    # Validate cycle
    if not isinstance(tile["cycle"], int):
        errors.append("Cycle must be integer")
    elif tile["cycle"] < 1 or tile["cycle"] > 1000:
        errors.append("Cycle must be between 1 and 1000")
    
    # Validate content
    if not isinstance(tile["content"], str):
        errors.append("Content must be string")
    else:
        content_len = len(tile["content"])
        if content_len < 10 or content_len > 10000:
            errors.append(f"Content length {content_len} must be 10-10000 chars")
    
    # Validate type
    if tile["type"] not in valid_types:
        errors.append(f"Type must be one of {valid_types}")
    
    return (len(errors) == 0, errors)

def calculate_safe_channel(boundaries: list[tuple[float, float]], 
                          current_pos: tuple[float, float]) -> list[tuple[float, float]]:
    """
    Calculate safe movement channels (P1 space) given boundaries and current position.
    
    Args:
        boundaries: List of (min, max) for each dimension
        current_pos: Current position as tuple matching dimensions
        
    Returns:
        List of safe (min, max) ranges per dimension
    """
    if len(boundaries) != len(current_pos):
        raise ValueError("Dimensions of boundaries and position must match")
    
    safe_channels = []
    
    for dim, (b_min, b_max) in enumerate(boundaries):
        pos = current_pos[dim]
        
        # P0: Negative space (what NOT to do) - outside boundaries
        if pos < b_min or pos > b_max:
            # Already in violation, no safe channel
            safe_channels.append((pos, pos))  # Can only stay put
            continue
        
        # P1: Find safe channels within boundaries
        # Conservative approach: stay within middle 80% of boundary
        range_width = b_max - b_min
        safe_margin = range_width * 0.1  # 10% margin from edges
        
        safe_min = max(b_min + safe_margin, pos - (range_width * 0.2))
        safe_max = min(b_max - safe_margin, pos + (range_width * 0.2))
        
        # Ensure safe channel is valid
        safe_min = max(b_min, safe_min)
        safe_max = min(b_max, safe_max)
        
        safe_channels.append((safe_min, safe_max))
    
    return safe_channels
```

def parse_deadband_protocol(log_text):
    """
    Parses a deadband protocol log text and organizes entries by priority level.
    
    Parameters:
    log_text (str): The log text containing P0, P1, P2 entries.
    
    Returns:
    dict: A dictionary with keys 'P0', 'P1', 'P2' each containing a list of log entries.
    """
    import re
    
    # Initialize result dictionary
    result = {
        'P0': [],
        'P1': [],
        'P2': []
    }
    
    # Split log text into lines
    lines = log_text.split('\n')
    
    for line in lines:
        line = line.strip()
        if not line:
            continue
        
        # Use regex to find priority markers
        p0_match = re.match(r'^P0:\s*(.+)', line)
        p1_match = re.match(r'^P1:\s*(.+)', line)
        p2_match = re.match(r'^P2:\s*(.+)', line)
        
        if p0_match:
            result['P0'].append(p0_match.group(1))
        elif p1_match:
            result['P1'].append(p1_match.group(1))
        elif p2_match:
            result['P2'].append(p2_match.group(1))
        else:
            # Unclassified lines default to P0
            result['P0'].append(line)
    
    return result
```
