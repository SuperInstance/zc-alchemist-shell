) -> Dict[str, List[float]]:
    """
    Apply the Deadband Protocol to a list of float values.

    Parameters:
    -----------
    values : List[float]
        Input list of float values.
    lower_bound : float, default=-10.0
        Lower bound for safe channel (P1).
    upper_bound : float, default=10.0
        Upper bound for safe channel (P1).
    scale_factor : float, default=1.0
        Scaling factor for optimization (P2).

    Returns:
    --------
    Dict[str, List[float]]
        Dictionary with keys:
        - "p0_rocks": values that are NaN, infinity, or outside bounds
        - "p1_safe": values within [lower_bound, upper_bound] (rocks removed)
        - "p2_optimized": p1_safe values multiplied by scale_factor

    Example:
    --------
    >>> deadband_filter([1.0, float('nan'), 15.0, -5.0, float('inf')])
    {'p0_rocks': [nan, 15.0, inf], 'p1_safe': [1.0, -5.0], 'p2_optimized': [1.0, -5.0]}
    """

    p1_safe = []

    for v in values:
        if math.isnan(v) or math.isinf(v):
            p0_rocks.append(v)
        elif v < lower_bound or v > upper_bound:
            p0_rocks.append(v)
        else:
            p1_safe.append(v)

    p2_optimized = [v * scale_factor for v in p1_safe]

    return {
        "p0_rocks": p0_rocks,
        "p1_safe": p1_safe,
        "p2_optimized": p2_optimized
    }

    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Identify rocks (NaN, Inf, out-of-bounds)
    P1: Extract safe values within bounds
    P2: Scale safe values by scale_factor
    
    Args:
        values: List of float values to process
        lower_bound: Lower bound for safe values
        upper_bound: Upper bound for safe values  
        scale_factor: Factor to scale safe values by
    
    Returns:
        Dictionary with three lists:
        - p0_rocks: Invalid/out-of-bounds values
        - p1_safe: Valid values within bounds
        - p2_optimized: Safe values scaled by scale_factor
    
    Example:
        >>> result = deadband_filter([1.0, float('nan'), 20.0, -3.0])
        >>> result['p0_rocks']
        [nan, 20.0]
        >>> result['p1_safe']
        [1.0, -3.0]
        >>> result['p2_optimized']
        [1.0, -3.0]
    """
    p0_rocks = []
    p1_safe = []
    
    for value in values:
        # Check for NaN or infinity
        if math.isnan(value) or math.isinf(value):
            p0_rocks.append(value)
        # Check bounds
        elif value < lower_bound or value > upper_bound:
            p0_rocks.append(value)
        else:
            p1_safe.append(value)
    
    # Apply scaling to safe values
    p2_optimized = [value * scale_factor for value in p1_safe]
    
    return {
        "p0_rocks": p0_rocks,
        "p1_safe": p1_safe, 
        "p2_optimized": p2_optimized
    }
