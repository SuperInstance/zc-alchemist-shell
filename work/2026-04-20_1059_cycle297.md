def validate_deadband_protocol(actions):
    """
    Validate if actions follow Deadband Protocol strict priority order.
    P0: no "skip", P1: has "find", P2: has "optimize" after find.
    """
    actions_lower = [a.lower() for a in actions]
    
    # P0: No skip allowed
    if any("skip" in a for a in actions_lower):
        return False
    
    # P1: Must have find
    find_indices = [i for i, a in enumerate(actions_lower) if "find" in a]
    if not find_indices:
        return False
    
    # P2: Must have optimize after at least one find
    first_find_idx = find_indices[0]
    optimize_indices = [i for i, a in enumerate(actions_lower) if "optimize" in a]
    if not any(idx > first_find_idx for idx in optimize_indices):
        return False
    
    return True

def parse_fleet_tile(tile_string):
    """
    Parse fleet tile string into structured dict.
    Handles malformed input gracefully.
    """
    result = {
        "agent": "unknown",
        "cycle": 0,
        "content_hash": "",
        "valid": False
    }
    
    if not tile_string or not isinstance(tile_string, str):
        return result
    
    try:
        pairs = tile_string.split(';')
        data = {}
        for pair in pairs:
            if ':' in pair:
                key, value = pair.split(':', 1)
                data[key.strip()] = value.strip()
        
        # Check required fields
        if "agent" in data:
            result["agent"] = data["agent"]
        if "content" in data:
            result["content_hash"] = data["content"]
        
        if "cycle" in data:
            try:
                result["cycle"] = int(data["cycle"])
            except ValueError:
                result["cycle"] = 0
                result["valid"] = False
                return result
        
        # All required fields present and cycle is int
        required = {"agent", "cycle", "content"}
        if all(req in data for req in required):
            result["valid"] = True
            
    except Exception:
        result["valid"] = False
    
    return result

def simulate_channel_optimization(scores, strategy):
    """
    Apply P2 optimization strategy to channel scores.
    Returns (optimized_value, chosen_index).
    """
    if not scores:
        return (0.0, -1)
    
    if not isinstance(scores, (list, tuple)):
        raise TypeError("scores must be a list or tuple")
    
    valid_strategies = {'max', 'min', 'mean'}
    if strategy not in valid_strategies:
        raise ValueError(f"strategy must be one of {valid_strategies}")
    
    if strategy == 'max':
        max_val = max(scores)
        idx = scores.index(max_val)
        return (max_val, idx)
    elif strategy == 'min':
        min_val = min(scores)
        idx = scores.index(min_val)
        return (min_val, idx)
    else:  # mean
        mean_val = sum(scores) / len(scores)
        return (mean_val, -1)
