## **3. Model Outputs**

### **DeepSeek-chat Implementation**

```python
import math
from typing import Optional

def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol filtering.
    
    P0: Remove values that are 'rocks' (None, NaN, inf).
    P1: Find safe channels (values within ±deadband of zero become zero).
    P2: Optimize (return filtered list).
    """
    result = []
    for v in values:
        # P0: Remove rocks
        if v is None:
            continue
        try:
            if math.isnan(v) or math.isinf(v):
                continue
        except TypeError:
            continue
        
        # P1: Safe channel
        if -deadband <= v <= deadband:
            result.append(0.0)
        else:
            result.append(v)
    
    return result


def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a fleet log line with format:
    [TIMESTAMP] AGENT:LEVEL > message @extra=data
    """
    import re
    
    pattern = r'\[([^\]]+)\]\s+([^:]+):([^\s>]+)\s*>\s*(.+?)(?:\s*@([^=]+)=(.+))?$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        return {}
    
    timestamp, agent, level, message, extra_key, extra_value = match.groups()
    
    result = {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': message.strip()
    }
    
    if extra_key and extra_value:
        result['extra'] = {extra_key: extra_value}
    else:
        result['extra'] = {}
    
    return result


def constraint_snap(point: tuple[float, float], 
                   grid_size: float = 1.0,
                   constraints: Optional[list[tuple[float, float]]] = None) -> tuple[float, float]:
    """
    Snap a 2D point to nearest grid intersection, respecting constraints.
    """
    x, y = point
    
    if constraints and len(constraints) > 0:
        # Snap to nearest constraint point
        min_dist = float('inf')
        snapped = (x, y)
        
        for cx, cy in constraints:
            dist = math.sqrt((x - cx)**2 + (y - cy)**2)
            if dist < min_dist:
                min_dist = dist
                snapped = (cx, cy)
        
        return snapped
    
    # Grid snapping
    snapped_x = round(x / grid_size) * grid_size
    snapped_y = round(y / grid_size) * grid_size
    
    return (snapped_x, snapped_y)
```

### **Groq Llama-70b Implementation**

```python
import math
from typing import List, Tuple, Optional, Dict, Any

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of float values.
    
    Implements the three priority levels:
    P0: Filter out invalid values (None, NaN, infinity)
    P1: Values within deadband of zero are set to zero
    P2: Return the processed list
    """
    filtered = []
    
    for val in values:
        # P0: Remove invalid values
        if val is None:
            continue
            
        # Check for NaN or infinity
        try:
            if math.isnan(val) or math.isinf(val):
                continue
        except (TypeError, ValueError):
            continue
        
        # P1: Apply deadband
        if abs(val) <= deadband:
            filtered.append(0.0)
        else:
            filtered.append(val)
    
    return filtered


def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line.
    
    Expected format: [TIMESTAMP] AGENT:LEVEL > message @extra=data
    
    Returns a dictionary with parsed components.
    """
    import re
    
    # Remove leading/trailing whitespace
    line = log_line.strip()
    
    # Regex pattern to match the log format
    pattern = r'^\[([^\]]+)\]\s+([^:]+):([^\s>]+)\s*>\s*(.+?)(?:\s+@([^=]+)=(.+))?$'
    
    match = re.match(pattern, line)
    if not match:
        return {}
    
    timestamp = match.group(1)
    agent = match.group(2)
    level = match.group(3)
    message = match.group(4).strip()
    extra_key = match.group(5)
    extra_value = match.group(6)
    
    result = {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': message
    }
    
    # Add extra data if present
    if extra_key and extra_value:
        result['extra'] = {extra_key: extra_value}
    else:
        result['extra'] = {}
    
    return result


def constraint_snap(point: Tuple[float, float], 
                   grid_size: float = 1.0,
                   constraints: Optional[List[Tuple[float, float]]] = None) -> Tuple[float, float]:
    """
    Snap a 2D point to a grid or to constraint points.
    
    If constraints are provided, snap to the nearest constraint point.
    Otherwise, snap to the nearest grid intersection.
    """
    x, y = point
    
    # If constraints are provided, use them
    if constraints:
        min_distance = float('inf')
        best_point = (x, y)
        
        for constraint_point in constraints:
            cx, cy = constraint_point
            distance = math.sqrt((x - cx)**2 + (y - cy)**2)
            
            if distance < min_distance:
                min_distance = distance
                best_point = (cx, cy)
        
        return best_point
    
    # Otherwise, snap to grid
    snapped_x = round(x / grid_size) * grid_size
    snapped_y = round(y / grid_size) * grid_size
    
    return (snapped_x, snapped_y)
```
