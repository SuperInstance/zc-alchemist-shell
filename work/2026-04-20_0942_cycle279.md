from typing import List, Union

def deadband_filter(
    values: List[Union[int, float]],
    lower: Union[int, float],
    upper: Union[int, float]
) -> List[Union[int, float]]:
    """
    Apply a deadband filter to a list of numeric values.
    
    The deadband protocol (P0, P1, P2) is implemented as:
    - P0 (negative space): values outside [lower, upper] are clamped to nearest bound.
    - P1 (safe channel): values inside the band pass unchanged.
    - P2 (optimization): not applied here (strict priority: never skip to P2).
    
    Example:
        >>> deadband_filter([1, 5, 10, 15, 20], lower=5, upper=15)
        [5, 5, 10, 15, 15]
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    filtered = []
    for v in values:
        if v < lower:
            filtered.append(lower)
        elif v > upper:
            filtered.append(upper)
        else:
            filtered.append(v)
    return filtered
```

from typing import Tuple
import math

def constraint_snap(
    point: Tuple[float, float],
    line_start: Tuple[float, float],
    line_end: Tuple[float, float],
    tolerance: float = 1e-6
) -> Tuple[float, float]:
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Based on geometric constraint snapping (constraint-theory-core).
    Uses vector projection to find the closest point on the infinite line,
    then clamps to the segment endpoints.
    
    Example:
        >>> constraint_snap((2, 3), (0, 0), (5, 5), tolerance=0.5)
        (2.5, 2.5)
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector from start to end
    dx = x2 - x1
    dy = y2 - y1
    seg_len_sq = dx*dx + dy*dy
    
    # If segment is degenerate, return start point
    if seg_len_sq < tolerance:
        return (x1, y1)
    
    # Vector from start to point
    vx = px - x1
    vy = py - y1
    
    # Projection parameter
    t = (vx*dx + vy*dy) / seg_len_sq
    
    # Clamp to segment
    t = max(0.0, min(1.0, t))
    
    # Closest point on segment
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    
    # Snap only if within tolerance
    dist = math.hypot(px - closest_x, py - closest_y)
    if dist <= tolerance:
        return (closest_x, closest_y)
    else:
        return (px, py)
```

from typing import Dict, Any, Union
import json
import hashlib
from datetime import datetime

def tile_encoder(
    interaction_data: Dict[str, Any],
    metadata: Dict[str, Union[str, int, float]]
) -> Dict[str, Any]:
    """
    Encode an agent interaction into a PLATO training tile format.
    
    A training tile contains:
    - interaction: the raw interaction data
    - metadata: context about the agent, cycle, phase, etc.
    - tile_id: SHA‑256 hash of the canonicalized tile
    - timestamp: ISO‑8601 creation time
    - version: tile format version (currently 1)
    
    Example:
        >>> tile = tile_encoder(
        ...     {"prompt": "test", "response": "ok"},
        ...     {"agent": "Alchemist", "cycle": 279}
        ... )
        >>> "tile_id" in tile
        True
    """
    # Ensure required metadata fields
    required = ["agent", "cycle", "phase"]
    for field in required:
        if field not in metadata:
            raise ValueError(f"metadata missing required field: {field}")
    
    # Construct tile skeleton
    tile = {
        "version": 1,
        "timestamp": datetime.utcnow().isoformat() + "Z",
        "interaction": interaction_data,
        "metadata": metadata
    }
    
    # Generate deterministic tile_id
    # Canonicalize by sorting keys and using json.dumps with separators
    canonical = json.dumps(tile, sort_keys=True, separators=(',', ':'))
    tile_id = hashlib.sha256(canonical.encode('utf-8')).hexdigest()
    
    tile["tile_id"] = tile_id
    return tile
```

import math

def constraint_snap(point, line_start, line_end, tolerance=1e-6):
    """
    Snap point to line segment if within tolerance.
    
    point: (x, y)
    line_start: (x1, y1)
    line_end: (x2, y2)
    tolerance: max distance for snapping
    
    Returns snapped point (or original if outside tolerance).
    """
    x, y = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Line vector
    dx = x2 - x1
    dy = y2 - y1
    
    # Vector from start to point
    vx = x - x1
    vy = y - y1
    
    # Projection
    dot = vx*dx + vy*dy
    len_sq = dx*dx + dy*dy
    
    if len_sq == 0:
        # Zero‑length segment
        closest = (x1, y1)
    else:
        t = dot / len_sq
        t = max(0, min(1, t))
        closest = (x1 + t*dx, y1 + t*dy)
    
    # Check distance
    dist = math.sqrt((x - closest[0])**2 + (y - closest[1])**2)
    if dist <= tolerance:
        return closest
    return (x, y)
```
