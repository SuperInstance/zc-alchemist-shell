```python
from typing import List, Union

def deadband_filter(data: List[Union[int, float]], lower: float, upper: float) -> List[Union[int, float]]:
    """
    Apply a deadband filter to a list of numeric values.
    
    Values within the inclusive range [lower, upper] are set to zero.
    Values outside the range pass through unchanged.
    
    Args:
        data: List of numeric values (int or float).
        lower: Lower bound of deadband (inclusive).
        upper: Upper bound of deadband (inclusive).
    
    Returns:
        Filtered list with deadband values zeroed.
    
    Raises:
        ValueError: If lower > upper.
    """
    if lower > upper:
        raise ValueError(f"Lower bound ({lower}) cannot be greater than upper bound ({upper})")
    
    result = []
    for value in data:
        if lower <= value <= upper:
            result.append(0)
        else:
            result.append(value)
    return result
```

```python
import re

from typing import Dict, Optional, Union

def parse_fleet_log(log_text: str) -> Dict[str, Union[str, int]]:
    """
    Parse a semi-structured fleet log entry.
    
    Expected format: "[YYYY-MM-DD HH:MM:SS] AGENT_NAME CYCLE_NUMBER: MESSAGE"
    Example: "[2026-04-19 23:34:00] Alchemist 102: Starting phase 4"
    
    Args:
        log_text: A single log line as string.
    
    Returns:
        Dictionary with keys: timestamp (str), agent (str), cycle (int), message (str).
        If parsing fails, returns dict with 'error' key.
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+) (\d+): (.+)'
    match = re.match(pattern, log_text)
    
    if not match:
        return {"error": "Invalid log format"}
    
    timestamp_str, agent, cycle_str, message = match.groups()
    
    # Validate timestamp format (optional, could parse to datetime)
    try:
        datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        return {"error": "Invalid timestamp format"}
    
    return {
        "timestamp": timestamp_str,
        "agent": agent,
        "cycle": int(cycle_str),
        "message": message.strip()
    }
```

```python
import re

def parse_fleet_log(log_text: str) -> dict:
    """
    Parses a fleet log string.
    
    Format: "[timestamp] agent cycle: message"
    Example: "[2026-04-19 23:34:00] Alchemist 102: Starting phase 4"
    
    Args:
        log_text (str): The log string.
    
    Returns:
        dict: Contains 'timestamp', 'agent', 'cycle', 'message'.
        Returns {'error': 'parse failed'} on failure.
    """
    pattern = r'^\[([^\]]+)\]\s+(\w+)\s+(\d+):\s+(.+)$'
    match = re.match(pattern, log_text)
    if match:
        timestamp, agent, cycle, message = match.groups()
        return {
            'timestamp': timestamp,
            'agent': agent,
            'cycle': int(cycle),
            'message': message
        }
    else:
        return {'error': 'parse failed'}
```

- DeepSeek includes datetime validation, more robust.
- DeepSeek uses explicit type hints (`Dict[str, Union[str, int]]`).
- Groq's regex `[^\]]+` is more flexible for timestamps but less strict.
- DeepSeek returns timestamp as string (preserves format), Groq same.
- DeepSeek strips message whitespace.

### **Function 3: constraint_snap**

**DeepSeek-chat output:**

```python
import math
from typing import Tuple

def constraint_snap(point: Tuple[float, float], 
                    line_start: Tuple[float, float], 
                    line_end: Tuple[float, float], 
                    tolerance: float) -> Tuple[float, float]:
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Uses vector projection to find closest point on the infinite line,
    then clamps to segment endpoints. If distance > tolerance, returns original point.
    
    Args:
        point: (x, y) coordinate to snap.
        line_start: (x, y) start of line segment.
        line_end: (x, y) end of line segment.
        tolerance: Maximum distance for snapping.
    
    Returns:
        Snapped point if within tolerance, else original point.
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector from line_start to line_end
    dx = x2 - x1
    dy = y2 - y1
    
    # If line segment is degenerate (zero length), snap to start if within tolerance
    if dx == 0 and dy == 0:
        dist = math.hypot(px - x1, py - y1)
        return (x1, y1) if dist <= tolerance else (px, py)
    
    # Vector from line_start to point
    ax = px - x1
    ay = py - y1
    
    # Projection parameter t (0 at start, 1 at end)
    t = (ax * dx + ay * dy) / (dx * dx + dy * dy)
    
    # Clamp t to [0, 1] to stay on segment
    t_clamped = max(0.0, min(1.0, t))
    
    # Closest point on segment
        closest_x = x1 + t_clamped * dx
        closest_y = y1 + t_clamped * dy
    
    # Distance to closest point
    dist = math.hypot(px - closest_x, py - closest_y)
    
    return (closest_x, closest_y) if dist <= tolerance else (px, py)
```
