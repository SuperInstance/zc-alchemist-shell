```python
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol.
    Protocol requires steps to be in order: P0 (negative space) -> P1 (safe channels) -> P2 (optimization).
    Steps should be prefixed with 'P0:', 'P1:', or 'P2:'.
    
    Args:
        steps: List of step descriptions with protocol prefixes.
    
    Returns:
        Tuple of (is_valid, error_message). Error message is empty if valid.
    """
    expected_phase = 0  # 0 for P0, 1 for P1, 2 for P2
    phase_names = {0: "P0", 1: "P1", 2: "P2"}
    
    for i, step in enumerate(steps):
        step_upper = step.strip().upper()
        
        # Check prefix
        if step_upper.startswith("P0:"):
            current_phase = 0
        elif step_upper.startswith("P1:"):
            current_phase = 1
        elif step_upper.startswith("P2:"):
            current_phase = 2
        else:
            return False, f"Step {i}: '{step}' does not start with P0:, P1:, or P2:"
        
        # Check phase order
        if current_phase < expected_phase:
            return False, f"Step {i}: '{step}' is {phase_names[current_phase]} but expected at least {phase_names[expected_phase]}"
        elif current_phase > expected_phase:
            expected_phase = current_phase
    
    return True, ""
```

#### **Function 2: `parse_fleet_tile`**
```python
import re
from datetime import datetime
from typing import Dict, Optional

def parse_fleet_tile(tile_text: str) -> Dict[str, str]:
    """
    Parses a fleet training tile text into structured data.
    
    Expected format includes headers like:
        # Cycle X â€” AgentName
        **Phase:** Y
        **Task:** Description
    
    Args:
        tile_text: Raw text of a training tile.
    
    Returns:
        Dictionary with keys: agent, cycle, phase, task, content, timestamp.
        Missing values will be empty strings.
    """
    result = {
        "agent": "",
        "cycle": "",
        "phase": "",
        "task": "",
        "content": tile_text.strip(),
        "timestamp": datetime.utcnow().isoformat() + "Z"
    }
    
    # Extract agent and cycle from first line
    first_line_match = re.match(r"# Cycle (\d+) â€” (\w+)", tile_text)
    if first_line_match:
        result["cycle"] = first_line_match.group(1)
        result["agent"] = first_line_match.group(2)
    
    # Extract phase
    phase_match = re.search(r"\*\*Phase:\*\*\s*(\d+)", tile_text)
    if phase_match:
        result["phase"] = phase_match.group(1)
    
    # Extract task - capture until next header or end
    task_match = re.search(r"\*\*Task:\*\*\s*(.+?)(?=\n\s*\*\*|\n\s*#|\n\s*$)", tile_text, re.DOTALL)
    if task_match:
        result["task"] = task_match.group(1).strip()
    
    return result
```
