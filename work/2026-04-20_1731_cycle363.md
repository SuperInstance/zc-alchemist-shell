**DeepSeek-chat Output:**
```python
def deadband_filter(data: float, lower: float, upper: float) -> float | None:
    """
    Implements the Deadband Protocol (P0: avoid, P1: channel, P2: optimize).
    If data is within [lower, upper] (deadband), return None.
    If below lower, return lower - data (negative error).
    If above upper, return data - upper (positive error).

    Args:
        data: Input value to filter.
        lower: Lower bound of deadband.
        upper: Upper bound of deadband.

    Returns:
        Error magnitude outside deadband, or None if within.
    """
    if not (isinstance(data, (int, float)) and
            isinstance(lower, (int, float)) and
            isinstance(upper, (int, float))):
        raise TypeError("All inputs must be numeric")

    if lower >= upper:
        raise ValueError("lower must be less than upper")

    if lower <= data <= upper:
        return None
    elif data < lower:
        return lower - data
    else:  # data > upper
        return data - upper
```

```python
    """
    Apply deadband filtering to a data point.
    Returns None if data is within [lower, upper].
    Otherwise returns the distance from the nearest bound.
    """
    # Input validation
    try:
        data = float(data)
        lower = float(lower)
        upper = float(upper)
    except (ValueError, TypeError):
        raise ValueError("Inputs must be convertible to float")

    if lower > upper:
        raise ValueError("lower must be <= upper")

        return None
```

*   **Clarity:** DeepSeek includes the fleet doctrine (P0/P1/P2) in the docstring, linking code to context. Both have clear docstrings.
*   **Style:** DeepSeek uses type hints (`float | None`), Groq does not. DeepSeek's validation is more granular.

```python
import re

def parse_fleet_log_line(line: str) -> dict[str, str] | None:
    """
    Parse a structured fleet log line.
    Expected format: 'YYYY-MM-DD HH:MM:SS [LEVEL] service: message key=value...'

    Args:
        line: Log line string.

    Returns:
        Dictionary with keys: timestamp, level, service, message.
        Returns None if line cannot be parsed.
    """
    if not line or not isinstance(line, str):
        return None

    # Regex matches: timestamp, level in brackets, service (up to colon), rest as message
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\] (\w+(-vessel)?): (.+)$'
    match = re.match(pattern, line.strip())

    if not match:
        # Fallback: try to extract at least timestamp and level
        alt_pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\]'
        alt_match = re.match(alt_pattern, line.strip())
        if alt_match:
            return {
                'timestamp': alt_match.group(1),
                'level': alt_match.group(2),
                'service': 'unknown',
                'message': line.strip()
            }
        return None

    return {
        'timestamp': match.group(1),
        'level': match.group(2),
        'service': match.group(3),
        'message': match.group(5)
    }
```

```python
import re

def parse_fleet_log_line(line):
    """
    Parse a fleet log line into its components.
    Format: timestamp [level] service: message
    Example: "2026-04-20 17:30:23 [INFO] oracle1-vessel: tile accepted sha=abc123"
    """
    if not line:
        return {}

    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[([A-Z]+)\] ([^:]+): (.+)$'
    match = re.search(pattern, line)

    if match:
        return {
            'timestamp': match.group(1),
            'level': match.group(2),
            'service': match.group(3),
            'message': match.group(4)
        }
    else:
        # Try a more lenient parse
        parts = line.split()
        if len(parts) >= 4:
            return {
                'timestamp': ' '.join(parts[0:2]),
                'level': parts[2].strip('[]'),
                'service': parts[3].rstrip(':'),
                'message': ' '.join(parts[4:]) if len(parts) > 4 else ''
            }
        return {}
```

**Comparison:**
*   **Correctness:** Both handle the example format. DeepSeek's regex is stricter (`\w+` for service, allows `-vessel` suffix). Groq's regex uses `[^:]+` which is more permissive.
*   **Robustness:** DeepSeek provides a two-tier fallback (full parse, then partial). It returns `None` on complete failure. Groq returns an empty dict `{}` and has a "lenient parse" fallback that could produce misleading results.
*   **Clarity:** DeepSeek's docstring specifies the return type can be `None`. Groq's does not mention the empty dict fallback.
*   **Style:** DeepSeek imports `datetime` (unused in this snippet but suggests future validation). Groq's "lenient parse" is a simpler split approach.
