def deadband_filter(data, lower, upper):
    """
    Implements a deadband filter.
    If a value is within [lower, upper] inclusive, map it to 0.
    Otherwise, pass it through unchanged.
    Works on single numbers or lists/arrays.
    """
    import numpy as np
    
    # Handle single value
    if isinstance(data, (int, float)):
        if lower <= data <= upper:
            return 0
        return data
    
    # Handle list/array
    data = np.asarray(data)
    result = np.zeros_like(data)
    for i, val in enumerate(data):
        if lower <= val <= upper:
            result[i] = 0
        else:
            result[i] = val
    return result.tolist() if isinstance(data, list) else result
```

def parse_fleet_log_line(line):
    """
    Parses a fleet log line format:
    "YYYY-MM-DD HH:MM UTC | AgentName | Cycle N | Key: Value | ..."
    Returns dict with keys: timestamp, agent, cycle, status, etc.
    """
    if not line or '|' not in line:
        return {}
    
    parts = [p.strip() for p in line.split('|')]
    result = {}
    
    # First three parts are fixed
    if len(parts) >= 1:
        result['timestamp'] = parts[0]
    if len(parts) >= 2:
        result['agent'] = parts[1]
    if len(parts) >= 3:
        # Extract cycle number
        cycle_part = parts[2]
        if 'Cycle' in cycle_part:
            try:
                result['cycle'] = int(cycle_part.split('Cycle')[1].strip())
            except:
                result['cycle'] = None
    
    # Remaining parts are key-value pairs
    for part in parts[3:]:
        if ':' in part:
            key, val = part.split(':', 1)
            result[key.strip()] = val.strip()
    
    return result
```

```python
def constraint_snap(point, constraints):
    """
    Given a 2D point (x,y) and list of line constraints,
    each as ((x1,y1), (x2,y2)),
    returns the closest point on any constraint line.
    """
    import math
    
    if not constraints:
        return point
    
    px, py = point
    best_point = None
    best_dist = float('inf')
    
    for (x1, y1), (x2, y2) in constraints:
        # Line vector
        dx = x2 - x1
        dy = y2 - y1
        
        # Handle zero-length segment
        if dx == 0 and dy == 0:
            candidate = (x1, y1)
        else:
            # Project point onto line
            t = ((px - x1) * dx + (py - y1) * dy) / (dx*dx + dy*dy)
            # Clamp to segment
            t = max(0, min(1, t))
            candidate = (x1 + t * dx, y1 + t * dy)
        
        # Distance
        dist = math.hypot(candidate[0] - px, candidate[1] - py)
        if dist < best_dist:
            best_dist = dist
            best_point = candidate
    
    return best_point
```

def deadband_filter(data, lower, upper):
    """
    Deadband filter: values in [lower, upper] become 0.
    """
    import numpy as np
    data = np.asarray(data)
    mask = (data >= lower) & (data <= upper)
    result = data.copy()

```python
def parse_fleet_log_line(line):
    """
    Parse fleet log line.
    """
    import re
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2} UTC) \| ([^|]+) \| Cycle (\d+) \| (.+)'
    match = re.match(pattern, line)
    if not match:
        return {}
    timestamp, agent, cycle, rest = match.groups()
    result = {'timestamp': timestamp, 'agent': agent, 'cycle': int(cycle)}
    # Parse key-value pairs
    pairs = re.findall(r'([^:|]+): ([^|]+)', rest)
    for k, v in pairs:
        result[k.strip()] = v.strip()
    return result
```

```python
def constraint_snap(point, constraints):
    """
    Snap point to closest constraint line.
    """
    import numpy as np
    px, py = point
    best_dist = float('inf')
    best_point = point
    
    for (x1, y1), (x2, y2) in constraints:
        # Line direction
        dir_vec = np.array([x2 - x1, y2 - y1])
        pt_vec = np.array([px - x1, py - y1])
        line_len = np.dot(dir_vec, dir_vec)
        
        if line_len == 0:
            # Zero-length segment
            candidate = (x1, y1)
        else:
            t = np.dot(pt_vec, dir_vec) / line_len
            t = np.clip(t, 0, 1)
            candidate = (x1 + t * dir_vec[0], y1 + t * dir_vec[1])
        
        dist = np.linalg.norm(np.array(candidate) - np.array(point))
        if dist < best_dist:
            best_dist = dist
            best_point = candidate
    
    return best_point
```
