### **Function 1: `validate_deadband_protocol`**

**DeepSeek-chat Implementation:**
```python
from typing import List, Dict, Any

def validate_deadband_protocol(steps: List[str]) -> Dict[str, Any]:
    """
    Validates if a sequence of agent steps follows the Deadband Protocol.
    Protocol: P0 (negative space) must come before P1 (safe channels), 
    which must come before P2 (optimization). Steps are strings starting with P0/P1/P2.

    Args:
        steps: List of step strings (e.g., ['P0: map rocks', 'P1: find channel'])

    Returns:
        Dictionary with:
        - valid: bool indicating if protocol was followed
        - violations: list of indices where protocol was violated
        - phase_summary: dict counting occurrences of each phase
    """
    phase_order = {'P0': 0, 'P1': 1, 'P2': 2}
    current_phase = -1
    violations = []
    phase_counts = {'P0': 0, 'P1': 0, 'P2': 0}

    for i, step in enumerate(steps):
        # Extract phase prefix
        step_upper = step.strip().upper()
        found_phase = None
        for phase in phase_order:
            if step_upper.startswith(phase):
                found_phase = phase
                break
        
        if found_phase is None:
            # Step without clear phase marker - treat as continuation of current phase
            continue
        
        phase_counts[found_phase] += 1
        phase_num = phase_order[found_phase]
        
        # Check ordering
        if phase_num < current_phase:
            violations.append(i)
        current_phase = max(current_phase, phase_num)  # Allow staying in same phase

    return {
        'valid': len(violations) == 0,
        'violations': violations,
        'phase_summary': phase_counts
    }
```

def validate_deadband_protocol(steps):
    """
    Validate Deadband Protocol ordering.
    Input: list of steps as strings, each prefixed with P0, P1, or P2.
    Output: dict with keys:
        valid: boolean
        violations: list of indices where order is broken
        phase_summary: dict with counts per phase
    """
    order = {"P0": 0, "P1": 1, "P2": 2}
    violations = []
    counts = {"P0": 0, "P1": 0, "P2": 0}
    last_seen = -1
    
    for idx, step in enumerate(steps):
        # get phase prefix
        parts = step.strip().split(":")
        if len(parts) == 0:
            continue
        prefix = parts[0].strip().upper()
        if prefix not in order:
            # skip non‑phase lines
            continue
        
        counts[prefix] += 1
        current = order[prefix]
        if current < last_seen:
            violations.append(idx)
        last_seen = current
    
    return {
        "valid": len(violations) == 0,
        "violations": violations,
        "phase_summary": counts
    }
```

**Analysis for Function 1:**
*   **Correctness:** Both are logically correct. DeepSeek's `max(current_phase, phase_num)` allows same-phase repetition, which matches protocol. Groq's `last_seen = current` is stricter but also valid.
*   **Clarity:** DeepSeek uses explicit type hints (`List[str]`, `Dict[str, Any]`) and a more detailed docstring. Groq's docstring is adequate but less formal.
*   **Edge Cases:** DeepSeek handles non‑phase lines by `continue`ing without updating `current_phase`. Groq skips them entirely. DeepSeek's approach of treating them as continuation might be debatable but is specified.
*   **Winner:** **DeepSeek-chat** for better documentation and explicit type safety, aligning with fleet's emphasis on clear specs.

import re
from datetime import datetime
from typing import Optional, Dict, Any

def parse_fleet_tile(tile_text: str) -> Dict[str, Any]:
    """
    Parses a fleet training tile text into structured data.
    Expected format includes headers like '# Cycle 213 — Alchemist',
    '**Phase:** 4', '**Task:** ...', and a free‑text output section.

    Args:
        tile_text: Raw tile text content

    Returns:
        Dictionary with keys: agent, cycle, phase, task, output, timestamp.
        Missing fields will be set to None.
    """
    # Initialize with defaults
    result = {
        'agent': None,
        'cycle': None,
        'phase': None,
        'task': None,
        'output': None,
        'timestamp': datetime.utcnow().isoformat()  # default to now
    }
    
    # Extract agent and cycle from first line pattern: "# Cycle N — Agent"
    first_line_match = re.match(r'# Cycle (\d+) — (\w+)', tile_text.strip().split('\n')[0])
    if first_line_match:
        result['cycle'] = int(first_line_match.group(1))
        result['agent'] = first_line_match.group(2)
    
    # Extract phase: "**Phase:** X" or "Phase: X"
    phase_match = re.search(r'\*\*?Phase\*\*?:?\s*(\d+)', tile_text, re.IGNORECASE)
    if phase_match:
        result['phase'] = int(phase_match.group(1))
    
    # Extract task: "**Task:** ..." up to next markdown header or blank line
    task_match = re.search(r'\*\*?Task\*\*?:?\s*(.+?)(?=\n#|\n\n|\n\*\*|$)', tile_text, re.DOTALL | re.IGNORECASE)
    if task_match:
        result['task'] = task_match.group(1).strip()
    
    # Output is everything after the task (or whole text if no task found)
    # Simple heuristic: split by '---' sections and take the last substantial one
    sections = [s.strip() for s in tile_text.split('---') if s.strip()]
    if sections:
        result['output'] = sections[-1]
    
    # Try to find timestamp in text (e.g., "2026-04-20 04:55 UTC")
    ts_match = re.search(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}) UTC', tile_text)
    if ts_match:
        try:
            result['timestamp'] = datetime.strptime(ts_match.group(1), "%Y-%m-%d %H:%M").isoformat() + "Z"
        except ValueError:
            pass
    
    return result
```
