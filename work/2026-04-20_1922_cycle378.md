## **2. Function 1: `deadband_filter`**

**Prompt to both models:**
```
Write a Python function `deadband_filter(signals, deadband_min, deadband_max)` that implements the Cocapn fleet's Deadband Protocol.

Input: 
- `signals`: list of floats (e.g., sensor readings)
- `deadband_min`: float, lower bound of "safe channel" (P1 zone)
- `deadband_max`: float, upper bound of "safe channel"

Protocol:
- P0 (negative space): If signal < deadband_min or > deadband_max, return None (invalid).
- P1 (safe channel): If signal within [deadband_min, deadband_max], return signal unchanged.
- P2 (optimize): Not applied here (per fleet doctrine: never skip to P2).

Return: List of filtered signals (or None where invalid). Preserve original list length.

Include a docstring and a simple example.
```

### **DeepSeek-chat output:**
```python
def deadband_filter(signals, deadband_min, deadband_max):
    """
    Apply Deadband Protocol P0/P1 filtering to a list of signals.
    
    Parameters:
    signals (list of float): Input sensor readings.
    deadband_min (float): Lower safe bound.
    deadband_max (float): Upper safe bound.
    
    Returns:
    list: Filtered signals where invalid (P0) are replaced with None,
          valid (P1) remain unchanged.
    """
    if deadband_min > deadband_max:
        raise ValueError("deadband_min must be <= deadband_max")
    
    filtered = []
    for s in signals:
        if s is None:
            filtered.append(None)
        elif deadband_min <= s <= deadband_max:
            filtered.append(s)  # P1: safe channel
        else:
            filtered.append(None)  # P0: negative space
    return filtered
```

```python
def deadband_filter(signals, deadband_min, deadband_max):
    """
    Implements the Deadband Protocol for signal filtering.
    
    Args:
        signals: List of numerical signal values.
        deadband_min: Minimum acceptable value (inclusive).
        deadband_max: Maximum acceptable value (inclusive).
    
    Returns:
        List where signals outside deadband are replaced with None,
        signals inside are kept.
    """
    # Validate bounds
    if deadband_min >= deadband_max:
        raise ValueError("deadband_min should be less than deadband_max")
    
    result = []
    for val in signals:
        # Check if value is within safe channel (P1)
        if deadband_min <= val <= deadband_max:
            result.append(val)
        else:
            # P0: negative space - mark as invalid
            result.append(None)
    return result
```

import re
from datetime import datetime

def parse_fleet_log(log_text):
    """
    Parse Cocapn fleet log lines into structured events.
    
    Parameters:
    log_text (str): Multiline log string.
    
    Returns:
    list of dict: Parsed events, each with timestamp, cycle, agent, event_type, message.
    """
    lines = log_text.strip().split('\n')
    events = []
    
    # Regex to match log format
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2} UTC) \[(.*?)\] (.*)'
    
    for line in lines:
        line = line.strip()
        if not line:
            continue
            
        match = re.match(pattern, line)
        if not match:
            # Skip malformed lines
            continue
            
        timestamp_str, bracket_content, rest = match.groups()
        
        # Parse timestamp
        try:
            timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M UTC')
        except ValueError:
            continue
        
        # Extract cycle and agent from bracket content
        cycle = None
        agent = None
        event_type = None
        
        # Check for cycle pattern like "CYCLE 377"
        cycle_match = re.search(r'CYCLE\s+(\d+)', bracket_content)
        if cycle_match:
            cycle = int(cycle_match.group(1))
        
        # Agent names: Alchemist, JetsonClaw1, Oracle1, etc.
        agent_match = re.search(r'(Alchemist|JetsonClaw1|Oracle1|Forgemaster|Zeroclaw)', bracket_content)
        if agent_match:
            agent = agent_match.group(1)
        
        # Event type: first word in rest
        event_parts = rest.split()
        event_type = event_parts[0] if event_parts else "UNKNOWN"
        
        # Message is the rest after event_type
        message = ' '.join(event_parts[1:]) if len(event_parts) > 1 else ""
        
        events.append({
            'timestamp': timestamp,
            'cycle': cycle,
            'agent': agent,
            'event_type': event_type,
            'message': message
        })
    
    return events
