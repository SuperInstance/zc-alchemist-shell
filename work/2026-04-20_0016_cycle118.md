### **Function 1: `clean_and_normalize_text`**

**DeepSeek-chat Output:**
```python
import re
import unicodedata

def clean_and_normalize_text(text: str) -> str:
    """
    Clean and normalize a text string.
    
    Steps:
    1. Normalize Unicode to NFKC form (compatibility decomposition followed by composition)
    2. Convert to lowercase
    3. Remove extra whitespace (newlines, tabs, multiple spaces)
    4. Remove non-alphanumeric characters except basic punctuation and spaces
    
    Args:
        text: Input text string
        
    Returns:
        Cleaned and normalized text string
    """
    if not isinstance(text, str):
        raise TypeError(f"Expected string, got {type(text).__name__}")
    
    # Normalize Unicode
    text = unicodedata.normalize('NFKC', text)
    
    # Convert to lowercase
    text = text.lower()
    
    # Replace various whitespace with single space
    text = re.sub(r'\s+', ' ', text)
    
    # Remove non-alphanumeric except basic punctuation and space
    # Keep: letters, numbers, basic punctuation (. , ! ?), and space
    text = re.sub(r'[^a-z0-9\s.,!?]', '', text)
    
    # Strip leading/trailing whitespace
    return text.strip()
```

**DeepSeek-chat Output:**

def find_longest_palindromic_substring(s: str) -> str:
    """
    Find the longest palindromic substring in a string using expansion around center.
    
    Time complexity: O(n^2)
    Space complexity: O(1)
    
    Args:
        s: Input string
        
    Returns:
        Longest palindromic substring
    """
    if not s:
        return ""
    
    start = 0
    end = 0
    
    def expand_around_center(left: int, right: int) -> int:
        """Return length of palindrome centered at left/right."""
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
        return right - left - 1
    
    for i in range(len(s)):
        # Odd length palindrome
        len1 = expand_around_center(i, i)
        # Even length palindrome
        len2 = expand_around_center(i, i + 1)
        
        max_len = max(len1, len2)
        if max_len > end - start:
            start = i - (max_len - 1) // 2
            end = i + max_len // 2
    
    return s[start:end + 1]
```

def find_longest_palindromic_substring(s):
    """
    Returns the longest palindromic substring.
    Brute force approach for clarity.
    """
    if not s:
        return ""
    
    longest = ""
    n = len(s)
    
    for i in range(n):
        for j in range(i, n):
            substr = s[i:j+1]
            if substr == substr[::-1] and len(substr) > len(longest):
                longest = substr
                
    return longest
```
