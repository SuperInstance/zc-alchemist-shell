import re
from datetime import datetime
from math import sqrt

def deadband_filter(values, threshold):
    """
    Apply a deadband filter to a list of float values.
    
    Args:
        values: list of float values
        threshold: positive float, minimum change required to update output
    
    Returns:
        list of filtered values
    """
    if not values:
        return []
    
    result = [values[0]]
    for v in values[1:]:
        if abs(v - result[-1]) < threshold:
            result.append(result[-1])
        else:
            result.append(v)
    return result

def parse_fleet_log(log_line):
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format:
        "YYYY‑MM‑DD HH:MM:SS [LEVEL] agent=NAME cycle=N msg='TEXT'"
    
    Args:
        log_line: string containing log entry
    
    Returns:
        dict with keys: timestamp, level, agent, cycle, message
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\]\s*(?:agent=(\w+))?\s*(?:cycle=(\d+))?\s*(?:msg=\'([^\']*)\')?'
    match = re.match(pattern, log_line)
    if not match:
        return {
            'timestamp': None,
            'level': None,
            'agent': None,
            'cycle': None,
            'message': None
        }
    
    timestamp_str, level, agent, cycle_str, message = match.groups()
    try:
        timestamp = datetime.strptime(timestamp_str, '%Y‑%m‑d %H:%M:%S')
    except ValueError:
        timestamp = None
    
    cycle = int(cycle_str) if cycle_str else None
    
    return {
        'timestamp': timestamp,
        'level': level,
        'agent': agent,
        'cycle': cycle,
        'message': message
    }

def constraint_snap(seg1, seg2, tolerance):
    """
    Snap the closest endpoints of two line segments if within tolerance.
    
    Args:
        seg1: tuple of two points ((x1,y1), (x2,y2))
        seg2: tuple of two points ((x3,y3), (x4,y4))
        tolerance: maximum distance for snapping
    
    Returns:
        tuple of (snapped_seg1, snapped_seg2)
    """
    def dist(p1, p2):
        return sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)
    
    # All endpoint pairs
    endpoints1 = [seg1[0], seg1[1]]
    endpoints2 = [seg2[0], seg2[1]]
    
    min_dist = float('inf')
    snap_pair = None
    
    for i, p1 in enumerate(endpoints1):
        for j, p2 in enumerate(endpoints2):
            d = dist(p1, p2)
            if d < min_dist:
                min_dist = d
                snap_pair = (i, j)
    
    if min_dist <= tolerance:
        i, j = snap_pair
        p1 = endpoints1[i]
        p2 = endpoints2[j]
        # Average position
        avg_x = (p1[0] + p2[0]) / 2
        avg_y = (p1[1] + p2[1]) / 2
        
        # Create new segments
        new_seg1 = list(seg1)
        new_seg2 = list(seg2)
        new_seg1[i] = (avg_x, avg_y)
        new_seg2[j] = (avg_x, avg_y)
        
        return (tuple(new_seg1), tuple(new_seg2))
    
    return (seg1, seg2)
