# Function 1: Validate Deadband Protocol
from typing import List

def validate_deadband_protocol(actions: List[str]) -> bool:
    """
    Validates if a sequence of agent actions follows the Deadband Protocol.
    Protocol: Must contain P0 (map negative space) before P1 (find safe channels)
    before P2 (optimize). No skipping phases. Actions are strings like 'P0', 'P1', 'P2'.

    Args:
        actions: List of action strings.

    Returns:
        True if sequence is valid, False otherwise.
    """
    expected_phase = 0  # Start looking for P0
    for action in actions:
        if action == f"P{expected_phase}":
            expected_phase += 1  # Move to next expected phase
        elif action in ["P0", "P1", "P2"]:
            # Found a phase out of order or a skip
            return False
        # Ignore non-phase actions (could be other strings)
    return True

# Function 2: Parse Fleet Tile
import re
from typing import Dict, Optional

def parse_fleet_tile(raw_tile: str) -> Dict[str, Optional[str]]:
    """
    Parses a raw training tile submission string into a structured dictionary.

    Expected format (simplified):
    AGENT: Alchemist
    CYCLE: 140
    CONTENT: Some text here.

    Args:
        raw_tile: The raw tile text.

    Returns:
        Dictionary with keys: agent_id, cycle, content, is_valid_format.
    """
    result = {
        "agent_id": None,
        "cycle": None,
        "content": None,
        "is_valid_format": False
    }

    agent_match = re.search(r"AGENT:\s*(.+)", raw_tile)
    cycle_match = re.search(r"CYCLE:\s*(\d+)", raw_tile)
    content_match = re.search(r"CONTENT:\s*(.+)", raw_tile, re.DOTALL)

    if agent_match and cycle_match and content_match:
        result["agent_id"] = agent_match.group(1).strip()
        result["cycle"] = int(cycle_match.group(1))
        # Get content from the match to end of string, clean up
        result["content"] = content_match.group(1).strip()
        result["is_valid_format"] = True

    return result

# Function 3: Simulate Channel Optimization
import random
from typing import List

def simulate_channel_optimization(channel_scores: List[float]) -> int:
    """
    Finds the index of the channel with the highest quality score.
    In case of a tie, selects randomly among the tied indices.

    Args:
        channel_scores: List of non-negative quality scores.

    Returns:
        Index of the best channel.
    """
    if not channel_scores:
        raise ValueError("Channel scores list cannot be empty.")

    max_score = max(channel_scores)
    # Find all indices that have the max score
    best_indices = [i for i, score in enumerate(channel_scores) if score == max_score]

    # Random tie-breaker
    return random.choice(best_indices)

# Function 1: Validate Deadband Protocol
from typing import List

def validate_deadband_protocol(actions: List[str]) -> bool:
    """
    Check if actions follow Deadband Protocol: P0 -> P1 -> P2, no skipping.

    Args:
        actions (List[str]): List of action strings like 'P0', 'P1', 'P2'.

    Returns:
        bool: True if valid, False if invalid.
    """
    phases_seen = set()
    for action in actions:
        if action == 'P0':
            phases_seen.add('P0')
        elif action == 'P1':
            if 'P0' not in phases_seen:
                return False
            phases_seen.add('P1')
        elif action == 'P2':
            if 'P0' not in phases_seen or 'P1' not in phases_seen:
                return False
            phases_seen.add('P2')
        else:
            # Ignore other actions
            continue
    return True

# Function 2: Parse Fleet Tile
import re
from typing import Dict, Any

def parse_fleet_tile(raw_tile: str) -> Dict[str, Any]:
    """
    Parse a fleet training tile string into structured data.

    Args:
        raw_tile (str): Raw tile text.

    Returns:
        Dict[str, Any]: Dictionary with agent_id, cycle, content, is_valid_format.
    """
    pattern = r'AGENT:\s*(?P<agent_id>\w+)\s*\nCYCLE:\s*(?P<cycle>\d+)\s*\nCONTENT:\s*(?P<content>.*)'
    match = re.search(pattern, raw_tile, re.DOTALL)

    if match:
        return {
            'agent_id': match.group('agent_id'),
            'cycle': int(match.group('cycle')),
            'content': match.group('content').strip(),
            'is_valid_format': True
        }
    else:
        return {
            'agent_id': None,
            'cycle': None,
            'content': None,
            'is_valid_format': False
        }

# Function 3: Simulate Channel Optimization
import random
from typing import List

def simulate_channel_optimization(channel_scores: List[float]) -> int:
    """
    Find index of best channel score, random tie-break.

    Args:
        channel_scores (List[float]): List of channel quality scores.

    Returns:
        int: Index of best channel.
    """
    if len(channel_scores) == 0:
        raise ValueError("Empty channel_scores list")

    max_score = max(channel_scores)
    best_indices = [idx for idx, score in enumerate(channel_scores) if score == max_score]
    return random.choice(best_indices)
