```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.
    Protocol: Must be a strict sequence of P0 -> P1 -> P2.
    No skipping phases, no going backwards, no repeats.
    
    Args:
        steps: List of step identifiers (e.g., ["P0", "P1", "P2"])
    
    Returns:
        Tuple of (is_valid: bool, errors: list[str])
    """
    valid_sequence = ["P0", "P1", "P2"]
    errors = []
    
    # Track the highest phase we've seen
    last_index = -1
    
    for i, step in enumerate(steps):
        try:
            current_index = valid_sequence.index(step)
        except ValueError:
            errors.append(f"Step {i}: '{step}' is not a valid phase. Must be P0, P1, or P2.")
            continue
            
        # Check for repeats or going backwards
        if current_index < last_index:
            errors.append(f"Step {i}: '{step}' violates order. Cannot go back to earlier phase.")
        elif current_index == last_index:
            errors.append(f"Step {i}: '{step}' is a repeat. Phases cannot repeat.")
        
        last_index = current_index
    
    # Check if we completed the sequence properly
    if last_index != len(valid_sequence) - 1 and not errors:
        errors.append(f"Incomplete sequence. Ended at {valid_sequence[last_index] if last_index >=0 else 'start'}, need full P0->P1->P2.")
    
    return (len(errors) == 0, errors)
```

```python
import re

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet agent's tile text to extract metadata.
    Expected format includes headers like '# Cycle X — AgentName'
    
    Args:
        tile_text: Raw text content of a tile
    
    Returns:
        Dictionary with agent, cycle, phase, task, content_length
        Missing values will be empty string or 0.
    """
    result = {
        "agent": "",
        "cycle": 0,
        "phase": 0,
        "task": "",
        "content_length": len(tile_text)
    }
    
    # Try to extract agent and cycle from first line pattern
    first_line = tile_text.split('\n')[0] if tile_text else ""
    cycle_match = re.search(r'Cycle (\d+)', first_line)
    agent_match = re.search(r'— (\w+)', first_line)
    
    if agent_match:
        result["agent"] = agent_match.group(1)
    
    if cycle_match:
        try:
            result["cycle"] = int(cycle_match.group(1))
        except ValueError:
            pass
    
    # Look for Phase pattern
    phase_match = re.search(r'Phase:\s*(\d+)', tile_text)
    if phase_match:
        try:
            result["phase"] = int(phase_match.group(1))
        except ValueError:
            pass
    
    # Extract task - look for "Task:" line
    task_lines = []
    in_task_section = False
    
    for line in tile_text.split('\n'):
        if line.startswith('**Task:**') or 'Task:' in line:
            in_task_section = True
            task_text = line.split('Task:')[-1].strip()
            if task_text:
                task_lines.append(task_text)
        elif in_task_section and line.strip() and not line.startswith('#'):
            task_lines.append(line.strip())
        elif in_task_section and not line.strip():
            break
    
    if task_lines:
        result["task"] = ' '.join(task_lines)
    
    return result
```

```python
import math

def generate_safe_channel(boundaries: list[tuple[float, float]], 
                         current_pos: tuple[float, float]) -> list[tuple[float, float]]:
    """
    Generates a safe channel from current position to the nearest boundary point.
    Simple straight-line implementation for P1 channel finding.
    
    Args:
        boundaries: List of (x,y) points defining a polygon (negative space)
        current_pos: Starting (x,y) position inside the polygon
    
    Returns:
        List of points forming a path [current_pos, nearest_boundary_point]
    """
    if not boundaries:
        return [current_pos]
    
    # Find closest boundary point using Euclidean distance
    min_distance = float('inf')
    nearest_point = boundaries[0]
    
    for point in boundaries:
        distance = math.sqrt((point[0] - current_pos[0])**2 + 
                           (point[1] - current_pos[1])**2)
        if distance < min_distance:
            min_distance = distance
            nearest_point = point
    
    # Return simple straight-line channel
    return [current_pos, nearest_point]
```
