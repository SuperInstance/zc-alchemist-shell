```python
# Function Set 1: DeepSeek-chat style
# Focus: Concise, direct logic, minimal boilerplate.

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """Validate strict P0->P1->P2 sequence with no skips."""
    required = ["P0", "P1", "P2"]
    idx = 0
    for step in steps:
        if step == required[idx]:
            idx += 1
            if idx == len(required):
                break
        elif step in required:
            return False, f"Protocol violation: Found {step} out of order. Expected {required[idx]}."
    return (idx == len(required), "") if idx == len(required) else (False, f"Missing priority steps: {required[idx:]}")

def parse_fleet_tile(content: str) -> dict:
    """Extract basic metadata from a fleet training tile."""
    lines = content.split('\n')
    has_doctrine = any('P0' in line or 'deadband' in line.lower() for line in lines)
    phase = None
    for line in lines:
        if 'phase' in line.lower():
            words = line.split()
            for w in words:
                if w.isdigit():
                    phase = int(w)
                    break
    word_count = len(content.split())
    is_structured = (content.count('#') >= 2 or content.count('- ') > 3)
    return {
        "has_doctrine_ref": has_doctrine,
        "phase": phase,
        "word_count": word_count,
        "is_structured": is_structured
    }

def simulate_channel_optimization(bandwidth: list[float], noise: list[float]) -> list[int]:
    """Select non-adjacent channels with good signal-to-noise ratio."""
    if len(bandwidth) != len(noise):
        return []
    selected = []
    i = 0
    while i < len(bandwidth):
        snr = bandwidth[i] / (noise[i] + 0.1)
        if snr > 5.0:
            selected.append(i)
            i += 2  # Skip next channel (adjacent)
        else:
            i += 1
    return selected
```

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates that a sequence of steps adheres to the Fleet's Deadband Protocol.

    The Deadband Protocol requires the exact sequence ["P0", "P1", "P2"] to be
    present, in that order, with no interleaving of other priority codes between them.
    Skipping a priority level is a violation.

    Args:
        steps: A list of strings representing the steps.

    Returns:
        A tuple (is_valid: bool, error_message: str). If valid, error_message is empty.

    Example:
        >>> validate_deadband_protocol(["P0", "P1", "P2"])
        (True, "")
        >>> validate_deadband_protocol(["P0", "P2"])
        (False, "Missing priority steps: ['P1', 'P2']")
    """
    # Define the mandatory sequence
    mandatory_sequence = ["P0", "P1", "P2"]
    sequence_index = 0

    for i, step in enumerate(steps):
        # If we've already found the full sequence, we can stop checking.
        if sequence_index >= len(mandatory_sequence):
            break
        # Check if the current step matches the next required priority
        if step == mandatory_sequence[sequence_index]:
            sequence_index += 1
        # If it's a priority code but the wrong one, that's an out-of-order error.
        elif step in mandatory_sequence:
            expected = mandatory_sequence[sequence_index]
            return False, f"Deadband Protocol sequence error at step {i}. Found '{step}' but expected '{expected}'. Priorities cannot be reordered or skipped."

    # After processing, check if we found the complete sequence.
    if sequence_index == len(mandatory_sequence):
        return True, ""
    else:
        missing = mandatory_sequence[sequence_index:]
        return False, f"Missing priority steps: {missing}"

def parse_fleet_tile(content: str) -> dict:
    """
    Parses a text content string believed to be a Fleet training tile and extracts
    key metadata attributes useful for automatic curation and scoring.

    Args:
        content: The full text content of the tile.

    Returns:
        A dictionary with the following keys:
        - has_doctrine_ref (bool): True if content mentions P0/P1/P2 or 'deadband'.
        - phase (int | None): The phase number if mentioned, else None.
        - word_count (int): Total words in the content.
        - is_structured (bool): Heuristic for whether content uses markdown headers or lists.
    """
    result = {
        "has_doctrine_ref": False,
        "phase": None,
        "word_count": 0,
        "is_structured": False
    }

    # Convert to lower case for case-insensitive checks
    content_lower = content.lower()

    # Check for doctrine references
    doctrine_indicators = ["p0", "p1", "p2", "deadband"]
    for indicator in doctrine_indicators:
        if indicator in content_lower:
            result["has_doctrine_ref"] = True
            break

    # Extract phase number
    import re
    phase_pattern = r'phase\s*(\d+)'
    match = re.search(phase_pattern, content_lower)
    if match:
        try:
            result["phase"] = int(match.group(1))
        except ValueError:
            pass  # Keep as None if conversion fails

    # Word count (simple split on whitespace)
    result["word_count"] = len(content.split())

    # Structured content heuristic
    header_count = content.count('#')
    list_item_count = content.count('- ') + content.count('* ')
    result["is_structured"] = (header_count >= 2) or (list_item_count > 3)

    return result
