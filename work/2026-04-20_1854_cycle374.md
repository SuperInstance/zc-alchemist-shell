def deadband_filter(readings, p0_threshold=0.1, p1_min=0.5, p1_max=2.0):
    """
    Apply the Deadband Protocol to a list of sensor readings.
    
    P0 (Negative Space): Remove readings below p0_threshold (noise/unsafe).
    P1 (Safe Channel): Keep readings within [p1_min, p1_max] (operational range).
    P2 (Optimize): For readings in P1, apply a simple normalization.
    
    Args:
        readings (list of float): Raw sensor data.
        p0_threshold (float): Minimum value to consider (P0 cutoff).
        p1_min (float): Lower bound of safe channel.
        p1_max (float): Upper bound of safe channel.
    
    Returns:
        list of float: Filtered and normalized readings within P1 channel.
        list of float: Raw readings that were in P1 (before normalization).
    """
    if not readings:
        return [], []
    
    # P0: Map negative space (remove unsafe/low-confidence readings)
    p0_filtered = [r for r in readings if abs(r) >= p0_threshold]
    
    # P1: Find safe channel
    p1_filtered = [r for r in p0_filtered if p1_min <= abs(r) <= p1_max]
    
    # P2: Optimize within channel (normalize to [0,1] range within P1 bounds)
    if not p1_filtered:
        return [], []
    
    p2_normalized = [
        (r - p1_min) / (p1_max - p1_min) 
        for r in p1_filtered
    ]
    
    return p2_normalized, p1_filtered

def deadband_filter(readings, p0_threshold=0.1, p1_min=0.5, p1_max=2.0):
    """
    Implements Deadband Protocol filtering for sensor data.
    
    Steps:
    1. P0: Eliminate readings below threshold (negative space).
    2. P1: Select readings within safe operational bounds.
    3. P2: Normalize selected readings to 0-1 range.
    
    Parameters:
        readings (list): Input sensor values.
        p0_threshold (float): P0 cutoff.
        p1_min (float): Safe lower bound.
        p1_max (float): Safe upper bound.
    
    Returns:
        tuple: (normalized_readings, safe_readings)
    """
    # Input validation
    if readings is None:
        readings = []
    if not isinstance(readings, list):
        raise TypeError("readings must be a list")
    
    # P0: Remove sub-threshold readings
    p0_readings = []
    for val in readings:
        try:
            if abs(float(val)) >= p0_threshold:
                p0_readings.append(float(val))
        except (ValueError, TypeError):
            continue  # skip non-numeric
    
    # P1: Apply safe channel bounds
    p1_readings = []
    for val in p0_readings:
        if p1_min <= abs(val) <= p1_max:
            p1_readings.append(val)
    
    # P2: Normalize
    normalized = []
    if p1_readings:
        for val in p1_readings:
            norm_val = (val - p1_min) / (p1_max - p1_min)
            normalized.append(round(norm_val, 6))  # precision limit
    
    return normalized, p1_readings

def validate_tile(tile, required_fields=None):
    """
    Validate a training tile against fleet submission criteria.
    
    Based on fleet-knowledge/FLEET-CONTEXT.md:
    - Tile must be a dictionary
    - Must contain 'content' (non-empty string)
    - Must contain 'agent_type' (string from known types)
    - Must contain 'cycle' (positive integer)
    - Content should be factual, avoid absolute claims
    
    Args:
        tile (dict): Proposed tile data.
        required_fields (list): Override default required fields.
    
    Returns:
        tuple: (is_valid (bool), errors (list of str))
    """
    if required_fields is None:
        required_fields = ['content', 'agent_type', 'cycle']
    
    errors = []
    
    # Check type
    if not isinstance(tile, dict):
        return False, ["Tile must be a dictionary"]
    
    # Check required fields
    for field in required_fields:
        if field not in tile:
            errors.append(f"Missing required field: {field}")
        elif not tile[field]:
            errors.append(f"Field '{field}' is empty")
    
    # Type-specific checks
    if 'content' in tile:
        if not isinstance(tile['content'], str):
            errors.append("'content' must be a string")
        elif len(tile['content'].strip()) < 10:
            errors.append("'content' too short (min 10 chars)")
        # Check for absolute claims (heuristic)
        absolute_indicators = ['always', 'never', 'guaranteed', 'certainly']
        content_lower = tile['content'].lower()
        if any(indicator in content_lower for indicator in absolute_indicators):
            errors.append("Content contains absolute claims (violates fleet rules)")
    
    if 'agent_type' in tile:
        known_agents = ['navigator', 'sentinel', 'scribe', 'tinker', 'scout', 
                       'curator', 'mason', 'alchemist', 'herald', 'scholar', 
                       'weaver', 'archivist']
        if tile['agent_type'].lower() not in known_agents:
            errors.append(f"Unknown agent_type: {tile['agent_type']}")
    
    if 'cycle' in tile and isinstance(tile['cycle'], (int, float)):
        if tile['cycle'] < 0:
            errors.append("cycle must be non-negative")
    
    return len(errors) == 0, errors
