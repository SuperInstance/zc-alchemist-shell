#### **Function 1: Deadband Protocol Validator**
```python
from typing import List, Tuple, Optional

def validate_deadband_sequence(actions: List[Tuple[str, str]]) -> Tuple[bool, Optional[str]]:
    """
    Validate that a sequence of actions respects Deadband Protocol priorities.
    
    Priority levels:
    - P0: "avoid" (negative space mapping)
    - P1: "explore" (safe channel finding)
    - P2: "optimize" (within-channel optimization)
    
    Rules:
    1. P0 must come before P1, P1 before P2
    2. Can skip levels only if no actions at that level
    3. Cannot regress to earlier priority once advanced
    
    Args:
        actions: List of (priority, description) tuples
        
    Returns:
        (is_valid, error_message) where error_message is None if valid
    """
    priority_order = {"P0": 0, "P1": 1, "P2": 2}
    current_level = -1
    
    for priority, _ in actions:
        if priority not in priority_order:
            return False, f"Unknown priority: {priority}"
        
        level = priority_order[priority]
        
        # Check for regression
        if level < current_level:
            return False, f"Priority regression from {list(priority_order.keys())[current_level]} to {priority}"
        
        # Update current level
        if level > current_level:
            current_level = level
    
    return True, None
```

#### **Function 2: Training Tile Formatter**
```python
import json
from datetime import datetime
from typing import Dict, Any, Optional

def format_training_tile(
    raw_text: str,
    agent_id: str,
    room_id: str,
    metadata: Optional[Dict[str, Any]] = None
) -> Dict[str, Any]:
    """
    Convert raw agent output into a structured training tile for PLATO room ingestion.
    
    Tile structure:
    - id: Unique tile identifier
    - timestamp: ISO format
    - agent: Agent identifier
    - room: Target room
    - content: Processed text
    - metadata: Additional context
    - version: Tile format version
    
    Args:
        raw_text: Raw agent output text
        agent_id: Identifier of the producing agent
        room_id: Target PLATO room
        metadata: Optional additional data
        
    Returns:
        Structured tile dictionary
    """
    # Clean and normalize text
    content = raw_text.strip()
    if not content:
        raise ValueError("Raw text cannot be empty")
    
    # Generate tile ID (simple hash-based)
    import hashlib
    tile_id = hashlib.sha256(
        f"{agent_id}:{room_id}:{datetime.utcnow().isoformat()}".encode()
    ).hexdigest()[:16]
    
    tile = {
        "id": tile_id,
        "timestamp": datetime.utcnow().isoformat() + "Z",
        "agent": agent_id,
        "room": room_id,
        "content": content,
        "metadata": metadata or {},
        "version": "1.0"
    }
    
    return tile
```

import os
import re
from pathlib import Path
from typing import Dict, List, Set
import networkx as nx

def map_repo_dependencies(repo_dir: str) -> Dict[str, List[str]]:
    """
    Parse a directory containing git repositories and extract dependency relationships.
    
    Looks for:
    - requirements.txt (Python)
    - package.json (Node.js)
    - Cargo.toml (Rust)
    - .gitmodules (Git submodules)
    
    Args:
        repo_dir: Path to directory containing git repositories
        
    Returns:
        Dictionary mapping repo names to list of dependencies
    """
    dependencies = {}
    repo_dir_path = Path(repo_dir)
    
    if not repo_dir_path.exists():
        raise FileNotFoundError(f"Directory not found: {repo_dir}")
    
    for repo_path in repo_dir_path.iterdir():
        if not repo_path.is_dir():
            continue
            
        repo_name = repo_path.name
        repo_deps = set()
        
        # Check for Python dependencies
        req_file = repo_path / "requirements.txt"
        if req_file.exists():
            with open(req_file) as f:
                for line in f:
                    line = line.strip()
                    if line and not line.startswith("#"):
                        # Extract package name (ignore versions)
                        pkg = line.split('==')[0].split('>=')[0].split('<=')[0].strip()
                        repo_deps.add(f"python:{pkg}")
        
        # Check for Node.js dependencies
        package_file = repo_path / "package.json"
        if package_file.exists():
            try:
                import json
                with open(package_file) as f:
                    package_data = json.load(f)
                    deps = package_data.get("dependencies", {})
                    for dep in deps:
                        repo_deps.add(f"npm:{dep}")
            except json.JSONDecodeError:
                pass
        
        # Check for Rust dependencies
        cargo_file = repo_path / "Cargo.toml"
        if cargo_file.exists():
            with open(cargo_file) as f:
                content = f.read()
                # Simple regex for dependencies (not production-ready)
                dep_matches = re.findall(r'\[dependencies\.(.*?)\]', content)
                for dep in dep_matches:
                    repo_deps.add(f"cargo:{dep}")
        
        # Check for git submodules
        gitmodules = repo_path / ".gitmodules"
        if gitmodules.exists():
            with open(gitmodules) as f:
                for line in f:
                    if "path =" in line:
                        submodule_path = line.split("=")[1].strip()
                        repo_deps.add(f"git:{submodule_path}")
        
        dependencies[repo_name] = sorted(list(repo_deps))
    
    return dependencies
```

def check_deadband_priority(actions):
    """
    Validate Deadband Protocol priority sequence.
    
    Args:
        actions: list of tuples (priority, action)
        
    Returns:
        (valid, message)
    """
    priorities = ['P0', 'P1', 'P2']
    seen = set()
    last_index = -1
    
    for priority, _ in actions:
        if priority not in priorities:
            return False, f"Invalid priority {priority}"
        
        curr_index = priorities.index(priority)
        
        # Check if we're going backward
        if curr_index < last_index:
            return False, f"Cannot go from {priorities[last_index]} to {priority}"
        
        # Update tracking
        seen.add(priority)
        last_index = curr_index
    
    # Check if we skipped a required priority
    for i in range(len(priorities)):
        if priorities[i] not in seen:
            # Check if any higher priority was seen
            if any(p in seen for p in priorities[i+1:]):
                return False, f"Skipped required priority {priorities[i]}"
    
    return True, "Valid sequence"
```
