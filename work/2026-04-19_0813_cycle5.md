def validate_deadband_steps(steps: list[str]) -> bool:
    priority_order = {"P0": 0, "P1": 1, "P2": 2}
    last_seen = -1
    for step in steps:
        # Extract priority prefix
        if step.startswith("P0"):
            curr = 0
        elif step.startswith("P1"):
            curr = 1
        elif step.startswith("P2"):
            curr = 2
        else:
            return False  # Invalid step format
        if curr < last_seen:
            return False  # Priority decreased
        if curr > last_seen + 1:
            return False  # Skipped a priority
        last_seen = curr
    return True
