def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if steps follow P0->P1->P2 priority order.

    Args:
        steps: List of step description strings.

    Returns:
        Tuple of (is_valid, message). Message explains violation if invalid.
    """
    # Track earliest occurrence of each priority marker
    positions = {}
    for i, step in enumerate(steps):
        step_lower = step.lower()
        if 'p0' in step_lower and 'p0' not in positions:
            positions['p0'] = i
        if 'p1' in step_lower and 'p1' not in positions:
            positions['p1'] = i

            positions['p2'] = i

    # Check for missing required priorities
    if 'p0' not in positions:
        return False, "Missing P0 step"
    if 'p1' not in positions:
        return False, "Missing P1 step"
    if 'p2' not in positions:
        return False, "Missing P2 step"

    # Check order
    if positions['p1'] <= positions['p0']:
        return False, "P1 found before P0"
    if positions['p2'] <= positions['p1']:
        return False, "P2 found before P1"

    return True, "OK"


def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a raw fleet training tile into structured data.

    Args:
        tile_text: Raw tile text with headers and content.

    Returns:
        Dictionary with agent, cycle, phase, task, content fields.
    """
    result = {
        'agent': None,
        'cycle': None,
        'phase': None,
        'task': None,
        'content': None
    }

    lines = tile_text.strip().split('\n')
    if not lines:
        return result

    # Parse first line for cycle and agent
    first_line = lines[0].strip()
    cycle_match = re.search(r'# Cycle (\d+) — (\w+)', first_line)
    if cycle_match:
        result['cycle'] = int(cycle_match.group(1))
        result['agent'] = cycle_match.group(2)

    # Parse subsequent lines for phase and task
    for line in lines[1:]:
        line_stripped = line.strip()
        if line_stripped.startswith('---'):
            break  # Start of content
        if '**Phase' in line_stripped:
            phase_match = re.search(r'\*\*Phase[^:]*:\s*(.*?)\s*\*\*', line_stripped)
            if phase_match:
                result['phase'] = phase_match.group(1).strip()
        if '**Task' in line_stripped:
            task_match = re.search(r'\*\*Task[^:]*:\s*(.*?)\s*\*\*', line_stripped)
            if task_match:
                result['task'] = task_match.group(1).strip()

    # Find content separator
    for i, line in enumerate(lines):
        if line.strip().startswith('---'):
            content_lines = lines[i+1:]
            result['content'] = '\n'.join(content_lines).strip()
            break

    return result
