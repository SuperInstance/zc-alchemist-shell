#### **Function 1: `validate_deadband_protocol`**
```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of agent steps against the Deadband Protocol.
    Protocol requires strict order: P0 -> P1 -> P2. No phase skipping allowed.

    Args:
        steps: List of step strings, e.g., ["P0: check", "P1: explore"].

    Returns:
        Tuple of (is_valid: bool, message: str). Message is "OK" if valid,
        otherwise an error describing the violation.
    """
    if not steps:
        return True, "OK"  # Empty sequence is trivially valid

    # Define phase order
    phase_order = {"P0": 0, "P1": 1, "P2": 2}
    current_phase = -1

    for i, step in enumerate(steps):
        # Extract phase prefix (assumes format "PX: ...")
        step_upper = step.strip().upper()
        if not step_upper.startswith(("P0:", "P1:", "P2:")):
            return False, f"Step {i}: '{step}' does not start with P0:, P1:, or P2:"

        # Get phase identifier
        phase_key = step_upper[:2]  # "P0", "P1", "P2"
        phase_num = phase_order.get(phase_key)
        if phase_num is None:
            return False, f"Step {i}: Unknown phase '{phase_key}'"

        # Check order
        if phase_num < current_phase:
            return False, f"Step {i}: Phase regression from {list(phase_order.keys())[current_phase]} to {phase_key}"
        if phase_num > current_phase + 1:
            skipped = list(phase_order.keys())[current_phase + 1:phase_num]
            return False, f"Step {i}: Skipped phase(s) {skipped} from {list(phase_order.keys())[current_phase]} to {phase_key}"

        current_phase = phase_num

    return True, "OK"
```

#### **Function 2: `parse_fleet_tile`**

```python
import re

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a raw fleet log tile text into a structured dictionary.

    Expected format (lines may vary):
        # Cycle X — AgentName
        **Phase Y: ...**
        **Task:** ...
        ---
        Content...

    Returns dict with keys: agent, cycle, phase, task, content.
    Missing values are set to empty string.
    """
    result = {
        "agent": "",
        "cycle": "",
        "phase": "",
        "task": "",
        "content": ""
    }

    lines = tile_text.strip().split('\n')
    if not lines:
        return result

    # Parse header line: "# Cycle X — AgentName"
    header_match = re.match(r'# Cycle\s+(\d+)\s*—\s*(\w+)', lines[0])
    if header_match:
        result["cycle"] = header_match.group(1)
        result["agent"] = header_match.group(2)

    # Look for Phase and Task lines (bold markdown **)
    for line in lines:
        stripped = line.strip()
        if stripped.startswith('**Phase'):
            # Extract phase description after "Phase X: "
            phase_match = re.search(r'Phase\s*(\d+)(?::\s*(.*?))?\*\*', stripped)
            if phase_match:
                result["phase"] = phase_match.group(1)
        elif stripped.startswith('**Task:'):
            # Remove "**Task:** " prefix
            task_text = re.sub(r'^\*\*Task:\*\*\s*', '', stripped)
            result["task"] = task_text

    # Find content separator "---"
    try:
        sep_index = lines.index('---')
        content_lines = lines[sep_index + 1:]
        result["content"] = '\n'.join(content_lines).strip()
    except ValueError:
        # No separator, content is everything after header
        if len(lines) > 1:
            result["content"] = '\n'.join(lines[1:]).strip()

    return result
```

#### **Function 3: `snap_to_constraint`**
```python
import math

def snap_to_constraint(point: tuple[float, float], line_points: list[tuple]) -> tuple[float, float]:
    """
    Snaps a 2D point to the nearest point on a polyline defined by line_points.

    Args:
        point: (x, y) coordinate to snap.
        line_points: List of (x, y) tuples defining the polyline segments.
                     Must have at least 2 points.

    Returns:
        (x, y) of the nearest point on the polyline.
        If line_points has < 2 points, returns the original point.
    """
    if len(line_points) < 2:
        return point

    px, py = point
    min_dist_sq = float('inf')
    snap_x, snap_y = point

    for i in range(len(line_points) - 1):
        x1, y1 = line_points[i]
        x2, y2 = line_points[i + 1]

        # Vector from segment start to end
        dx, dy = x2 - x1, y2 - y1
        seg_len_sq = dx*dx + dy*dy

        if seg_len_sq == 0:
            # Zero-length segment, just distance to point
            dist_sq = (px - x1)**2 + (py - y1)**2
            if dist_sq < min_dist_sq:
                min_dist_sq = dist_sq
                snap_x, snap_y = x1, y1
            continue

        # Project point onto segment line parameter t
        t = ((px - x1) * dx + (py - y1) * dy) / seg_len_sq
        t_clamped = max(0.0, min(1.0, t))

        # Closest point on segment
        closest_x = x1 + t_clamped * dx
        closest_y = y1 + t_clamped * dy

        dist_sq = (px - closest_x)**2 + (py - closest_y)**2
        if dist_sq < min_dist_sq:
            min_dist_sq = dist_sq
            snap_x, snap_y = closest_x, closest_y

    return (snap_x, snap_y)
```
