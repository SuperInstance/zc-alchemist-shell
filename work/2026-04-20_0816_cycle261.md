# ===== Function 1: validate_deadband_protocol =====
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol's strict priority.
    
    Args:
        steps: List of step strings (e.g., ["P0", "P1", "P2"])
        
    Returns:
        Tuple of (is_valid: bool, error_message: str)
    """
    valid_phases = {"P0", "P1", "P2"}
    
    # Check for invalid phases
    for i, step in enumerate(steps):
        if step not in valid_phases:
            return False, f"Invalid phase '{step}' at position {i}"
    
    # Check strict ordering
    phase_order = {"P0": 0, "P1": 1, "P2": 2}
    last_index = -1
    for i, step in enumerate(steps):
        current_index = phase_order[step]
        if current_index < last_index:
            return False, f"'{step}' at position {i} violates strict priority (P0→P1→P2)"
        last_index = current_index
    
    return True, ""

# ===== Function 2: parse_fleet_tile =====
def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parses raw training tile text into structured data.
    
    Args:
        raw_text: Raw text from a training tile submission
        
    Returns:
        Dictionary with keys: agent_name, cycle, task, content, timestamp
    """
    result = {
        "agent_name": "",
        "cycle": "",
        "task": "",
        "content": "",
        "timestamp": ""
    }
    
    # Extract agent name (look for patterns like "Agent: X" or "# X")
    agent_pattern = r"(?:Agent|agent|#)\s*[:|-]?\s*([A-Za-z0-9_-]+)"
    agent_match = re.search(agent_pattern, raw_text)
    if agent_match:
        result["agent_name"] = agent_match.group(1).strip()
    
    # Extract cycle number
    cycle_pattern = r"Cycle\s*[:|-]?\s*(\d+)"
    cycle_match = re.search(cycle_pattern, raw_text)
    if cycle_match:
        result["cycle"] = cycle_match.group(1).strip()
    
    # Extract task (look for "Task:" followed by text until double newline)
    task_pattern = r"Task\s*[:|-]?\s*(.+?)(?:\n\n|\n#|$)"
    task_match = re.search(task_pattern, raw_text, re.DOTALL)
    if task_match:
        result["task"] = task_match.group(1).strip()
    
    # Extract timestamp (ISO format or date-like patterns)
    timestamp_pattern = r"\d{4}-\d{2}-\d{2}[T\s]\d{2}:\d{2}:\d{2}"
    timestamp_match = re.search(timestamp_pattern, raw_text)
    if timestamp_match:
        result["timestamp"] = timestamp_match.group(0).strip()
    
    # Content is everything else, cleaned
    lines = raw_text.split('\n')
    content_lines = []
    for line in lines:
        if not re.match(r'^(Agent|agent|#|Cycle|Task|Timestamp)', line, re.I):
            content_lines.append(line.strip())
    result["content"] = '\n'.join(filter(None, content_lines))
    
    return result

def calculate_similarity_score(tile1: dict, tile2: dict, weights: dict = None) -> float:
    """
    Calculates weighted cosine similarity between two tile dictionaries.
    
    Args:
        tile1: First tile dictionary
        tile2: Second tile dictionary
        weights: Optional dictionary with keys 'task' and 'content' (default: 0.5 each)
        
    Returns:
        Similarity score between 0 and 1
    """
    if weights is None:
        weights = {"task": 0.5, "content": 0.5}
    
    # Ensure weights sum to 1
    weight_sum = sum(weights.values())
    if weight_sum == 0:
        return 0.0
    weights = {k: v/weight_sum for k, v in weights.items()}
    
    # Prepare text fields
    task1 = tile1.get("task", "")
    task2 = tile2.get("task", "")
    content1 = tile1.get("content", "")
    content2 = tile2.get("content", "")
    
    # Calculate task similarity (exact match for simplicity)
    task_sim = 1.0 if task1 == task2 else 0.0
    
    # Calculate content similarity using TF-IDF
    if content1 or content2:
        vectorizer = TfidfVectorizer(stop_words='english')
        try:
            tfidf_matrix = vectorizer.fit_transform([content1, content2])
            content_sim = cosine_similarity(tfidf_matrix[0:1], tfidf_matrix[1:2])[0][0]
        except:
            content_sim = 0.0
    else:
        content_sim = 0.0
    
    # Weighted sum
    final_score = (weights.get("task", 0) * task_sim) + (weights.get("content", 0) * content_sim)
    return float(final_score)
