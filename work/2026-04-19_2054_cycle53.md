import datetime
from typing import Dict, List, Tuple, Any, Optional

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol to a list of float values.
    
    Args:
        values: List of input floats.
        deadband: Threshold around zero (default 0.1).
        
    Returns:
        List where values within [-deadband, +deadband] are set to 0.0.
    """
    return [0.0 if -deadband <= v <= deadband else v for v in values]


def parse_fleet_log(log_line: str) -> Dict[str, Optional[Any]]:
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format: "YYYY-MM-DD HH:MM:SS | AGENT name | CYCLE N | PHASE M | STATUS status"
    Fields may be missing or out of order.
    
    Args:
        log_line: A single log line string.
        
    Returns:
        Dictionary with keys: timestamp, agent, cycle, phase, status.
        Missing fields become None.
    """
    result = {
        "timestamp": None,
        "agent": None,
        "cycle": None,
        "phase": None,
        "status": None
    }
    
    parts = log_line.split(" | ")
    for part in parts:
        part = part.strip()
        if part.startswith("AGENT "):
            result["agent"] = part[6:].strip()
        elif part.startswith("CYCLE "):
            try:
                result["cycle"] = int(part[6:].strip())
            except ValueError:
                result["cycle"] = None
        elif part.startswith("PHASE "):
            try:
                result["phase"] = int(part[6:].strip())
            except ValueError:
                result["phase"] = None
        elif part.startswith("STATUS "):
            result["status"] = part[7:].strip()
        else:
            # Try to parse as timestamp
            try:
                dt = datetime.datetime.strptime(part, "%Y-%m-%d %H:%M:%S")
                result["timestamp"] = dt
            except ValueError:
                pass
    
    return result


def validate_tile(tile: Dict[str, Any], schema: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile against a JSON Schema-like schema.
    
    Args:
        tile: The tile dictionary to validate.
        schema: Schema dict with "type", "properties", "required".
        
    Returns:
        (is_valid, list_of_errors)
    """
    errors = []
    
    # Check required fields
    required = schema.get("required", [])
    for field in required:
        if field not in tile:
            errors.append(f"Missing required field: {field}")
    
    # Check field types
    properties = schema.get("properties", {})
    for field, value in tile.items():
        if field in properties:
            expected_type = properties[field].get("type")
            if expected_type == "string":
                if not isinstance(value, str):
                    errors.append(f"Field '{field}' should be string, got {type(value).__name__}")
            elif expected_type == "integer":
                if not isinstance(value, int):
                    errors.append(f"Field '{field}' should be integer, got {type(value).__name__}")
            elif expected_type == "boolean":
                if not isinstance(value, bool):
                    errors.append(f"Field '{field}' should be boolean, got {type(value).__name__}")
            elif expected_type == "array":
                if not isinstance(value, list):
                    errors.append(f"Field '{field}' should be array, got {type(value).__name__}")
            elif expected_type == "object":
                if not isinstance(value, dict):
                    errors.append(f"Field '{field}' should be object, got {type(value).__name__}")
    
    return (len(errors) == 0, errors)
