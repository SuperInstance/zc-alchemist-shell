def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Remove NaN and infinite values (negative space)
    P1: Keep values within [-deadband, deadband] (safe channel)
    P2: Sort by absolute value (optimize within channel)
    
    Args:
        values: List of float values
        deadband: Deadband threshold (non-negative)
    
    Returns:
        Filtered and sorted list of values
    """

        raise ValueError("deadband must be non-negative")

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line into a dictionary.
    
    Format: "YYYY-MM-DD HH:MM:SS | AGENT=X | CYCLE=N | STATUS=X | MSG=text"
    
    Args:
        log_line: Log line string
    
    Returns:
        Dictionary with parsed fields
    """
    result = {
        "timestamp": None,
        "agent": None,
        "cycle": None,
        "status": None,
        "message": None
    }
    
    try:
        # Split by pipe
        parts = log_line.split(" | ")
        
        # Parse timestamp (first part)
        if len(parts) > 0:
            result["timestamp"] = datetime.strptime(parts[0], "%Y-%m-%d %H:%M:%S")
        
        # Parse key-value pairs
        for part in parts[1:]:
            if "=" in part:
                key, value = part.split("=", 1)
                key = key.strip().lower()
                
                if key == "agent":
                    result["agent"] = value
                elif key == "cycle":
                    try:
                        result["cycle"] = int(value)
                    except ValueError:
                        result["cycle"] = None
                elif key == "status":
                    result["status"] = value
                elif key == "msg":
                    result["message"] = value
    
    except Exception as e:
        # Graceful degradation - return what we have
        pass
    
    return result

def validate_training_tile(tile: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile object against schema.
    
    Required keys: id, agent, cycle, content, timestamp
    """
    errors = []
    
    # Check required keys
    required = ["id", "agent", "cycle", "content", "timestamp"]
    for key in required:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    # Validate id (UUID)
    if "id" in tile:
        try:
            uuid.UUID(tile["id"])
        except (ValueError, TypeError):
            errors.append("Invalid UUID format for id")
    
    # Validate agent (non-empty string)
    if "agent" in tile and not isinstance(tile["agent"], str):
        errors.append("agent must be a string")
    elif "agent" in tile and not tile["agent"].strip():
        errors.append("agent cannot be empty")
    
    # Validate cycle (non-negative integer)
    if "cycle" in tile:
        if not isinstance(tile["cycle"], int):
            errors.append("cycle must be an integer")
        elif tile["cycle"] < 0:
            errors.append("cycle must be non-negative")
    
    # Validate content (non-empty string)
    if "content" in tile and not isinstance(tile["content"], str):
        errors.append("content must be a string")
    elif "content" in tile and not tile["content"].strip():
        errors.append("content cannot be empty")
    
    # Validate timestamp (ISO format)
    if "timestamp" in tile:
        try:
            datetime.fromisoformat(tile["timestamp"].replace('Z', '+00:00'))
        except (ValueError, TypeError):
            errors.append("timestamp must be valid ISO format")
    
    return len(errors) == 0, errors
