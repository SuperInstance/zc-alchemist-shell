```python
def deadband_filter(data, lower, upper):
    """
    Apply a deadband filter to input data.
    
    Parameters:
    data (float or list): Input value or list of values
    lower (float): Lower deadband threshold
    upper (float): Upper deadband threshold
    
    Returns:
    float or list: Filtered output where values within [lower, upper] 
                   are mapped to zero, others pass through linearly.
    
    Example:
    >>> deadband_filter([-2, -1, 0, 1, 2, 3, 4], 1, 3)
    [-2, -1, 0, 0, 0, 0, 4]
    """
    if lower > upper:
        lower, upper = upper, lower
    
    def _filter_single(x):
        if lower <= x <= upper:
            return 0.0
        return x
    
    if isinstance(data, list):
        return [_filter_single(x) for x in data]
    return _filter_single(data)
```

#### **Function 2: parse_fleet_log**
```python
import re
from datetime import datetime

def parse_fleet_log(log_text):
    """
    Parse a structured fleet log entry.
    
    Expected format: "[TIMESTAMP] AGENT CYCLE: MESSAGE"
    Example: "[2026-04-20 14:39:42] Alchemist 345: Testing deadband filter"
    
    Parameters:
    log_text (str): Log line string
    
    Returns:
    dict: Parsed components with keys: timestamp, agent, cycle, message
          Returns None if format doesn't match
    """
    pattern = r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+) (\d+): (.+)$'
    match = re.match(pattern, log_text)
    
    if not match:
        return None
    
    timestamp_str, agent, cycle_str, message = match.groups()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
        cycle = int(cycle_str)
    except ValueError:
        return None
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "message": message.strip()
    }
```

```python
import math

def constraint_snap(point, line_start, line_end, tolerance):
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Parameters:
    point (tuple): (x, y) coordinates of point to snap
    line_start (tuple): (x, y) start of line segment
    line_end (tuple): (x, y) end of line segment
    tolerance (float): Maximum distance for snapping
    
    Returns:
    tuple: (x, y) of snapped point if within tolerance, else original point
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector math for projection
    dx = x2 - x1
    dy = y2 - y1
    line_len_sq = dx*dx + dy*dy
    
    # Handle zero-length segment
    if line_len_sq == 0:
        dist = math.hypot(px - x1, py - y1)
        return (x1, y1) if dist <= tolerance else point
    
    # Project point onto line
    t = ((px - x1) * dx + (py - y1) * dy) / line_len_sq
    t = max(0, min(1, t))  # Clamp to segment
    
    # Nearest point on segment
    nearest_x = x1 + t * dx
    nearest_y = y1 + t * dy
    
    # Check distance
    dist = math.hypot(px - nearest_x, py - nearest_y)
    
    if dist <= tolerance:
        return (nearest_x, nearest_y)
    return point
```

```python
def deadband_filter(data, lower, upper):
    """
    Implements a deadband filter.
    
    Args:
        data: A single numeric value or list of numeric values.
        lower: Lower threshold of deadband.
        upper: Upper threshold of deadband.
        
    Returns:
        Filtered data where values within [lower, upper] are set to 0,
        otherwise the original value is returned.
        
    Example:
        >>> deadband_filter([1, 2, 3, 4, 5], 2, 4)
        [1, 0, 0, 0, 5]
    """
    # Ensure lower <= upper
    if lower > upper:
        lower, upper = upper, lower
        
    if isinstance(data, list):
        return [0 if lower <= x <= upper else x for x in data]
    else:
        return 0 if lower <= data <= upper else data
```

```python
            'agent': agent,
            'cycle': int(cycle),
```
