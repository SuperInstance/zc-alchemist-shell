import math
from datetime import datetime, timezone

def deadband_filter(readings, threshold):
    """
    Apply Deadband Protocol to sensor readings.
    
    P0: Filter out NaN/infinite values (negative space).
    P1: Keep readings within safe channel [-threshold, threshold].
    P2: From safe readings, pick closest to zero (optimal).
    
    Args:
        readings: List of float sensor readings.
        threshold: Positive float defining safe channel.
        
    Returns:
        Optimal safe reading (float) or None if no safe readings.
    """
    # P0: Remove negative space
    valid_readings = []
    for r in readings:
        if not (math.isnan(r) or math.isinf(r)):
            valid_readings.append(r)
    
    # P1: Find safe channel
    safe_readings = [r for r in valid_readings if -threshold <= r <= threshold]
    
    # P2: Optimize within channel
    if not safe_readings:
        return None
    
    # Find reading with minimum absolute value
    optimal = min(safe_readings, key=abs)
    return optimal


def constraint_snap(point, constraints):
    """
    Snap a 2D point to geometric constraints.
    
    Args:
        point: Tuple (x, y).
        constraints: List of dicts with keys:
            - "type": "horizontal", "vertical", or "distance"
            - "value": float
            
    Returns:
        Snapped point (x, y).
    """
    x, y = point
    
    for constraint in constraints:
        c_type = constraint["type"]
        c_val = constraint["value"]
        
        if c_type == "horizontal":
            y = c_val
        elif c_type == "vertical":
            x = c_val
        elif c_type == "distance":
            # Snap to maintain distance c_val from origin
            current_dist = math.sqrt(x*x + y*y)
            if current_dist > 0:
                scale = c_val / current_dist
                x *= scale
                y *= scale
            else:
                # Point at origin, can't snap distance
                pass
        else:
            raise ValueError(f"Unknown constraint type: {c_type}")
    
    return (x, y)


def validate_tile(tile):
    """
    Validate a training tile against fleet standards.
    
    Args:
        tile: Dict with keys "content", "author", "timestamp", "tags".
        
    Returns:
        Tuple (is_valid, list_of_errors).
    """
    errors = []
    
    # Check content
    if "content" not in tile:
        errors.append("Missing 'content' field")
    else:
        if not isinstance(tile["content"], str):
            errors.append("'content' must be a string")
        elif len(tile["content"].strip()) == 0:
            errors.append("'content' cannot be empty")
    
    # Check author
    fleet_agents = ["Oracle1", "JetsonClaw1", "Forgemaster", "Zeroclaw"]
    if "author" not in tile:
        errors.append("Missing 'author' field")
    elif tile["author"] not in fleet_agents:
        errors.append(f"Author '{tile['author']}' not in known fleet agents")
    
    # Check timestamp
    if "timestamp" not in tile:
        errors.append("Missing 'timestamp' field")
    else:
        try:
            dt = datetime.fromisoformat(tile["timestamp"].replace('Z', '+00:00'))
            now = datetime.now(timezone.utc)
            if dt > now:
                errors.append("Timestamp cannot be in the future")
        except ValueError:
            errors.append("Timestamp must be valid ISO 8601 format")
    
    # Check tags
    if "tags" not in tile:
        errors.append("Missing 'tags' field")
    else:
        if not isinstance(tile["tags"], list):
            errors.append("'tags' must be a list")
        else:
            for i, tag in enumerate(tile["tags"]):
                if not isinstance(tag, str):
                    errors.append(f"Tag at index {i} must be a string")
                elif len(tag) > 20:
                    errors.append(f"Tag '{tag[:10]}...' exceeds 20 characters")
    
    is_valid = len(errors) == 0
    return (is_valid, errors)
