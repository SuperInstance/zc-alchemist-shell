"""
Code generated by DeepSeek-chat for Alchemist model comparison task.
"""

from typing import Any
import re
from datetime import datetime

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if steps follow Deadband Protocol: P0 -> P1 -> P2.
    
    Args:
        steps: List of step strings, e.g., ["P0: map rocks", "P1: find channel"]
    
    Returns:
        Tuple of (is_valid, error_message). Error message empty if valid.
    """
    if not isinstance(steps, list):
        return False, "Input must be a list"
    
    # Extract protocol prefixes
    prefixes = []
    for step in steps:
        if not isinstance(step, str):
            return False, f"Step must be string, got {type(step)}"
        match = re.match(r'^(P[0-2]):', step.strip())
        if match:
            prefixes.append(match.group(1))
    
    # Check ordering
    expected_order = ['P0', 'P1', 'P2']
    seen = set()
    last_index = -1
    
    for prefix in prefixes:
        if prefix in seen:
            return False, f"Duplicate protocol phase: {prefix}"
        current_index = expected_order.index(prefix)
        if current_index < last_index:
            return False, f"Protocol violation: {prefix} appears after {expected_order[last_index]}"
        last_index = current_index
        seen.add(prefix)
    
    return True, ""

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses fleet training tile text into structured dictionary.
    
    Expected format:
        # Cycle X — Agent
        **Phase Y: ...**
        **Task:** ...
        Content...
    
    Args:
        tile_text: Raw tile text
    
    Returns:
        Dictionary with keys: cycle, agent, phase, task, content, timestamp
    """
    result = {
        'cycle': None,
        'agent': None,
        'phase': None,
        'task': None,
        'content': '',
        'timestamp': datetime.utcnow().isoformat()
    }
    
    lines = tile_text.strip().split('\n')
    
    # Parse header line: # Cycle X — Agent
    if lines and lines[0].startswith('# Cycle'):
        header_match = re.match(r'# Cycle\s+(\d+)\s*—\s*(\w+)', lines[0])
        if header_match:
            result['cycle'] = int(header_match.group(1))
            result['agent'] = header_match.group(2)
    
    # Look for Phase and Task markers
    for i, line in enumerate(lines):
        if '**Phase' in line:
            phase_match = re.search(r'Phase\s+(\d+):\s*(.+)', line)
            if phase_match:
                result['phase'] = int(phase_match.group(1))
        elif '**Task:**' in line:
            result['task'] = line.replace('**Task:**', '').strip()
    
    # Content is everything else
    content_lines = []
    in_content = False
    for line in lines:
        if line.startswith('---') or '**Task:**' in line:
            in_content = True
            continue
        if in_content and line.strip() and not line.startswith('#'):
            content_lines.append(line)
    
    result['content'] = '\n'.join(content_lines).strip()
    return result

def calculate_similarity_score(tile_a: dict, tile_b: dict, weights: dict = None) -> float:
    """
    Calculates weighted similarity between two parsed tile dictionaries.
    
    Args:
        tile_a: First tile dict from parse_fleet_tile()
        tile_b: Second tile dict
        weights: Optional dict with keys 'content', 'task', 'agent'
                 Default: {'content': 0.6, 'task': 0.3, 'agent': 0.1}
    
    Returns:
        Float score from 0.0 (no similarity) to 1.0 (identical)
    """
    default_weights = {'content': 0.6, 'task': 0.3, 'agent': 0.1}
    if weights is None:
        weights = default_weights
    else:
        # Ensure all weights are present
        for key in default_weights:
            weights.setdefault(key, default_weights[key])
    
    # Normalize weights to sum to 1.0
    weight_sum = sum(weights.values())
    if weight_sum == 0:
        return 0.0
    normalized_weights = {k: v/weight_sum for k, v in weights.items()}
    
    score = 0.0
    
    # Content similarity (simple Jaccard on words)
    if 'content' in tile_a and 'content' in tile_b:
        words_a = set(tile_a['content'].lower().split())
        words_b = set(tile_b['content'].lower().split())
        if words_a or words_b:
            content_sim = len(words_a & words_b) / len(words_a | words_b)
            score += content_sim * normalized_weights['content']
    
    # Task similarity (exact match)
    if tile_a.get('task') == tile_b.get('task'):
        score += normalized_weights['task']
    
    # Agent similarity (exact match)
    if tile_a.get('agent') == tile_b.get('agent'):
        score += normalized_weights['agent']
    
    return min(score, 1.0)  # Cap at 1.0
