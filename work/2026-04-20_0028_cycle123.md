def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of steps against the Deadband Protocol priority order.
    Protocol: All P0 steps must precede any P1, which must precede any P2.
    Steps must start with 'P0:', 'P1:', or 'P2:'.
    Returns (is_valid, error_message).
    """
    expected_phase = 0  # Start expecting P0
    for i, step in enumerate(steps):
        step = step.strip()
        if not (step.startswith('P0:') or step.startswith('P1:') or step.startswith('P2:')):
            return False, f"Invalid step prefix at index {i}: '{step[:3]}'"
        
        if step.startswith('P0:'):
            current_phase = 0
        elif step.startswith('P1:'):
            current_phase = 1
        else:  # P2
            current_phase = 2

        if current_phase < expected_phase:
            return False, f"Phase regression at index {i}: {step[:3]} found after higher phase was expected"
        if current_phase > expected_phase:
            # Advance phase, but cannot go back
            expected_phase = current_phase
    
    return True, "Sequence valid"
```

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validate that steps follow Deadband Protocol ordering (P0 -> P1 -> P2).
    """
    seen_p1 = False
    seen_p2 = False
    for idx, step in enumerate(steps):
        if not step.startswith(('P0:', 'P1:', 'P2:')):
            return False, f"Invalid step prefix in '{step}' at position {idx}"
        
        if step.startswith('P0:'):
            if seen_p1 or seen_p2:
                return False, f"P0 step at index {idx} appears after P1/P2"
        elif step.startswith('P1:'):
            seen_p1 = True
            if seen_p2:
                return False, f"P1 step at index {idx} appears after P2"
        elif step.startswith('P2:'):
            seen_p2 = True
    return True, "Valid"
```

import re

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parse a fleet training tile markdown text into structured data.
    Expected format includes headers like '# Cycle N — Agent' and lines like '**Key:** value'.
    """
    lines = tile_text.split('\n')
    result = {
        'agent': 'Unknown',
        'cycle': -1,
        'phase': -1,
        'task': '',
        'content': ''
    }
    
    # Parse headers and key-value lines
    content_start = 0
    for i, line in enumerate(lines):
        line = line.strip()
        # Agent and cycle from header
        if line.startswith('# Cycle'):
            match = re.match(r'# Cycle\s+(\d+)\s+—\s+(\w+)', line)
            if match:
                result['cycle'] = int(match.group(1))
                result['agent'] = match.group(2)
        # Key-value patterns
        elif line.startswith('**Phase:**'):
            result['phase'] = int(line.replace('**Phase:**', '').strip())
        elif line.startswith('**Task:**'):
            result['task'] = line.replace('**Task:**', '').strip()
            content_start = i + 1
        elif line.startswith('## Task:'):
            result['task'] = line.replace('## Task:', '').strip()
            content_start = i + 1
    
    # Content is everything after task line or from a certain point
    if content_start < len(lines):
        content_lines = []
        for line in lines[content_start:]:
            if line.strip() and not line.strip().startswith('#'):
                content_lines.append(line)
        result['content'] = '\n'.join(content_lines).strip()
    
    return result
```

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parse fleet tile markdown into structured dict.
    """
    data = {"agent": "", "cycle": None, "phase": None, "task": "", "content": ""}
    
    # Simple line-by-line parsing
    in_content = False
    content_lines = []
    for line in tile_text.split('\n'):
        if line.startswith('# Cycle'):
            parts = line.split('—')
            if len(parts) == 2:
                data['agent'] = parts[1].strip()
                # extract cycle number
                import re
                cycle_match = re.search(r'Cycle\s+(\d+)', parts[0])
                if cycle_match:
                    data['cycle'] = int(cycle_match.group(1))
        elif line.startswith('**Phase:**'):
            data['phase'] = int(line.split('**Phase:**')[1].strip())
        elif line.startswith('**Task:**'):
            data['task'] = line.split('**Task:**')[1].strip()
            in_content = True
        elif in_content and line.strip() and not line.startswith('#'):
            content_lines.append(line)
    
    data['content'] = '\n'.join(content_lines).strip()
    return data
```
