#### **DeepSeek-chat Output:**

def deadband_filter(decisions):
    """
    Apply the Deadband Protocol (P0, P1, P2) to a list of decision values.

    P0: Remove values <= 0.0 (negative space).
    P1: Keep values where 0.0 < value <= 100.0 (safe channels).
    P2: Return the top 3 highest values from P1 results (optimize within channels).

    Args:
        decisions (list of float): Input decision values.

    Returns:
        list of float: Filtered and optimized decision values.
    """
    # P0: Map negative space
    p0_results = [d for d in decisions if d > 0.0]

    # P1: Find safe channels
    p1_results = [d for d in p0_results if d <= 100.0]

    # P2: Optimize within channels
    p1_results.sort(reverse=True)
    return p1_results[:3]

# Example usage

### **Function 2: `parse_fleet_context`**

**Prompt:**
"Write a Python function `parse_fleet_context` that takes a string `md_text` containing markdown in the format of FLEET-CONTEXT.md. It should parse the text and return a dictionary with the following keys: 'agents' (list of strings from the 'Who We Are' section), 'technologies' (list of strings from the 'Key Technologies' section), and 'rules' (list of strings from the 'Rules' section). Assume section headers are exactly '## Who We Are', '## Key Technologies', and '## Rules'. Extract the list items from each section. Handle potential markdown list syntax ('- ' or '* '). Return an empty list for a section if not found. Include a docstring."

#### **DeepSeek-chat Output:**

import re

def parse_fleet_context(md_text):
    """
    Parse a fleet context markdown string and extract structured data.

    Args:
        md_text (str): Markdown text following FLEET-CONTEXT.md format.

    Returns:
        dict: Dictionary with keys 'agents', 'technologies', 'rules'.
              Each value is a list of strings extracted from list items
              in the corresponding section.
    """
    def extract_section_items(section_header):
        pattern = rf'{re.escape(section_header)}\s*\n(.*?)(?=\n##|\Z)'
        match = re.search(pattern, md_text, re.DOTALL)
        if not match:
            return []
        section_content = match.group(1).strip()
        # Find list items starting with '- ' or '* '
        items = re.findall(r'^[-\*]\s*(.+)$', section_content, re.MULTILINE)
        # Also handle lines that might be part of a list but without markdown
        if not items:
            # Fallback: split by newline and clean
            lines = [line.strip() for line in section_content.split('\n') if line.strip()]
            items = lines
        return items

    return {
        'agents': extract_section_items('## Who We Are'),
        'technologies': extract_section_items('## Key Technologies'),
        'rules': extract_section_items('## Rules')
    }
