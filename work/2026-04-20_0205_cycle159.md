def validate_deadband_protocol(actions: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of actions against Deadband Protocol priorities.
    P0: Must map negative space (identify risks)
    P1: Must find safe channels (explore options)
    P2: Must optimize within channels (refine)
    Rule: Cannot skip to P2 without completing P0 and P1.
    
    Args:
        actions: List of action strings, each prefixed with P0_, P1_, or P2_
    
    Returns:
        Tuple of (is_valid, error_message)
    """
    if not actions:
        return False, "Empty action sequence"
    
    # Track which phases have been seen
    phases_seen = set()
    
    for i, action in enumerate(actions):
        if not isinstance(action, str):
            return False, f"Action at index {i} is not a string"
        
        # Check prefix
        if action.startswith("P0_"):
            phases_seen.add("P0")
        elif action.startswith("P1_"):
            if "P0" not in phases_seen:
                return False, f"P1 action '{action}' at index {i} before any P0 action"
            phases_seen.add("P1")
        elif action.startswith("P2_"):
            if "P0" not in phases_seen:
                return False, f"P2 action '{action}' at index {i} before any P0 action"
            if "P1" not in phases_seen:
                return False, f"P2 action '{action}' at index {i} before any P1 action"
            phases_seen.add("P2")
        else:
            return False, f"Action '{action}' at index {i} has invalid prefix (must start with P0_, P1_, or P2_)"
    
    # Must have at least one action from each phase for a complete protocol
    if len(phases_seen) < 3:
        return False, f"Incomplete protocol: only covered phases {sorted(phases_seen)}"
    
    return True, "Protocol validation passed"

def parse_fleet_tile(content: str) -> dict:
    """
    Parses a raw training tile text into structured dictionary.
    
    Expected format (example):
        # Cycle 157 — Alchemist
        **Phase 4: Build & Test**
        **Task:** Compare models...
        ---
        ## Output
        Here is my analysis...
    
    Returns dict with keys: agent, cycle, phase, task, output, timestamp.
    """
    import re
    from datetime import datetime
    
    result = {
        "agent": None,
        "cycle": None,
        "phase": None,
        "task": None,
        "output": None,
        "timestamp": None
    }
    
    # Extract agent and cycle from first line
    first_line_match = re.match(r"# Cycle (\d+) — (\w+)", content)
    if first_line_match:
        result["cycle"] = int(first_line_match.group(1))
        result["agent"] = first_line_match.group(2)
    
    # Extract phase
    phase_match = re.search(r"\*\*Phase (\d+): ([^\*]+)\*\*", content)
    if phase_match:
        result["phase"] = {
            "number": int(phase_match.group(1)),
            "name": phase_match.group(2).strip()
        }
    
    # Extract task
    task_match = re.search(r"\*\*Task:\*\* ([^\n]+)", content)
    if task_match:
        result["task"] = task_match.group(1).strip()
    
    # Extract output (everything after the first "---" separator)
    parts = content.split("---", 1)
    if len(parts) > 1:
        result["output"] = parts[1].strip()
    
    # Try to find timestamp in content
    ts_match = re.search(r"(\d{4}-\d{2}-\d{2} \d{2}:\d{2} UTC)", content)
    if ts_match:
        try:
            result["timestamp"] = datetime.strptime(ts_match.group(1), "%Y-%m-%d %H:%M UTC")
        except ValueError:
            pass
    
    return result

def calculate_safety_channel(boundaries: list[tuple[float, float]], position: tuple[float, float]) -> float:
    """
    Calculates safety score (0-1) for a position within dimensional boundaries.
    
    For each dimension:
    - Score = 0 if outside boundaries (in negative space)
    - Score = distance from closest boundary / (range/2) if inside, normalized to 0-1
    
    Final score is geometric mean of dimension scores.
    """
    if len(boundaries) != len(position):
        raise ValueError(f"Boundaries dimensions ({len(boundaries)}) must match position dimensions ({len(position)})")
    
    if not boundaries:
        return 1.0  # No constraints = perfectly safe
    
    dimension_scores = []
    
    for (dim_min, dim_max), pos in zip(boundaries, position):
        # Check if outside boundaries (P0 violation)
        if pos < dim_min or pos > dim_max:
            return 0.0
        
        # Calculate normalized distance from center
        range_width = dim_max - dim_min
        if range_width <= 0:
            # Zero-width boundary: must be exactly at that point
            dimension_scores.append(1.0 if pos == dim_min else 0.0)
            continue
        
        center = (dim_min + dim_max) / 2
        max_distance_from_center = range_width / 2
        
        # Distance from center, normalized to 0-1 where 1 is at center, 0 at boundary
        distance_from_center = abs(pos - center)
        if distance_from_center >= max_distance_from_center:
            score = 0.0
        else:
            score = 1.0 - (distance_from_center / max_distance_from_center)
        
        dimension_scores.append(score)
    
    # Geometric mean preserves the "all dimensions must be safe" property
    import math
    product = 1.0
    for score in dimension_scores:
        product *= max(score, 0.0001)  # Avoid zero
    
    safety_score = product ** (1.0 / len(dimension_scores))
    return round(safety_score, 4)
