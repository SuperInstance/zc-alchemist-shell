```python
def deadband_p0_checker(actions, p0_violations):
    """
    Check if any action in the provided list matches a known P0 violation.
    
    Args:
        actions (list of str): List of proposed actions.
        p0_violations (set of str): Set of strings representing forbidden actions.
        
    Returns:
        bool: True if no action violates P0, False otherwise.
        
    Example:
        >>> violations = {"skip_p0", "ignore_constraints", "hallucinate"}
        >>> deadband_p0_checker(["plan_path", "skip_p0"], violations)
        False
        >>> deadband_p0_checker(["plan_path", "validate"], violations)
        True
    """
    if not isinstance(actions, list) or not isinstance(p0_violations, set):
        raise TypeError("actions must be list, p0_violations must be set")
    
    for action in actions:
        if action in p0_violations:
            return False
    return True
```

def validate_tile(tile):
    """
    Validate a training tile against fleet submission criteria.
    
    Args:
        tile (dict): Candidate tile dictionary.
        
    Returns:
        tuple: (is_valid (bool), message (str))
        
    Example:
        >>> tile = {"id": "tile_001", "agent": "Alchemist", 
        ...         "cycle": 231, "content": "Test output", 
        ...         "type": "doc"}
        >>> validate_tile(tile)
        (True, "Tile is valid")
    """
    required_keys = {"id", "agent", "cycle", "content", "type"}
    valid_types = {"code", "doc", "data"}
    
    # Check all required keys present

        missing = required_keys - set(tile.keys())
        return False, f"Missing keys: {missing}"
    
    # Check content non‑empty string
    if not isinstance(tile["content"], str) or not tile["content"].strip():
        return False, "Content must be non‑empty string"
    
    # Check type is valid
    if tile["type"] not in valid_types:
        return False, f"Type must be one of {valid_types}"
    
    # Check cycle is non‑negative int
    if not isinstance(tile["cycle"], int) or tile["cycle"] < 0:
        return False, "Cycle must be non‑negative integer"
    
    return True, "Tile is valid"

**Prompt:** *“Write a Python function that scans a directory (like fleet‑knowledge/) and builds a dependency graph. Assume each .md file may contain lines like ‘depends: repo1, repo2’. Return a dict {file: [deps]}. Use pathlib.”*

```python
from pathlib import Path
import re

def repo_dependency_mapper(root_dir):
    """
    Map dependencies between markdown files in a directory.
    
    Args:
        root_dir (str or Path): Path to the directory to scan.
        
    Returns:
        dict: Mapping from file path (str) to list of dependency strings.
        
    Example:
        >>> deps = repo_dependency_mapper("./fleet-knowledge")
        >>> print(deps.get("FLEET-CONTEXT.md", []))
    """
    root = Path(root_dir)
    if not root.is_dir():
        raise ValueError(f"{root_dir} is not a valid directory")
    
    dep_pattern = re.compile(r"depends:\s*(.+)", re.IGNORECASE)
    graph = {}
    
    for md_file in root.rglob("*.md"):
        deps = []
        try:
            content = md_file.read_text(encoding="utf-8")
            for line in content.splitlines():
                match = dep_pattern.search(line)
                if match:
                    # Split by comma, strip whitespace
                    deps.extend([d.strip() for d in match.group(1).split(",")])
        except (UnicodeDecodeError, IOError):
            continue  # skip unreadable files
        
        graph[str(md_file.relative_to(root))] = deps
    
    return graph
```
