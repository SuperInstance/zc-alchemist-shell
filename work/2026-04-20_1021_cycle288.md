from typing import List

def deadband_filter(values: List[float], lower: float, upper: float) -> List[float]:
    """
    Apply a deadband filter to a list of values.
    
    For each value:
    - If value is within [lower, upper] (inclusive), output the previous output value.
    - If it's the first value and within deadband, output 0.0.
    - Otherwise, output the value itself.
    
    Args:
        values: List of input float values
        lower: Lower bound of deadband
        upper: Upper bound of deadband
    
    Returns:
        Filtered list of floats
    """
    if not values:
        return []
    
    filtered = []
    prev_output = 0.0
    
    for val in values:
        if lower <= val <= upper:
            filtered.append(prev_output)
        else:
            filtered.append(val)
            prev_output = val
    
    return filtered
```

from datetime import datetime
from typing import Dict, Any
import re

def parse_fleet_log_line(line: str) -> Dict[str, Any]:
    """
    Parse a fleet log line in format:
    [TIMESTAMP] [AGENT] [LEVEL] message key=value key2=value2
    
    Returns dict with keys: timestamp, agent, level, message, params
    
    Args:
        line: Log line string
    
    Returns:
        Dictionary with parsed components
    """
    # Basic pattern for the fixed parts
    pattern = r'\[([^\]]+)\] \[([^\]]+)\] \[([^\]]+)\] (.+)'
    match = re.match(pattern, line)
    
    if not match:
        return {"error": "Malformed log line", "original": line}
    
    timestamp_str, agent, level, rest = match.groups()
    
    # Parse timestamp
    try:
        # Assuming ISO format or similar
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        timestamp = timestamp_str  # fallback
    
    # Split message and key-value pairs
    parts = rest.strip().split()
    message_parts = []
    params = {}
    
    for part in parts:
        if '=' in part:
            key, val = part.split('=', 1)
            params[key] = val
        else:
            message_parts.append(part)
    
    message = ' '.join(message_parts)
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "level": level,
        "message": message,
        "params": params
    }
```

def constraint_snap(point: Tuple[float, float], 
                    constraints: List[Tuple[float, float, float]]) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest constraint within radius.
    
    Args:
        point: (x, y) tuple
        constraints: List of (x, y, radius) tuples
    
    Returns:
        Snapped (x, y) point
    """
    px, py = point
    min_dist = float('inf')
    snap_point = point
    
    for cx, cy, radius in constraints:
        dx = px - cx
        dy = py - cy
        dist = math.sqrt(dx*dx + dy*dy)
        
        if dist <= radius and dist < min_dist:
            min_dist = dist
            snap_point = (cx, cy)
    
    return snap_point

from typing import List

def deadband_filter(values: List[float], lower: float, upper: float) -> List[float]:
    """
    Deadband filter implementation.
    
    Parameters:
    - values: input list of floats
    - lower: lower deadband threshold
    - upper: upper deadband threshold
    
    Returns:
    - List of filtered floats where values within [lower, upper] are replaced
      by the last value outside the deadband (or 0.0 if none).
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    result = []
    last_valid = 0.0
    
    for v in values:
        if lower <= v <= upper:
            result.append(last_valid)
        else:
            result.append(v)
            last_valid = v
    
    return result

    Returns dict with timestamp (datetime), agent, level, message, and params dict.

        # Try common formats
        for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%dT%H:%M:%S", "%Y-%m-%d %H:%M:%S.%f"):
            try:
                timestamp = datetime.strptime(ts_str, fmt)
                break
            except ValueError:
                continue
        else:
            timestamp = ts_str  # fallback string
    except Exception:
        timestamp = ts_str
    
    # Split rest into message and key=value pairs
    parts = rest.split()
    message_parts = []
    params = {}
    
    for p in parts:
        if '=' in p:
            k, v = p.split('=', 1)
            params[k] = v
        else:
            message_parts.append(p)
    
    message = ' '.join(message_parts).strip()
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "level": level,
        "message": message,
        "params": params
    }
