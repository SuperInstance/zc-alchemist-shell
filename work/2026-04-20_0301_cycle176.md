import re
from typing import Tuple, List

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.
    
    Rules:
    1. Only valid steps are "P0", "P1", "P2".
    2. Must start with P0.
    3. Must not skip phases (P0 -> P1 -> P2).
    4. Must not repeat phases out of order.
    
    Returns:
        Tuple[bool, str]: (True, "") if valid, (False, error_message) otherwise.
    """
    valid_steps = {"P0", "P1", "P2"}
    expected_order = ["P0", "P1", "P2"]
    last_index = -1
    
    for i, step in enumerate(steps):
        if step not in valid_steps:
            return False, f"Invalid step '{step}' at position {i}. Must be P0, P1, or P2."
        
        current_index = expected_order.index(step)
        if current_index < last_index:
            return False, f"Protocol regression at position {i}: {step} after {expected_order[last_index]}."
        if current_index > last_index + 1:
            missing = expected_order[last_index + 1:current_index]
            return False, f"Protocol skip at position {i}: jumped from {expected_order[last_index]} to {step}, missing {missing}."
        
        last_index = current_index
    
    if last_index == -1:
        return False, "Empty step list."
    
    return True, ""
```

**Analysis:**
*   **Correctness:** Logic appears sound. Handles empty list, invalid steps, skips, and regressions.
*   **Style:** Clear, uses type hints from `typing`. Good error messages.
*   **Potential Issue:** `expected_order.index(step)` is O(n) for each step. For 3 steps it's fine, but not optimal for large lists. Could use a mapping dict.

import re

from typing import Dict, Any

def parse_fleet_log_line(line: str) -> Dict[str, Any]:
    """
    Parses a structured fleet log line into its components.
    
    Format: "YYYY-MM-DD HH:MM:SS [AGENT] [LEVEL] Message @key=value"
    Metadata (@key=value) is optional.
    
    Returns:
        Dictionary with keys: timestamp, agent, level, message, metadata.
    """
    # Base regex for timestamp, agent, level, and the start of the message
    base_pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[([^\]]+)\] \[([^\]]+)\] (.+)$'
    base_match = re.match(base_pattern, line.strip())
    if not base_match:
        # Fallback: return minimal structure if line doesn't match
        return {
            "timestamp": None,
            "agent": None,
            "level": None,
            "message": line.strip(),
            "metadata": {}
        }
    
    timestamp_str, agent, level, remainder = base_match.groups()
    
    # Parse metadata (key=value pairs after @)
    metadata = {}
    message_parts = []
    for part in remainder.split():
        if part.startswith('@'):
            kv = part[1:].split('=', 1)
            if len(kv) == 2:
                metadata[kv[0]] = kv[1]
            else:
                # Malformed metadata, keep as part of message?
                message_parts.append(part)
        else:
            message_parts.append(part)
    
    message = ' '.join(message_parts).strip()
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        timestamp = timestamp_str  # Keep as string if parsing fails
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "level": level,
        "message": message,
        "metadata": metadata
    }
```

        "cycle": None,  # Should be filled by the room server

        "task": task,
        "output": output,
        "tags": tags,

            "model": "deepseek-chat",  # Could be parameterized
            "language": "python"
        }
    }
    
    return tile
