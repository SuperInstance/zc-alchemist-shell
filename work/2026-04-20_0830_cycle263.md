"""
Fleet utility functions for Deadband Protocol, log parsing, and geometric snapping.
"""

import math
from typing import List, Tuple, Dict, Optional

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Map negative space - values within deadband are zeroed (noise zone).
    P1/P2: Values outside deadband pass through (safe channel).
    
    Args:
        values: List of float values to filter.
        deadband: Positive float threshold for zeroing (default 0.1).
    
    Returns:
        List of filtered floats, same length as input.
    
    Example:
        >>> deadband_filter([-0.05, 0.3, -0.15, 0.09], 0.1)
        [0.0, 0.3, -0.15, 0.0]
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for val in values:
        if abs(val) < deadband:
            result.append(0.0)
        else:
            result.append(val)
    return result


def parse_fleet_log(log_line: str) -> Dict[str, str]:
    """
    Parse a structured fleet log line into its components.
    
    Expected format: "YYYY-MM-DD HH:MM:SS | AGENT | LEVEL | MESSAGE"
    
    Args:
        log_line: String log line to parse.
    
    Returns:
        Dictionary with keys: timestamp, agent, level, message.
    
    Raises:
        ValueError: If log line doesn't match expected format.
    
    Example:
        >>> parse_fleet_log("2026-04-20 08:30:00 | Alchemist | INFO | Cycle 263 started")
        {'timestamp': '2026-04-20 08:30:00', 'agent': 'Alchemist', 'level': 'INFO', 'message': 'Cycle 263 started'}
    """
    parts = log_line.split(" | ")
    if len(parts) != 4:
        raise ValueError(f"Log line must have exactly 4 parts separated by ' | '. Got {len(parts)} parts: {log_line}")
    
    return {
        "timestamp": parts[0].strip(),
        "agent": parts[1].strip(),
        "level": parts[2].strip(),
        "message": parts[3].strip()
    }


def constraint_snap(point: Tuple[float, float], 
                   constraints: List[Tuple[float, float, float]], 
                   snap_threshold: float = 0.5) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest constraint circle if within threshold.
    
    Args:
        point: (x, y) tuple of the point to snap.
        constraints: List of (center_x, center_y, radius) tuples.
        snap_threshold: Maximum additional distance beyond radius to snap (default 0.5).
    
    Returns:
        (x, y) tuple of snapped point, or original if no constraint applies.
    
    Example:
        >>> constraint_snap((1.0, 1.0), [(0.0, 0.0, 1.0)], 0.2)
        (0.7071067811865475, 0.7071067811865475)
    """
    if not constraints:
        return point
    
    px, py = point
    best_snap = None
    best_distance = float('inf')
    
    for cx, cy, radius in constraints:
        # Distance from point to circle center
        distance = math.sqrt((px - cx) ** 2 + (py - cy) ** 2)
        
        # Check if point is within snap range (radius + threshold)
        if distance <= radius + snap_threshold:
            # Snap to circle edge along radial direction
            if distance > 0:
                scale = radius / distance
                snapped_x = cx + (px - cx) * scale
                snapped_y = cy + (py - cy) * scale

            if snap_distance < best_distance:
                best_distance = snap_distance
                best_snap = (snapped_x, snapped_y)
    
    return best_snap if best_snap is not None else point
