import re
from typing import List, Dict, Any, Optional

def deadband_filter(values: List[float], p0_threshold: float, 
                    p1_safe_min: float, p1_safe_max: float) -> List[float]:
    """
    Apply fleet Deadband Protocol (P0/P1) to filter a list of values.
    P0: Remove values within threshold of zero (ambiguous).
    P1: Keep only values within the safe channel.
    """
    # P0: Map negative space - remove values too close to zero
    p0_filtered = [v for v in values if abs(v) >= p0_threshold]
    
    # P1: Find safe channels - keep values within bounds
    p1_filtered = [v for v in p0_filtered if p1_safe_min <= v <= p1_safe_max]
    
    return p1_filtered

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a semi-structured fleet log line into structured data.
    """
    # Base pattern for timestamp, level, and message
    base_pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\] (.+)'
    base_match = re.match(base_pattern, log_line)
    
    if not base_match:
        return {"timestamp": None, "level": None, "message": log_line, "tile_id": None, "status": None}
    
    timestamp, level, message = base_match.groups()
    
    # Extract optional tile# and status
    tile_id = None
    status = None
    
    tile_match = re.search(r'tile#(\d+)', message)
    if tile_match:
        tile_id = int(tile_match.group(1))
    
    status_match = re.search(r'status=(\w+)', message)
    if status_match:
        status = status_match.group(1)
    
    return {
        "timestamp": timestamp,
        "level": level,
        "message": message,
        "tile_id": tile_id,
        "status": status
    }

def validate_tile(tile_candidate: Dict[str, Any]) -> bool:
    """
    Validate if a dictionary meets the minimum requirements for a training tile.
    """
    required_keys = ["agent", "cycle", "content", "timestamp"]
    
    for key in required_keys:
        if key not in tile_candidate:
            print(f"Invalid tile: missing required key '{key}'")
            return False
    
    # Type checks
    if not isinstance(tile_candidate["agent"], str):
        print("Invalid tile: 'agent' must be a string")
        return False
    if not isinstance(tile_candidate["cycle"], int):
        print("Invalid tile: 'cycle' must be an integer")
        return False
    if not isinstance(tile_candidate["content"], str):
        print("Invalid tile: 'content' must be a string")
        return False
    if not isinstance(tile_candidate["timestamp"], str):
        print("Invalid tile: 'timestamp' must be a string")
        return False
    
    return True
