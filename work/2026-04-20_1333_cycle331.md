from typing import List, Dict
import math

def deadband_filter(values: List[float], p0_threshold: float, p1_margin: float) -> Dict[str, List[float]]:
    """
    Implements the Cocapn fleet's Deadband Protocol (P0/P1/P2) for numeric values.
    
    Args:
        values: List of numeric values to categorize.
        p0_threshold: Absolute threshold below which values are P0 (negative space).
        p1_margin: Absolute margin above which values are P2 (optimization zone).
    
    Returns:
        Dictionary with keys 'p0', 'p1', 'p2' containing lists of values in each category.
    
    Raises:
        ValueError: If p0_threshold >= p1_margin (invalid deadband).
    """
    if p0_threshold >= p1_margin:
        raise ValueError("p0_threshold must be less than p1_margin for valid deadband")
    
    result = {"p0": [], "p1": [], "p2": []}
    for v in values:
        abs_v = abs(v)
        if abs_v < p0_threshold:
            result["p0"].append(v)
        elif abs_v <= p1_margin:
            result["p1"].append(v)
        else:
            result["p2"].append(v)
    return result

    P0: |value| < p0_threshold (avoid)
    P1: p0_threshold <= |value| <= p1_margin (safe)
    P2: |value| > p1_margin (optimize)
    """

        elif abs_val <= p1_margin:
            p1_vals.append(val)
        else:
            p2_vals.append(val)
