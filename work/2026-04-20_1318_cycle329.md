from typing import List, Dict, Any, Tuple
import re
from datetime import datetime

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of float values.
    
    Args:
        values: List of input float values
        deadband: Threshold around zero (default 0.1)
        
    Returns:
        List of filtered values where:
        - Values within [-deadband, deadband] map to 0.0
        - Values outside deadband are shifted toward zero by deadband magnitude
    """
    result = []
    for val in values:
        if -deadband <= val <= deadband:
            result.append(0.0)
        elif val > deadband:
            result.append(val - deadband)
        else:  # val < -deadband
            result.append(val + deadband)
    return result


def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line into components.
    
    Expected format: "[TIMESTAMP] [AGENT] [LEVEL] message key=value key2=value2"
    
    Args:
        log_line: A single log line string
        
    Returns:
        Dictionary with keys: timestamp, agent, level, message, kwargs
    """
    # Regex to capture the structured parts
    pattern = r'^\[([^\]]+)\] \[([^\]]+)\] \[([^\]]+)\] (.+?)(?:\s+(\S+=\S+.*))?$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        raise ValueError(f"Log line does not match expected format: {log_line}")
    
    timestamp, agent, level, message, kv_pairs = match.groups()
    
    # Parse key=value pairs if present
    kwargs = {}
    if kv_pairs:
        for pair in kv_pairs.split():
            if '=' in pair:
                key, value = pair.split('=', 1)
                kwargs[key] = value
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': message.strip(),
        'kwargs': kwargs
    }


def validate_tile_json(tile: dict) -> Tuple[bool, List[str]]:
    """
    Validate a training tile JSON structure.
    
    Required keys: 'agent', 'cycle', 'phase', 'content', 'timestamp'
    'content' must be non-empty string.
    
    Args:
        tile: Dictionary representing a training tile
        
    Returns:
        Tuple of (is_valid, list_of_error_messages)
    """
    required_keys = {'agent', 'cycle', 'phase', 'content', 'timestamp'}
    errors = []
    
    # Check required keys
    missing_keys = required_keys - set(tile.keys())
    if missing_keys:
        errors.append(f"Missing required keys: {sorted(missing_keys)}")
    
    # Validate content
    if 'content' in tile:
        if not isinstance(tile['content'], str):
            errors.append("'content' must be a string")
        elif not tile['content'].strip():
            errors.append("'content' cannot be empty")
    
    # Validate timestamp format (basic check)
    if 'timestamp' in tile:
        try:
            datetime.fromisoformat(tile['timestamp'].replace('Z', '+00:00'))
        except ValueError:
            errors.append("'timestamp' must be valid ISO format")
    
    return len(errors) == 0, errors

    """

    Args:

    Returns:

    """
