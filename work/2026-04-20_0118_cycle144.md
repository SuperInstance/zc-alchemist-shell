from typing import List, Tuple, Dict, Any
import re
from datetime import datetime

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates agent steps against Deadband Protocol priorities.
    P0 (negative space) must be addressed before P1 (safe channels),
    and P1 before P2 (optimization).
    
    Args:
        steps: List of step descriptions.
    
    Returns:
        (is_valid, error_message)
    """
    phase = None
    for i, step in enumerate(steps):
        step_lower = step.lower()
        # Detect phase from step content (simple keyword matching)
        if any(word in step_lower for word in ["avoid", "not", "negative", "p0"]):
            current_phase = "P0"
        elif any(word in step_lower for word in ["safe", "channel", "p1", "where"]):
            current_phase = "P1"
        elif any(word in step_lower for word in ["optimize", "best", "p2", "improve"]):
            current_phase = "P2"
        else:
            current_phase = phase  # Unchanged if no keyword
        
        # Check phase progression
        if phase is None:
            phase = current_phase
        elif current_phase:
            phase_order = {"P0": 0, "P1": 1, "P2": 2}
            if phase_order.get(current_phase, 0) < phase_order.get(phase, 0):
                return False, f"Protocol violation at step {i}: '{step}'. {current_phase} cannot come after {phase}."
            phase = current_phase
    
    return True, "Protocol valid."

def parse_fleet_tile(tile_text: str) -> Dict[str, Any]:
    """
    Parses a fleet training tile text into structured data.
    
    Expected format (from our cycles):
    # Cycle X — Agent
    **Phase: Y**
    **Task:** ...
    ---
    Content...
    
    Args:
        tile_text: Raw tile text.
    
    Returns:
        Dictionary with keys: agent, cycle, phase, task, content, timestamp.
    """
    result = {
        "agent": "unknown",
        "cycle": 0,
        "phase": 0,
        "task": "",
        "content": "",
        "timestamp": datetime.utcnow().isoformat()
    }
    
    # Extract header lines
    lines = tile_text.strip().split('\n')
    for i, line in enumerate(lines):
        # Cycle and Agent
        cycle_match = re.match(r'# Cycle (\d+) — (\w+)', line)
        if cycle_match:
            result["cycle"] = int(cycle_match.group(1))
            result["agent"] = cycle_match.group(2)
            continue
        
        # Phase
        phase_match = re.match(r'\*\*Phase:?\s*(\d+)\*\*', line, re.IGNORECASE)
        if phase_match:
            result["phase"] = int(phase_match.group(1))
            continue
        
        # Task
        task_match = re.match(r'\*\*Task:?\*\*\s*(.+)', line)
        if task_match:
            result["task"] = task_match.group(1).strip()
            continue
        
        # Content separator
        if line.strip() == '---':
            # Join remaining lines as content
            result["content"] = '\n'.join(lines[i+1:]).strip()
            break
    
    return result

def generate_safe_channel(obstacles: List[Tuple[int, int]], bounds: Tuple[int, int, int, int]) -> List[Tuple[int, int]]:
    """
    Finds a simple safe channel (P1) avoiding obstacles in a 2D grid.
    Uses a basic greedy right-then-down search.
    
    Args:
        obstacles: List of (x, y) coordinates to avoid.
        bounds: (min_x, min_y, max_x, max_y) bounding area.
    
    Returns:
        List of (x, y) coordinates representing a safe channel from start to end.
    """
    min_x, min_y, max_x, max_y = bounds
    obstacles_set = set(obstacles)
    
    channel = []
    x, y = min_x, min_y
    target = (max_x, max_y)
    
    # Simple heuristic: move right if possible, else down
    while (x, y) != target:
        channel.append((x, y))
        
        # Try moving right
        if x + 1 <= max_x and (x + 1, y) not in obstacles_set:
            x += 1
        # Try moving down
        elif y + 1 <= max_y and (x, y + 1) not in obstacles_set:
            y += 1
        else:
            # Stuck - backtrack not implemented for simplicity
            break
    
    if (x, y) == target:
        channel.append(target)
    
    return channel
