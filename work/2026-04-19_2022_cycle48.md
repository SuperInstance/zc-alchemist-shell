```python
def deadband_filter(values, epsilon):
    """
    Apply Deadband Protocol to a list of values.
    
    Args:
        values: List of numeric values (int, float).
        epsilon: Threshold for deadband (non‑negative).
    
    Returns:
        List of filtered values where any value in [-epsilon, epsilon]
        is replaced by 0.0.
    """
    if not values:
        return []
    
    result = []
    for v in values:
        try:
            val = float(v)
        except (TypeError, ValueError):
            # If conversion fails, keep original (or raise?).
            # Per fleet robustness, we keep as‑is.
            result.append(v)
            continue
        
        if -epsilon <= val <= epsilon:
            result.append(0.0)
        else:
            result.append(val)
    return result
```

```python
import re

def parse_fleet_log(line):
    """
    Parse a structured fleet log line.
    
    Format: [TIMESTAMP] AGENT: MESSAGE | KEY1=VALUE1 KEY2=VALUE2...
    
    Returns:
        Dictionary with keys: timestamp, agent, message, params.
        params is a dict of key‑value pairs (str -> str).
    """
    if not line or not line.strip():
        return {}
    
    # Match [TIMESTAMP] AGENT: MESSAGE
    main_pattern = r'\[([^\]]+)\]\s*([^:]+):\s*(.*?)(?:\s*\|\s*(.*))?$'
    match = re.match(main_pattern, line.strip())
    if not match:
        return {}
    
    timestamp, agent, message, kv_block = match.groups()
    params = {}
    if kv_block:
        # Split by spaces, each part KEY=VALUE
        for pair in kv_block.split():
            if '=' in pair:
                k, v = pair.split('=', 1)
                params[k.strip()] = v.strip()
    
    return {
        'timestamp': timestamp.strip(),
        'agent': agent.strip(),
        'message': message.strip(),
        'params': params
    }
```

```python
import math

def point_to_segment_dist(px, py, x1, y1, x2, y2):
    """Return closest point on segment and squared distance."""
    dx = x2 - x1
    dy = y2 - y1
    if dx == 0 and dy == 0:
        # Segment is a point
        return (x1, y1), (px - x1)**2 + (py - y1)**2
    
    t = ((px - x1) * dx + (py - y1) * dy) / (dx*dx + dy*dy)
    t = max(0, min(1, t))
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    dist_sq = (px - closest_x)**2 + (py - closest_y)**2
    return (closest_x, closest_y), dist_sq

def constraint_snap(point, segments, tol):
    """
    Snap point to nearest line segment within tolerance.
    
    Args:
        point: Tuple (x, y).
        segments: List of tuples (x1, y1, x2, y2).
        tol: Tolerance (positive float).
    
    Returns:
        Tuple (snapped_x, snapped_y) if within tol, else original point.
    """
    if not segments:
        return point
    
    px, py = point
    best_dist_sq = float('inf')
    best_point = point
    
    for seg in segments:
        (cx, cy), dist_sq = point_to_segment_dist(px, py, *seg)
        if dist_sq < best_dist_sq:
            best_dist_sq = dist_sq
            best_point = (cx, cy)
    
    if math.sqrt(best_dist_sq) <= tol:
        return best_point
    return point
```

```python
        if isinstance(v, (int, float)):

                filtered.append(0.0)
```
