from typing import List, Optional, Dict, Any, Tuple

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply Deadband Protocol to filter noise around zero.
    
    P0: Map values within +/- deadband of zero to zero (negative space).
    P1: Otherwise pass through (safe channel).
    
    Args:
        values: List of floating-point numbers.
        deadband: Absolute threshold around zero.
    
    Returns:
        Filtered list where values within deadband are zero.
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for v in values:
        if -deadband <= v <= deadband:
            result.append(0.0)
        else:
            result.append(v)
    return result


def parse_fleet_log(log_line: str) -> Optional[Dict[str, Any]]:
    """
    Parse a structured fleet log line into components.
    
    Expected format: 
    "[YYYY-MM-DD HH:MM:SS] Agent Name (cycle N): message"
    
    Args:
        log_line: A single line of fleet log.
    
    Returns:
        Dict with keys: timestamp, agent, cycle, message.
        Returns None if format does not match.
    """
    import re
    
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] Agent (\w+) \(cycle (\d+)\): (.+)'
    match = re.match(pattern, log_line)
    
    if not match:
        return None
    
    timestamp, agent, cycle_str, message = match.groups()
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': int(cycle_str),
        'message': message.strip()
    }


def constraint_snap(point: Tuple[float, float], constraints: List[Dict[str, Any]]) -> Tuple[float, float]:
    """
    Simple 2D geometric constraint snapping.
    
    Currently supports:
    - 'axis' snapping: snap coordinate to zero on specified axis.
    
    Args:
        point: (x, y) tuple.
        constraints: List of constraint dicts.
            Example: {'type': 'axis', 'target': 'x'} snaps x to 0.
    
    Returns:
        Snapped (x, y) tuple.
    """
    x, y = point
    
    for constraint in constraints:
        c_type = constraint.get('type')
        target = constraint.get('target')
        
        if c_type == 'axis':
            if target == 'x':
                x = 0.0
            elif target == 'y':
                y = 0.0
            else:
                raise ValueError(f"Unsupported axis target: {target}")
        else:
            raise NotImplementedError(f"Constraint type {c_type} not implemented")
    
    return (x, y)

from typing import List, Optional, Dict, Any, Tuple
import re

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Implements the Deadband Protocol (P0/P1) for signal filtering.
    
    P0: If value is within [-deadband, deadband], set to 0 (avoid noise zone).
    P1: Otherwise, keep original value (safe channel).
    
    Parameters:
        values (List[float]): input list of floats.
        deadband (float): non‑negative deadband width.
    
    Returns:
        List[float]: filtered values.
    
    Raises:
        ValueError: if deadband < 0.
    """
    if deadband < 0:
        raise ValueError("deadband must be >= 0")
    
    filtered = []
    for v in values:
        if abs(v) <= deadband:
            filtered.append(0.0)
        else:
            filtered.append(v)
    return filtered


def parse_fleet_log(log_line: str) -> Optional[Dict[str, Any]]:
    """
    Parses a standardized fleet log entry.
    
    Format: '[timestamp] Agent name (cycle N): message'
    
    Parameters:
        log_line (str): a single log line.
    
    Returns:
        Optional[Dict]: dictionary with keys timestamp, agent, cycle, message
        if the line matches; otherwise None.

    match = re.match(pattern, log_line.strip())
    
    if not match:
        return None
    
    timestamp, agent, cycle_str, message = match.groups()
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': int(cycle_str),
        'message': message
    }

    Applies simple 2D snapping constraints to a point.
    
    Supported constraint types:
        'axis': target must be 'x' or 'y'; snaps that coordinate to 0.
    
    Parameters:
        point (Tuple[float, float]): (x, y) coordinates.
        constraints (List[Dict]): list of constraint dictionaries.

        Tuple[float, float]: snapped point.

        ValueError: for unsupported axis target.
        NotImplementedError: for unsupported constraint type.
