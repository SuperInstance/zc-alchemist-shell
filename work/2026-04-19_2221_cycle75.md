```python
import re
from datetime import datetime
from typing import Dict, Optional

def parse_log_line(log_line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured log line with format: 
    [TIMESTAMP] LEVEL: MESSAGE
    
    Args:
        log_line: A string log line (e.g., "[2023-10-05 14:30:22] INFO: Service started")
    
    Returns:
        Dictionary with keys 'timestamp', 'level', 'message' if successful, else None.
    """
    pattern = r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+): (.+)$'
    match = re.match(pattern, log_line)
    if not match:
        return None
    
    timestamp_str, level, message = match.groups()
    # Validate timestamp format (basic check)
    try:
        datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        return None
    
    return {
        'timestamp': timestamp_str,
        'level': level.upper(),
        'message': message.strip()
    }

# Example usage
if __name__ == "__main__":
    sample = "[2023-10-05 14:30:22] INFO: Service started"
    result = parse_log_line(sample)
    print(result)  # {'timestamp': '2023-10-05 14:30:22', 'level': 'INFO', 'message': 'Service started'}
```

## **3. Function 2: deadband_filter**

**DeepSeek-chat output:**
```python
from typing import List, Tuple

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply a deadband filter to a list of float values.
    
    The deadband filter ignores small changes within ±deadband of the last output value.
    Only values that deviate more than ±deadband from the last output are passed through.
    
    Args:
        values: List of input float values.
        deadband: Threshold for ignoring changes (default 0.1).
    
    Returns:
        Filtered list of values.
    
    Example:
        >>> deadband_filter([1.0, 1.05, 1.15, 1.14, 1.25], 0.1)
        [1.0, 1.15, 1.25]
    """
    if not values:
        return []
    
    filtered = [values[0]]
    last_output = values[0]
    
    for v in values[1:]:
        if abs(v - last_output) > deadband:
            filtered.append(v)
            last_output = v
    
    return filtered

# Example usage
if __name__ == "__main__":
    data = [1.0, 1.05, 1.15, 1.14, 1.25]
    result = deadband_filter(data, 0.1)
    print(result)  # [1.0, 1.15, 1.25]
```

```python
    Args:
        values: Input list of floats.
```
