def deadband_filter(values, deadband=0.1):
    if not values:
        return []
    # P0: map negative space (values within deadband of zero)
    # P1: safe channel is outside deadband
    safe = [v for v in values if abs(v) > deadband]
    # P2: return filtered
    return safe

def parse_fleet_log(log_line):
    import re
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \| (\w+) \| Cycle (\d+) \| (.+)'
    match = re.match(pattern, log_line)
    if match:
        timestamp, agent, cycle, message = match.groups()
        return {
            'timestamp': timestamp,
            'agent': agent,
            'cycle': int(cycle),
            'message': message
        }
    else:
        return {'timestamp': None, 'agent': None, 'cycle': None, 'message': None}

def constraint_snap(point, constraints, tolerance=0.01):
    if not constraints:
        return point
    px, py = point
    best_dist = float('inf')
    best_point = point
    for (cx, cy) in constraints:
        dist = ((px - cx) ** 2 + (py - cy) ** 2) ** 0.5
        if dist < tolerance and dist < best_dist:
            best_dist = dist
            best_point = (cx, cy)
    return best_point

def deadband_filter(values, deadband=0.1):
    if values is None:
        return []
    filtered = []
    for v in values:
        if abs(v) > deadband:
            filtered.append(v)
    return filtered

def parse_fleet_log(log_line):
    if not log_line:
        return {'timestamp': None, 'agent': None, 'cycle': None, 'message': None}
    parts = log_line.split(' | ')
    if len(parts) != 4:
        return {'timestamp': None, 'agent': None, 'cycle': None, 'message': None}
    timestamp, agent, cycle_str, message = parts
    try:
        cycle = int(cycle_str.split()[1])
    except:
        cycle = None
    return {'timestamp': timestamp, 'agent': agent, 'cycle': cycle, 'message': message}

def constraint_snap(point, constraints, tolerance=0.01):
    x, y = point
    min_dist = tolerance
    snapped = point
    for (cx, cy) in constraints:
        dist = ((x - cx) ** 2 + (y - cy) ** 2) ** 0.5
        if dist < min_dist:
            min_dist = dist
            snapped = (cx, cy)
    return snapped
