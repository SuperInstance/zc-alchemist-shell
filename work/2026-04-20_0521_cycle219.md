```python
from typing import List, Dict, Any

def validate_deadband_protocol(steps: List[str]) -> Dict[str, Any]:
    """
    Validates a list of procedural steps against the Deadband Protocol.
    Protocol: P0 (map negative) must come before P1 (find channels),
    P1 before P2 (optimize). Skipping phases is a violation.

    Args:
        steps: List of step descriptions (strings).

    Returns:
        Dict with keys:
        - valid (bool): True if protocol is followed.
        - violations (List[str]): Descriptions of any violations found.
        - phase (str): 'P0', 'P1', 'P2', or 'UNKNOWN' based on last valid step.
    """
    violations = []
    expected_phase = "P0"
    phase_map = {"P0": 0, "P1": 1, "P2": 2}
    current_phase = "UNKNOWN"

    for i, step in enumerate(steps):
        step_upper = step.upper()
        detected = None

        # Simple keyword detection - in practice, use more robust NLP
        if "P0" in step_upper or "NEGATIVE" in step_upper or "NOT" in step_upper:
            detected = "P0"
        elif "P1" in step_upper or "SAFE" in step_upper or "CHANNEL" in step_upper:
            detected = "P1"
        elif "P2" in step_upper or "OPTIMIZE" in step_upper or "BEST" in step_upper:
            detected = "P2"

        if detected:
            current_phase = detected
            if phase_map[detected] < phase_map.get(expected_phase, -1):
                violations.append(f"Step {i}: Phase regression ({detected} after {expected_phase})")
            elif phase_map[detected] > phase_map[expected_phase]:
                # Advance phase
                expected_phase = detected
            # If equal, same phase continues
        else:
            # No phase detected, assume it belongs to current expected phase
            pass

    # Final check: did we reach P2?
    if "P2" not in [s.upper() for s in steps] and current_phase != "P2":
        violations.append("Protocol incomplete: P2 (optimize) phase not reached.")

    return {
        "valid": len(violations) == 0,
        "violations": violations,
        "phase": current_phase
    }
```

```python
from typing import List, Dict, Any

def validate_deadband_protocol(steps: List[str]) -> Dict[str, Any]:
    """
    Validate steps against Deadband Protocol (P0→P1→P2).

    Args:
        steps: list of step descriptions

    Returns:
        dict with keys: valid (bool), violations (list), phase (str)
    """
    # Phase detection
    phase_keywords = {
        'P0': ['p0', 'negative', 'not', 'avoid', 'map negative'],
        'P1': ['p1', 'safe', 'channel', 'find channel', 'feasible'],
        'P2': ['p2', 'optimize', 'best', 'optimal', 'improve']
    }

    violations = []
    detected_phases = []
    last_phase_index = -1

    for idx, step in enumerate(steps):
        step_lower = step.lower()
        current_phase = None

        for phase, keywords in phase_keywords.items():
            if any(keyword in step_lower for keyword in keywords):
                current_phase = phase
                break

        if current_phase:
            phase_index = {'P0': 0, 'P1': 1, 'P2': 2}[current_phase]
            detected_phases.append(current_phase)

            # Check ordering
            if phase_index < last_phase_index:
                violations.append(f"Step {idx}: phase out of order ({current_phase} after {detected_phases[-2]})")
            last_phase_index = phase_index
        else:
            # No phase marker, assume continues current context
            pass

    # Check for missing phases
    if 'P0' not in detected_phases:
        violations.append("Missing P0 (map negative) phase")
    if 'P1' not in detected_phases:
        violations.append("Missing P1 (find safe channels) phase")
    if 'P2' not in detected_phases:
        violations.append("Missing P2 (optimize) phase")

    final_phase = detected_phases[-1] if detected_phases else 'UNKNOWN'

    return {
        'valid': len(violations) == 0,
        'violations': violations,
        'phase': final_phase
    }
```

**Comparison (Function 1):**
*   **Correctness:** Both implement core logic. DeepSeek's "phase regression" check is subtle. Groq's missing phase check is stricter.
*   **Clarity:** DeepSeek has more detailed docstring. Groq is concise.
*   **Robustness:** DeepSeek tracks `expected_phase` advancement. Groq tracks index of last detected phase. Both handle non-phase steps.
*   **Fleet Context:** Both reference protocol correctly. DeepSeek's final P2 check is advisory; Groq's is a violation.
*   **Edge Cases:** Groq would flag `["P1", "P2"]` as missing P0. DeepSeek would flag it as a regression (P1 after expected P0). DeepSeek's logic aligns closer to "never skip to P2" doctrine.
