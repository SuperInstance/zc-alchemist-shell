import re
from typing import List, Callable, Dict, Any, Tuple

def find_safe_channels(data: List[float], constraints: List[Callable[[float], bool]]) -> List[int]:
    """
    Implements P1 of the Deadband Protocol: find indices where all constraints are satisfied.
    
    Args:
        data: List of numeric values to evaluate.
        constraints: List of constraint functions, each taking a value and returning bool.
    
    Returns:
        List of indices where all constraints return True for the corresponding data value.
    """
    safe_indices = []
    for idx, value in enumerate(data):
        # Check all constraints for this value
        if all(constraint(value) for constraint in constraints):
            safe_indices.append(idx)
    return safe_indices


def parse_fleet_log(log_text: str) -> List[Dict[str, str]]:
    """
    Parse a fleet log string into structured event dictionaries.
    
    Log line format: "[YYYY-MM-DD HH:MM:SS] <AGENT_ID> <LEVEL> - <message>"
    
    Args:
        log_text: Multiline string containing log entries.
    
    Returns:
        List of dictionaries with keys: 'timestamp', 'agent_id', 'level', 'message'.
    """
    log_entries = []
    # Regex to match the log format
    log_pattern = re.compile(r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\S+) (\S+) - (.+)$')
    
    for line_num, line in enumerate(log_text.strip().split('\n'), 1):
        line = line.strip()
        if not line:
            continue
        
        match = log_pattern.match(line)
        if match:
            timestamp, agent_id, level, message = match.groups()
            log_entries.append({
                'timestamp': timestamp,
                'agent_id': agent_id,
                'level': level,
                'message': message
            })
        else:
            # Graceful handling: skip malformed line with warning
            print(f"Warning: Line {line_num} malformed, skipping: {line[:50]}...")
    
    return log_entries


def validate_tile(tile_data: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile against the expected schema.
    
    Required fields:
    - id: str
    - agent: str
    - cycle: int
    - content: str
    - type: must be one of ['code', 'doc', 'data']
    
    Args:
        tile_data: Dictionary containing tile data.
    
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    """
    errors = []
    required_fields = ['id', 'agent', 'cycle', 'content', 'type']
    
    # Check required fields exist
    for field in required_fields:
        if field not in tile_data:
            errors.append(f"Missing required field: '{field}'")
    
    # Type validation if fields exist
    if 'id' in tile_data and not isinstance(tile_data['id'], str):
        errors.append("Field 'id' must be a string")
    if 'agent' in tile_data and not isinstance(tile_data['agent'], str):
        errors.append("Field 'agent' must be a string")
    if 'cycle' in tile_data and not isinstance(tile_data['cycle'], int):
        errors.append("Field 'cycle' must be an integer")
    if 'content' in tile_data and not isinstance(tile_data['content'], str):
        errors.append("Field 'content' must be a string")
    
    # Validate 'type' enum
    valid_types = {'code', 'doc', 'data'}
    if 'type' in tile_data:
        if not isinstance(tile_data['type'], str):
            errors.append("Field 'type' must be a string")
        elif tile_data['type'] not in valid_types:
            errors.append(f"Field 'type' must be one of {list(valid_types)}, got '{tile_data['type']}'")
    
    return (len(errors) == 0, errors)
