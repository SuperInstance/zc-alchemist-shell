```python
from typing import List, Dict, Any, Tuple, Optional
import base64
import math

def deadband_filter(readings: List[float], safe_min: float, safe_max: float, target: float) -> Dict[str, Any]:
    """
    Apply the Deadband Protocol to sensor readings.
    
    P0: Filter out unsafe readings (outside safe_min/safe_max).
    P1: Identify safe readings within [safe_min, safe_max].
    P2: Select optimal safe reading closest to target.
    
    Args:
        readings: List of sensor readings.
        safe_min: Minimum safe value.
        safe_max: Maximum safe value.
        target: Target optimal value.
    
    Returns:
        Dict with:
            p0_filtered: Unsafe readings.
            p1_safe: Safe readings.
            p2_optimal: Optimal safe reading (or None).
            applied_order: Order of protocol application.
    """
    # P0: Map negative space
    unsafe = [r for r in readings if r < safe_min or r > safe_max]
    safe = [r for r in readings if safe_min <= r <= safe_max]
    
    # P1: Find safe channels (already done)
    # P2: Optimize within channels
    optimal = None
    if safe:
        optimal = min(safe, key=lambda x: abs(x - target))
    
    return {
        "p0_filtered": unsafe,
        "p1_safe": safe,
        "p2_optimal": optimal,
        "applied_order": ["P0", "P1", "P2"]
    }

def constraint_snap(point: Tuple[float, float], existing_points: List[Tuple[float, float]], tolerance: float = 0.1) -> Tuple[float, float]:
    """
    Snap a point to the nearest existing point within tolerance.
    
    Args:
        point: (x, y) to snap.
        existing_points: List of (x, y) reference points.
        tolerance: Max distance for snapping.
    
    Returns:
        Snapped point (or original if none within tolerance).
    """
    if not existing_points:
        return point
    
    x, y = point
    best_point = None
    best_dist = float('inf')
    
    for ex, ey in existing_points:
        dist = math.hypot(x - ex, y - ey)
        if dist <= tolerance and dist < best_dist:
            best_dist = dist
            best_point = (ex, ey)
    
    return best_point if best_point is not None else point

def tile_encoder(tile_text: str, code_map: Optional[Dict[str, str]] = None) -> str:
    """
    Encode a text tile into a compact base64 representation.
    
    Args:
        tile_text: Original tile text.
        code_map: Mapping of fleet terms to short codes.
    
    Returns:
        Base64 encoded compressed string.
    """
    if code_map is None:
        code_map = {
            "Cocapn": "CC",
            "Deadband": "DB",
            "Protocol": "P",
            "PLATO": "PL",
            "room": "RM",
            "tile": "TL"
        }
    
    # Step 1: Remove extra whitespace
    import re
    compressed = re.sub(r'\s+', ' ', tile_text.strip())
    
    # Step 2: Replace terms (longest first to avoid partial matches)
    for term, code in sorted(code_map.items(), key=lambda x: -len(x[0])):
        compressed = compressed.replace(term, code)
    
    # Step 3: Base64 encode
    encoded = base64.b64encode(compressed.encode('utf-8')).decode('utf-8')
    
    return encoded
```
