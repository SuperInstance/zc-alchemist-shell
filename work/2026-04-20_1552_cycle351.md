from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.
    Protocol: Strict P0 -> P1 -> P2 priority. No skipping. Only valid steps are 'P0','P1','P2'.
    
    Args:
        steps: List of step identifiers as strings.
    
    Returns:
        Tuple of (is_valid: bool, error_message: str). Error message is empty if valid.
    """
    valid_steps = {"P0", "P1", "P2"}
    seen = {"P0": False, "P1": False, "P2": False}
    
    for i, step in enumerate(steps):
        # Check step is valid
        if step not in valid_steps:
            return False, f"Invalid step '{step}' at position {i}. Must be 'P0','P1', or 'P2'."
        
        # Mark step as seen
        seen[step] = True
        
        # Check for skipping: P2 cannot appear before P1, P1 cannot appear before P0
        if step == "P2" and not seen["P1"]:
            return False, f"Protocol violation at position {i}: 'P2' appears before 'P1'."
        if step == "P1" and not seen["P0"]:
            return False, f"Protocol violation at position {i}: 'P1' appears before 'P0'."
        
        # Check for regression (e.g., P1 after P2) - optional strict ordering
        if step == "P0" and seen["P1"]:
            return False, f"Protocol violation at position {i}: 'P0' appears after 'P1' (no regression)."
        if step == "P1" and seen["P2"]:
            return False, f"Protocol violation at position {i}: 'P1' appears after 'P2' (no regression)."
    
    return True, ""
```

import re
from typing import Dict, Any

def parse_fleet_tile(raw_text: str) -> Dict[str, Any]:
    """
    Parses a raw fleet training tile text into structured fields.
    
    Expected format (loose):
        # Cycle <number>
        # Agent <name>
        # Phase <number>
        <content>
    
    Args:
        raw_text: Raw tile text.
    
    Returns:
        Dictionary with keys: 'cycle', 'agent', 'phase', 'content', 'is_valid_format'.
        Missing fields are set to None. is_valid_format is True if at least Cycle is found.
    """
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "content": "",
        "is_valid_format": False
    }
    
    lines = raw_text.strip().split('\n')
    content_lines = []
    
    for line in lines:
        line_stripped = line.strip()
        # Try to match header lines
        if line_stripped.startswith('# Cycle'):
            match = re.search(r'# Cycle\s+(\d+)', line_stripped)
            if match:
                result["cycle"] = int(match.group(1))
                result["is_valid_format"] = True
            continue
        elif line_stripped.startswith('# Agent'):
            match = re.search(r'# Agent\s+(\w+)', line_stripped)
            if match:
                result["agent"] = match.group(1)
            continue
        elif line_stripped.startswith('# Phase'):
            match = re.search(r'# Phase\s+(\d+)', line_stripped)
            if match:
                result["phase"] = int(match.group(1))
            continue
        else:
            # Not a header, treat as content
            content_lines.append(line.rstrip())
    
    result["content"] = '\n'.join(content_lines).strip()
    return result
```

#### **Function 3: `simulate_channel_optimization`**
```python
from typing import List, Tuple

def simulate_channel_optimization(resources: List[float],
                                  constraints: List[Tuple[float, float]]) -> List[float]:
    """
    Optimizes resource allocation across channels (P2 optimization).
    Allocates available resources sequentially to maximize usage within min/max constraints.
    
    Args:
        resources: List of available resource units (e.g., [10.0, 5.0]).
        constraints: List of (min, max) for each channel. Length must match resources.
    
    Returns:
        List of allocated amounts per channel.
    """
    if len(resources) != len(constraints):
        raise ValueError("Resources list and constraints list must have same length.")
    
    allocation = [0.0] * len(resources)
    remaining_resources = sum(resources)
    
    # First pass: satisfy minimum constraints
    for i, (min_req, max_allowed) in enumerate(constraints):
        if min_req > max_allowed:
            raise ValueError(f"Channel {i}: min ({min_req}) > max ({max_allowed})")
        if remaining_resources >= min_req:
            allocation[i] = min_req
            remaining_resources -= min_req
        else:
            # Cannot meet minimums
            allocation[i] = remaining_resources
            remaining_resources = 0
            break
    
    # Second pass: distribute remaining resources up to max
    for i, (min_req, max_allowed) in enumerate(constraints):
        if remaining_resources <= 0:
            break
        room = max_allowed - allocation[i]
        if room > 0:
            give = min(room, remaining_resources)
            allocation[i] += give
            remaining_resources -= give
    
    return allocation
```
