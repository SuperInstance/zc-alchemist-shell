#### **Function 1: `validate_deadband_protocol`**
```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validate that a list of agent steps follows Deadband Protocol priority: P0 → P1 → P2.
    
    Rules:
    - P0 steps must come before P1, P1 before P2.
    - Steps are identified by prefix: 'P0:', 'P1:', 'P2:'.
    - Non‑prefix steps are ignored (treated as comments).
    
    Args:
        steps: List of step descriptions.
    
    Returns:
        Tuple of (is_valid, violations) where violations is a list of error messages.
    
    Example:
        >>> validate_deadband_protocol(['P0: map rocks', 'P1: find channel', 'P2: optimize'])
        (True, [])
        >>> validate_deadband_protocol(['P1: skip', 'P0: late'])
        (False, ['P1: skip found before P0'])
    """
    violations = []
    last_priority = -1  # -1: none, 0: P0, 1: P1, 2: P2
    priority_map = {'P0': 0, 'P1': 1, 'P2': 2}
    
    for i, step in enumerate(steps):
        # Extract prefix
        if ':' in step:
            prefix = step.split(':', 1)[0].strip()
            if prefix in priority_map:
                current = priority_map[prefix]
                if current < last_priority:
                    violations.append(f"{step} found out of order (P{current} after P{last_priority})")
                last_priority = current
    
    is_valid = len(violations) == 0
    return is_valid, violations
```

#### **Function 2: `extract_training_tiles`**
```python
import re
from typing import Optional

def extract_training_tiles(log_text: str) -> list[dict]:
    """
    Extract potential training tiles from a fleet agent log.
    
    Log format assumptions:
    - Each cycle starts with 'Cycle NNN — AgentName'
    - Tile content appears after a separator (---) or in indented blocks.
    - Tile type is inferred from keywords: 'error', 'test', 'output', 'thought'.
    
    Args:
        log_text: Raw log text.
    
    Returns:
        List of tile dicts with keys: cycle, agent, content, type.
    
    Example:
        >>> log = 'Cycle 42 — Sentinel\\n---\\nError: service down\\n---'
        >>> extract_training_tiles(log)
        [{'cycle': 42, 'agent': 'Sentinel', 'content': 'Error: service down', 'type': 'error'}]
    """
    tiles = []
    
    # Regex to capture cycle and agent
    header_pattern = r'Cycle\s+(\d+)\s*—\s*(\w+)'
    headers = list(re.finditer(header_pattern, log_text, re.IGNORECASE))
    
    for i, match in enumerate(headers):
        cycle = int(match.group(1))
        agent = match.group(2)
        start_pos = match.end()
        end_pos = headers[i + 1].start() if i + 1 < len(headers) else len(log_text)
        section = log_text[start_pos:end_pos].strip()
        
        # Simple tile splitting by '---' lines
        raw_tiles = re.split(r'\n---+\n', section)
        for raw in raw_tiles:
            if not raw.strip():
                continue
            content = raw.strip()
            # Infer type
            tile_type = 'output'
            if re.search(r'\berror\b', content, re.I):
                tile_type = 'error'
            elif re.search(r'\btest\b', content, re.I):
                tile_type = 'test'
            elif re.search(r'\bthought\b', content, re.I):
                tile_type = 'thought'
            
            tiles.append({
                'cycle': cycle,
                'agent': agent,
                'content': content,
                'type': tile_type
            })
    
    return tiles
```
