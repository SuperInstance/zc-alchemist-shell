**Objective:** Generate three distinct functions using both DeepSeek-chat (via API) and Groq Llama-70b (via Groq API). Compare outputs on correctness, conciseness, and adherence to specifications.

**Functions to implement:**
1. **deadband_filter(values: list[float], deadband: float) -> list[float]**
   - Implements a deadband filter: if abs(value) ≤ deadband, return 0.0, else return value.
   - Edge cases: empty list, negative deadband (should raise ValueError).

2. **parse_fleet_message(raw: str) -> dict**
   - Parses a simple fleet protocol message format: `"TYPE|SENDER|PAYLOAD"`.
   - Returns dict with keys: `type`, `sender`, `payload`.
   - If format invalid, raises `ValueError`.

3. **weighted_random_select(items: list[tuple[str, float]]) -> str**
   - Selects a string based on provided weights (floats).
   - Weights can be any non-negative float; zero weight means never selected.
   - Returns the selected string.

```python
def deadband_filter(values: list[float], deadband: float) -> list[float]:
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    if not values:
        return []
    result = []
    for v in values:
        if abs(v) <= deadband:
            result.append(0.0)
        else:
            result.append(v)
    return result

def parse_fleet_message(raw: str) -> dict:
    parts = raw.split('|', 2)
    if len(parts) != 3:
        raise ValueError("Message must have exactly three parts separated by '|'")
    msg_type, sender, payload = parts
    if not msg_type or not sender:
        raise ValueError("Type and sender cannot be empty")
    return {"type": msg_type, "sender": sender, "payload": payload}

import random

def weighted_random_select(items: list[tuple[str, float]]) -> str:
    if not items:
        raise ValueError("items list cannot be empty")
    strings, weights = zip(*items)
    # Normalize weights
    total = sum(weights)
    if total <= 0:
        raise ValueError("total weight must be positive")
    r = random.random() * total
    cumulative = 0
    for s, w in zip(strings, weights):
        cumulative += w
        if r < cumulative:
            return s
    # Fallback (should not happen with floating point)
    return strings[-1]
```

```python
        raise ValueError("Invalid message format: expected TYPE|SENDER|PAYLOAD")

    if msg_type.strip() == '' or sender.strip() == '':

# Function 3: weighted_random_select
import random

def weighted_random_select(items):
    if not items:
        raise ValueError("Items list is empty")
    # Extract weights
    weights = [w for _, w in items]
    # Check for non-negative weights
    if any(w < 0 for w in weights):
        raise ValueError("Weights must be non-negative")
    # Normalize
    total = sum(weights)
    if total == 0:
        raise ValueError("Total weight is zero")
    r = random.uniform(0, total)
    cumulative = 0
    for item, weight in items:
        cumulative += weight
        if r <= cumulative:
            return item
    # Fallback
    return items[-1][0]
```
