import re
from datetime import datetime

def parse_fleet_log(log_text: str) -> dict:
    """
    Parse fleet log entry with regex.
    Returns structured dict or empty dict on failure.
    """
    pattern = r'\[(.*?)\] \[(.*?)\] \[(.*?)\] (.*?) \((.*?)\)'
    match = re.match(pattern, log_text.strip())
    
    if not match:
        return {}
    
    timestamp_str, agent, level, message, params_str = match.groups()
    
    # Parse timestamp
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    except ValueError:
        timestamp = timestamp_str
    
    # Parse key=value pairs
    params = {}
    if params_str:
        for pair in params_str.split(', '):
            if '=' in pair:
                key, value = pair.split('=', 1)
                params[key.strip()] = value.strip()
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "level": level,
        "message": message,
        "params": params
    }

def validate_tile_json(tile_data: dict, schema: dict = None) -> tuple[bool, list[str]]:
    """
    Validate training tile against schema.
    Simple validation without full JSON Schema implementation.
    """
    if schema is None:
        # Default fleet tile schema
        schema = {
            "required": ["id", "agent", "cycle", "content", "timestamp"],
            "optional": ["score", "tags", "parent_id"]
        }
    
    errors = []
    
    # Check required fields
    for field in schema.get("required", []):
        if field not in tile_data:
            errors.append(f"Missing required field: {field}")
        elif tile_data[field] is None or tile_data[field] == "":
            errors.append(f"Required field empty: {field}")
    
    # Check for unknown fields (optional)
    all_fields = set(schema.get("required", []) + schema.get("optional", []))
    for field in tile_data.keys():
        if field not in all_fields:
            errors.append(f"Unknown field: {field}")
    
    # Type checks (basic)
    if "cycle" in tile_data and not isinstance(tile_data["cycle"], int):
        errors.append("Field 'cycle' must be integer")
    
    return len(errors) == 0, errors

    P1: values between p1_safe_min and p1_safe_max (inclusive) are SAFE.

            results.append("DANGER")

            results.append("SAFE")
        else:
            results.append("OPTIMIZE")
    return results

def parse_fleet_log(log_text):
    """
    Parses a fleet log string and returns a dictionary with extracted fields.
    Expected format: [timestamp] [agent] [level] message (key1=value1, key2=value2)

    return {
