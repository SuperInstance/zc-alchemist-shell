## **3. Generated Code**

### **DeepSeek-chat (via Ollama) Outputs**

**Function 1: deadband_filter**
```python
def deadband_filter(values: list[float], threshold: float) -> list[float]:
    """
    Apply Deadband Protocol filtering.
    P0: Remove values within deadband (abs <= threshold).
    Returns values in safe channels (abs > threshold).
    """
    if not values:
        return []
    
    safe_channels = [v for v in values if abs(v) > threshold]
    return safe_channels
```

**Function 2: parse_fleet_log**

def parse_fleet_log(log_line: str) -> dict[str, str] | None:
    """
    Parse a structured fleet log entry.
    Format: "TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE"
    Returns dict with keys: timestamp, agent, cycle, status, message.
    Returns None if malformed.
    """
    if not log_line or '|' not in log_line:
        return None
    
    parts = [part.strip() for part in log_line.split('|')]
    if len(parts) != 5:
        return None
    
    return {
        'timestamp': parts[0],
        'agent': parts[1],
        'cycle': parts[2],
        'status': parts[3],
        'message': parts[4]
    }
```

**Function 3: validate_tile**

def validate_tile(tile: dict) -> tuple[bool, str]:
    """
    Validate a training tile against fleet schema.
    Required keys: agent, cycle, task, output.
    Optional: notes.
    All values must be non-empty strings.
    """
    required_keys = ['agent', 'cycle', 'task', 'output']
    
    for key in required_keys:
        if key not in tile:
            return False, f"Missing required key: {key}"
        
        value = tile[key]
        if not isinstance(value, str) or not value.strip():
            return False, f"Invalid value for {key}: must be non-empty string"
    
    # Check optional notes if present
    if 'notes' in tile and tile['notes']:
        if not isinstance(tile['notes'], str):
            return False, "Notes must be a string if provided"
    
    return True, "Tile is valid"
```

### **Groq Llama-3.3-70b (via GroqCloud) Outputs**

**Function 1: deadband_filter**
```python
def deadband_filter(values: list[float], threshold: float) -> list[float]:
    """
    Apply Deadband Protocol: filter out values within deadband (|value| <= threshold).
    Implements P0 (avoid negatives) by removing values in the deadband zone.
    Returns values in safe channels (|value| > threshold).
    """
    if threshold < 0:
        raise ValueError("Threshold must be non-negative")
    
    result = []
    for v in values:
        if abs(v) > threshold:
            result.append(v)
    return result
```

**Function 2: parse_fleet_log**
```python
def parse_fleet_log(log_line: str) -> dict[str, str] | None:
    """
    Parse fleet log line with format: TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE
    Returns dictionary with parsed components or None if format invalid.
    """
    if not isinstance(log_line, str):
        return None
    
    segments = log_line.split('|')
    if len(segments) != 5:
        return None
    
    # Clean each segment
    cleaned = [s.strip() for s in segments]
    
    # Ensure no empty required fields
    if any(not f for f in cleaned[:4]):  # message can be empty
        return None
    
    return {
        'timestamp': cleaned[0],
        'agent': cleaned[1],
        'cycle': cleaned[2],
        'status': cleaned[3],
        'message': cleaned[4]
    }
```

**Function 3: validate_tile**
```python
def validate_tile(tile: dict) -> tuple[bool, str]:
    """
    Validate a training tile dictionary.
    Required keys: agent, cycle, task, output (all non-empty strings).
    Optional key: notes (string, can be empty).
    Returns (is_valid, error_message).
    """
    if not isinstance(tile, dict):
        return False, "Tile must be a dictionary"
    
    required = ['agent', 'cycle', 'task', 'output']
    for field in required:
        if field not in tile:
            return False, f"Missing required field: {field}"
        
        val = tile[field]
        if not isinstance(val, str):
            return False, f"Field '{field}' must be a string"
        if not val.strip():
            return False, f"Field '{field}' cannot be empty"
    
    # Validate notes if present
    if 'notes' in tile and tile['notes'] is not None:
        if not isinstance(tile['notes'], str):
            return False, "Notes must be a string or None"
    
    return True, "Validation passed"
```
