```python
def deadband_filter(values: list[float], lower: float, upper: float) -> list[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Values below lower or above upper are set to 0 (negative space).
    P1: Values within [lower, upper] are passed through (safe channel).
    P2: No optimization applied in this basic filter.
    
    Args:
        values: List of float values to filter.
        lower: Lower deadband threshold.
        upper: Upper deadband threshold.
    
    Returns:
        Filtered list where values outside deadband are zeroed.
    
    Raises:
        ValueError: If lower >= upper.
    """
    if lower >= upper:
        raise ValueError("lower must be less than upper")
    
    filtered = []
    for v in values:
        if v < lower or v > upper:
            # P0: Map negative space – zero out unsafe values
            filtered.append(0.0)
        else:
            # P1: Safe channel – pass through
            filtered.append(v)
    return filtered
```

```python
def parse_plato_tile(tile_text: str) -> dict[str, str | list[str]]:
    """
    Parse a PLATO training‑tile text block into structured fields.
    
    Expected format:
        --- header ---
        key: value
        key: value
        --- body ---
        free‑text content...
    
    Args:
        tile_text: Raw tile text with header/body sections.
    
    Returns:
        Dictionary with keys: 'header' (dict), 'body' (str), 'tags' (list).
    
    Raises:
        ValueError: If tile format is invalid (missing headers).
    """
    lines = tile_text.strip().splitlines()
    if "--- header ---" not in lines or "--- body ---" not in lines:
        raise ValueError("Tile missing required section markers")
    
    # Split into header and body sections
    header_start = lines.index("--- header ---")
    body_start = lines.index("--- body ---")
    
    header_lines = lines[header_start + 1:body_start]
    body_lines = lines[body_start + 1:]
    
    # Parse header key‑value pairs
    header = {}
    tags = []
    for line in header_lines:
        line = line.strip()
        if not line:
            continue
        if ":" in line:
            key, value = line.split(":", 1)
            key = key.strip()
            value = value.strip()
            if key == "tags":
                # Tags are comma‑separated
                tags = [t.strip() for t in value.split(",") if t.strip()]
            else:
                header[key] = value
    
    return {
        "header": header,
        "body": "\n".join(body_lines).strip(),
        "tags": tags
    }
```

```python
def constraint_snap(point: tuple[float, float], 
                    line: tuple[tuple[float, float], tuple[float, float]], 
                    tolerance: float = 1e-6) -> tuple[float, float]:
    """
    Snap a point to the nearest position on a line segment within tolerance.
    
    Based on constraint‑theory‑core geometric snapping.
    If the perpendicular projection falls outside segment, snap to nearest endpoint.
    
    Args:
        point: (x, y) coordinates of point to snap.
        line: ((x1, y1), (x2, y2)) defining the line segment.
        tolerance: Maximum distance for snapping (default 1e-6).
    
    Returns:
        (x, y) of snapped point.
    """
    (x, y) = point
    (x1, y1), (x2, y2) = line
    
    # Vector math for projection
    dx = x2 - x1
    dy = y2 - y1
    seg_len_sq = dx*dx + dy*dy
    
    if seg_len_sq < tolerance:
        # Line segment is degenerate – treat as single point
        return (x1, y1)
    
    # Projection parameter t (0 at start, 1 at end)
    t = ((x - x1) * dx + (y - y1) * dy) / seg_len_sq
    
    if t < 0:
        # Before start – snap to start point
        snapped = (x1, y1)
    elif t > 1:
        # After end – snap to end point
        snapped = (x2, y2)
    else:
        # Within segment – compute projected point
        snapped = (x1 + t * dx, y1 + t * dy)
    
    # Check if within tolerance
    dist_sq = (snapped[0] - x)**2 + (snapped[1] - y)**2
    if dist_sq <= tolerance**2:
        return snapped
    else:
        # Outside tolerance – return original point (no snap)
        return (x, y)
```

```python
def deadband_filter(values: list[float], lower: float, upper: float) -> list[float]:
    """
    Implements the Deadband Protocol: zero values outside [lower, upper].
    
    Args:
        values: Input list of floats.
        lower: Lower bound of deadband.
        upper: Upper bound of deadband.
    
    Returns:
        List where values outside deadband are replaced with 0.0.
    """
    if lower >= upper:
        raise ValueError("lower must be less than upper")
    
    result = []
    for v in values:
        if lower <= v <= upper:
            result.append(v)   # inside deadband → keep
        else:
            result.append(0.0) # outside → zero
    return result
```

```python
            val = val.strip()

    return {
        "header": header,
        "body": body_section,
        "tags": tags
    }
```
