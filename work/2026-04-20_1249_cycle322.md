def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of Deadband Protocol steps.

    The Deadband Protocol requires strict priority:
    1. P0 must come before P1 and P2.
    2. P1 must come before P2.
    3. No step can be skipped.

    Args:
        steps: List of step strings, e.g., ["P0", "P1", "P2"].

    Returns:
        Tuple of (is_valid, error_message). Error message is empty if valid.
    """
    expected_order = ["P0", "P1", "P2"]
    seen = set()

    for i, step in enumerate(steps):
        if step not in expected_order:
            return False, f"Invalid step '{step}' at position {i}. Must be one of {expected_order}"

        # Check ordering
        for higher_priority in expected_order:
            if step == higher_priority:
                break
            if higher_priority not in seen:
                return False, f"Step '{higher_priority}' missing before '{step}' at position {i}"

        seen.add(step)

    # Ensure all required steps are present
    for required in expected_order:
        if required not in seen:
            return False, f"Missing required step '{required}'"

    return True, ""


def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a raw fleet training tile text into a structured dictionary.

    Expected format (sections marked with '#' headers):
        # Cycle X
        # Phase Y
        # Agent Name
        # Task Description
        # Content
        ... free text ...

    Args:
        tile_text: The raw text of the tile.

    Returns:
        Dictionary with keys: 'cycle', 'phase', 'agent', 'task', 'content'.
        Missing sections will have value None.
    """
    result = {
        'cycle': None,
        'phase': None,
        'agent': None,
        'task': None,
        'content': None
    }

    lines = tile_text.strip().split('\n')
    current_section = None
    content_lines = []

    for line in lines:
        line_stripped = line.strip()
        if line_stripped.startswith('# Cycle'):
            result['cycle'] = line_stripped.replace('# Cycle', '').strip()
        elif line_stripped.startswith('# Phase'):
            result['phase'] = line_stripped.replace('# Phase', '').strip()
        elif line_stripped.startswith('# Agent'):
            result['agent'] = line_stripped.replace('# Agent', '').strip()
        elif line_stripped.startswith('# Task'):
            result['task'] = line_stripped.replace('# Task', '').strip()
        elif line_stripped.startswith('# Content'):
            current_section = 'content'
        elif current_section == 'content':
            content_lines.append(line)

    if content_lines:
        result['content'] = '\n'.join(content_lines).strip()

    return result


def simulate_channel_optimization(start: float, end: float,
                                  obstacles: list[tuple[float, float]]) -> list[float]:
    """
    Simulates P2 optimization within a 1D safe channel.

    Finds a path from start to end avoiding circular obstacles.
    Uses a simple greedy algorithm: move forward, deviate only when
    an obstacle blocks the direct path.

    Args:
        start: Starting position (float).
        end: Ending position (float). Must be > start.
        obstacles: List of (center, radius) tuples representing obstacles to avoid.

    Returns:
        List of waypoint positions from start to end.
    """
    if start >= end:
        return [start, end]  # No meaningful optimization in reversed channel

    waypoints = [start]
    current_pos = start

    # Sort obstacles by their center for processing
    sorted_obstacles = sorted(obstacles, key=lambda x: x[0])

    for center, radius in sorted_obstacles:
        # If obstacle is behind us or far ahead, ignore
        if center + radius < current_pos or center - radius > end:
            continue

        # If current path intersects obstacle
        if current_pos < center + radius:
            # Deviate to the right of the obstacle
            deviation_point = center + radius + 0.1  # Small buffer
            if deviation_point < end:
                waypoints.append(deviation_point)
                current_pos = deviation_point

    waypoints.append(end)
    return waypoints
