def deadband_filter(values, deadband=0.1, mode='center'):
    """
    Apply Deadband Protocol filtering to a list of float values.
    
    P0: Remove values that would cause instability (outside safe range).
    P1: Find values within deadband tolerance.
    P2: Optimize by snapping to nearest safe channel.
    
    Args:
        values: List of float input values.
        deadband: Tolerance band (default 0.1).
        mode: 'center' (deadband around zero) or 'relative' (around each value).
    
    Returns:
        Filtered list of floats with deadband applied.
    """
    if not values:
        return []
    
    filtered = []
    for v in values:
        # P0: Map negative space — reject NaN/inf
        if not isinstance(v, (int, float)) or not math.isfinite(v):
            continue
        
        # P1: Find safe channels
        if mode == 'center':
            if abs(v) <= deadband:
                # Within deadband, snap to zero
                filtered.append(0.0)
            else:
                # Outside deadband, keep as is (P2 optimize)
                filtered.append(v)
        elif mode == 'relative':
            # Placeholder for relative deadband logic
            # This is a simplified version
            snapped = round(v / deadband) * deadband
            filtered.append(snapped)
        else:
            raise ValueError(f"Unknown mode: {mode}")
    
    return filtered
```

def deadband_filter(values, deadband=0.1):
    """
    Implements deadband filtering on a list of numerical values.
    Values within +/- deadband of zero are set to zero.
    Other values are passed through.
    
    Parameters:
    values (list): Input numbers.
    deadband (float): Deadband threshold.
    
    Returns:
    list: Filtered values.
    """
    result = []
    for v in values:
        try:
            num = float(v)
        except (ValueError, TypeError):
            continue  # skip non-numeric
        
        if -deadband <= num <= deadband:
            result.append(0.0)
        else:
            result.append(num)
    return result
```

### **Function 2: `parse_fleet_log`**

**DeepSeek-chat version:**

import re
from datetime import datetime
from typing import List, Dict, Any

def parse_fleet_log(log_text: str) -> List[Dict[str, Any]]:
    """
    Parse raw fleet log text into structured event dictionaries.
    
    Expected log format:
    [2026-04-20 00:15:32] AGENT Oracle1 STATUS online CYCLES 142
    [2026-04-20 00:15:30] AGENT Zeroclaw ERROR tile_rejected REASON format
    [2026-04-20 00:15:28] SYSTEM plato-torch ROOM deadband ACTIVE agents 3
    
    Returns:
        List of event dicts with keys: timestamp, agent, event_type, details.
    """
    events = []
    # Regex matches timestamp, agent/component, event type, and message
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+) (\w+) (.+)'
    
    for line in log_text.strip().split('\n'):
        match = re.match(pattern, line)
        if not match:
            # Try fallback pattern for system messages
            fallback = re.match(r'\[(.+?)\] (.+)', line)
            if fallback:
                events.append({
                    'timestamp': fallback.group(1),
                    'agent': 'UNKNOWN',
                    'event_type': 'LOG',
                    'details': fallback.group(2)
                })
            continue
        
        timestamp_str, agent, event_type, details = match.groups()
        try:
            timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
        except ValueError:
            timestamp = timestamp_str
        
        # Extract key-value pairs from details if present
        kv_pairs = {}
        for part in details.split():
            if '=' in part:
                k, v = part.split('=', 1)
                kv_pairs[k] = v
        
        events.append({
            'timestamp': timestamp,
            'agent': agent,
            'event_type': event_type,
            'details': details,
            'metadata': kv_pairs if kv_pairs else None
        })
    
    return events
```

import re

def parse_fleet_log(log_text):
    """
    Parses fleet log lines into structured data.
    
    Input format example:
    [2026-04-20 00:15:32] AGENT Oracle1 STATUS online
    
    Returns list of dictionaries.
    """
    lines = log_text.split('\n')
    parsed = []
    
    for line in lines:
        line = line.strip()
        if not line:
            continue
            
        # Simple regex for timestamp and rest
        match = re.search(r'\[(.*?)\]\s*(.*)', line)
        if match:
            timestamp = match.group(1)
            message = match.group(2)
            
            # Try to extract agent and status
            agent = None
            status = None
            if 'AGENT' in message:
                parts = message.split()
                for i, part in enumerate(parts):
                    if part == 'AGENT' and i+1 < len(parts):
                        agent = parts[i+1]
                    if part == 'STATUS' and i+1 < len(parts):
                        status = parts[i+1]
            
            parsed.append({
                'time': timestamp,
                'full_message': message,
                'agent': agent,
                'status': status
            })
    
    return parsed
```
