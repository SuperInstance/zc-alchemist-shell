```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of steps against the Deadband Protocol.
    The protocol requires the exact sequence ['P0', 'P1', 'P2'] with no
    intervening steps. Skipping or reordering is invalid.

    Args:
        steps: List of step identifiers (e.g., ['P0', 'P1', 'P2']).

    Returns:
        (True, '') if valid, (False, error_message) if invalid.
    """
    required = ['P0', 'P1', 'P2']
    if len(steps) != len(required):
        return (False, f"Length mismatch. Expected {len(required)} steps, got {len(steps)}.")
    for i, (expected, actual) in enumerate(zip(required, steps)):
        if expected != actual:
            return (False, f"Step {i}: expected '{expected}', got '{actual}'.")
    return (True, "")

# Example
print(validate_deadband_protocol(["P0", "P1", "P2"]))  # (True, "")
print(validate_deadband_protocol(["P0", "P2"]))        # (False, "Length mismatch...")
```

```python
import re

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parses a raw fleet training tile text into a structured dictionary.
    Expects markdown headers like '# Cycle N' and '**Agent:** Name'.

    Args:
        raw_text: The full text of the tile.

    Returns:
        Dictionary with keys: cycle, agent, phase, content, is_valid_format.
    """
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "content": raw_text.strip(),
        "is_valid_format": False
    }

    # Extract cycle
    cycle_match = re.search(r'^# Cycle\s+(\d+)', raw_text, re.MULTILINE)
    if cycle_match:
        result["cycle"] = int(cycle_match.group(1))

    # Extract agent
    agent_match = re.search(r'\*\*Agent:\*\*\s*(\w+)', raw_text)
    if agent_match:
        result["agent"] = agent_match.group(1)

    # Extract phase (if present)
    phase_match = re.search(r'\*\*Phase:\*\*\s*(\d+)', raw_text)
    if phase_match:
        result["phase"] = int(phase_match.group(1))

    # Valid if we found at least cycle and agent
    result["is_valid_format"] = (result["cycle"] is not None and result["agent"] is not None)
    return result
```

#### **Function 3: `snap_to_constraint`**

```python
import math

def snap_to_constraint(point: tuple[float, float],
                       line_points: list[tuple],
                       tolerance: float = 0.1) -> tuple[float, float]:
    """
    Snaps a 2D point to the nearest point on a polyline defined by line_points,
    if within tolerance. Otherwise returns the original point.

    Args:
        point: (x, y) to snap.
        line_points: List of (x, y) defining polyline segments.
        tolerance: Maximum distance for snapping.

    Returns:
        Snapped (x, y) or original point.
    """
    if len(line_points) < 2:
        return point

    px, py = point
    min_dist = float('inf')
    snap_x, snap_y = px, py

    for i in range(len(line_points) - 1):
        x1, y1 = line_points[i]
        x2, y2 = line_points[i + 1]

        # Vector math for projection onto segment
        dx, dy = x2 - x1, y2 - y1
        seg_len_sq = dx*dx + dy*dy
        if seg_len_sq == 0:
            # Points are coincident
            dist = math.hypot(px - x1, py - y1)
            if dist < min_dist:
                min_dist = dist
                snap_x, snap_y = x1, y1
            continue

        # Projection parameter t
        t = ((px - x1) * dx + (py - y1) * dy) / seg_len_sq
        t = max(0, min(1, t))  # Clamp to segment
        proj_x = x1 + t * dx
        proj_y = y1 + t * dy

        dist = math.hypot(px - proj_x, py - proj_y)
        if dist < min_dist:
            min_dist = dist
            snap_x, snap_y = proj_x, proj_y

    if min_dist <= tolerance:
        return (snap_x, snap_y)
    return point

# Example
point = (1.0, 1.0)
line = [(0, 0), (2, 0), (2, 2)]
print(snap_to_constraint(point, line, 0.5))  # (1.0, 0.0)
```

def validate_deadband_protocol(steps):
    """
    Check if the steps list follows Deadband Protocol: must be exactly
    ['P0', 'P1', 'P2'] in that order.
    """
    if not isinstance(steps, list):
        return False, "Input must be a list"
    if steps == ['P0', 'P1', 'P2']:
        return True, ""
    else:
        # Provide some diagnostic
        if len(steps) != 3:
            return False, f"Expected 3 steps, got {len(steps)}"
        # Check order
        expected = ['P0', 'P1', 'P2']
        for i in range(3):
            if steps[i] != expected[i]:
                return False, f"Step {i+1} should be {expected[i]}, got {steps[i]}"
        return False, "Unknown error"
