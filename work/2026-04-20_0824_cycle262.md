def deadband_filter(readings: list[float], deadband: float, mode: str = "center") -> list[float]:
    """
    Apply a deadband filter to a list of sensor readings, following the Cocapn fleet's P0-P1-P2 protocol analogy.
    The deadband creates a 'zone of no response' around zero (or the center).

    Args:
        readings: List of raw sensor values (e.g., [-1.2, 0.3, 2.5, -0.05, 0.0, 1.8]).
        deadband: Positive float. The width of the deadband zone.
        mode: Either "center" (deadband symmetric around 0) or "threshold" (deadband from 0 to +deadband).
              For 'center', any reading where |value| <= deadband/2 is mapped to 0.0.
              For 'threshold', any reading where 0 <= value <= deadband is mapped to 0.0.

    Returns:
        Filtered list where values inside the deadband are set to 0.0, and values outside are
        adjusted by subtracting the deadband boundary (for 'center') or the deadband (for 'threshold').

    Example (center, deadband=1.0):
        Input:  [-1.2, 0.3, 2.5, -0.05]
        Output: [-0.7, 0.0, 2.0, 0.0]
        Explanation: |0.3| and |-0.05| are <= 0.5, so become 0.0. -1.2 becomes -1.2 - (-0.5) = -0.7. 2.5 becomes 2.5 - 0.5 = 2.0.
    """
    if deadband <= 0:
        raise ValueError("deadband must be positive")
    if mode not in ("center", "threshold"):
        raise ValueError("mode must be 'center' or 'threshold'")

    result = []
    half = deadband / 2.0

    for val in readings:
        if mode == "center":
            if abs(val) <= half:
                result.append(0.0)
            else:
                # Subtract half-band in the direction of the sign
                adjustment = half if val > 0 else -half
                result.append(val - adjustment)
        else:  # threshold mode
            if 0 <= val <= deadband:
                result.append(0.0)
            elif val > deadband:
                result.append(val - deadband)
            else:  # val < 0
                # Negative values are outside deadband in threshold mode
                result.append(val)
    return result
```

### **Groq Llama-70b Implementation:**

def deadband_filter(readings: list[float], deadband: float, mode: str = "center") -> list[float]:
    """
    Apply a deadband filter to a list of sensor readings, following the Cocapn fleet's P0-P1-P2 protocol analogy.
    The deadband creates a 'zone of no response' around zero (or the center).

    Args:
        readings: List of raw sensor values (e.g., [-1.2, 0.3, 2.5, -0.05, 0.0, 1.8]).
        deadband: Positive float. The width of the deadband zone.
        mode: Either "center" (deadband symmetric around 0) or "threshold" (deadband from 0 to +deadband).
              For 'center', any reading where |value| <= deadband/2 is mapped to 0.0.
              For 'threshold', any reading where 0 <= value <= deadband is mapped to 0.0.

    Returns:
        Filtered list where values inside the deadband are set to 0.0, and values outside are
        adjusted by subtracting the deadband boundary (for 'center') or the deadband (for 'threshold').

    Example (center, deadband=1.0):
        Input:  [-1.2, 0.3, 2.5, -0.05]
        Output: [-0.7, 0.0, 2.0, 0.0]
        Explanation: |0.3| and |-0.05| are <= 0.5, so become 0.0. -1.2 becomes -1.2 - (-0.5) = -0.7. 2.5 becomes 2.5 - 0.5 = 2.0.
    """
    if deadband <= 0:
        raise ValueError("deadband must be positive")

    if mode == "center":
        half_deadband = deadband / 2.0
        return [
            0.0 if abs(x) <= half_deadband else (x - half_deadband if x > 0 else x + half_deadband)
            for x in readings
        ]
    elif mode == "threshold":
        return [
            0.0 if 0 <= x <= deadband else (x - deadband if x > deadband else x)
            for x in readings
        ]
    else:
        raise ValueError("mode must be 'center' or 'threshold'")
```
