```python
def deadband_filter(values, lower, upper):
    """
    Apply a deadband filter to input values.
    
    Parameters:
    values (float or list): Single value or list of values
    lower (float): Lower bound of deadband
    upper (float): Upper bound of deadband
    
    Returns:
    Filtered value(s) where:
    - value < lower → lower
    - value > upper → upper
    - lower ≤ value ≤ upper → value unchanged
    
    Raises:
    ValueError: If lower > upper
    """
    if lower > upper:
        raise ValueError(f"Lower bound ({lower}) cannot exceed upper bound ({upper})")
    
    # Handle single value
    if isinstance(values, (int, float)):
        if values < lower:
            return lower
        elif values > upper:
            return upper
        return values
    
    # Handle list
    result = []
    for val in values:
        if val < lower:
            result.append(lower)
        elif val > upper:
            result.append(upper)
        else:
            result.append(val)
    return result
```

### **Function 2: parse_fleet_log_line**
```python
import re
from datetime import datetime

def parse_fleet_log_line(line):
    """
    Parse a structured fleet log line.
    
    Expected format: [TIMESTAMP] [AGENT] [LEVEL] message
    
    Parameters:
    line (str): Log line to parse
    
    Returns:
    dict: Parsed components with keys: timestamp, agent, level, message
    
    Raises:
    ValueError: If line doesn't match expected format
    """
    pattern = r'^\[([^\]]+)\] \[([^\]]+)\] \[([^\]]+)\] (.+)$'
    match = re.match(pattern, line.strip())
    
    if not match:
        raise ValueError(f"Log line doesn't match expected format: {line}")
    
    timestamp_str, agent, level, message = match.groups()
    
    # Try to parse timestamp
    timestamp = None
    for fmt in ('%Y-%m-%d %H:%M:%S', '%Y-%m-%dT%H:%M:%S', '%Y%m%d_%H%M%S'):
        try:
            timestamp = datetime.strptime(timestamp_str, fmt)
            break
        except ValueError:
            continue
    
    return {
        'timestamp': timestamp or timestamp_str,
        'agent': agent,
        'level': level.upper(),
        'message': message.strip()
    }
```

### **Function 3: weighted_random_choice**
```python
import random
import bisect

def weighted_random_choice(items, weights):
    """
    Select a random item with probability proportional to weights.
    
    Parameters:
    items (list): List of items to choose from
    weights (list): List of non-negative weights corresponding to items
    
    Returns:
    Selected item
    
    Raises:
    ValueError: If lengths don't match or weights are invalid
    """
    if len(items) != len(weights):
        raise ValueError(f"Items ({len(items)}) and weights ({len(weights)}) must have same length")
    
    if not weights:
        raise ValueError("Weights list cannot be empty")
    
    if any(w < 0 for w in weights):
        raise ValueError("Weights must be non-negative")
    
    # Handle all zero weights
    if all(w == 0 for w in weights):
        return random.choice(items)
    
    # Build cumulative distribution
    cumulative = []
    total = 0
    for w in weights:
        total += w
        cumulative.append(total)
    
    # Select random point and find corresponding item
    point = random.uniform(0, total)
    idx = bisect.bisect_left(cumulative, point)
    return items[idx]
```

```python
        return lower
```

### **Function 3: weighted_random_choice**
```python
import random

def weighted_random_choice(items, weights):
    """
    Weighted random selection from items.
    
    Args:
        items (list): Items to choose from
        weights (list): Corresponding weights (non-negative)
    
    Returns:
        Randomly selected item according to weights
    
    Raises:
        ValueError: If invalid inputs
    """
    if len(items) != len(weights):
        raise ValueError("items and weights must have same length")
    
    if not items:
        raise ValueError("items cannot be empty")
    
    # Normalize weights
    total = sum(weights)
    if total <= 0:
        raise ValueError("weights must sum to positive value")
    
    normalized = [w / total for w in weights]
    
    # Roulette wheel selection
    r = random.random()
    cumulative = 0
    for item, weight in zip(items, normalized):
        cumulative += weight
        if r <= cumulative:
            return item
    
    # Fallback (should rarely happen due to floating point)
    return items[-1]
```
