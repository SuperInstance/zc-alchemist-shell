def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a sequence of steps follows the Deadband Protocol priority:
    P0 (map negative space) must come before P1 (find safe channels),
    and P1 must come before P2 (optimize). No skipping allowed.

    Args:
        steps: List of step identifiers as strings (e.g., ["P0", "P1", "P2"]).

    Returns:
        Tuple of (is_valid: bool, error_message: str).
        If valid, error_message is an empty string.
    """
    expected_order = ["P0", "P1", "P2"]
    seen = set()

    for step in steps:
        if step not in expected_order:
            return False, f"Invalid step '{step}'. Must be one of {expected_order}."

        # Check for correct order: must not see a step before its predecessors
        for higher_priority in expected_order:
            if step == higher_priority:
                break
            if higher_priority not in seen:
                return False, f"Step '{step}' encountered before '{higher_priority}'."

        seen.add(step)

    # Check for missing mandatory steps? (Protocol doesn't forbid partial execution)
    # For now, any sequence that follows order is valid.
    return True, ""

import re

def parse_training_tile(raw_text: str) -> dict:
    """
    Parses a raw training tile text into structured fields.
    Expected format (loose):
        Agent: <name>
        Cycle: <number>
        Content: <free text>
        Tags: <comma-separated list>

    Args:
        raw_text: Raw tile text.

    Returns:
        Dictionary with keys: "agent", "cycle", "content", "tags".
        Missing fields are set to empty/default. Tags are split into a list.
    """
    result = {
        "agent": "",
        "cycle": 0,
        "content": "",
        "tags": []
    }

    # Try to extract Agent
    agent_match = re.search(r"Agent:\s*(.+)", raw_text, re.IGNORECASE)
    if agent_match:
        result["agent"] = agent_match.group(1).strip()

    # Try to extract Cycle
    cycle_match = re.search(r"Cycle:\s*(\d+)", raw_text, re.IGNORECASE)
    if cycle_match:
        try:
            result["cycle"] = int(cycle_match.group(1))
        except ValueError:
            pass  # keep default 0

    # Try to extract Tags
    tags_match = re.search(r"Tags:\s*(.+)", raw_text, re.IGNORECASE)
    if tags_match:
        tags_str = tags_match.group(1).strip()
        result["tags"] = [tag.strip() for tag in tags_str.split(",") if tag.strip()]

    # Content is everything else, minus the parsed header lines
    # Simple approach: take text after the last header match
    lines = raw_text.split('\n')
    content_lines = []
    in_content = True
    for line in lines:
        if re.match(r"^(Agent|Cycle|Tags):", line, re.IGNORECASE):
            continue
        content_lines.append(line)
    result["content"] = '\n'.join(content_lines).strip()

    return result

def generate_safe_channel(boundaries: list[tuple[float, float]], current_pos: tuple[float, float]) -> list[tuple[float, float]]:
    """
    Generates a safe channel (P1) as a series of waypoints from current_pos to (10.0, 10.0),
    avoiding all given x-axis boundaries (negative space).

    Args:
        boundaries: List of (x_min, x_max) tuples representing unsafe bands.
        current_pos: Starting (x, y) position.

    Returns:
        List of waypoints [(x1, y1), (x2, y2), ...] ending at (10.0, 10.0).
    """
    target = (10.0, 10.0)
    waypoints = [current_pos]

    # Sort boundaries for processing
    sorted_bounds = sorted(boundaries, key=lambda b: b[0])

    current_x, current_y = current_pos
    target_x, target_y = target

    # Simple strategy: move vertically first to avoid x-axis bands, then horizontally.
    # Find a safe x corridor: an x value not inside any boundary.
    safe_x = current_x
    # If current_x is inside a boundary, nudge right
    for x_min, x_max in sorted_bounds:
        if x_min <= current_x <= x_max:
            safe_x = x_max + 1.0  # arbitrary safe margin
            break

    # If safe_x is inside another boundary, find next gap (simplified)
    for x_min, x_max in sorted_bounds:
        if x_min <= safe_x <= x_max:
            safe_x = x_max + 1.0

    # Waypoint 1: move vertically to dodge x-bands (y change)
    waypoints.append((current_x, target_y))
    # Waypoint 2: move horizontally in safe corridor
    waypoints.append((safe_x, target_y))
    # Waypoint 3: move to target (if x different)
    if abs(safe_x - target_x) > 1e-9:
        waypoints.append((target_x, target_y))

    # Remove duplicates if any
    unique_waypoints = []
    for wp in waypoints:
        if not unique_waypoints or wp != unique_waypoints[-1]:
            unique_waypoints.append(wp)
    return unique_waypoints
