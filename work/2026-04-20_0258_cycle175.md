from typing import List, Dict, Any, Tuple
import re

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol: values within ±deadband of zero are mapped to zero.
    
    Args:
        values: List of float values
        deadband: Absolute threshold around zero (default 0.1)
        
    Returns:
        List of filtered values
        
    Example:
        >>> deadband_filter([0.05, 0.15, -0.07, -0.2], 0.1)
        [0.0, 0.15, 0.0, -0.2]
    """
    return [0.0 if -deadband <= v <= deadband else v for v in values]


def parse_fleet_log(line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line.
    
    Format: "TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE"
    
    Args:
        line: Log line string
        
    Returns:
        Dictionary with keys: timestamp, agent, cycle, status, message
        
    Raises:
        ValueError: If line doesn't match expected format
        
    Example:
        >>> parse_fleet_log("2026-04-20 03:00 | Alchemist | 175 | ACTIVE | Testing models")
        {'timestamp': '2026-04-20 03:00', 'agent': 'Alchemist', 'cycle': 175, 
         'status': 'ACTIVE', 'message': 'Testing models'}
    """
    pattern = r"^(.+?) \| (.+?) \| (\d+) \| (.+?) \| (.+)$"
    match = re.match(pattern, line)
    if not match:
        raise ValueError(f"Log line doesn't match expected format: {line}")
    
    timestamp, agent, cycle_str, status, message = match.groups()
    return {
        "timestamp": timestamp.strip(),
        "agent": agent.strip(),
        "cycle": int(cycle_str),
        "status": status.strip(),
        "message": message.strip()
    }


def constraint_snap(point: Tuple[float, float], 
                    constraints: List[Tuple[float, float, float]], 
                    threshold: float = 1.0) -> Tuple[float, float]:
    """
    Snap 2D point to nearest constraint line if within threshold.
    
    Args:
        point: (x, y) tuple
        constraints: List of (A, B, C) for line Ax + By + C = 0
        threshold: Maximum distance for snapping (default 1.0)
        
    Returns:
        Snapped (x, y) point or original if no snap
        
    Example:
        >>> constraints = [(0, 1, -5), (1, 0, -3)]  # y=5, x=3
        >>> constraint_snap((3.2, 5.1), constraints, 0.5)
        (3.0, 5.0)
    """
    x, y = point
    min_distance = float('inf')
    best_constraint = None
    
    for A, B, C in constraints:
        # Perpendicular distance to line
        denominator = (A**2 + B**2) ** 0.5
        if denominator == 0:
            continue  # Invalid constraint, skip
        distance = abs(A*x + B*y + C) / denominator
        
        if distance < min_distance:
            min_distance = distance
            best_constraint = (A, B, C)
    
    if min_distance <= threshold and best_constraint is not None:
        A, B, C = best_constraint
        # Project point onto line
        t = -(A*x + B*y + C) / (A**2 + B**2)
        x_snap = x + A * t
        y_snap = y + B * t
        return (x_snap, y_snap)
    
    return point

    """
