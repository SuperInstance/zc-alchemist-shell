**Prompt:**

Write a Python function `parse_log_line` that takes a string like "2026-04-20 14:30:23 INFO Service started" and returns a dict with keys: timestamp (datetime), level (str), message (str). Handle malformed lines gracefully.

**Output:**
```python
import re
from datetime import datetime

def parse_log_line(line: str) -> dict:
    """
    Parse a structured log line into timestamp, level, and message.
    Expected format: "YYYY-MM-DD HH:MM:SS LEVEL Message here"
    Returns dict with keys: timestamp (datetime), level (str), message (str).
    On malformed input, returns {'error': 'malformed line'}.
    """
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (\w+) (.+)$'
    match = re.match(pattern, line.strip())
    if not match:
        return {'error': 'malformed line'}
    
    ts_str, level, message = match.groups()
    try:
        timestamp = datetime.strptime(ts_str, '%Y-%m-%d %H:%M:%S')
    except ValueError:
        return {'error': 'invalid timestamp'}
    
    return {
        'timestamp': timestamp,
        'level': level.upper(),
        'message': message.strip()
    }
```

def deadband_filter(current: float, previous: float, threshold: float) -> str:
    """
    Simple deadband filter based on fleet protocol.
    
    P0: Negative space (invalid reading)
    P1: Changed beyond threshold → new channel
    P2: Within threshold → optimize in existing channel
    
    Negative space defined as reading < 0 or > 100.
    """
    # P0: Map negative space
    if current < 0 or current > 100:
        return 'P0'
    
    # P1/P2: Check change magnitude
    delta = abs(current - previous)
    if delta > threshold:
        return 'P1'  # New channel
    else:
        return 'P2'  # Same channel, optimize
```

def validate_fleet_tile(tile: dict) -> tuple[bool, list]:
    """
    Validate a fleet training tile against required schema.
    
    Required keys:
    - agent: str
    - cycle: int >= 0
    - content: str (non-empty)
    - type: one of ['log', 'code', 'analysis']
    
    Returns (is_valid, list_of_error_messages).
    """
    errors = []
    required_keys = ['agent', 'cycle', 'content', 'type']
    
    # Check all keys present
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    if errors:
        return False, errors
    
    # Validate types and constraints
    if not isinstance(tile['agent'], str) or not tile['agent'].strip():
        errors.append("'agent' must be non-empty string")
    
    if not isinstance(tile['cycle'], int) or tile['cycle'] < 0:
        errors.append("'cycle' must be non-negative integer")
    
    if not isinstance(tile['content'], str) or not tile['content'].strip():
        errors.append("'content' must be non-empty string")
    
    valid_types = ['log', 'code', 'analysis']
    if tile['type'] not in valid_types:
        errors.append(f"'type' must be one of {valid_types}")
    
    return len(errors) == 0, errors

def deadband_filter(current_reading, previous_reading, threshold):
    """
    Implements deadband filtering for sensor readings.
    
    Parameters:
    current_reading (float): current sensor value
    previous_reading (float): previous sensor value
    threshold (float): deadband threshold
    
    Returns:
    str: 'P0', 'P1', or 'P2' based on deadband protocol.
    
    P0: reading in negative space (invalid)
    P1: reading changed beyond threshold
    P2: reading within threshold
    """
    # Negative space: readings outside 0-100 range
    if current_reading < 0 or current_reading > 100:
        return 'P0'
    
    # Calculate absolute difference
    diff = abs(current_reading - previous_reading)
    
    if diff > threshold:
        return 'P1'
    else:
        return 'P2'
```
