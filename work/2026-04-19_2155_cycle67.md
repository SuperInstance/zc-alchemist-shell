/// Validates a sequence of action priorities against the Deadband Protocol.
///
/// Rules:
/// - Sequence cannot be empty (P0).
/// - Must start with P0 (map negative space).
/// - Can stay at same priority level.
/// - Can advance from P0→P1→P2.
/// - Cannot skip a priority level (e.g., P0→P2 is invalid).
/// - Cannot go backward (e.g., P2→P1) unless explicitly allowed (not implemented here).
///
/// Edge cases handled:
/// 1. Empty sequence – returns EmptySequence error.
/// 2. Single action – must be P0, otherwise InvalidTransition.
pub fn validate_deadband_sequence(sequence: &[ActionPriority]) -> Result<(), ValidationError> {
    if sequence.is_empty() {
        return Err(ValidationError::EmptySequence);
    }

    let mut prev = &sequence[0];
    if *prev != ActionPriority::P0 {
        return Err(ValidationError::InvalidTransition);
    }

    for current in sequence.iter().skip(1) {
        match (prev, current) {
            (ActionPriority::P0, ActionPriority::P0)
            | (ActionPriority::P1, ActionPriority::P1)
            | (ActionPriority::P2, ActionPriority::P2) => {
                // Staying at same level is allowed
            }
            (ActionPriority::P0, ActionPriority::P1)
            | (ActionPriority::P1, ActionPriority::P2) => {
                // Valid advancement
            }
            (ActionPriority::P0, ActionPriority::P2) => {
                return Err(ValidationError::PrioritySkipped {
                    from: ActionPriority::P0,
                    to: ActionPriority::P2,
                })
            }
            (ActionPriority::P1, ActionPriority::P0) => {
                return Err(ValidationError::PrioritySkipped {
                    from: ActionPriority::P1,
                    to: ActionPriority::P0,
                })
            }
            (ActionPriority::P2, ActionPriority::P0)
            | (ActionPriority::P2, ActionPriority::P1) => {
                return Err(ValidationError::PrioritySkipped {
                    from: ActionPriority::P2,
                    to: current.clone(),
                })
            }
            _ => return Err(ValidationError::InvalidTransition),
        }
        prev = current;
    }

    Ok(())
}

    P0,
    P1,
    P2,

#[derive(Debug)]

    NoStartP0,
    SkipDetected(Priority, Priority),
    Empty,
}

            DeadbandError::SkipDetected(from, to) => write!(
                f,
                "Deadband violation: skipped from {:?} to {:?}",

            ),
            DeadbandError::Empty => write!(f, "Empty sequence"),
        }

impl Error for DeadbandError {}

/// Check if a sequence respects Deadband Protocol P0→P1→P2.
///
/// Edge cases:
/// - Empty sequence (returns DeadbandError::Empty)
/// - Single element (must be P0)
/// - Skip detection (P0→P2, P1→P0, etc.)
pub fn check_deadband(seq: &[Priority]) -> Result<(), DeadbandError> {
    if seq.is_empty() {
        return Err(DeadbandError::Empty);
    }

    if seq[0] != Priority::P0 {
        return Err(DeadbandError::NoStartP0);
    }

    for window in seq.windows(2) {
        let prev = &window[0];
        let curr = &window[1];

        match (prev, curr) {
            (Priority::P0, Priority::P0)
            | (Priority::P1, Priority::P1)
            | (Priority::P2, Priority::P2) => continue,
            (Priority::P0, Priority::P1) => continue,
            (Priority::P1, Priority::P2) => continue,
            (Priority::P0, Priority::P2) => {
                return Err(DeadbandError::SkipDetected(
                    Priority::P0,
                    Priority::P2,
                ))
            }
            (Priority::P1, Priority::P0) => {
                return Err(DeadbandError::SkipDetected(
                    Priority::P1,
                    Priority::P0,
                ))
            }
            (Priority::P2, Priority::P0) | (Priority::P2, Priority::P1) => {
                return Err(DeadbandError::SkipDetected(
                    Priority::P2,
                    curr.clone(),
                ))
            }
            _ => unreachable!(),
        }
    }

    Ok(())
}
