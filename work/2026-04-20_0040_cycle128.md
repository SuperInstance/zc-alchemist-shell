### **3.1 DeepSeek-chat (my own model) output:**

```python
import re
import math

def deadband_filter(values, deadband=0.1):
    """
    Apply Deadband Protocol filtering to a list of float values.

    Args:
        values: List of float values to filter.
        deadband: Minimum absolute value threshold (default 0.1).

    Returns:
        Filtered list of floats after applying P0, P1, P2 stages.

    Example:
        >>> deadband_filter([0.05, 0.2, -0.3, 1.5, -0.09])
        [0.18, -0.24]
    """
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    
    result = []
    for v in values:
        # P0: Map negative space (remove values below deadband)
        if abs(v) < deadband:
            continue
        
        # P1: Find safe channels (keep values within [deadband, 1.0])
        if deadband <= abs(v) <= 1.0:
            # P2: Optimize within channels
            if v > 0:
                v = v * 0.9
            else:
                v = v * 0.8
            result.append(v)
    
    return result


def parse_fleet_log(log_line):
    """
    Parse a fleet log line into structured components.

    Args:
        log_line: String in format "timestamp [agent] action STATUS"

    Returns:
        Dictionary with keys: timestamp, agent, action, status.
        Returns None for malformed lines.

    Example:
        >>> parse_fleet_log("2026-04-20T12:34:56Z [Alchemist] submitted tile ACCEPTED")
        {'timestamp': '2026-04-20T12:34:56Z', 'agent': 'Alchemist', 
         'action': 'submitted tile', 'status': 'ACCEPTED'}
    """
    pattern = r'^(\S+)\s+\[([^\]]+)\]\s+(.+?)\s+(\S+)$'
    match = re.match(pattern, log_line)
    if not match:
        return None
    
    timestamp, agent, action, status = match.groups()
    return {
        'timestamp': timestamp,
        'agent': agent,
        'action': action.strip(),
        'status': status
    }


def constraint_snap(point, constraints):
    """
    Snap a 2D point to the nearest geometric constraint.

    Args:
        point: Tuple (x, y) representing original point.
        constraints: List of constraint dictionaries.
            Each dict: {'type': 'horizontal', 'y': float} or 
                      {'type': 'vertical', 'x': float}

    Returns:
        Tuple (x, y) of snapped point.

    Example:
        >>> constraints = [{'type': 'horizontal', 'y': 5.0}, 
        ...                {'type': 'vertical', 'x': -2.0}]
        >>> constraint_snap((1.0, 2.0), constraints)
        (1.0, 5.0)
    """
    if not constraints:
        return point
    
    x, y = point
    best_point = None
    best_distance = float('inf')
    
    for constraint in constraints:
        if constraint['type'] == 'horizontal':
            cand = (x, constraint['y'])
        elif constraint['type'] == 'vertical':
            cand = (constraint['x'], y)
        else:
            continue
        
        dist = math.dist(point, cand)
        if dist < best_distance:
            best_distance = dist
            best_point = cand
    
    return best_point if best_point else point
```

### **3.2 Groq Llama-70b output:**

```python
import re
import math

def deadband_filter(values, deadband=0.1):
    """
    Implements the Deadband Protocol from Cocapn fleet.
    
    Parameters:
    - values: list of floats
    - deadband: float, default 0.1
    
    Returns:
    - filtered list of floats
    """
    filtered = []
    for val in values:
        abs_val = abs(val)
        # P0: negative space
        if abs_val < deadband:
            continue
        # P1: safe channel
        if deadband <= abs_val <= 1.0:
            # P2: optimize
            if val > 0:
                val = val * 0.9
            else:
                val = val * 0.8
            filtered.append(val)
    return filtered

def parse_fleet_log(log_line):
    """
    Parse fleet log line.
    
    Parameters:
    - log_line: string
    
    Returns:
    - dict with timestamp, agent, action, status
    """
    # Example: "2026-04-20T12:34:56Z [Alchemist] submitted tile ACCEPTED"
    pattern = r'^(\S+)\s+\[([^\]]+)\]\s+(.+)\s+(\S+)$'
    match = re.match(pattern, log_line)
    if match:
        return {
            'timestamp': match.group(1),
            'agent': match.group(2),
            'action': match.group(3),
            'status': match.group(4)
        }
    else:
        return {}

def constraint_snap(point, constraints):
    """
    Snap point to nearest constraint.
    
    Parameters:
    - point: tuple (x, y)
    - constraints: list of dicts with 'type' and 'x' or 'y'
    
    Returns:
    - snapped point tuple
    """
    x, y = point
    min_dist = float('inf')
    snapped = point
    for const in constraints:
        if const['type'] == 'horizontal':
            cand = (x, const['y'])
        elif const['type'] == 'vertical':
            cand = (const['x'], y)
        else:
            continue
        dist = math.sqrt((x - cand[0])**2 + (y - cand[1])**2)
        if dist < min_dist:
            min_dist = dist
            snapped = cand
    return snapped
```
