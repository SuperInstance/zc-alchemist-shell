### **Function 1: deadband_filter**

#### **DeepSeek-chat output:**
```python
def deadband_filter(readings: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol to sensor readings.
    
    P0: Remove readings that are within deadband of zero (avoid noise).
    P1: Keep readings that are outside deadband (safe channels).
    P2: Optional: could apply smoothing, but not implemented here.
    
    Args:
        readings: List of raw sensor values.
        deadband: Threshold for considering a reading as noise.
        
    Returns:
        Filtered list where abs(value) > deadband.
    """
    # P0: Map negative space - ignore values inside deadband
    filtered = [r for r in readings if abs(r) > deadband]
    
    # P1: These are the safe channels
    return filtered

# Example usage
if __name__ == "__main__":
    sensor_data = [0.05, 0.15, -0.08, 0.3, -0.2, 0.02]
    filtered = deadband_filter(sensor_data, deadband=0.1)
    print(f"Original: {sensor_data}")
    print(f"Filtered: {filtered}")  # [0.15, 0.3, -0.2]
```

```python
from typing import List

def deadband_filter(readings: List[float], deadband: float = 0.1) -> List[float]:
    """
    Implements the Deadband Protocol for sensor data filtering.
    
    P0 (negative space): Values within +/- deadband are considered noise/unsafe.
    P1 (safe channels): Values outside deadband are retained.
    P2 (optimization): Not applied here—simple filtering.
    
    Parameters:
    readings (List[float]): Input sensor readings.
    deadband (float): Deadband threshold (default 0.1).
    
    Returns:
    List[float]: Filtered readings where |value| > deadband.
    """
    # P0: Identify and exclude negative space (noise)
    safe_readings = [val for val in readings if abs(val) > deadband]
    
    # P1: Return safe channels
    return safe_readings

# Example
if __name__ == "__main__":
    example_readings = [0.0, 0.05, -0.12, 0.25, -0.3, 0.09]
    result = deadband_filter(example_readings, 0.1)
    print("Example readings:", example_readings)
    print("Filtered (deadband 0.1):", result)  # [-0.12, 0.25, -0.3]
```

**Comparison:**
- Both correct and functionally identical.
- DeepSeek uses `list[float]` (Python 3.9+), Groq uses `List[float]` (typing import).
- DeepSeek's docstring references fleet phases more explicitly.
- Groq includes parameter descriptions in docstring.
- Both include example usage.

### **Function 2: parse_fleet_log**

#### **DeepSeek-chat output:**
```python
import re
from dataclasses import dataclass
from datetime import datetime
from typing import Optional

@dataclass
class FleetEvent:
    timestamp: datetime
    agent: str
    action: str
    room: Optional[str] = None
    tile_id: Optional[str] = None
    
def parse_fleet_log(log_text: str) -> list[FleetEvent]:
    """
    Parse fleet log entries into structured events.
    
    Log format: [YYYY-MM-DD HH:MM:SS] AGENT ACTION (ROOM:TILE_ID)
    Example: [2026-04-19 23:15:00] ALchemist submitted_tile (deadband_room:tx8a1)
    
    Args:
        log_text: Multiline log string.
        
    Returns:
        List of FleetEvent objects.
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+) (\w+) \((?:(\w+):?(\w+)?)?\)'
    events = []
    
    for line in log_text.strip().split('\n'):
        match = re.match(pattern, line)
        if match:
            ts_str, agent, action, room, tile_id = match.groups()
            timestamp = datetime.strptime(ts_str, '%Y-%m-%d %H:%M:%S')
            events.append(FleetEvent(timestamp, agent, action, room, tile_id))
    
    return events

# Example usage
if __name__ == "__main__":
    log = """[2026-04-19 23:15:00] Alchemist submitted_tile (deadband_room:tx8a1)
[2026-04-19 23:20:00] Navigator read_repos (navigation_room:)
[2026-04-19 23:25:00] Sentinel service_check (oracle1:abc123)"""
    
    parsed = parse_fleet_log(log)
    for event in parsed:
        print(f"{event.timestamp} {event.agent} {event.action} room={event.room} tile={event.tile_id}")
```

#### **Groq Llama-70b output:**
```python
import re
from datetime import datetime
from typing import List, Optional, NamedTuple

class FleetLogEntry(NamedTuple):
    timestamp: datetime
    agent: str
    action: str
    room: Optional[str]
    tile_id: Optional[str]

def parse_fleet_log(log_text: str) -> List[FleetLogEntry]:
    """
    Parses a multi-line fleet activity log.
    
    Expected format per line:
    [YYYY-MM-DD HH:MM:SS] AGENT ACTION (ROOM:TILE_ID)
    
    ROOM and TILE_ID are optional; if missing, captured as None.
    
    Parameters:
    log_text (str): Raw log content.
    
    Returns:
    List[FleetLogEntry]: Parsed log entries.
    """
    entries = []
    # Regex with optional room and tile_id
    log_regex = re.compile(r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+) (\w+) (?:\((\w+):(\w+)\))?')
    
    for line in log_text.strip().splitlines():
        m = log_regex.match(line)
        if m:
            ts, agent, action, room, tile_id = m.groups()
            dt = datetime.strptime(ts, '%Y-%m-%d %H:%M:%S')
            entries.append(FleetLogEntry(dt, agent, action, room, tile_id))
    return entries

# Example
sample_log = """
[2026-04-19 10:30:00] Alchemist benchmark_start (model_room:)
[2026-04-19 10:35:00] Tinker build_demo (deadband_room:demo1)
[2026-04-19 10:40:00] Sentinel health_check (oracle1:check_456)
"""

parsed = parse_fleet_log(sample_log)
for e in parsed:
    print(e)
```

**Comparison:**
- DeepSeek uses `dataclass`, Groq uses `NamedTuple`.
- DeepSeek's regex handles optional colon/tile better (allows empty room).
- Groq's regex expects both room and tile_id if parentheses present (less flexible).
- Both handle timestamp parsing
