def deadband_filter(signal: list[float], p0_threshold: float, p1_margin: float) -> list[float]:
    """
    Apply deadband filtering based on P0/P1/P2 protocol.

    Args:
        signal: Input signal values
        p0_threshold: Threshold for P0 avoidance (absolute value)
        p1_margin: Margin for P1 safe channel (absolute value)

    Returns:
        Filtered signal values
    """
    result = []
    for value in signal:
        abs_val = abs(value)
        
        # P0: Negative space - complete avoidance
        if abs_val >= p0_threshold:
            result.append(0.0)
            continue
            
        # P1: Safe channel - pass through unchanged
        if abs_val <= p1_margin:
            result.append(value)
            continue
            
        # P2: Optimization - apply scaling
        scale = (p0_threshold - abs_val) / (p0_threshold - p1_margin)
        result.append(value * scale)
    
    return result


def parse_fleet_context(filepath: str) -> dict:
    """
    Parse fleet context markdown file into structured data.

    Args:
        filepath: Path to markdown file

    Returns:
        Dictionary with extracted structure
    """
    result = {
        'title': '',
        'doctrine': '',
        'key_technologies': [],
        'rules': []
    }
    
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()
    except FileNotFoundError:
        return result
    
    lines = content.split('\n')
    current_section = None
    in_list = False
    
    for i, line in enumerate(lines):
        line = line.rstrip()
        
        # Extract title from first top-level heading
        if line.startswith('# ') and not result['title']:
            result['title'] = line[2:].strip()
            continue
            
        # Detect section headers
        if line.startswith('## '):
            header = line[3:].strip().lower()
            if 'doctrine' in header:
                current_section = 'doctrine'
                in_list = False
            elif 'key technology' in header:
                current_section = 'key_technologies'
                in_list = True
            elif 'rules' in header:
                current_section = 'rules'
                in_list = True
            else:
                current_section = None
                in_list = False
            continue
        
        # Process content based on current section
        if current_section == 'doctrine' and line and not line.startswith('#'):
            result['doctrine'] += line + '\n'
        elif current_section == 'key_technologies' and line.strip().startswith('- '):
            result['key_technologies'].append(line.strip()[2:].strip())
        elif current_section == 'rules' and line.strip().startswith('- '):
            result['rules'].append(line.strip()[2:].strip())
    
    # Clean up doctrine text
    result['doctrine'] = result['doctrine'].strip()
    
    return result


def validate_training_tile(tile: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile against fleet requirements.

    Args:
        tile: Dictionary containing tile data

    Returns:
        Tuple of (is_valid, list_of_errors)
    """
    errors = []
    
    # Check required keys exist
    required_keys = ['id', 'agent', 'cycle', 'content', 'phase']
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    if errors:
        return False, errors
    
    # Validate id
    if not isinstance(tile['id'], str) or not tile['id'].strip():
        errors.append("id must be a non-empty string")
    
    # Validate agent
    valid_agents = ["Navigator", "Sentinel", "Scribe", "Tinker", "Scout", "Curator", 
                   "Mason", "Alchemist", "Herald", "Scholar", "Weaver", "Archivist"]
    if tile['agent'] not in valid_agents:
        errors.append(f"agent must be one of: {valid_agents}")
    
    # Validate cycle
    if not isinstance(tile['cycle'], int) or tile['cycle'] <= 0:
        errors.append("cycle must be a positive integer")
    
    # Validate content
    if not isinstance(tile['content'], str):
        errors.append("content must be a string")
    elif len(tile['content']) < 50 or len(tile['content']) > 10000:
        errors.append("content length must be between 50 and 10000 characters")
    
    # Validate phase
    if tile['phase'] not in [1, 2, 3, 4]:
        errors.append("phase must be 1, 2, 3, or 4")
    
    return len(errors) == 0, errors
