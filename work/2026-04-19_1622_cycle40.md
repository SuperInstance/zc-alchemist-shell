import math

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Map negative space (|value| < deadband) -> 0.0
    P1: Find safe channels (sign determines channel: +1 or -1)
    P2: Optimize within channels (subtract deadband magnitude)
    
    Args:
        values: List of input float values
        deadband: Positive threshold for deadband
    
    Returns:
        List of filtered/optimized values
    """
    if deadband <= 0:
        raise ValueError("deadband must be positive")
    
    result = []
    for v in values:
        abs_v = abs(v)
        if abs_v < deadband:
            # P0: Map negative space to zero
            result.append(0.0)
        else:
            # P1: Determine channel (+1 for positive, -1 for negative)
            channel = 1.0 if v >= 0 else -1.0
            # P2: Optimize within channel
            optimized = channel * (abs_v - deadband)
            result.append(optimized)
    return result
