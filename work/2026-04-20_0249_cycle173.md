```python
import re

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol sequence P0->P1->P2 consecutively.

    Args:
        steps: List of step identifiers (strings).

    Returns:
        Tuple of (is_valid: bool, message: str). Message explains violation if invalid.
    """
    # Find the first occurrence of P0
    try:
        p0_index = steps.index("P0")
    except ValueError:
        return False, "Missing P0 phase"

    # Check if P1 and P2 follow immediately after P0 in order
    required = ["P1", "P2"]
    for i, req in enumerate(required, start=1):
        if p0_index + i >= len(steps):
            return False, f"Missing {req} phase"
        if steps[p0_index + i] != req:
            return False, f"Expected {req} but found {steps[p0_index + i]}"

    # Check for extra steps between the required sequence
    if len(steps) > p0_index + 3:
        # If there are more steps after P2, that's okay per spec? Spec says "no other elements between them"
        # Let's check between P0 and P2
        subset = steps[p0_index:p0_index+3]
        if subset != ["P0", "P1", "P2"]:
            return False, "Extra elements between protocol phases"
    
    return True, "Protocol valid"

if __name__ == "__main__":
    # Test cases
    print(validate_deadband_protocol(["P0", "P1", "P2"]))  # (True, "Protocol valid")
    print(validate_deadband_protocol(["A", "P0", "P1", "P2", "B"]))  # (True, "Protocol valid")? Wait, has extra before/after
    print(validate_deadband_protocol(["P0", "P2"]))  # (False, "Expected P1 but found P2")
    print(validate_deadband_protocol(["P1", "P2"]))  # (False, "Missing P0 phase")
    print(validate_deadband_protocol(["P0", "P1", "X", "P2"]))  # (False, "Extra elements between protocol phases")
```

#### **Function 2: `parse_fleet_log_line`**
```python
def parse_fleet_log_line(line: str) -> dict:
    """
    Parses a standardized fleet log line into its components.

    Expected format: "YYYY-MM-DD HH:MM:SS [AGENT] [CYCLE N] Message text"

    Args:
        line: A log line string.

    Returns:
        Dictionary with keys: timestamp, agent, cycle, message.
        Returns empty dict for malformed lines.
    """
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[([^\]]+)\] \[CYCLE (\d+)\] (.+)$'
    match = re.match(pattern, line)
    if not match:
        return {}  # Or raise an exception, but spec says "handle gracefully"
    
    timestamp, agent, cycle, message = match.groups()
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": int(cycle),
        "message": message.strip()
    }

if __name__ == "__main__":
```
