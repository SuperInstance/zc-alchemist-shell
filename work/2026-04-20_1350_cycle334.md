```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validate that a list of steps follows the Deadband Protocol.
    
    Rules:
    1. Must contain exactly one each of "P0", "P1", "P2"
    2. Must appear in order: P0 before P1 before P2
    3. No skipping allowed (e.g., P0 → P2 is invalid)
    4. No repeats allowed
    
    Args:
        steps: List of step strings (e.g., ["P0", "P1", "P2"])
    
    Returns:
        Tuple of (is_valid, error_messages)
        error_messages is empty list if valid
    """
    required = ["P0", "P1", "P2"]
    errors = []
    
    # Check for missing required steps
    for req in required:
        if req not in steps:
            errors.append(f"Missing required step: {req}")
    
    # Check order
    seen_indices = []
    for req in required:
        if req in steps:
            idx = steps.index(req)
            seen_indices.append(idx)
    
    if len(seen_indices) == 3:
        if not (seen_indices[0] < seen_indices[1] < seen_indices[2]):
            errors.append("Steps not in correct order: P0 must come before P1 before P2")
    
    # Check for duplicates
    for req in required:
        if steps.count(req) > 1:
            errors.append(f"Duplicate step: {req}")
    
    # Check for skipping
    if "P0" in steps and "P2" in steps:
        p0_idx = steps.index("P0")
        p2_idx = steps.index("P2")
        if "P1" not in steps[p0_idx:p2_idx]:
            errors.append("Skipping detected: P1 missing between P0 and P2")
    
    return (len(errors) == 0, errors)
```

```python
import re
from datetime import datetime

def extract_training_tiles(log_text: str) -> list[dict]:
    """
    Extract training tiles from log text.
    
    Expected format per tile:
    [TILE:<id>:<type>] <content> (<timestamp>)
    
    Args:
        log_text: Multi-line string containing log entries
    
    Returns:
        List of dictionaries with keys: id, type, content, timestamp
        timestamp converted to datetime object
    """
    pattern = r'\[TILE:([^:]+):([^]]+)\]\s*(.+?)\s*\(([^)]+)\)'
    tiles = []
    
    for line in log_text.strip().split('\n'):
        match = re.match(pattern, line.strip())
        if match:
            tile_id, tile_type, content, timestamp_str = match.groups()
            try:
                # Try parsing ISO format timestamp
                timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
            except ValueError:
                # Fallback to string if parsing fails
                timestamp = timestamp_str
            
            tiles.append({
                'id': tile_id,
                'type': tile_type,
                'content': content.strip(),
                'timestamp': timestamp
            })
    
    return tiles
```

```python
import math
from typing import List, Tuple

def constraint_snap(points: List[Tuple[float, float]], tolerance: float) -> List[Tuple[float, float]]:
    """
    Snap points within tolerance to their centroid.
    
    Args:
        points: List of (x, y) coordinates
        tolerance: Maximum distance for snapping
    
    Returns:
        New list of snapped points
    """
    if not points or tolerance <= 0:
        return points.copy()
    
    # Create clusters
    clusters = []
    assigned = [-1] * len(points)
    
    for i, (x1, y1) in enumerate(points):
        if assigned[i] == -1:
            # Start new cluster
            cluster_idx = len(clusters)
            clusters.append([i])
            assigned[i] = cluster_idx
            
            # Find all points within tolerance
            for j, (x2, y2) in enumerate(points[i+1:], start=i+1):
                if assigned[j] == -1:
                    distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
                    if distance <= tolerance:
                        clusters[cluster_idx].append(j)
                        assigned[j] = cluster_idx
    
    # Compute centroids and create snapped points
    snapped = []
    for cluster_indices in clusters:
        if len(cluster_indices) == 1:
            # Single point, no snapping
            snapped.append(points[cluster_indices[0]])
        else:
            # Compute centroid
            sum_x, sum_y = 0.0, 0.0
            for idx in cluster_indices:
                x, y = points[idx]
                sum_x += x
                sum_y += y
            centroid = (sum_x / len(cluster_indices), sum_y / len(cluster_indices))
            for _ in cluster_indices:
                snapped.append(centroid)
    
    # Return in original order
    result = [None] * len(points)
    for cluster_idx, cluster_indices in enumerate(clusters):
        centroid = snapped[cluster_idx]
        for idx in cluster_indices:
            result[idx] = centroid
    
    return result
```

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validate Deadband Protocol steps.
    
    Args:
        steps: list of step strings
    
    Returns:
        (is_valid, list of error messages)
    """
    errors = []
    required_order = ["P0", "P1", "P2"]
    
    # Check for presence
    for req in required_order:
        if req not in steps:
            errors.append(f"Missing {req}")
    
    # Check order
    indices = []
    for req in required_order:
        if req in steps:
            indices.append(steps.index(req))
    
    if len(indices) == 3:
        if indices != sorted(indices):
            errors.append("Steps out of order")
    
    # Check duplicates
    for step in set(steps):
        if steps.count(step) > 1:
            errors.append(f"Duplicate {step}")
    
    # Check skipping
    if "P0" in steps and "P2" in steps:
        p0_idx = steps.index("P0")
        p2_idx = steps.index("P2")
        if p2_idx - p0_idx > 1:
            middle = steps[p0_idx+1:p2_idx]
            if "P1" not in middle:
                errors.append("Skipped P1")
