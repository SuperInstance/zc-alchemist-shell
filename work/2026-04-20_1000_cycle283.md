from typing import List, Tuple

def deadband_filter(values: List[float], threshold: float = 0.05) -> Tuple[List[float], List[float], List[float]]:
    """
    Apply Deadband Protocol to filter values.
    
    P0: Identify invalid inputs (values <= 0)
    P1: Find valid values within ±threshold of an integer
    P2: Round valid values to nearest 0.1
    
    Args:
        values: List of float values
        threshold: Tolerance for integer proximity (default 0.05)
    
    Returns:
        Tuple of (p0_errors, p1_valid, p2_optimized)
    
    Example:
        >>> deadband_filter([-1.0, 0.0, 0.98, 1.02, 2.1, 3.0])
        ([-1.0, 0.0], [0.98, 1.02, 3.0], [1.0, 1.0, 3.0])
    """
    p0_errors = []
    p1_valid = []
    
    for v in values:
        if v <= 0:
            p0_errors.append(v)
        else:
            # Check if within threshold of any integer
            nearest_int = round(v)
            if abs(v - nearest_int) <= threshold:
                p1_valid.append(v)
    
    # P2: Optimize valid values
    p2_optimized = [round(v, 1) for v in p1_valid]
    
    return p0_errors, p1_valid, p2_optimized

import math
from typing import List, Dict, Tuple, Optional

def constraint_snap(point: Tuple[float, float], 
                    constraints: List[Dict], 
                    tolerance: float = 0.01) -> Tuple[float, float]:
    """
    Snap a 2D point to nearest geometric constraint.
    
    Args:
        point: (x, y) coordinates
        constraints: List of constraint dicts with 'type' and 'params'
        tolerance: Maximum distance for snapping
    
    Returns:
        Snapped (x, y) coordinates
    
    Example:
        >>> constraints = [
        ...     {'type': 'line', 'params': (1, 0)},  # y = x
        ...     {'type': 'circle', 'params': (0, 0, 1)}
        ... ]
        >>> constraint_snap((0.1, 0.1), constraints)
        (0.0, 0.0)
    """
    x, y = point
    
    if not constraints:
        return point
    
    best_point = None
    best_dist = float('inf')
    
    for constraint in constraints:
        c_type = constraint['type']
        params = constraint['params']
        
        if c_type == 'line':
            m, b = params
            # Project point to line
            # Line: y = m*x + b
            # Perpendicular: y = (-1/m)*x + c (if m != 0)
            if m == 0:  # Horizontal line
                snap_x = x
                snap_y = b
            else:
                # Solve intersection of line and perpendicular through point
                m_perp = -1 / m
                c_perp = y - m_perp * x
                
                # Intersection: m*x + b = m_perp*x + c_perp
                snap_x = (c_perp - b) / (m - m_perp)
                snap_y = m * snap_x + b
            
            dist = math.hypot(snap_x - x, snap_y - y)
            if dist <= tolerance and dist < best_dist:
                best_dist = dist
                best_point = (snap_x, snap_y)
                
        elif c_type == 'circle':
            cx, cy, r = params
            # Vector from center to point
            dx = x - cx
            dy = y - cy
            length = math.hypot(dx, dy)
            
            if length == 0:
                # Point at center, snap to any point on circle
                snap_x = cx + r
                snap_y = cy
            else:
                # Snap to nearest point on circle
                snap_x = cx + (dx / length) * r
                snap_y = cy + (dy / length) * r
            
            dist = math.hypot(snap_x - x, snap_y - y)
            if dist <= tolerance and dist < best_dist:
                best_dist = dist
                best_point = (snap_x, snap_y)
    
    return best_point if best_point is not None else point
