```python
import re
from datetime import datetime

def parse_log_line(log_line: str) -> dict:
    """
    Parse a structured log line in format:
    '[TIMESTAMP] LEVEL: MESSAGE'
    
    Example: '[2026-04-19 22:10:45] INFO: Agent started successfully'
    
    Returns dict with keys: timestamp, level, message.
    Returns None if parsing fails.
    """
    # Regex pattern: [timestamp] level: message
    pattern = r'^\[(.+?)\] (\w+): (.+)$'
    match = re.match(pattern, log_line.strip())
    
    if not match:
        return None
    
    timestamp_str, level, message = match.groups()
    
    # Try to parse timestamp
    timestamp = None
    for fmt in ('%Y-%m-%d %H:%M:%S', '%Y/%m/%d %H:%M:%S', '%d-%b-%Y %H:%M:%S'):
        try:
            timestamp = datetime.strptime(timestamp_str, fmt)
            break
        except ValueError:
            continue
    
    return {
        'timestamp': timestamp,
        'level': level.upper(),
        'message': message.strip()
    }
```

```python
import re
from datetime import datetime

def parse_log_line(log_line):
    """
    Parses a log line with format [timestamp] level: message.
    
    Args:
        log_line (str): The log line to parse.
        
    Returns:
        dict: Contains 'timestamp' (datetime object), 'level' (str), 
              'message' (str). Returns None if parsing fails.
    """
    # Remove leading/trailing whitespace
    log_line = log_line.strip()
    
    # Pattern: [timestamp] level: message
    pattern = r'^\[([^\]]+)\]\s+(\w+):\s+(.+)$'
    match = re.match(pattern, log_line)
    
    if not match:
        return None
    
    timestamp_str, level, message = match.groups()
    
    # Try common timestamp formats
    timestamp = None
    formats = [
        '%Y-%m-%d %H:%M:%S',
        '%Y/%m/%d %H:%M:%S', 
        '%d.%m.%Y %H:%M:%S',
        '%b %d %H:%M:%S'
    ]
    
    for fmt in formats:
        try:
            timestamp = datetime.strptime(timestamp_str, fmt)
            break
        except ValueError:
            continue
    
    if timestamp is None:
        # Could not parse timestamp, keep as string
        timestamp = timestamp_str
    
    return {
        'timestamp': timestamp,
        'level': level.upper(),
        'message': message
    }
```

### **Function 2: `deadband_filter`**

**DeepSeek-chat output:**
```python
from typing import List, Tuple

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of values.
    
    P0: Remove values that would cause instability (outside safe range ±2.0)
    P1: Find values within deadband threshold of previous output
    P2: Return filtered values
    
    Args:
        values: List of float values to filter
        deadband: Minimum change threshold (default 0.1)
    
    Returns:
        Filtered list where only significant changes are kept
    """
    if not values:
        return []
    
    # P0: Remove values that would cause instability
    safe_values = [v for v in values if -2.0 <= v <= 2.0]
    
    if not safe_values:
        return []
    
    # Initialize with first safe value
    filtered = [safe_values[0]]
    last_output = safe_values[0]
    
    # P1/P2: Apply deadband filtering
    for value in safe_values[1:]:
        # Check if change exceeds deadband threshold
        if abs(value - last_output) >= deadband:
            filtered.append(value)
            last_output = value
    
    return filtered
```

```python
def deadband_filter(data, deadband=0.1, max_range=2.0):
    """
    Implements deadband filtering based on fleet protocol.
    
    P0: Values outside [-max_range, max_range] are discarded (unsafe)
    P1: Values within deadband of previous output are ignored
    P2: Remaining values form the output sequence
    
    Parameters:
        data (list): Input values
        deadband (float): Deadband threshold
        max_range (float): Maximum allowed absolute value
        
    Returns:
        list: Filtered values
    """
    if not data:
        return []
    
    # P0: Remove unsafe values
    safe_data = [x for x in data if abs(x) <= max_range]
    
    if len(safe_data) == 0:
        return []
    
    result = [safe_data[0]]
    last_val = safe_data[0]
    
    # Process remaining values
    for val in safe_data[1:]:
        # Check if change exceeds deadband
        if abs(val - last_val) > deadband:
            result.append(val)
            last_val = val
    
    return result
```

### **Function 3: `validate_github_webhook`**

**DeepSeek-chat output:**
```python
import hmac
import hashlib

def validate_github_webhook(payload_body: bytes, secret_token: str, 
                           signature_header: str) -> bool:
    """
    Validate a GitHub webhook signature.
    
    Args:
        payload_body: Raw request body as bytes
        secret_token: GitHub webhook secret token
        signature_header: 'X-Hub-Signature-256' header value
    
    Returns:
        bool: True if signature is valid, False otherwise
    """
    if not signature_header or not secret_token:
        return False
    
    # Extract signature from header (format: "sha256=...")
    if not signature_header.startswith('sha256='):
        return False
    
    expected_signature = signature_header[7:]  # Remove 'sha256='
    
    # Calculate HMAC
    calculated_signature = hmac.new(
        key=secret_token.encode('utf-8'),
        msg=payload_body,
        digestmod=hashlib.sha256
    ).hexdigest()
    
    # Use constant-time comparison to prevent timing attacks
    return hmac.compare_digest(expected_signature, calculated_signature)
```

```python
import hmac
import hashlib

def validate_github_webhook(payload, secret, signature):
    """
    Validates GitHub webhook signature to ensure payload authenticity.
    
    Args:
        payload (bytes or str): The request payload
        secret (str): Webhook secret configured in GitHub
        signature (str): The 'X-Hub-Signature-256' header value
        
    Returns:
        bool: True if valid, False otherwise
    """
    if not all([payload, secret, signature]):
        return False
    
    # Convert payload to bytes if needed
    if isinstance(payload, str):
        payload = payload.encode('utf-8')
    
    # Check signature format
    if not signature.startswith('sha256='):
        return False
    
    # Extract signature hash
    received_hash = signature[7:]
    
    # Compute expected hash
    expected_hash = hmac.new(
        key=secret.encode('utf-8'),
        msg=payload,
        digestmod=hashlib.sha256
    ).hexdigest()
    
    # Secure comparison
    return hmac.compare_digest(received_hash, expected_hash)
```
