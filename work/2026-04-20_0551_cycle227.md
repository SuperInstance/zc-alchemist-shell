```python
from typing import List, Tuple, Dict, Any
import re
from datetime import datetime

def deadband_filter(values: List[float], deadband: Tuple[float, float]) -> List[str]:
    """
    Apply deadband protocol classification to a list of values.
    
    Args:
        values: List of float values to classify
        deadband: Tuple (low, high) defining the deadband boundaries
        
    Returns:
        List of classification strings: "P0", "P1", or "P2"
    """
    low, high = deadband
    classifications = []
    
    for val in values:
        if val <= low:
            classifications.append("P0")
        elif low < val < high:
            classifications.append("P1")
        else:  # val >= high
            classifications.append("P2")
    
    return classifications


def parse_fleet_context(md_text: str) -> Dict[str, Any]:
    """
    Parse fleet context markdown into structured dictionary.
    
    Args:
        md_text: Markdown text from FLEET-CONTEXT.md
        
    Returns:
        Dictionary with section titles as keys and content as lists of lines
    """
    sections = {
        "Who We Are": [],
        "Fleet Doctrine": [],
        "Your Purpose": [],
        "Key Technologies": [],
        "Architecture": [],
        "Rules": []
    }
    
    current_section = None
    lines = md_text.split('\n')
    
    for line in lines:
        # Check for section headers (## followed by section name)
        if line.startswith('## '):
            header = line[3:].strip()
            if header in sections:
                current_section = header
                sections[current_section] = []
            else:
                current_section = None
        elif current_section and line.strip():
            # Add non-empty lines to current section
            sections[current_section].append(line.strip())
    
    # Convert lists to single strings for cleaner output
    for section in sections:
        if sections[section]:
            sections[section] = '\n'.join(sections[section])
        else:
            sections[section] = ""
    
    return sections


def validate_training_tile(tile: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile dictionary.
    
    Args:
        tile: Dictionary containing training tile data
        
    Returns:
        Tuple of (is_valid, list_of_error_messages)
    """
    errors = []
    required_keys = ['agent_id', 'cycle', 'task', 'content', 'timestamp_utc']
    
    # Check required keys
    for key in required_keys:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    # Check content
    if 'content' in tile:
        if not isinstance(tile['content'], str):
            errors.append("Content must be a string")
        elif not tile['content'].strip():
            errors.append("Content cannot be empty")
    
    # Check timestamp format
    if 'timestamp_utc' in tile:
        try:
            datetime.fromisoformat(tile['timestamp_utc'].replace('Z', '+00:00'))
        except (ValueError, AttributeError):
            errors.append("timestamp_utc must be valid ISO 8601 format")
    
    # Check cycle is integer
    if 'cycle' in tile and not isinstance(tile.get('cycle'), int):
        errors.append("Cycle must be an integer")
    
    return len(errors) == 0, errors
```

    """
