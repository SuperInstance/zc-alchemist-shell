from typing import List, Dict, Any
import re
from datetime import datetime, timezone
import json

def deadband_filter(values: List[float], p0_thresh: float, p1_margin: float) -> Dict[str, List[int]]:
    """
    Apply Deadband Protocol P0/P1/P2 classification to a list of values.
    
    Args:
        values: List of float values to classify
        p0_thresh: Threshold for P0 (negative space, avoid)
        p1_margin: Upper bound for P1 (safe channel)
    
    Returns:
        Dictionary with lists of indices for each category:
        - p0_negative: indices where value <= p0_thresh
        - p1_safe: indices where p0_thresh < value <= p1_margin  
        - p2_optimal: indices where value > p1_margin
    
    Raises:
        ValueError: If p1_margin <= p0_thresh
    """
    if p1_margin <= p0_thresh:
        raise ValueError("p1_margin must be greater than p0_thresh")
    
    result = {
        'p0_negative': [],
        'p1_safe': [],
        'p2_optimal': []
    }
    
    for idx, val in enumerate(values):
        if val <= p0_thresh:
            result['p0_negative'].append(idx)
        elif val <= p1_margin:
            result['p1_safe'].append(idx)
        else:
            result['p2_optimal'].append(idx)
    
    return result


def parse_fleet_context(markdown_text: str) -> Dict[str, str]:
    """
    Parse FLEET-CONTEXT.md structured markdown into section dictionary.
    
    Args:
        markdown_text: Complete markdown text of FLEET-CONTEXT.md
    
    Returns:
        Dictionary with section headers (without '## ') as keys and
        section content (stripped) as values.
    """
    sections = {}
    current_section = None
    current_content = []
    
    # Expected fleet sections in order
    expected_sections = [
        'Who We Are',
        'Fleet Doctrine', 
        'Your Purpose',
        'Key Technologies',
        'Architecture',
        'Rules'
    ]
    
    for line in markdown_text.split('\n'):
        if line.startswith('## '):
            if current_section is not None:
                sections[current_section] = '\n'.join(current_content).strip()
            
            section_name = line[3:].strip()
            if section_name in expected_sections:
                current_section = section_name
                current_content = []
            else:
                current_section = None
        elif current_section is not None:
            current_content.append(line)
    
    # Add the last section
    if current_section is not None:
        sections[current_section] = '\n'.join(current_content).strip()
    
    # Ensure all expected sections exist (empty if not found)
    for section in expected_sections:
        if section not in sections:
            sections[section] = ""
    
    return sections


def generate_training_tile(agent_role: str, cycle: int, content: str, task: str) -> Dict[str, Any]:
    """
    Generate a PLATO room training tile in JSON-serializable format.
    
    Args:
        agent_role: Role of the submitting agent (e.g., 'Alchemist')
        cycle: Current training cycle number
        content: The work output/content of the tile
        task: Description of the task performed
    
    Returns:
        Dictionary ready for JSON serialization to PLATO room server
    """
    tile = {
        'agent': agent_role,
        'cycle': cycle,
        'timestamp': datetime.now(timezone.utc).isoformat(),
        'content': content,
        'task': task,
        'phase': 4,  # Current boot camp phase
        'version': '1.0',
        'fleet': 'Cocapn',
        'type': 'training_tile'
    }
    return tile
