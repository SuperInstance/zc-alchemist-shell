from typing import List, Tuple

def validate_deadband_protocol(actions: List[str]) -> Tuple[bool, str]:
    """
    Validates a sequence of actions against the Deadband Protocol (P0->P1->P2).
    
    Args:
        actions: List of action strings, e.g., ["P0", "P1", "P2"]
    
    Returns:
        Tuple of (is_valid, error_message). Valid if sequence follows P0, then P1,
        then P2 strictly without skipping or regression.
    """
    expected_sequence = ["P0", "P1", "P2"]
    seen = set()
    
    for action in actions:
        if action not in expected_sequence:
            return False, f"Invalid action '{action}'. Must be one of {expected_sequence}"
        
        # Check for regression: if we've seen a later phase, can't go back
        for later_phase in expected_sequence[expected_sequence.index(action) + 1:]:
            if later_phase in seen:
                return False, f"Protocol regression: '{action}' after '{later_phase}'"
        
        # Check for skipping: must see phases in order
        for prev_phase in expected_sequence[:expected_sequence.index(action)]:
            if prev_phase not in seen:
                return False, f"Protocol skip: '{action}' before '{prev_phase}'"
        
        seen.add(action)
    
    return True, "Sequence valid"

import re
from datetime import datetime
from typing import Dict, Optional

def extract_training_tile(text: str) -> Dict[str, str]:
    """
    Extract structured training tile fields from raw agent output text.
    
    Fields extracted:
    - agent_id: Matches pattern like "Alchemist", "Navigator"
    - cycle: Number after "Cycle" or "cycle"
    - task: Text after "Task:" or "task:"
    - content: Main body content (first 500 chars)
    - timestamp: Current UTC ISO format
    
    Returns dict with extracted values or placeholders.
    """
    result = {
        "agent_id": "Unknown",
        "cycle": "0",
        "task": "Unknown",
        "content": text[:500].strip(),
        "timestamp": datetime.utcnow().isoformat() + "Z"
    }
    
    # Extract agent_id
    agent_match = re.search(r'# (Cycle \d+ — )?(\w+)', text)
    if agent_match:
        result["agent_id"] = agent_match.group(2)
    
    # Extract cycle number
    cycle_match = re.search(r'Cycle (\d+)', text, re.IGNORECASE)
    if cycle_match:
        result["cycle"] = cycle_match.group(1)
    
    # Extract task
    task_match = re.search(r'Task[:\s]+(.+?)(?=\n\n|\n#|$)', text, re.IGNORECASE | re.DOTALL)
    if task_match:
        result["task"] = task_match.group(1).strip()
    
    return result

    """
    Simulate simple resource-aware task assignment for Cocapn fleet.
    
    Priority: active > idle > offline agents.
    Checks GPU memory constraint if task requires it.
    
    Args:
        agent_states: Dict mapping agent name to state ("active", "idle", "offline")
        resource_constraints: Dict of constraints, e.g., {"gpu_mem_mb": 4096}
    
    Returns:
        List of assignment dicts [{"agent": name, "task": task_name}]
    """
    # Define available tasks and their requirements
    tasks = [

        {"name": "log_analysis", "requires_gpu": False},
        {"name": "api_monitor", "requires_gpu": False},

    ]
    
    # Priority order
    state_priority = {"active": 0, "idle": 1, "offline": 2}
    
    # Sort agents by state priority
    sorted_agents = sorted(agent_states.items(), 
                          key=lambda x: state_priority.get(x[1], 99))
    
    assignments = []
    available_gpu_mb = resource_constraints.get("gpu_mem_mb", 0)

        if state == "offline":
            continue

            # Check GPU requirement

                    continue  # Not enough GPU

            assignments.append({
                "agent": agent,
                "task": task["name"],
                "state": state
            })
            break  # Assign one task per agent
    
    return assignments
