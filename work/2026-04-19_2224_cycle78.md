```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validate that steps follow Deadband Protocol ordering: P0, P1, P2 exactly once each.
    
    Args:
        steps: List of step identifiers as strings.
        
    Returns:
        Tuple of (is_valid, error_message). Valid if steps == ["P0", "P1", "P2"].
    """
    expected = ["P0", "P1", "P2"]
    if steps == expected:
        return True, ""
    if len(steps) != 3:
        return False, f"Expected 3 steps, got {len(steps)}"
    for i, (exp, got) in enumerate(zip(expected, steps)):
        if exp != got:
            return False, f"Step {i}: expected '{exp}', got '{got}'"
    # Check for duplicates
    if len(set(steps)) != 3:
        return False, "Duplicate steps detected"
    return False, "Unknown validation error"
```

```python
import math

def constraint_snap(points: list[tuple[float, float]], tolerance: float = 0.01) -> list[tuple[float, float]]:
    """
    Snap points within tolerance to the first point in each cluster.
    
    Args:
        points: List of (x, y) coordinates.
        tolerance: Maximum Euclidean distance for snapping.
        
    Returns:
        List of snapped points preserving original order.
    """
    if not points:
        return []
    
    snapped = []
    clusters = []  # list of cluster centers
    
    for x, y in points:
        snapped_point = (x, y)
        for cx, cy in clusters:
            dist = math.hypot(x - cx, y - cy)
            if dist <= tolerance:
                snapped_point = (cx, cy)
                break
        else:
            clusters.append((x, y))
        snapped.append(snapped_point)
    return snapped
```

```python
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if the given steps list strictly follows the Deadband Protocol:
    must contain exactly one each of "P0", "P1", "P2" in that order.
    
    Args:
        steps: A list of step strings.
    
    Returns:
        A tuple (is_valid, error_message). is_valid is True if the protocol is followed,
        False otherwise. error_message provides details on failure.
    """
    expected_order = ["P0", "P1", "P2"]
    
    # Check length
    if len(steps) != len(expected_order):
        return False, f"Expected {len(expected_order)} steps, got {len(steps)}"
    
    # Check exact order
    for i, (expected, actual) in enumerate(zip(expected_order, steps)):
        if expected != actual:
            return False, f"At position {i}: expected '{expected}', got '{actual}'"
    
    # Check for duplicates (though order check already ensures uniqueness)
    if len(set(steps)) != len(steps):
        return False, "Duplicate steps found"
    
    return True, "Protocol valid"
```

```python
from typing import Dict

def parse_fleet_tile(tile_text: str) -> Dict[str, str]:
    """
    Parses a fleet tile text into a dictionary.
    
    Each line should be in the format 'key: value'. Lines without a colon are ignored.
    Keys are converted to lowercase and stripped of whitespace.
    Values are stripped of leading/trailing whitespace.
    
    Args:
        tile_text: A multiline string containing key-value pairs.
    
    Returns:
        A dictionary with parsed key-value pairs.
    """
    parsed = {}
    
    for line in tile_text.splitlines():
        line = line.strip()
        if not line:
            continue
        
        if ':' not in line:
            # Silently skip lines without colon as per spec
            continue
        
        key, value = line.split(':', 1)
        key = key.strip().lower()
        value = value.strip()
        
        parsed[key] = value
    
    return parsed
```

```python
import math
from typing import List, Tuple

def constraint_snap(points: List[Tuple[float, float]], tolerance: float = 0.01) -> List[Tuple[float, float]]:
    """
    Snaps points that are within a given tolerance of each other to the first point in their cluster.
    
    Args:
        points: A list of (x, y) tuples.
        tolerance: The maximum Euclidean distance for points to be considered the same cluster.
    
    Returns:
        A list of snapped points in the same order as input.
    """
    if not points:
        return []
    
    snapped_points = []
    cluster_centers = []
    
    for point in points:
        x, y = point
        snapped = point  # default to original
        
        for center in cluster_centers:
            cx, cy = center
            distance = math.sqrt((x - cx) ** 2 + (y - cy) ** 2)
            if distance <= tolerance:
                snapped = center
                break
        
        if snapped == point:
            # This point is a new cluster center
            cluster_centers.append(point)
        
        snapped_points.append(snapped)
    
    return snapped_points
```
