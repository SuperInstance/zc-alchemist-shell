def deadband_filter(readings: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol P0/P1/P2 logic to sensor readings.
    
    P0: Ignore readings within deadband of zero (negative space)
    P1: Keep readings outside deadband (safe channels)
    P2: Apply smoothing to remaining readings (optimization)
    
    Args:
        readings: List of raw sensor values
        deadband: Absolute threshold for ignoring near-zero values
    
    Returns:
        Filtered and smoothed list of readings
    """
    if not readings:
        return []
    
    # P0: Map negative space (ignore near-zero)
    safe_readings = []
    for val in readings:
        if abs(val) >= deadband:
            safe_readings.append(val)
    
    # P1: Find safe channels (already done in filtering)
    if not safe_readings:
        return []
    
    # P2: Optimize within channels (simple moving average)
    optimized = []
    window_size = min(3, len(safe_readings))
    for i in range(len(safe_readings)):
        start = max(0, i - window_size + 1)
        window = safe_readings[start:i+1]
        optimized.append(sum(window) / len(window))
    
    return optimized
```

import re
from typing import Dict, List, Optional

def parse_fleet_context(markdown_text: str) -> Dict[str, any]:
    """
    Parse fleet context markdown into structured data.
    
    Extracts:
    - Fleet members
    - Doctrine lines
    - Key technologies
    - Architecture notes
    - Rules
    
    Args:
        markdown_text: Raw markdown from FLEET-CONTEXT.md
    
    Returns:
        Dictionary with parsed sections
    """
    result = {
        "fleet_members": [],
        "doctrine": [],
        "technologies": [],
        "architecture": [],
        "rules": []
    }
    
    lines = markdown_text.split('\n')
    current_section = None
    
    for line in lines:
        # Detect section headers
        if line.startswith('## '):
            section = line[3:].strip().lower()
            if 'who we are' in section:
                current_section = 'fleet_members'
            elif 'doctrine' in section:
                current_section = 'doctrine'
            elif 'technologies' in section:
                current_section = 'technologies'
            elif 'architecture' in section:
                current_section = 'architecture'
            elif 'rules' in section:
                current_section = 'rules'
            else:
                current_section = None
        
        # Parse content based on current section
        if current_section and line.strip() and not line.startswith('#') and not line.startswith('```'):
            clean_line = line.strip()
            if clean_line and not clean_line.startswith('---'):
                if current_section == 'fleet_members':
                    if ':' in clean_line and 'agent' in clean_line.lower():
                        result['fleet_members'].append(clean_line)
                elif current_section == 'doctrine':
                    if clean_line.startswith('P') and ':' in clean_line:
                        result['doctrine'].append(clean_line)
                elif current_section == 'technologies':
                    if clean_line.startswith('- **'):
                        tech = clean_line[4:].split('**')[0]
                        result['technologies'].append(tech)
                elif current_section == 'rules':
                    if clean_line.startswith('- '):
                        result['rules'].append(clean_line[2:])
    
    return result
```

from datetime import datetime
from typing import Tuple, List

def tile_validator(tile_content: str, tile_metadata: dict) -> Tuple[bool, List[str]]:
    """
    Validate a training tile against fleet submission rules.
    
    Checks:
    1. Content is non-empty
    2. Has proper cycle header format
    3. Contains actual work (not just placeholders)
    4. Follows factual style (no absolutes)
    5. Metadata has required fields
    
    Args:
        tile_content: The tile text/markdown
        tile_metadata: Dict with agent, cycle, timestamp
    
    Returns:
        (is_valid, list_of_errors)
    """
    errors = []
    
    # Rule 1: Non-empty content
    if not tile_content or tile_content.strip() == '':
        errors.append("Tile content is empty")
    
    # Rule 2: Cycle header format
    if not re.search(r'Cycle \d+ — \w+', tile_content):
        errors.append("Missing proper cycle header (Cycle X — AgentName)")
    
    # Rule 3: Contains actual work
    placeholder_phrases = ['TODO', 'FIXME', 'coming soon', 'to be added', 'placeholder']
    content_lower = tile_content.lower()
    if any(phrase in content_lower for phrase in placeholder_phrases):
        errors.append("Contains placeholder text, not actual work")
    
    # Rule 4: Factual style check
    absolute_phrases = ['always', 'never', 'guaranteed', 'certainly', 'definitely']
    for phrase in absolute_phrases:
        if phrase in content_lower:
            errors.append(f"Contains absolute claim: '{phrase}'")
    
    # Rule 5: Metadata validation
    required_fields = ['agent', 'cycle', 'timestamp']
    for field in required_fields:
        if field not in tile_metadata:
            errors.append(f"Missing metadata field: {field}")
        elif not tile_metadata[field]:
            errors.append(f"Empty metadata field: {field}")
    
    # Special: Timestamp format check
    if 'timestamp' in tile_metadata:
        try:
            datetime.fromisoformat(tile_metadata['timestamp'].replace('Z', '+00:00'))
        except (ValueError, TypeError):
            errors.append("Invalid timestamp format")
    
    return (len(errors) == 0, errors)
```

def deadband_filter(sensor_readings, deadband_threshold=0.1):
    """
    Implements deadband filtering on sensor data.
    
    Parameters:
    sensor_readings (list): List of floating point sensor readings
    deadband_threshold (float): Threshold for deadband filtering
    
    Returns:
    list: Filtered readings after deadband application
    """
    filtered_readings = []
    
    # Remove readings within deadband of zero
    for reading in sensor_readings:
        if abs(reading) > deadband_threshold:
            filtered_readings.append(reading)
    
    # Apply smoothing if we have enough data
    if len(filtered_readings) > 1:
        smoothed = []
        for i in range(len(filtered_readings)):
            if i == 0:
                smoothed.append(filtered_readings[i])
            else:
                # Simple exponential smoothing
                smoothed.append(0.7 * filtered_readings[i] + 0.3 * filtered_readings[i-1])
        return smoothed
    
    return filtered_readings
```
