def deadband_filter(values, deadband_threshold=0.1):
    """
    Apply the Deadband Protocol to a list of numeric values.
    
    Parameters:
    - values: list of floats
    - deadband_threshold: float, default 0.1
    
    Returns:
    - list of strings: 'P0', 'P1', or 'P2' for each value
    
    Protocol:
    - P0 (negative space): value <= -deadband_threshold
    - P1 (safe channel): -deadband_threshold < value < deadband_threshold
    - P2 (optimization): value >= deadband_threshold

    result = []
    for v in values:
        if v <= -deadband_threshold:
            result.append('P0')
        elif v >= deadband_threshold:
            result.append('P2')
        else:
            result.append('P1')
    return result

## **3. Function 2: `parse_fleet_log`**

**Prompt:** "Write a Python function `parse_fleet_log(log_line)` that parses a fleet log line formatted as: '2026-04-20 14:30:45 | AGENT=Alchemist | CYCLE=339 | STATUS=ACTIVE | MSG=Testing tile generation'. Return a dict with keys: timestamp, agent, cycle, status, message. Handle missing fields gracefully (default None). Use datetime parsing for timestamp."

### **DeepSeek-chat output:**
```python
from datetime import datetime

def parse_fleet_log(log_line):
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format:
    'YYYY-MM-DD HH:MM:SS | AGENT=name | CYCLE=number | STATUS=status | MSG=message'
    
    Parameters:
    - log_line: string
    
    Returns:
    - dict with keys: timestamp (datetime), agent (str), cycle (int), status (str), message (str)
      Missing fields become None.
    """
    parts = log_line.split(' | ')
    result = {
        'timestamp': None,
        'agent': None,
        'cycle': None,
        'status': None,
        'message': None
    }
    
    for part in parts:
        if '=' in part:
            key, value = part.split('=', 1)
            key = key.strip().lower()
            if key == 'agent':
                result['agent'] = value
            elif key == 'cycle':
                try:
                    result['cycle'] = int(value)
                except ValueError:
                    result['cycle'] = None
            elif key == 'status':
                result['status'] = value
            elif key == 'msg':
                result['message'] = value
        else:
            # Try to parse as timestamp
            try:
                result['timestamp'] = datetime.strptime(part, '%Y-%m-%d %H:%M:%S')
            except ValueError:
                pass
    
    return result

# Example
if __name__ == "__main__":
    line = "2026-04-20 14:30:45 | AGENT=Alchemist | CYCLE=339 | STATUS=ACTIVE | MSG=Testing tile generation"
    print(parse_fleet_log(line))
```
