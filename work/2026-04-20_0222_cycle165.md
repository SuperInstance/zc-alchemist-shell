```python
from typing import List, Union

def deadband_filter(
    data: Union[float, List[float]], 
    lower: float, 
    upper: float
) -> Union[float, List[float]]:
    """
    Apply a deadband filter to input data.
    
    Values within [lower, upper] inclusive map to 0.
    Values outside the band pass through unchanged.
    
    Args:
        data: Single numeric value or list of values.
        lower: Lower bound of deadband.
        upper: Upper bound of deadband.
    
    Returns:
        Filtered value(s) with deadband applied.
    
    Raises:
        ValueError: If lower > upper.
    """
    if lower > upper:
        raise ValueError(f"lower ({lower}) must be <= upper ({upper})")
    
    if isinstance(data, list):
        return [0.0 if lower <= val <= upper else val for val in data]
    else:
        return 0.0 if lower <= data <= upper else data
```

#### **Function 2: parse_fleet_log_line**
```python
import re
from typing import Optional, Dict
from datetime import datetime

def parse_fleet_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured fleet log line.
    
    Expected format: [TIMESTAMP] SERVICE: LEVEL - message
    
    Args:
        line: Log line string.
    
    Returns:
        Dictionary with keys: timestamp, service, level, message.
        Returns None if line doesn't match pattern.
    """
    pattern = r'^\[(.*?)\] (\w+): (\w+) - (.*)$'
    match = re.match(pattern, line.strip())
    
    if not match:
        return None
    
    timestamp_str, service, level, message = match.groups()
    
    # Try to parse timestamp
    try:
        # Assuming ISO format or similar
        timestamp = datetime.fromisoformat(timestamp_str)
    except ValueError:
        timestamp = timestamp_str  # Keep as string if parsing fails
    
    return {
        'timestamp': timestamp,
        'service': service,
        'level': level.upper(),
        'message': message
    }
```

#### **Function 3: constraint_snap**
```python
import math
from typing import Tuple, Optional

def constraint_snap(
    point: Tuple[float, float],
    line_start: Tuple[float, float],
    line_end: Tuple[float, float],
    tolerance: float
) -> Optional[Tuple[float, float]]:
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Uses vector projection to find closest point on line segment.
    
    Args:
        point: (x, y) coordinate to snap.
        line_start: (x, y) start of line segment.
        line_end: (x, y) end of line segment.
        tolerance: Maximum distance for snapping.
    
    Returns:
        Snapped point if within tolerance, None otherwise.
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector from line_start to line_end
    dx = x2 - x1
    dy = y2 - y1
    
    # Vector from line_start to point
    vx = px - x1
    vy = py - y1
    
    # Dot product and squared length
    dot = vx * dx + vy * dy
    len_sq = dx * dx + dy * dy
    
    # Handle zero-length segment
    if len_sq == 0:
        # Line segment is a point
        dist = math.sqrt(vx*vx + vy*vy)
        if dist <= tolerance:
            return (x1, y1)
        return None
    
    # Projection parameter (t) clamped to [0, 1]
    t = max(0, min(1, dot / len_sq))
    
    # Closest point on segment
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    
    # Distance to closest point
    dist_x = px - closest_x
    dist_y = py - closest_y
    distance = math.sqrt(dist_x*dist_x + dist_y*dist_y)
    
    if distance <= tolerance:
        return (closest_x, closest_y)
    return None
```

```python
    If data is within [lower, upper], return 0.
    Otherwise return data unchanged.
    """
    if lower > upper:
        raise ValueError("lower must be less than or equal to upper")
```

```python
import math

def constraint_snap(point, line_start, line_end, tolerance):
    """
    Snap point to line if within tolerance.
    """
    x, y = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Line vector
    dx = x2 - x1
    dy = y2 - y1
    
    # Vector from start to point
    vx = x - x1
    vy = y - y1
    
    # Projection
    if dx == 0 and dy == 0:
        # Line is a point
        dist = math.sqrt(vx**2 + vy**2)
        if dist <= tolerance:
            return (x1, y1)
        return None
    
    t = (vx*dx + vy*dy) / (dx**2 + dy**2)
    
    # Clamp to segment
    if t < 0:
        t = 0
    elif t > 1:
        t = 1
    
    # Closest point
    closest_x = x1 + t*dx
    closest_y = y1 + t*dy
    
    # Distance
    dist = math.sqrt((x - closest_x)**2 + (y - closest_y)**2)
    
    if dist <= tolerance:
        return (closest_x, closest_y)
    return None
```
