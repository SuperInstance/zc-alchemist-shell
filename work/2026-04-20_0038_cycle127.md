from typing import List, Union

def deadband_filter(decisions: List[Union[int, float]]) -> List[float]:
    """
    Apply the fleet's Deadband Protocol to a list of numeric decisions.
    
    Protocol:
    - P0 (negative space): Remove values < -10 (unsafe/forbidden)
    - P1 (safe channel): Keep values -10 <= x <= 10 (acceptable range)
    - P2 (optimization): Scale values > 10 by 0.5 (optimize within channel)
    
    Args:
        decisions: List of numeric decision values
        
    Returns:
        Filtered list with P0 removed, P1 kept, P2 scaled
    """
    filtered = []
    for val in decisions:
        if val < -10:
            continue  # P0: remove unsafe values
        elif -10 <= val <= 10:
            filtered.append(float(val))  # P1: keep as-is
        else:  # val > 10
            filtered.append(val * 0.5)  # P2: scale down for optimization
    return filtered
```

import math
from typing import Tuple, List

def constraint_snap(point: Tuple[float, float], 
                    constraints: List[Tuple[float, float, float]],
                    tolerance: float = 0.1) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest geometric constraint line if within tolerance.
    
    Based on fleet's constraint-theory-core concepts (simplified).
    
    Args:
        point: (x, y) coordinate to snap
        constraints: List of (a, b, c) where line is ax + by + c = 0
        tolerance: Maximum distance for snapping
        
    Returns:
        Snapped (x, y) coordinates, or original if no constraint within tolerance
    """
    x, y = point
    min_distance = float('inf')
    nearest_snap = None
    
    for a, b, c in constraints:
        # Distance from point to line |ax + by + c| / sqrt(a² + b²)
        denominator = math.sqrt(a*a + b*b)
        if denominator == 0:
            continue  # Invalid constraint, skip
        distance = abs(a*x + b*y + c) / denominator
        
        if distance < min_distance and distance <= tolerance:
            min_distance = distance
            # Project point onto line
            t = -(a*x + b*y + c) / (a*a + b*b)
            nearest_snap = (x + a*t, y + b*t)
    
    return nearest_snap if nearest_snap is not None else point
```

from typing import Dict, Any, Tuple
import time

def tile_validator(tile: Dict[str, Any]) -> Tuple[bool, str]:
    """
    Validate a training tile structure against fleet schema.
    
    Required fields:
    - agent_id: str identifier
    - cycle: int > 0
    - content: str (non-empty)
    - room_id: str (format 'room_xxx')
    - timestamp: str (ISO format)
    
    Args:
        tile: Dictionary containing tile data
        
    Returns:
        Tuple of (is_valid: bool, error_message: str)
    """
    required = {'agent_id', 'cycle', 'content', 'room_id', 'timestamp'}
    
    # Check all required fields present
    missing = required - set(tile.keys())
    if missing:
        return False, f"Missing required fields: {missing}"
    
    # Validate types and constraints
    if not isinstance(tile['agent_id'], str) or not tile['agent_id']:
        return False, "agent_id must be non-empty string"
    
    if not isinstance(tile['cycle'], int) or tile['cycle'] <= 0:
        return False, "cycle must be positive integer"
    
    if not isinstance(tile['content'], str) or not tile['content'].strip():
        return False, "content must be non-empty string"
    
    if not isinstance(tile['room_id'], str) or not tile['room_id'].startswith('room_'):
        return False, "room_id must start with 'room_'"
    
    # Validate timestamp (basic ISO format check)
    ts = tile['timestamp']
    if not isinstance(ts, str):
        return False, "timestamp must be string"
    try:
        # Try parsing as ISO format (simplified)
        time.strptime(ts, '%Y-%m-%dT%H:%M:%S')
    except ValueError:
        return False, f"timestamp not valid ISO format: {ts}"
    
    return True, "Tile valid"
