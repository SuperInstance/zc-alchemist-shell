```python
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of steps against the Deadband Protocol.
    Protocol: Must contain P0, then P1, then P2 in order. No phase skipping.
    
    Args:
        steps: List of step strings, e.g., ["P0: map rocks", "P1: find channel"]
    
    Returns:
        Tuple of (is_valid: bool, error_message: str).
        Error message is empty string if valid.
    """
    # Define the required phase order
    required_order = ["P0", "P1", "P2"]
    # Track the highest phase index we've seen
    last_seen_index = -1
    
    for i, step in enumerate(steps):
        step_upper = step.upper()
        # Check for phase marker at start of step
        for phase_idx, phase in enumerate(required_order):
            if step_upper.startswith(phase):
                # Found a phase marker
                if phase_idx < last_seen_index:
                    return False, f"Phase regression at step {i}: '{step}'. {required_order[phase_idx]} appears after {required_order[last_seen_index]}."
                elif phase_idx == last_seen_index:
                    # Same phase repeated is okay
                    continue
                elif phase_idx == last_seen_index + 1:
                    # Correct next phase
                    last_seen_index = phase_idx
                    break
                else:
                    # Skipped a phase (e.g., P0 -> P2)
                    skipped = required_order[last_seen_index + 1:phase_idx]
                    return False, f"Phase skip at step {i}: '{step}'. Skipped {skipped}."
                break
    
    # Check if we reached P2 (optional based on strictness)
    if last_seen_index < 2:
        return False, f"Incomplete protocol. Last phase reached: {required_order[last_seen_index] if last_seen_index >=0 else 'None'}. Need P0->P1->P2."
    
    return True, ""
```

```python
import re
from datetime import datetime

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet training tile text into a structured dictionary.
    Expected format lines like:
        Agent: Alchemist
        Cycle: 145
        Task: Compare models
        Output: <multi-line>
    
    Args:
        tile_text: Raw tile text content.
    
    Returns:
        Dictionary with keys: agent, cycle, phase, task, output, timestamp.
        Missing keys will be None.
    """
    # Default result
    result = {
        "agent": None,
        "cycle": None,
        "phase": None,
        "task": None,
        "output": None,
        "timestamp": datetime.utcnow().isoformat() + "Z"  # default to now
    }
    
    # Simple line-by-line parser
    lines = tile_text.strip().split('\n')
    current_key = None
    output_lines = []
    
    for line in lines:
        # Check for key-value pair (key: value)
        match = re.match(r'^\s*([A-Za-z]+):\s*(.+?)\s*$', line)
        if match:
            # Save any accumulated output lines
            if current_key == "output" and output_lines:
                result["output"] = '\n'.join(output_lines)
                output_lines = []
            
            key, value = match.groups()
            key_lower = key.lower()
            
            if key_lower in result:
                # Handle special cases
                if key_lower == "cycle" or key_lower == "phase":
                    try:
                        result[key_lower] = int(value)
                    except ValueError:
                        result[key_lower] = value
                elif key_lower == "timestamp":
                    # Try to parse, keep as string if fails
                    try:
                        datetime.fromisoformat(value.replace('Z', '+00:00'))
                        result[key_lower] = value
                    except ValueError:
                        result[key_lower] = value
                else:
                    result[key_lower] = value
            current_key = key_lower
        elif current_key == "output":
            # Accumulate output lines
            output_lines.append(line)
        elif line.strip() and not current_key:
            # If no key yet and line isn't empty, treat as start of output
            current_key = "output"
            output_lines.append(line)
    
    # Capture any remaining output
    if current_key == "output" and output_lines:
        result["output"] = '\n'.join(output_lines)
    
    return result
```

```python
import math

def simulate_constraint_snap(points: list[tuple[float, float]], tolerance: float) -> list[tuple[float, float]]:
    """
    Snaps points together if they are within tolerance distance.
    When points snap, they become their average position.
    
    Args:
        points: List of (x, y) coordinate tuples.
        tolerance: Maximum distance for snapping.
    
    Returns:
        New list of snapped points.
    """
    if not points or tolerance <= 0:
        return points.copy()
    
    # Group points into clusters
    clusters = []
    
    for x, y in points:
        added_to_cluster = False
        
        # Try to add to existing cluster
        for cluster in clusters:
            # Check distance to cluster centroid
            cx, cy = cluster["centroid"]
            dx, dy = x - cx, y - cy
            distance = math.sqrt(dx*dx + dy*dy)
            
            if distance <= tolerance:
                # Add to this cluster
                cluster["points"].append((x, y))
                # Update centroid immediately (could defer)
                count = len(cluster["points"])
                cluster["centroid"] = (
                    (cx * (count-1) + x) / count,
                    (cy * (count-1) + y) / count
                )
                added_to_cluster = True
                break
        
        # Create new cluster if not added
        if not added_to_cluster:
            clusters.append({
                "points": [(x, y)],
                "centroid": (x, y)
            })
    
    # Return centroids of clusters
    return [cluster["centroid"] for cluster in clusters]
```
