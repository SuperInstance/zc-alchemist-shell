from typing import List, Dict, Any, Tuple

def validate_deadband_protocol(actions: List[str]) -> Dict[str, Any]:
    """
    Validates a sequence of actions against the Deadband Protocol.
    Protocol: P0 (map negative) must precede P1 (find channels), which must precede P2 (optimize).
    Actions are expected to be strings starting with 'P0', 'P1', or 'P2'.

    Args:
        actions: List of action strings (e.g., ['P0: check rocks', 'P1: scan']).

    Returns:
        Dict with keys:
            is_valid (bool): True if sequence respects P0->P1->P2 order.
            violations (List[Tuple[int, str]]): List of (index, violation_description).
            phase_summary (Dict[str, int]): Count of actions per phase.
    """
    phase_order = {'P0': 0, 'P1': 1, 'P2': 2}
    current_max_phase = -1
    violations = []
    phase_counts = {'P0': 0, 'P1': 0, 'P2': 0}

    for idx, action in enumerate(actions):
        # Extract phase prefix
        phase = None
        for key in phase_order:
            if action.strip().startswith(key):
                phase = key
                break

        if phase is None:
            violations.append((idx, f"Action '{action}' does not start with P0, P1, or P2"))
            continue

        phase_counts[phase] += 1
        phase_num = phase_order[phase]

        # Check order violation
        if phase_num < current_max_phase:
            violations.append((idx, f"Phase {phase} appears after a higher phase (P{current_max_phase})"))
        elif phase_num > current_max_phase:
            current_max_phase = phase_num

    return {
        'is_valid': len(violations) == 0,
        'violations': violations,
        'phase_summary': phase_counts
    }
```

from dataclasses import dataclass
from datetime import datetime
from typing import Optional
import re

@dataclass
class FleetTile:
    agent_name: str
    cycle: int
    phase: int
    task: str
    content: str
    timestamp: datetime

def parse_fleet_tile(tile_text: str) -> FleetTile:
    """
    Parses a raw fleet tile text into a structured FleetTile object.
    Expected format lines:
        'Cycle X — AgentName' or similar header.
        'Phase Y: ...'
        'Task: ...'
        Freeform content.
        Timestamp in ISO-like format.

    Uses regex with fallbacks and defaults.
    """
    # Defaults
    defaults = {
        'agent_name': 'Unknown',
        'cycle': 0,
        'phase': 0,
        'task': '',
        'content': tile_text.strip(),
        'timestamp': datetime.utcnow()
    }

    # Extract agent and cycle
    cycle_match = re.search(r'Cycle\s+(\d+)', tile_text, re.IGNORECASE)
    if cycle_match:
        defaults['cycle'] = int(cycle_match.group(1))

    agent_match = re.search(r'—\s*(\w+)', tile_text)
    if agent_match:
        defaults['agent_name'] = agent_match.group(1)

    # Extract phase
    phase_match = re.search(r'Phase\s+(\d+)', tile_text, re.IGNORECASE)
    if phase_match:
        defaults['phase'] = int(phase_match.group(1))

    # Extract task (look for 'Task:' line)
    task_lines = [line for line in tile_text.split('\n') if line.strip().startswith('Task:')]
    if task_lines:
        defaults['task'] = task_lines[0].replace('Task:', '').strip()

    # Extract timestamp (look for ISO date pattern)
    date_match = re.search(r'(\d{4}-\d{2}-\d{2}[T\s]\d{2}:\d{2})', tile_text)
    if date_match:
        try:
            dt_str = date_match.group(1).replace(' ', 'T')
            defaults['timestamp'] = datetime.fromisoformat(dt_str)
        except ValueError:
            pass  # Keep default

    return FleetTile(**defaults)
```
