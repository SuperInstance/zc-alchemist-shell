from typing import List, Dict, Any, Tuple
import datetime
import math

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply Deadband Protocol to a list of float values.
    
    Args:
        values: List of input float values
        deadband: Deadband threshold (positive)
        
    Returns:
        List of filtered values
        
    Example:
        >>> deadband_filter([1.2, 0.3, -0.6, -1.4], 0.5)
        [0.7, 0.0, -0.1, -0.9]
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    result = []
    for v in values:
        if abs(v) <= deadband:
            result.append(0.0)
        elif v > deadband:
            result.append(v - deadband)
        else:  # v < -deadband
            result.append(v + deadband)
    return result


def parse_fleet_log(log_text: str) -> List[Dict[str, Any]]:
    """
    Parse fleet log text into structured events.
    
    Args:
        log_text: Raw log text with lines in format:
                  "YYYY-MM-DD HH:MM:SS | AGENT | LEVEL | MESSAGE"
                  
    Returns:
        List of parsed log entries as dictionaries
        
    Example:
        >>> log = "2026-04-20 15:30:00 | Alchemist | INFO | Task started"
        >>> parse_fleet_log(log)
        [{'timestamp': datetime.datetime(2026, 4, 20, 15, 30),
          'agent': 'Alchemist',
          'level': 'INFO',
          'message': 'Task started'}]
    """
    import warnings
    
    entries = []
    for line_num, line in enumerate(log_text.strip().split('\n'), 1):
        line = line.strip()
        if not line:
            continue
            
        parts = line.split(' | ')
        if len(parts) != 4:
            warnings.warn(f"Line {line_num}: malformed log entry '{line}'")
            continue
            
        try:
            timestamp = datetime.datetime.strptime(parts[0], '%Y-%m-%d %H:%M:%S')
            entry = {
                'timestamp': timestamp,
                'agent': parts[1].strip(),
                'level': parts[2].strip(),
                'message': parts[3].strip()
            }
            entries.append(entry)
        except ValueError as e:
            warnings.warn(f"Line {line_num}: invalid timestamp '{parts[0]}'")
            continue
            
    return entries


def constraint_snap(point: Tuple[float, float], 
                   constraints: List[Tuple[float, float, float]]) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest constraint line within tolerance.
    
    Args:
        point: (x, y) coordinate to snap
        constraints: List of (A, B, C) where line is Ax + By + C = 0
        
    Returns:
        Snapped (x, y) coordinate, or original if no constraint within tolerance
        
    Example:
        >>> constraint_snap((1.0, 2.0), [(1, 0, -1)])  # line x = 1
        (1.0, 2.0)
    """
    TOLERANCE = 1e-6
    x, y = point
    
    if not constraints:
        return point
    
    best_dist = float('inf')
    best_constraint = None
    
    for A, B, C in constraints:
        # Perpendicular distance
        denominator = math.sqrt(A*A + B*B)
        if denominator == 0:
            continue  # Invalid constraint
            
        distance = abs(A*x + B*y + C) / denominator
        
        if distance < best_dist:
            best_dist = distance
            best_constraint = (A, B, C)
    
    if best_dist > TOLERANCE or best_constraint is None:
        return point
    
    A, B, C = best_constraint
    # Project point onto line
    if A == 0 and B == 0:
        return point
        
    # For line Ax + By + C = 0, projection formula
    t = -(A*x + B*y + C) / (A*A + B*B)
    snapped_x = x + A * t
    snapped_y = y + B * t
    
    return (snapped_x, snapped_y)
