import re
from datetime import datetime
from typing import Optional, Dict

def parse_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured log line.
    Expected format: 'YYYY-MM-DD HH:MM:SS LEVEL [MODULE] Message'
    Returns dict with keys: timestamp, level, module, message.
    Returns None if line doesn't match pattern.
    """
    # Regex matches the standard fleet log format
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (\w+) \[([^\]]+)\] (.+)$'
    match = re.match(pattern, line.strip())
    
    if not match:
        # P0: Don't crash on malformed lines, return None
        return None
    
    timestamp_str, level, module, message = match.groups()
    
    # Validate timestamp can be parsed
    try:
        # Keep as string; could parse to datetime object if needed
        datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
    except ValueError:
        return None
    
    return {
        'timestamp': timestamp_str,
        'level': level,
        'module': module,
        'message': message.strip()
    }

    """
    Parses a log line with format: YYYY-MM-DD HH:MM:SS LEVEL [MODULE] Message
    Returns a dictionary with keys: time, level, module, message.
    If the line doesn't match, returns an empty dict.
    """
    # Regular expression to match the log format

    match = log_pattern.match(log_line.strip())
    if match:

        # Validate timestamp format

            datetime.strptime(timestamp, '%Y-%m-%d %H:%M:%S')
        except ValueError:
            return {}
        return {
            'timestamp': timestamp,
            'level': level,
            'module': module,
            'message': message
        }

        return {}

def validate_deadband_protocol(actions: List[str]) -> Tuple[bool, str]:
    """
    Validate that actions follow Deadband Protocol priority.
    Actions are strings prefixed with P0:, P1:, or P2:.
    Rule: No P2 action may appear until all P0 and P1 actions are complete.
    Returns (is_valid, error_message).
    """
    if not actions:
        return True, "Empty action list is valid"
    
    seen_p2 = False
    for i, action in enumerate(actions):
        if not action.startswith(('P0:', 'P1:', 'P2:')):
            return False, f"Action '{action}' at index {i} lacks valid P-prefix"

            seen_p2 = True

            return False, f"Protocol violation at index {i}: '{action}' found after P2 action"
    
    # Additional check: ensure at least one P0 before any P1? (Not required by doctrine)

    return True, "Protocol valid"

def validate_deadband_protocol(actions):
    """
    Validates a list of actions against the Deadband Protocol.
    Actions should be strings with prefixes P0, P1, or P2.
    The rule: P2 actions cannot appear before all P0 and P1 actions are done.
    Returns a tuple (is_valid, message).
    """
    if not actions:
        return True, "No actions to validate"
    
    p0_done = False
    p1_done = False
    
    for idx, action in enumerate(actions):
        if action.startswith('P0'):
            p0_done = True
        elif action.startswith('P1'):
            if not p0_done:
                return False, f"P1 action '{action}' at index {idx} appears before P0 action"
            p1_done = True
        elif action.startswith('P2'):
            if not (p0_done and p1_done):
                return False, f"P2 action '{action}' at index {idx} appears before P0 and P1 are done"
        else:
            return False, f"Invalid action prefix in '{action}' at index {idx}"
    
    return True, "All actions comply with Deadband Protocol"
