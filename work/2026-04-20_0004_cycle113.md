```python
from typing import List, Optional
import math

def deadband_filter(values: List[float], threshold: float = 1.0) -> List[float]:
    """
    Apply the Deadband Protocol (P0/P1/P2) to filter numeric values.

    P0: Remove invalid values (NaN, infinite, outside safe range -1000 to 1000).
    P1: Keep values within `threshold` of the median of remaining values.
    P2: Sort the final list.

    Args:
        values: List of numeric values to filter.
        threshold: Deadband threshold for P1 filtering (default 1.0).

    Returns:
        Filtered and sorted list of values.

    Example:
        >>> deadband_filter([1.0, 2.0, 10000.0, float('nan'), 1.5, 2.5], 0.6)
        [1.5, 2.0]
    """
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    if not isinstance(threshold, (int, float)):
        raise TypeError("threshold must be numeric")
    if threshold < 0:
        raise ValueError("threshold must be non-negative")

    # P0: Remove rocks
    safe_vals = []
    for v in values:
        if not isinstance(v, (int, float)):
            continue
        if math.isnan(v) or math.isinf(v):
            continue
        if v < -1000 or v > 1000:
            continue
        safe_vals.append(v)

    if not safe_vals:
        return []

    # P1: Deadband around median
    sorted_vals = sorted(safe_vals)
    n = len(sorted_vals)
    median = sorted_vals[n // 2] if n % 2 == 1 else (sorted_vals[n // 2 - 1] + sorted_vals[n // 2]) / 2

    filtered = [v for v in sorted_vals if abs(v - median) <= threshold]

    # P2: Sort (already sorted from median calculation, but ensure)
    filtered.sort()
    return filtered
```

```python
    """
    Apply Deadband Protocol (P0/P1/P2) filtering to a list of numeric values.

    P0: Remove invalid values (NaN, Inf, outside [-1000, 1000]).
    P1: Keep values within `threshold` distance from median of remaining values.
    P2: Sort final values.

    Args:
        values (List[float]): Input list of floats.
        threshold (float): Deadband threshold for P1 step.

    Returns:
        List[float]: Filtered, sorted list.

    Example:
        >>> deadband_filter([1.0, float('nan'), 1001.0, 2.0, 1.8], 0.5)
        [1.8, 2.0]
    """
    # Input validation
    if not isinstance(values, list):
        raise TypeError("Input must be a list")
    if not isinstance(threshold, (int, float)):
        raise TypeError("Threshold must be a number")
    if threshold < 0:
        raise ValueError("Threshold must be non-negative")

    # P0: Remove rocks
    safe = []
    for v in values:
        if not isinstance(v, (int, float)):
            continue
        if math.isnan(v) or math.isinf(v):
            continue
        if v < -1000 or v > 1000:
            continue
        safe.append(v)

    if len(safe) == 0:
        return []

    # P1: Deadband around median
    safe_sorted = sorted(safe)
    n = len(safe_sorted)
    if n % 2 == 1:
        median = safe_sorted[n // 2]
    else:
        median = (safe_sorted[n // 2 - 1] + safe_sorted[n // 2]) / 2

    in_band = [v for v in safe_sorted if abs(v - median) <= threshold]

    # P2: Sort (already sorted, but ensure)
    in_band.sort()
    return in_band
```
