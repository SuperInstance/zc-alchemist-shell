from typing import Tuple, List, Optional

def deadband_protocol_simulator(
    sensor_readings: List[float],
    p0_threshold: float,
    p1_lower: float,
    p1_upper: float,
    p2_target: float
) -> List[Tuple[float, str, Optional[float]]]:
    """
    Simulate P0/P1/P2 decision flow for a list of sensor readings.
    
    P0: If reading <= p0_threshold, avoid (negative space).
    P1: If reading between p1_lower and p1_upper, safe channel.
    P2: If reading in P1 range, optimize toward p2_target.
    
    Args:
        sensor_readings: List of float sensor values.
        p0_threshold: Values <= this are P0 (unsafe).
        p1_lower: Lower bound of safe channel.
        p1_upper: Upper bound of safe channel.
        p2_target: Target value for optimization within P1.
    
    Returns:
        List of tuples (reading, phase, recommended_action).
        phase: "P0", "P1", or "P2".
        recommended_action: None for P0, current reading for P1,
            delta to target for P2.
    
    Example:
        >>> readings = [0.5, 1.5, 2.5, 3.5]
        >>> deadband_protocol_simulator(readings, 0.8, 1.0, 3.0, 2.0)
        [(0.5, 'P0', None), (1.5, 'P2', -0.5), (2.5, 'P2', 0.5), (3.5, 'P1', 3.5)]
    """
    results = []
    
    for reading in sensor_readings:
        # P0: Negative space
        if reading <= p0_threshold:
            results.append((reading, "P0", None))
            continue
        
        # P1: Safe channel check
        if p1_lower <= reading <= p1_upper:
            # P2: Optimization possible
            if p1_lower <= p2_target <= p1_upper:
                delta = p2_target - reading
                results.append((reading, "P2", delta))
            else:
                # Target outside P1, just report P1
                results.append((reading, "P1", reading))
        else:
            # Outside P1 but above P0 is undefined in protocol
            results.append((reading, "P1", reading))
    
    return results

def deadband_protocol_simulator(
    readings: List[float],
    p0_thresh: float,
    p1_low: float,
    p1_high: float,
    p2_target: float
) -> List[Tuple[float, str, Optional[float]]]:
    """
    Simulates the Deadband Protocol (P0, P1, P2) on sensor readings.
    
    Parameters:
        readings: list of sensor values
        p0_thresh: threshold for P0 (avoid)
        p1_low: lower bound of P1 safe channel
        p1_high: upper bound of P1 safe channel
        p2_target: target value for P2 optimization
    
    Returns:
        list of (value, phase, action) where:
            phase: 'P0', 'P1', or 'P2'
            action: None for P0, value for P1, delta for P2
    
    Example:
        >>> deadband_protocol_simulator([0.5, 1.2, 2.8], 0.8, 1.0, 3.0, 2.0)
        [(0.5, 'P0', None), (1.2, 'P2', 0.8), (2.8, 'P2', -0.8)]
    """
    result = []
    
    for val in readings:
        # P0 check
        if val <= p0_thresh:
            result.append((val, 'P0', None))
            continue
        
        # P1 safe channel check
        if p1_low <= val <= p1_high:
            # Check if target is within P1
            if p1_low <= p2_target <= p1_high:
                delta = p2_target - val
                result.append((val, 'P2', delta))
            else:
                result.append((val, 'P1', val))
        else:
            # Outside P1 range
            result.append((val, 'P1', val))
    
    return result
