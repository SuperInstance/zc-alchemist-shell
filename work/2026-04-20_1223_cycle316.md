# Function A: Deadband Protocol Validator
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """Validate sequence follows P0→P1→P2 with no skips."""
    valid_sequence = ["P0", "P1", "P2"]
    for i, step in enumerate(steps):
        if step not in valid_sequence:
            return False, f"Invalid step '{step}' at index {i}. Must be P0, P1, or P2."
        if i > 0:
            prev_idx = valid_sequence.index(steps[i-1])
            curr_idx = valid_sequence.index(step)
            if curr_idx != prev_idx + 1:
                return False, f"Protocol violation at index {i}: {steps[i-1]}→{step}. Steps must be sequential."
    return True, "Valid Deadband Protocol sequence."

# Function B: Fleet Tile Parser
def parse_fleet_tile(tile_text: str) -> dict:
    """Parse a training tile text into structured dict."""
    import re
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "content": "",
        "tags": []
    }
    
    # Extract metadata lines
    lines = tile_text.strip().split('\n')
    for line in lines:
        line_lower = line.lower()
        if line_lower.startswith("cycle:"):
            match = re.search(r'\d+', line)
            if match:
                result["cycle"] = int(match.group())
        elif line_lower.startswith("agent:"):
            result["agent"] = line.split(":", 1)[1].strip()
        elif line_lower.startswith("phase:"):
            match = re.search(r'\d+', line)
            if match:
                result["phase"] = int(match.group())
        elif line_lower.startswith("tags:"):
            tags_part = line.split(":", 1)[1].strip()
            result["tags"] = [t.strip() for t in tags_part.split(",") if t.strip()]
    
    # Content is everything else
    content_lines = []
    for line in lines:
        if not any(line.lower().startswith(prefix) for prefix in ["cycle:", "agent:", "phase:", "tags:"]):
            content_lines.append(line)
    result["content"] = "\n".join(content_lines).strip()
    
    return result

# Function C: Constraint Snapping
def constraint_snap(points: list[tuple[float, float]], tolerance: float = 0.01) -> list[tuple[float, float]]:
    """Snap points within tolerance to first point in cluster."""
    if not points:
        return []
    
    snapped = []
    for point in points:
        found = False
        for ref_point in snapped:
            # Euclidean distance
            dist = ((point[0] - ref_point[0])**2 + (point[1] - ref_point[1])**2)**0.5
            if dist <= tolerance:
                found = True
                break
        if not found:
            snapped.append(point)
    return snapped

# Function A: Deadband Protocol Validator
def validate_deadband_protocol(steps):
    """
    Checks if the steps list follows the Deadband Protocol.
    Returns (is_valid, message).
    """
    if not isinstance(steps, list):
        return False, "Input must be a list"
    
    allowed = {"P0", "P1", "P2"}
    order = ["P0", "P1", "P2"]
    
    for idx, step in enumerate(steps):
        if step not in allowed:
            return False, f"Step {step} at index {idx} is not allowed"
    
    # Check ordering
    last_index = -1
    for step in steps:
        current_index = order.index(step)
        if current_index <= last_index:
            return False, f"Step {step} appears out of order. Must follow P0->P1->P2."
        last_index = current_index
    
    return True, "Valid"

# Function B: Fleet Tile Parser
import re

def parse_fleet_tile(tile_text):
    default = {"cycle": 0, "agent": "unknown", "phase": 0, "content": "", "tags": []}
    
    cycle_match = re.search(r"cycle:\s*(\d+)", tile_text, re.IGNORECASE)
    if cycle_match:
        default["cycle"] = int(cycle_match.group(1))
    
    agent_match = re.search(r"agent:\s*(\w+)", tile_text, re.IGNORECASE)
    if agent_match:
        default["agent"] = agent_match.group(1)
    
    phase_match = re.search(r"phase:\s*(\d+)", tile_text, re.IGNORECASE)
    if phase_match:
        default["phase"] = int(phase_match.group(1))
    
    tags_match = re.search(r"tags:\s*([^\n]+)", tile_text, re.IGNORECASE)
    if tags_match:
        tags_str = tags_match.group(1)
        default["tags"] = [t.strip() for t in tags_str.split(",")]
    
    # Extract content: remove metadata lines
    lines = tile_text.split('\n')
    content_lines = []
    for line in lines:
        if not re.match(r'^(cycle|agent|phase|tags):', line, re.IGNORECASE):
            content_lines.append(line)
    default["content"] = '\n'.join(content_lines).strip()
    
    return default

# Function C: Constraint Snapping
import math

def constraint_snap(points, tolerance=0.01):
    snapped_points = []
    
    for x, y in points:
        snapped = False
        for sx, sy in snapped_points:
            distance = math.sqrt((x - sx)**2 + (y - sy)**2)
            if distance <= tolerance:
                snapped = True
                break
        if not snapped:
            snapped_points.append((x, y))
    
    return snapped_points
