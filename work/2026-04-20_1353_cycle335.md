### **Function 1: deadband_filter**

```python
def deadband_filter(data, lower, upper):
    """
    Apply a deadband filter to input data.
    
    If data is within the deadband range [lower, upper] (inclusive),
    return 0. Otherwise, return the original data.
    
    Args:
        data (float): Input value to filter
        lower (float): Lower bound of deadband
        upper (float): Upper bound of deadband
    
    Returns:
        float: 0 if within deadband, else original data
    
    Examples:
        >>> deadband_filter(5, 2, 8)
        0
        >>> deadband_filter(10, 2, 8)
        10
    """
    if not isinstance(data, (int, float)):
        raise TypeError("data must be numeric")
    if lower > upper:
        lower, upper = upper, lower  # Auto-correct range
    
    if lower <= data <= upper:
        return 0.0
    return float(data)
```

### **Function 2: parse_fleet_log_line**
```python
import re
from datetime import datetime

def parse_fleet_log_line(line):
    """
    Parse a structured fleet log line.
    
    Expected format: [TIMESTAMP] SERVICE: LEVEL - message
    Example: [2026-04-20 14:30:00] plato-torch: INFO - Training room started
    
    Args:
        line (str): Log line to parse
    
    Returns:
        dict: Parsed components with keys: timestamp, service, level, message
              Returns None if line doesn't match format
    """
    if not line or not isinstance(line, str):
        return None
    
    pattern = r'\[([^\]]+)\]\s+([^:]+):\s+(\w+)\s+-\s+(.+)'
    match = re.match(pattern, line.strip())
    
    if not match:
        return None
    
    timestamp_str, service, level, message = match.groups()
    
    # Try to parse timestamp
    timestamp = None
    try:
        timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
    except ValueError:
        timestamp = timestamp_str  # Keep as string if parsing fails
    
    return {
        'timestamp': timestamp,
        'service': service.strip(),
        'level': level.strip(),
        'message': message.strip()
    }
```

### **Function 3: validate_constraint_snapping**
```python
import math

def validate_constraint_snapping(points, tolerance):
    """
    Check if any two points are within snapping tolerance.
    
    Args:
        points (list): List of (x, y) tuples
        tolerance (float): Maximum distance for snapping
    
    Returns:
        bool: True if any pair within tolerance, False otherwise
    """
    if not points or len(points) < 2:
        return False
    
    if tolerance < 0:
        tolerance = abs(tolerance)
    
    n = len(points)
    for i in range(n):
        x1, y1 = points[i]
        for j in range(i + 1, n):
            x2, y2 = points[j]
            distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
            if distance <= tolerance:
                return True
    
    return False
```

```python
import math

def validate_constraint_snapping(points, tolerance):
    """
    Validates if any two points are within a given tolerance.
    
    Parameters:
    points (list of tuples): List of (x, y) points.
    tolerance (float): The maximum allowed distance between points.
    
    Returns:
    bool: True if any two points are within tolerance, False otherwise.
    """
    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            x1, y1 = points[i]
            x2, y2 = points[j]
            distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
            if distance <= tolerance:
                return True
    return False
```
