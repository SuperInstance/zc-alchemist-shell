**DeepSeek-chat implementation:**

def deadband_filter(values, p0_threshold=0.1, p1_margin=0.2):
    """
    Apply Deadband Protocol to a list of float values.
    
    P0: Values below threshold are unsafe (negative space)
    P1: Values within [threshold, threshold+margin] are safe channels
    P2: Values above threshold+margin are optimization zones
    
    Args:
        values: List of float values
        p0_threshold: Minimum safe value (P0 boundary)
        p1_margin: Additional margin for P1 safe channel
        
    Returns:
        dict with P0, P1, P2 lists and classification counts
    """
    if not values:
        return {"P0": [], "P1": [], "P2": [], "counts": {"P0": 0, "P1": 0, "P2": 0}}
    
    p0_vals = []
    p1_vals = []
    p2_vals = []
    
    p1_max = p0_threshold + p1_margin
    
    for v in values:
        if v < p0_threshold:
            p0_vals.append(v)
        elif v <= p1_max:
            p1_vals.append(v)
        else:
            p2_vals.append(v)
    
    return {
        "P0": p0_vals,
        "P1": p1_vals,
        "P2": p2_vals,
        "counts": {
            "P0": len(p0_vals),
            "P1": len(p1_vals),
            "P2": len(p2_vals)
        }
    }
```

def deadband_filter(values: list[float], p0_threshold: float = 0.1, p1_margin: float = 0.2) -> dict:
    """
    Implements the Deadband Protocol (P0/P1/P2) for signal processing.
    
    Protocol:
    - P0 (negative space): values < p0_threshold (unsafe, avoid)
    - P1 (safe channel): p0_threshold <= values <= p0_threshold + p1_margin
    - P2 (optimization): values > p0_threshold + p1_margin
    
    Parameters:
    -----------
    values : list[float]
        Input values to classify
    p0_threshold : float
        Minimum value for safe operation (P0 boundary)
    p1_margin : float
        Width of the safe channel (P1 region)
        
    Returns:
    --------
    dict with keys:
        - "P0": list of values in negative space
        - "P1": list of values in safe channel  
        - "P2": list of values in optimization zone
        - "summary": dict with counts and percentages
    """
    # Input validation
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    
    p0_list = []
    p1_list = []
    p2_list = []
    
    p1_upper = p0_threshold + p1_margin
    
    for val in values:
        if val < p0_threshold:
            p0_list.append(val)
        elif val <= p1_upper:
            p1_list.append(val)
        else:
            p2_list.append(val)
    
    total = len(values)
    return {
        "P0": p0_list,
        "P1": p1_list, 
        "P2": p2_list,
        "summary": {
            "total": total,
            "P0_count": len(p0_list),
            "P1_count": len(p1_list),
            "P2_count": len(p2_list),
            "P0_percent": len(p0_list) / total * 100 if total > 0 else 0,
            "P1_percent": len(p1_list) / total * 100 if total > 0 else 0,
            "P2_percent": len(p2_list) / total * 100 if total > 0 else 0
        }
    }

import json
from datetime import datetime

def validate_tile(tile_data, strict_mode=True):
    """
    Validate a training tile against fleet standards.
    
    Required fields:
    - agent_id: string identifier
    - cycle: integer
    - content: string (non-empty)
    - timestamp: ISO format string
    - tile_type: one of ['code', 'doc', 'analysis', 'test']
    
    Optional but recommended:
    - phase: integer
    - tags: list of strings
    - dependencies: list of tile IDs
    
    Returns:
        (is_valid, errors, warnings)
    """
    if not isinstance(tile_data, dict):
        return False, ["Tile must be a dictionary"], []
    
    errors = []
    warnings = []
    
    # Required fields
    required = ['agent_id', 'cycle', 'content', 'timestamp', 'tile_type']
    for field in required:
        if field not in tile_data:
            errors.append(f"Missing required field: {field}")
    
    if errors:
        return False, errors, warnings
    
    # Validate types and content
    if not isinstance(tile_data['agent_id'], str) or not tile_data['agent_id']:
        errors.append("agent_id must be non-empty string")
    
    if not isinstance(tile_data['cycle'], int) or tile_data['cycle'] < 0:
        errors.append("cycle must be non-negative integer")
    
    if not isinstance(tile_data['content'], str) or not tile_data['content'].strip():
        errors.append("content must be non-empty string")
    
    # Validate timestamp
    try:
        datetime.fromisoformat(tile_data['timestamp'].replace('Z', '+00:00'))
    except (ValueError, TypeError):
        errors.append("timestamp must be valid ISO format")
    
    # Validate tile_type
    valid_types = ['code', 'doc', 'analysis', 'test']
    if tile_data['tile_type'] not in valid_types:
        errors.append(f"tile_type must be one of {valid_types}")
    
    # Optional field validations
    if 'phase' in tile_data and not isinstance(tile_data['phase'], int):
        warnings.append("phase should be integer")
    
    if 'tags' in tile_data:
        if not isinstance(tile_data['tags'], list):
            warnings.append("tags should be a list")
        elif any(not isinstance(tag, str) for tag in tile_data['tags']):
            warnings.append("all tags should be strings")
    
    if strict_mode and warnings:
        # In strict mode, warnings become errors
        errors.extend(warnings)
        warnings = []
    
    return len(errors) == 0, errors, warnings
```
