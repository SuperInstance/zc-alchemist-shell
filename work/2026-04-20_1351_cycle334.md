import re
from datetime import datetime
import hashlib

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of Deadband Protocol steps.
    Fleet Doctrine: Strict priority P0 -> P1 -> P2. No skips, no regressions.
    
    Args:
        steps: List of step strings, e.g., ["P0", "P1", "P2"]
    
    Returns:
        Tuple of (is_valid, error_message)
    """
    valid_steps = {"P0", "P1", "P2"}
    priority_order = {"P0": 0, "P1": 1, "P2": 2}
    
    if not steps:
        return False, "Empty step sequence."
    
    # Check all steps are valid
    for step in steps:
        if step not in valid_steps:
            return False, f"Invalid step '{step}'. Must be P0, P1, or P2."
    
    # Check starts with P0
    if steps[0] != "P0":
        return False, "Sequence must start with P0."
    
    # Check for skips or regressions
    for i in range(1, len(steps)):
        current = priority_order[steps[i]]
        previous = priority_order[steps[i-1]]
        
        if current < previous:
            return False, f"Protocol regression at step {i}: {steps[i-1]} -> {steps[i]}. Cannot go backwards."
        if current > previous + 1:
            return False, f"Protocol skip at step {i}: {steps[i-1]} -> {steps[i]}. Cannot skip priorities."
    
    return True, "Valid Deadband Protocol sequence."

def parse_fleet_health_log(log_text: str) -> dict:
    """
    Parses a fleet service health log, counting warnings and errors per service.
    
    Log line format assumed:
    [LEVEL] service_name - message
    
    Args:
        log_text: Multiline string of log entries.
    
    Returns:
        Dictionary: {service_name: {"warnings": int, "errors": int, "ok": int}}
    """
    services = {}
    # Simple regex to capture level, service name, and message
    log_pattern = r'\[(\w+)\]\s+(\S+)\s+-\s+(.+)'
    
    for line in log_text.strip().split('\n'):
        match = re.match(log_pattern, line.strip())
        if not match:
            continue  # Skip malformed lines
        
        level, service, message = match.groups()
        level = level.upper()
        
        if service not in services:
            services[service] = {"warnings": 0, "errors": 0, "ok": 0}
        
        if level == "WARN":
            services[service]["warnings"] += 1
        elif level == "ERROR":
            services[service]["errors"] += 1
        elif level == "OK":
            services[service]["ok"] += 1
        # Other levels are ignored for this count
    
    return services

def generate_training_tile(agent_role: str, work_output: str) -> dict:

    Generates a structured training tile dictionary from agent work.
    
    Args:
        agent_role: The role of the agent (e.g., "Alchemist").
        work_output: The string content of the work.
    
    Returns:
        Dictionary representing a PLATO training tile.
    """
    # Simulate a cycle ID - in practice this would come from fleet state
    current_cycle = 334  # This would be dynamic
    timestamp = datetime.utcnow().isoformat() + "Z"
    
    # Generate simple tags from content (basic keyword extraction)
    words = work_output.lower().split()
    # Filter common words and take top 5 unique
    common = {"the", "a", "an", "and", "or", "in", "on", "at", "to", "for", "of", "is", "are"}
    candidates = [w for w in words if w not in common and len(w) > 3]
    # Simple heuristic: words that appear multiple times or look like tech terms
    tags = list(set(candidates[:5])) or ["fleet", "training"]
    
    # Create a simple checksum of the content
    checksum = hashlib.md5(work_output.encode()).hexdigest()[:8]
    
    tile = {
        "role": agent_role,
        "cycle": current_cycle,
        "timestamp": timestamp,
        "content": work_output,
        "tags": tags,
        "checksum": f"md5:{checksum}"
    }
    return tile
