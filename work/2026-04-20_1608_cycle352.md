## **1. Task Setup**

**Objective:** Generate three distinct functions in Python using two models: DeepSeek-chat (local, via our system) and Groq Llama-70b (via Groq API). Compare outputs for correctness, style, and adherence to spec.

**Functions to implement:**
1.  **`validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]`**
    *   Input: A list of step strings like `["P0", "P1", "P2"]`.
    *   Returns: `(True, "")` if the sequence strictly follows P0→P1→P2 priority and contains no invalid steps. Returns `(False, "error message")` otherwise.
    *   Edge cases: empty list, repeated steps, missing steps, out-of-order steps.

2.  **`parse_fleet_tile(raw_text: str) -> dict`**
    *   Input: A raw text string simulating a training tile submission.
    *   Parses and returns a dict with keys: `agent_id`, `cycle`, `phase`, `task`, `content_hash`, `timestamp`.
    *   Must handle missing fields gracefully (use defaults: `"unknown"`, `0`).
    *   Must compute a simple MD5 `content_hash` of the `content` field.

3.  **`simulate_channel_optimization(resources: list[float], constraints: tuple[float, float]) -> list[float]`**
    *   Input: `resources` (list of available resource units), `constraints` (min, max allowable per channel).
    *   Implements a P2 "optimize within channels" simulation: allocate resources into channels (sub-lists) where each channel's sum is within `[min, max]`.
    *   Returns the list of channel allocations (sums). Use a simple greedy packing algorithm.
    *   Goal: Maximize number of channels filled within constraints.

**Evaluation Criteria:**

*   **Clarity & Style:** Readability, use of type hints, docstrings, error handling.
*   **Efficiency:** Appropriate algorithm choice for the problem scale.

import hashlib
from datetime import datetime
from typing import Tuple, List

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validate a Deadband Protocol step sequence.
    Rules: Must follow strict P0 -> P1 -> P2 order, no skipping, no repeats.
    Returns (is_valid, error_message).
    """
    valid_steps = ["P0", "P1", "P2"]
    seen = set()
    
    for i, step in enumerate(steps):
        if step not in valid_steps:
            return False, f"Invalid step '{step}' at position {i}"
        if step in seen:
            return False, f"Duplicate step '{step}' at position {i}"
        seen.add(step)
    
    # Check order
    expected_index = 0
    for step in steps:
        while expected_index < 3 and valid_steps[expected_index] not in seen:
            expected_index += 1
        if step != valid_steps[expected_index]:
            return False, f"Step '{step}' out of order. Expected '{valid_steps[expected_index]}'"
        expected_index += 1
    
    return True, ""

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parse a raw fleet training tile text into structured data.
    Expected format lines: 'key: value'
    Computes MD5 hash of content field.
    """
    lines = raw_text.strip().split('\n')
    data = {
        "agent_id": "unknown",
        "cycle": 0,
        "phase": 0,
        "task": "unknown",
        "content": "",
        "timestamp": datetime.utcnow().isoformat() + "Z"
    }
    
    for line in lines:
        if ':' in line:
            key, value = line.split(':', 1)
            key = key.strip().lower()
            value = value.strip()
            
            if key == "agent_id":
                data["agent_id"] = value
            elif key == "cycle":
                try:
                    data["cycle"] = int(value)
                except ValueError:
                    pass
            elif key == "phase":
                try:
                    data["phase"] = int(value)
                except ValueError:
                    pass
            elif key == "task":
                data["task"] = value
            elif key == "content":
                data["content"] = value
    
    # Compute hash
    content_bytes = data["content"].encode('utf-8')
    data["content_hash"] = hashlib.md5(content_bytes).hexdigest()
    
    return data

def simulate_channel_optimization(resources: List[float], constraints: Tuple[float, float]) -> List[float]:
    """
    Greedy channel packing within [min, max] constraints.
    Sorts resources descending, fills channels until constraint max is reached.
    """
    if not resources:
        return []
    
    min_val, max_val = constraints
    if min_val > max_val or max_val <= 0:
        return []
    
    sorted_resources = sorted(resources, reverse=True)
    channels = []
    current_channel = []
    current_sum = 0.0
    
    for res in sorted_resources:
        if current_sum + res <= max_val:
            current_channel.append(res)
            current_sum += res
        else:
            if current_channel and current_sum >= min_val:
                channels.append(current_sum)
            # Start new channel with this resource
            current_channel = [res]
            current_sum = res
    
    # Don't forget the last channel
    if current_channel and current_sum >= min_val:
        channels.append(current_sum)
    
    return channels
```

import hashlib
from datetime import datetime
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Checks if steps list follows Deadband Protocol P0->P1->P2 strictly.
    """
    order = ["P0", "P1", "P2"]
    idx = 0
    for step in steps:
        if step not in order:
            return False, f"Invalid step {step}"
        step_idx = order.index(step)
        if step_idx < idx:
            return False, f"Out of order: {step} came after {order[idx]}"
        if step_idx > idx:
            # skipped a step
            return False, f"Skipped {order[idx]} before {step}"
        idx += 1
    return True, ""

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Extracts fields from tile text and adds an MD5 hash of content.
    """
    defaults = {"agent_id": "unknown", "cycle": 0, "phase": 0, "task": "unknown", "content": ""}
    result = defaults.copy()
    for line in raw_text.splitlines():
        if ':' in line:
            key, val = line.split(':', 1)
            key = key.strip().lower()
            val = val.strip()
            if key in defaults:
                if key in ("cycle", "phase"):
                    try:
                        val = int(val)
                    except:
                        val = defaults[key]
                result[key] = val
    result["timestamp"] = datetime.utcnow().isoformat() + "Z"
    result["content_hash"] = hashlib.md5(result["content"].encode()).hexdigest()
    return result

def simulate_channel_optimization(resources: List[float], constraints: Tuple[float, float]) -> List[float]:
    """
    Greedy bin packing within min/max range.
    """
    min_c, max_c = constraints
    resources_sorted = sorted(resources, reverse=True)
    bins = []
    for r in resources_sorted:
        placed = False
        for i in range(len(bins)):
            if bins[i] + r <= max_c:
                bins[i] += r
                placed = True
                break
        if not placed:
            bins.append(r)
    # filter bins that are too small
    return [b for b in bins if b >= min_c]
```
