import re
import numpy as np
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validate that steps follow strict P0→P1→P2 priority.
    
    Rules:
    1. Only "P0", "P1", "P2" allowed.
    2. Must start with P0.
    3. No skipping: P0→P1→P2 only.
    4. No repeats: each step appears at most once.
    
    Returns:
        (True, "") if valid, (False, error_message) otherwise.
    """
    valid_steps = {"P0", "P1", "P2"}
    
    if not steps:
        return False, "Empty step list"
    
    # Check all steps are valid
    for i, step in enumerate(steps):
        if step not in valid_steps:
            return False, f"Invalid step '{step}' at position {i}"
    
    # Check start
    if steps[0] != "P0":
        return False, "Must start with P0"
    
    # Check order and no repeats
    seen = set()
    expected_order = ["P0", "P1", "P2"]
    expected_index = 0
    
    for step in steps:
        if step in seen:
            return False, f"Duplicate step '{step}'"
        seen.add(step)
        
        # Find expected step for current position
        while expected_index < len(expected_order) and expected_order[expected_index] not in valid_steps:
            expected_index += 1
        
        if expected_index >= len(expected_order):
            return False, "Too many steps"
        
        if step != expected_order[expected_index]:
            return False, f"Expected '{expected_order[expected_index]}' but got '{step}'"
        
        expected_index += 1
    
    return True, ""

def extract_tile_metadata(log_text: str) -> dict:
    """
    Extract tile metadata from PLATO room submission log.
    
    Expected format:
    [YYYY-MM-DD HH:MM] AgentName@cycleXXX submitted tile HASH to room ROOM-ID
    
    Returns dict with keys: agent_name, cycle, timestamp, tile_hash, room_id
    Values are None if not found.
    """
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2})\] (\w+)@cycle(\d+) submitted tile (\w+) to room ([\w-]+)'
    match = re.search(pattern, log_text)
    
    if not match:
        # Return empty dict with Nones
        return {
            "agent_name": None,
            "cycle": None,
            "timestamp": None,
            "tile_hash": None,
            "room_id": None
        }
    
    timestamp, agent_name, cycle, tile_hash, room_id = match.groups()
    
    return {
        "agent_name": agent_name,
        "cycle": int(cycle),
        "timestamp": timestamp,
        "tile_hash": tile_hash,
        "room_id": room_id
    }

def batch_normalize_vectors(vectors: List[List[float]]) -> List[List[float]]:
    """
    L2-normalize a batch of vectors.
    
    For each vector v:
        norm = sqrt(sum(v_i^2))
        if norm > 0: normalized = v / norm
        else: normalized = v (zero vector unchanged)
    
    Uses NumPy if available for performance.
    """
    try:
        import numpy as np
        use_numpy = True
    except ImportError:
        use_numpy = False
    
    if not vectors:
        return []
    
    if use_numpy:
        np_vectors = np.array(vectors, dtype=np.float64)
        norms = np.linalg.norm(np_vectors, axis=1, keepdims=True)
        # Avoid division by zero
        norms[norms == 0] = 1
        normalized = np_vectors / norms
        return normalized.tolist()
    else:
        # Pure Python fallback
        result = []
        for vec in vectors:
            norm_sq = sum(x * x for x in vec)
            if norm_sq > 0:
                norm = norm_sq ** 0.5
                normalized = [x / norm for x in vec]
            else:
                normalized = vec[:]  # Copy zero vector
            result.append(normalized)
        return result
