```python
# Function 1: Deadband filter
def deadband_filter(data, lower, upper):
    """
    Apply a deadband filter to input data.
    
    Args:
        data: Numeric value or list of values
        lower: Lower bound of deadband
        upper: Upper bound of deadband
        
    Returns:
        Filtered value(s). Returns 0 if within [lower, upper],
        otherwise returns original value.
    """
    import numpy as np
    
    # Handle single value
    if isinstance(data, (int, float)):
        if lower <= data <= upper:
            return 0
        return data
    
    # Handle list/array
    data = np.asarray(data)
    result = np.zeros_like(data)
    mask = (data < lower) | (data > upper)
    result[mask] = data[mask]
    return result.tolist() if isinstance(data, list) else result
```

```python
def parse_fleet_log_line(line):
    """
    Parse a structured fleet log line into dictionary.
    
    Expected format: "YYYY-MM-DD HH:MM:SS [LEVEL] key1=value1 key2=value2"
    
    Args:
        line (str): Log line
        
    Returns:
        dict: Parsed components or empty dict if format invalid
    """
    import re
    
    if not line or not isinstance(line, str):
        return {}
    
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\] (.*)'
    match = re.match(pattern, line)
    
    if not match:
        return {}
    
    timestamp, level, rest = match.groups()
    result = {'timestamp': timestamp, 'level': level}
    
    # Parse key=value pairs
    kv_pattern = r'(\w+)=([^\s]+)'
    for key, value in re.findall(kv_pattern, rest):
        # Try to convert numeric values
        try:
            if value.isdigit():
                value = int(value)
            elif re.match(r'^\d+\.\d+$', value):
                value = float(value)
        except:
            pass
        result[key] = value
    
    return result
```

```python
# Function 3: Constraint snapping
def constraint_snap(point, constraints):
    """
    Snap a 2D point to the closest constraint line.
    
    Args:
        point: Tuple (x, y)
        constraints: List of line constraints, each as ((x1,y1), (x2,y2))
        
    Returns:
        Tuple (x, y) of closest point on any constraint line,
        or original point if no constraints.
    """
    import math
    
    if not constraints:
        return point
    
    px, py = point
    min_dist = float('inf')
    closest_point = point
    
    for (x1, y1), (x2, y2) in constraints:
        # Line vector
        dx, dy = x2 - x1, y2 - y1
        line_len_sq = dx*dx + dy*dy
        
        if line_len_sq == 0:  # Zero-length segment
            dist = math.hypot(px - x1, py - y1)
            candidate = (x1, y1)
        else:
            # Project point onto line
            t = ((px - x1) * dx + (py - y1) * dy) / line_len_sq
            t = max(0, min(1, t))  # Clamp to segment
            cx = x1 + t * dx
            cy = y1 + t * dy
            candidate = (cx, cy)
            dist = math.hypot(px - cx, py - cy)
        
        if dist < min_dist:
            min_dist = dist
            closest_point = candidate
    
    return closest_point
```

```python
# Function 2: Fleet log parser  
def parse_fleet_log_line(log_line):
    """
    Parses fleet log lines like:
    "2026-04-19 12:34:56 [INFO] agent=Alchemist cycle=25 tile_accepted"
    
    Returns dict with timestamp, level, and key-value pairs.
    """
    import re
    
    # Split into timestamp, level, and message parts
    parts = re.match(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(\w+)\] (.*)', log_line)
    if not parts:
        return {}
    
    timestamp, level, message = parts.groups()
    result = {'timestamp': timestamp, 'level': level}
    
    # Extract key=value pairs
    for pair in message.split():
        if '=' in pair:
            key, val = pair.split('=', 1)
            result[key] = val
    
    return result
```

```python
# Function 3: Constraint snapping
def constraint_snap(point, lines):
    """
    Snaps a point to the nearest line constraint.
    
    point: (x, y) tuple
    lines: list of ((x1, y1), (x2, y2)) tuples
    
    Returns the closest point on any line segment.
    """
    import numpy as np
    
    best_point = point
    best_dist = float('inf')
    
    for (x1, y1), (x2, y2) in lines:
        # Convert to numpy for vector math
        p = np.array(point)
        a = np.array([x1, y1])
        b = np.array([x2, y2])
        
        # Line vector
        ab = b - a
        # Vector from a to p
        ap = p - a
        
        # Projection parameter
        t = np.dot(ap, ab) / np.dot(ab, ab)
        t = max(0, min(1, t))  # Clamp to segment
        
        # Closest point on segment
        closest = a + t * ab
        dist = np.linalg.norm(p - closest)
        
        if dist < best_dist:
            best_dist = dist
            best_point = tuple(closest)
    
    return best_point
```
