**DeepSeek-chat Output:**

from typing import List, Dict, Any

def validate_deadband_protocol(actions: List[str]) -> Dict[str, Any]:
    """
    Validates a sequence of actions against the Deadband Protocol (P0->P1->P2).
    
    Args:
        actions: List of action strings, where each action starts with 'P0', 'P1', or 'P2'.
    
    Returns:
        Dictionary with validation results.
    """
    phases = {"P0": 0, "P1": 1, "P2": 2}
    violations = []
    current_phase = -1
    
    for i, action in enumerate(actions):
        # Extract phase prefix
        prefix = action[:2] if len(action) >= 2 else None
        if prefix not in phases:
            violations.append(f"Action '{action}' at index {i} has invalid phase prefix")
            continue
            
        phase_num = phases[prefix]
        if phase_num < current_phase:
            violations.append(f"Phase regression at index {i}: {action} appears after higher phase")
        current_phase = max(current_phase, phase_num)
    
    # Count actions per phase
    phase_summary = {phase: 0 for phase in phases}
    for action in actions:
        prefix = action[:2]
        if prefix in phase_summary:
            phase_summary[prefix] += 1
    
    return {
        "is_valid": len(violations) == 0,
        "violations": violations,
        "phase_summary": phase_summary
    }
```

def validate_deadband_protocol(actions: List[str]) -> Dict[str, Any]:
    """
    Validate that actions follow Deadband Protocol ordering: P0 (negative space),
    then P1 (safe channels), then P2 (optimization). No phase skipping allowed.
    """
    expected_order = ["P0", "P1", "P2"]
    phase_indices = {phase: i for i, phase in enumerate(expected_order)}
    
    violations = []
    highest_seen = -1
    
    for idx, action in enumerate(actions):
        # Get first two chars as phase identifier
        if len(action) < 2:
            violations.append(f"Action '{action}' too short at position {idx}")
            continue
            
        phase = action[:2]
        if phase not in phase_indices:
            violations.append(f"Unknown phase '{phase}' in action '{action}' at position {idx}")
            continue
            
        current_phase_idx = phase_indices[phase]
        if current_phase_idx < highest_seen:
            violations.append(
                f"Protocol violation at position {idx}: {phase} appears after {expected_order[highest_seen]}"
            )
        highest_seen = max(highest_seen, current_phase_idx)
    
    # Summary counts
    summary = {phase: 0 for phase in expected_order}
    for action in actions:
        if len(action) >= 2:
            phase = action[:2]
            if phase in summary:
                summary[phase] += 1
    
    return {
        "is_valid": not violations,
        "violations": violations,
        "phase_summary": summary
    }
```

*   **Best Practices:** Both use type hints and docstrings. Groq's docstring more explicitly mentions "no phase skipping," which is good. DeepSeek's variable naming (`phase_num`, `current_phase`) is very clear.
*   **Edge Cases:** Both handle malformed actions (too short, invalid prefix). DeepSeek's `prefix = action[:2] if len(action) >= 2 else None` is a clean one-liner.

### **Function B: `parse_fleet_tile`**

**DeepSeek-chat Output:**
```python
import re
from typing import Optional, Dict
from datetime import datetime

def parse_fleet_tile(tile_content: str) -> Optional[Dict]:
    """
    Parse a fleet training tile string into structured data.
    
    Expected format:
        # Cycle N — AgentName
        **Phase X: Description**
        **Task:** Task description
        ---
        Content...
    
    Returns None if parsing fails critically.
    """
    if not tile_content or not isinstance(tile_content, str):
        return None
    
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "task": None,
        "timestamp": None,
        "content": ""
    }
    
    lines = tile_content.strip().split('\n')
    if len(lines) < 2:
        return None  # Too short to be valid
    
    # Parse header line: "# Cycle N — AgentName"
    header_match = re.match(r'# Cycle (\d+) — (\w+)', lines[0])
    if header_match:
        result["cycle"] = int(header_match.group(1))
        result["agent"] = header_match.group(2)
    
    # Look for phase and task lines
    for line in lines[1:4]:  # Check first few lines after header
        if line.startswith("**Phase"):
            # Extract "X: Description"
            phase_match = re.search(r'Phase (\d+): (.+)', line)
            if phase_match:
                result["phase"] = {
                    "number": int(phase_match.group(1)),
                    "description": phase_match.group(2).strip('* ')
                }
        elif line.startswith("**Task:"):
            result["task"] = line.replace("**Task:**", "").strip()
    
    # Extract content after the first "---" separator
    try:
        sep_index = lines.index("---")
        result["content"] = "\n".join(lines[sep_index + 1:]).strip()
    except ValueError:
        result["content"] = "\n".join(lines[2:]).strip()
    
    # Add timestamp
    result["timestamp"] = datetime.utcnow().isoformat() + "Z"
    
    return result if result["cycle"] and result["agent"] else None
```
