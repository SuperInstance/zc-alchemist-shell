def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol strict priority.
    Rules: Must contain only P0, P1, P2. Must be in order P0 -> P1 -> P2.
    No skipping allowed (e.g., P0 to P2 is invalid). No repeats after moving forward.
    
    Args:
        steps: List of step labels as strings.
    
    Returns:
        Tuple of (is_valid: bool, error_message: str).
    """
    valid_steps = {"P0", "P1", "P2"}
    order = {"P0": 0, "P1": 1, "P2": 2}
    last_index = -1
    
    for i, step in enumerate(steps):
        if step not in valid_steps:
            return False, f"Invalid step '{step}' at position {i}. Must be P0, P1, or P2."
        
        current_index = order[step]
        if current_index < last_index:
            return False, f"Protocol violation at position {i}: '{step}' appears after a higher priority step."
        # Allow staying at same priority (e.g., multiple P0 steps)
        # But check for skip: if current_index > last_index + 1
        if last_index != -1 and current_index > last_index + 1:
            return False, f"Protocol skip at position {i}: jumped from {list(order.keys())[last_index]} to {step}."
        
        last_index = current_index
    
    return True, ""

def parse_fleet_manifest(manifest_text: str) -> dict:
    """
    Parses a simple fleet manifest text into a structured dictionary.
    Expected format per line: "key:value, key2:value2"
    Primary key is 'repo'. Converts 'cycles' to int.
    
    Args:
        manifest_text: Multiline string of manifest entries.
    
    Returns:
        Dictionary with repo names as keys, each containing a dict of fields.
    """
    result = {}
    lines = manifest_text.strip().split('\n')
    
    for line_num, line in enumerate(lines, 1):
        line = line.strip()
        if not line:
            continue
            
        repo_name = None
        entry = {}
        pairs = [p.strip() for p in line.split(',')]
        
        for pair in pairs:
            if ':' not in pair:
                continue  # Skip malformed pair
            key, value = pair.split(':', 1)
            key, value = key.strip(), value.strip()
            
            if key == 'repo':
                repo_name = value
            else:
                entry[key] = value
        
        if repo_name is None:
            # If no repo specified, skip entry with warning?
            # For robustness, we'll skip.
            continue
        
        # Type conversion for known numeric fields
        if 'cycles' in entry:
            try:
                entry['cycles'] = int(entry['cycles'])
            except ValueError:
                # Keep as string if conversion fails
                pass
        
        result[repo_name] = entry
    
    return result

def generate_tile_summary(work_output: str, agent_role: str) -> str:
    """
    Generates a concise summary from raw work output for a training tile.
    
    Args:
        work_output: Raw text output from an agent's work.
        agent_role: The role of the agent (e.g., 'Alchemist').
    
    Returns:
        A formatted summary string, approximately 200 characters.
    """
    # Extract first few sentences or key lines
    lines = work_output.strip().split('\n')
    content_lines = [l.strip() for l in lines if l.strip() and not l.strip().startswith('#')]
    
    if not content_lines:
        summary_text = "No content extracted."
    else:
        # Take first 2-3 non-empty, non-comment lines
        summary_text = ' '.join(content_lines[:3])
    
    # Truncate to ~180 chars for role tag
    max_len = 180
    if len(summary_text) > max_len:
        summary_text = summary_text[:max_len].rsplit(' ', 1)[0] + '...'
    
    # Format with role tag
    return f"[{agent_role}] {summary_text}"

    Groq is faster but more expensive.

    summary = generate_tile_summary(sample_output, "Alchemist")
    print(f"Summary: {summary}")
    print(f"Length: {len(summary)}")
