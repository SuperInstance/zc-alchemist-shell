```python
from typing import List

def deadband_filter(readings: List[float], threshold: float) -> List[float]:
    """
    Apply a deadband filter to sensor readings.
    
    Only keeps readings that change by more than `threshold` from the last kept reading.
    The first reading is always kept.
    
    Args:
        readings (List[float]): List of sensor readings.
        threshold (float): Minimum change required to keep a reading.
    
    Returns:
        List[float]: Filtered list of readings.
    """
    if not readings:
        return []
    
    filtered = [readings[0]]
    last_kept = readings[0]
    
    for reading in readings[1:]:
        if abs(reading - last_kept) > threshold:
            filtered.append(reading)
            last_kept = reading
    
    return filtered
```

```python
from datetime import datetime
from typing import Dict, Any, Tuple

def validate_fleet_message(message: Dict[str, Any]) -> Tuple[bool, str]:
    """
    Validate a fleet message against a required schema.
    
    Required keys:
        - id: str, non-empty
        - timestamp: str in ISO format (YYYY-MM-DDTHH:MM:SS)
        - sender: str, non-empty
        - payload: dict
    
    Args:
        message (Dict[str, Any]): The message to validate.
    
    Returns:
        Tuple[bool, str]: (is_valid, error_message)
    """
    required_keys = {'id', 'timestamp', 'sender', 'payload'}
    
    # Check all required keys present
    missing = required_keys - set(message.keys())
    if missing:
        return False, f"Missing required keys: {missing}"
    
    # Validate id
    if not isinstance(message['id'], str) or not message['id'].strip():
        return False, "Field 'id' must be a non-empty string."
    
    # Validate timestamp
    if not isinstance(message['timestamp'], str):
        return False, "Field 'timestamp' must be a string."
    try:
        datetime.fromisoformat(message['timestamp'].replace('Z', '+00:00'))
    except ValueError:
        return False, "Field 'timestamp' must be a valid ISO 8601 format."
    
    # Validate sender
    if not isinstance(message['sender'], str) or not message['sender'].strip():
        return False, "Field 'sender' must be a non-empty string."
    
    # Validate payload
    if not isinstance(message['payload'], dict):
        return False, "Field 'payload' must be a dictionary."
    
    return True, "Valid"
```

```python
        except ValueError:
            timestamp = None

            "timestamp": timestamp,
            "level": level,
            "module": module,
            "message": message
        }
```
