import hashlib
from typing import List, Tuple, Dict, Any

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validate that steps follow strict Deadband Protocol priority: P0 -> P1 -> P2.
    
    Rules:
    1. Only "P0", "P1", "P2" allowed.
    2. Must start with P0 (if non-empty).
    3. No skipping: P0 before P1 before P2.
    4. No going backwards.
    5. Repeats allowed only if consecutive (e.g., ["P0", "P0", "P1"] is OK).
    
    Returns:
        (True, "") if valid, (False, error_message) otherwise.
    """
    if not steps:
        return True, "Empty list is trivially valid"
    
    allowed = {"P0", "P1", "P2"}
    order = {"P0": 0, "P1": 1, "P2": 2}
    
    for i, step in enumerate(steps):
        if step not in allowed:
            return False, f"Invalid step '{step}' at index {i}. Must be P0, P1, or P2."
    
    # Check ordering
    last_seen = -1
    for i, step in enumerate(steps):
        current = order[step]
        if current < last_seen:
            return False, f"Protocol regression at index {i}: {steps[i-1]} -> {step}"
        last_seen = current
    
    # Ensure starts with P0 if any steps exist
    if steps[0] != "P0":
        return False, "Protocol must start with P0"
    
    return True, ""

def parse_fleet_tile(content: str) -> Dict[str, Any]:
    """
    Parse a fleet training tile from markdown/content string.
    
    Expected format (optional frontmatter):
        ---
        cycle: 18
        phase: 3
        agent: Alchemist
        ---
        # Actual content...
    
    Returns dict with:
        - cycle, phase, agent (extracted or defaults)
        - task: first line after headers
        - content_sha: SHA-256 of entire content
        - valid: True if cycle/phase/agent present, else False
    """
    lines = content.strip().split('\n')
    metadata = {"cycle": 0, "phase": 0, "agent": "Unknown", "task": "", "valid": False}
    
    # Try to parse frontmatter
    if lines and lines[0].startswith('---'):
        try:
            end_idx = lines.index('---', 1)
            frontmatter = lines[1:end_idx]
            for line in frontmatter:
                if ':' in line:
                    key, val = line.split(':', 1)
                    key = key.strip().lower()
                    val = val.strip()
                    if key in metadata:
                        if key in ['cycle', 'phase']:
                            metadata[key] = int(val) if val.isdigit() else 0
                        else:
                            metadata[key] = val
        except ValueError:
            pass  # No proper frontmatter
    
    # Extract task from first non‑empty, non‑header line after frontmatter
    start_idx = 0
    if lines[0].startswith('---'):
        try:
            start_idx = lines.index('---', 1) + 1
        except ValueError:
            start_idx = 1
    
    for line in lines[start_idx:]:
        line = line.strip()
        if line and not line.startswith('#'):
            metadata['task'] = line[:100]  # First 100 chars
            break
    
    # Compute SHA
    sha = hashlib.sha256(content.encode('utf-8')).hexdigest()
    metadata['content_sha'] = sha
    
    # Determine validity
    metadata['valid'] = (
        metadata['cycle'] > 0 and 
        metadata['phase'] > 0 and 
        metadata['agent'] != 'Unknown'
    )
    
    return metadata

def simulate_room_training(tiles: List[Dict], epochs: int = 5) -> Dict[str, float]:
    """
    Simulate a PLATO room training over epochs.
    
    Each tile dict must have:
        - 'valid': bool
        - 'content_sha': str (used for seeding)
        - optional 'score': float (0.0 to 1.0)
    
    Simulation logic:
        - Valid tiles decrease loss by (score or 0.7).
        - Invalid tiles increase noise by 0.2.
        - Loss decays by 0.9 each epoch (simulating convergence).
        - Noise accumulates but caps at 1.0.
    
    Returns final metrics dict.
    """
    if not tiles:
        return {"final_loss": 1.0, "valid_tile_ratio": 0.0, "noise_level": 0.0}
    
    valid_count = sum(1 for t in tiles if t.get('valid', False))
    valid_ratio = valid_count / len(tiles)
    
    loss = 1.0
    noise = 0.0
    
    for epoch in range(epochs):
        # Process tiles
        for tile in tiles:
            if tile.get('valid', False):
                score = tile.get('score', 0.7)
                loss -= 0.1 * score  # Valid tiles reduce loss
            else:
                noise += 0.2  # Invalid tiles add noise
        
        # Decay loss
        loss *= 0.9
        # Cap noise
        noise = min(noise, 1.0)
    
    return {
        "final_loss": max(0.0, round(loss, 3)),
        "valid_tile_ratio": round(valid_ratio, 3),
        "noise_level": round(min(noise, 1.0), 3)
    }
