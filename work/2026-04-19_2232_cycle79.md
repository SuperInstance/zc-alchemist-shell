```python
from typing import List, Dict, Any

def validate_deadband_protocol(steps: List[str]) -> Dict[str, Any]:
    """
    Validates a list of steps against the Deadband Protocol phase ordering.
    
    Args:
        steps: List of step descriptions, optionally prefixed with "P0:", "P1:", or "P2:".
    
    Returns:
        Dict with keys:
        - valid: True if protocol is followed (P0 before P1 before P2, no skips)
        - violations: List of indices where ordering is violated
        - phase_summary: Dict counting steps per phase
    """
    phase_order = {"P0": 0, "P1": 1, "P2": 2}
    current_phase = -1
    violations = []
    phase_summary = {"P0": 0, "P1": 0, "P2": 0, "unknown": 0}
    
    for i, step in enumerate(steps):
        step_phase = None
        for prefix in phase_order:
            if step.strip().startswith(prefix + ":"):
                step_phase = prefix
                break
        
        if step_phase is None:
            phase_summary["unknown"] += 1
            continue
        
        phase_summary[step_phase] += 1
        expected_next = current_phase + 1
        
        if phase_order[step_phase] < expected_next:
            # Regression to earlier phase is allowed (e.g., P1 after P2) per spec? Assuming not.
            violations.append(i)
        elif phase_order[step_phase] > expected_next:
            # Skipped a phase
            violations.append(i)
        
        current_phase = phase_order[step_phase]
    
    valid = len(violations) == 0
    return {
        "valid": valid,
        "violations": violations,
        "phase_summary": phase_summary
    }
```

```python
import re
from typing import Dict, Any
from datetime import datetime

def parse_fleet_tile(tile_text: str) -> Dict[str, Any]:
    """
    Parses a fleet training tile text into structured data.
    
    Expected format includes headers like '# Cycle X — Agent', free text,
    and optional timestamp.
    
    Args:
        tile_text: Raw markdown/text content of a tile.
    
    Returns:
        Dict with keys: cycle, agent, phase, task, content, timestamp.
        Missing values are set to None.
    """
    # Initialize with defaults
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "task": None,
        "content": "",
        "timestamp": None
    }
    
    lines = tile_text.split('\n')
    content_lines = []
    
    for line in lines:
        # Extract cycle and agent from header pattern: # Cycle X — Agent
        cycle_match = re.search(r'# Cycle (\d+) — (\w+)', line)
        if cycle_match:
            result["cycle"] = int(cycle_match.group(1))
            result["agent"] = cycle_match.group(2)
            continue
        
        # Extract phase pattern: **Phase X: Description**
        phase_match = re.search(r'\*\*Phase (\d+):\s*(.*?)\*\*', line)
        if phase_match:
            result["phase"] = int(phase_match.group(1))
            result["task"] = phase_match.group(2).strip()
            continue
        
        # Extract timestamp pattern: YYYY-MM-DD HH:MM UTC
        time_match = re.search(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}) UTC', line)
        if time_match and result["timestamp"] is None:
            try:
                result["timestamp"] = datetime.strptime(time_match.group(1), "%Y-%m-%d %H:%M")
            except ValueError:
                pass
            continue
        
        # Collect non-header lines as content
        if not line.startswith('#') and not line.startswith('---'):
            content_lines.append(line)
    
    result["content"] = '\n'.join(content_lines).strip()
    return result
```
