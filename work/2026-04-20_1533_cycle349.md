from typing import List, Dict, Any, Tuple
from datetime import datetime

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol to a list of float values.
    
    Args:
        values: List of input float values
        deadband: Threshold around zero (default 0.1)
        
    Returns:
        List of filtered values where:
        - Values within ±deadband of zero become 0.0
        - Values outside deadband are shifted toward zero by deadband magnitude
    """
    result = []
    for v in values:
        if -deadband <= v <= deadband:
            result.append(0.0)
        elif v > deadband:
            result.append(v - deadband)
        else:  # v < -deadband
            result.append(v + deadband)
    return result


def parse_fleet_log(line: str) -> Dict[str, Any]:
    """
    Parse a structured fleet log line.
    
    Format: "YYYY-MM-DD HH:MM:SS | AGENT | CYCLE | PHASE | MESSAGE"
    
    Args:
        line: Log line string
        
    Returns:
        Dictionary with keys: timestamp, agent, cycle, phase, message
        
    Raises:
        ValueError: If line format is invalid
    """
    parts = line.split(' | ')
    if len(parts) != 5:
        raise ValueError(f"Invalid log format. Expected 5 parts, got {len(parts)}")
    
    date_str, agent, cycle_str, phase, message = parts
    
    try:
        timestamp = datetime.strptime(date_str, '%Y-%m-%d %H:%M:%S')
        cycle = int(cycle_str)
    except ValueError as e:
        raise ValueError(f"Invalid timestamp or cycle: {e}")
    
    return {
        'timestamp': timestamp,
        'agent': agent.strip(),
        'cycle': cycle,
        'phase': phase.strip(),
        'message': message.strip()
    }


def validate_tile(tile: Dict[str, Any]) -> Tuple[bool, List[str]]:
    """
    Validate a training tile against schema.
    
    Required keys: id, agent, cycle, content, timestamp
    Optional keys: tags, score
    
    Args:
        tile: Dictionary to validate
        
    Returns:
        Tuple of (is_valid, list_of_error_messages)
    """
    errors = []
    required = ['id', 'agent', 'cycle', 'content', 'timestamp']
    
    # Check required keys
    for key in required:
        if key not in tile:
            errors.append(f"Missing required key: {key}")
    
    # Validate types if present
    if 'id' in tile and not isinstance(tile['id'], str):
        errors.append("'id' must be a string")
    
    if 'agent' in tile and not isinstance(tile['agent'], str):
        errors.append("'agent' must be a string")
    
    if 'cycle' in tile and not isinstance(tile['cycle'], int):
        errors.append("'cycle' must be an integer")
    
    if 'content' in tile and not isinstance(tile['content'], str):
        errors.append("'content' must be a string")
    
    if 'timestamp' in tile:
        try:
            datetime.fromisoformat(tile['timestamp'])
        except (ValueError, TypeError):
            errors.append("'timestamp' must be valid ISO format string")
    
    # Validate optional fields
    if 'tags' in tile:
        if not isinstance(tile['tags'], list):
            errors.append("'tags' must be a list")
        elif not all(isinstance(tag, str) for tag in tile['tags']):
            errors.append("All tags must be strings")
    
    if 'score' in tile:
        if not isinstance(tile['score'], (int, float)):
            errors.append("'score' must be numeric")
        elif not 0 <= tile['score'] <= 1:
            errors.append("'score' must be between 0 and 1")
    
    return len(errors) == 0, errors
