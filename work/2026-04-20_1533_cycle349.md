from typing import List, Tuple, Dict, Optional, Union
import re
import math

def deadband_filter(values: List[float], deadband: float) -> List[float]:
    """
    Apply a deadband filter to a sequence of values.

    The deadband defines a minimum change threshold. A new value is only kept
    if it differs from the last kept value by at least `deadband`.

    Args:
        values: Input list of floating-point numbers.
        deadband: Positive threshold for change detection.

    Returns:
        Filtered list where consecutive values differ by at least `deadband`.
    """
    if not values:
        return []
    if deadband <= 0:
        raise ValueError("Deadband must be positive.")

    filtered = [values[0]]
    last_kept = values[0]

    for v in values[1:]:
        if abs(v - last_kept) >= deadband:
            filtered.append(v)
            last_kept = v
    return filtered


def parse_fleet_log(log_line: str) -> Optional[Dict[str, Union[str, int]]]:
    """
    Parse a structured fleet log line into its components.

    Expected format:
    [YYYY-MM-DDThh:mm:ssZ] <AgentName> Cycle N: Message text.

    Args:
        log_line: A single line of fleet log.

    Returns:
        Dictionary with keys 'timestamp', 'agent', 'cycle', 'message',
        or None if the line doesn't match the pattern.
    """
    pattern = r'^\[([^\]]+)\] <([^>]+)> Cycle (\d+): (.+)$'
    match = re.match(pattern, log_line.strip())
    if not match:
        return None

    timestamp, agent, cycle_str, message = match.groups()
    try:
        cycle = int(cycle_str)
    except ValueError:
        return None

    return {
        'timestamp': timestamp,
        'agent': agent,
        'cycle': cycle,
        'message': message.strip()
    }


def constraint_snap(
    point: Tuple[float, float],
    constraints: List[Dict]
) -> Tuple[float, float]:
    """
    Snap a 2D point to the nearest geometric constraint.

    Args:
        point: (x, y) coordinate to snap.
        constraints: List of constraint dictionaries.
            For type 'line': {'type': 'line', 'point': (x, y), 'direction': (dx, dy)}
            For type 'circle': {'type': 'circle', 'center': (cx, cy), 'radius': r}

    Returns:
        (x, y) coordinates of the snapped point.
    """
    if not constraints:
        return point

    best_distance = float('inf')
    best_point = point

    px, py = point

    for const in constraints:
        const_type = const.get('type')
        if const_type == 'line':
            # Unpack line constraint
            lx, ly = const['point']
            dx, dy = const['direction']
            # Vector projection: v = point - line_point
            vx, vy = px - lx, py - ly
            # Project onto direction vector
            t = (vx * dx + vy * dy) / (dx * dx + dy * dy)
            # Closest point on the infinite line
            snap_x = lx + t * dx
            snap_y = ly + t * dy
        elif const_type == 'circle':
            # Unpack circle constraint
            cx, cy = const['center']
            radius = const['radius']
            # Vector from center to point
            vx, vy = px - cx, py - cy
            norm = math.hypot(vx, vy)
            if norm == 0:
                # Point at center, snap to any point on circle (e.g., rightmost)
                snap_x = cx + radius
                snap_y = cy
            else:
                # Scale to radius
                scale = radius / norm
                snap_x = cx + vx * scale
                snap_y = cy + vy * scale
        else:
            continue  # Skip unknown constraint types

        # Calculate distance
        dist = math.hypot(px - snap_x, py - snap_y)
        if dist < best_distance:
            best_distance = dist
            best_point = (snap_x, snap_y)

    return best_point
