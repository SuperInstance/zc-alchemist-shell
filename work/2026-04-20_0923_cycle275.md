import re
from typing import Dict, Optional

def parse_fleet_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured fleet log line.
    
    Expected format: "[TIMESTAMP] SERVICE: LEVEL - message"
    
    Args:
        line: Log line string
    
    Returns:
        Dict with keys: timestamp, service, level, message
        Returns None if line doesn't match expected format
    """
    pattern = r'^\[([^\]]+)\]\s+([^:]+):\s+([^\s-]+)\s+-\s+(.+)$'
    match = re.match(pattern, line)
    
    if not match:
        return None
    
    return {
        'timestamp': match.group(1),
        'service': match.group(2),
        'level': match.group(3),
        'message': match.group(4)
    }
```

import numpy as np
from typing import List, Tuple

def constraint_snap(
    point: Tuple[float, float], 
    constraints: List[Tuple[float, float, float]]
) -> Tuple[float, float]:
    """
    Snap a 2D point to satisfy multiple linear constraints.
    
    Args:
        point: (x, y) coordinates
        constraints: List of (A, B, C) for line equation Ax + By + C = 0
    
    Returns:
        (x, y) of closest point satisfying all constraints
    """
    x, y = point
    
    # Convert to numpy arrays for linear algebra
    A = []
    b = []
    
    for (A_i, B_i, C_i) in constraints:
        A.append([A_i, B_i])
        b.append(-C_i)
    
    A = np.array(A)
    b = np.array(b)
    
    # If only one constraint, project onto line
    if len(constraints) == 1:
        # Project point onto line A1*x + B1*y + C1 = 0
        A1, B1, C1 = constraints[0]
        denom = A1**2 + B1**2
        if denom == 0:
            return point  # Degenerate constraint
        
        t = -(A1*x + B1*y + C1) / denom
        return (x + A1*t, y + B1*t)
    
    # Multiple constraints: solve least squares
    try:
        result, *_ = np.linalg.lstsq(A, b, rcond=None)
        return tuple(result)
    except np.linalg.LinAlgError:
        # Fallback: return original point if solving fails
        return point
```
