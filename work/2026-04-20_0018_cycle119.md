import re
from typing import Any

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of steps follows the strict Deadband Protocol sequence.
    P0: Map negative space, P1: Find safe channels, P2: Optimize within channels.
    The sequence must be exactly ["P0", "P1", "P2"] with no other elements.

    Args:
        steps: List of step identifiers as strings.

    Returns:
        Tuple of (is_valid, error_message). Error message is empty string if valid.
    """
    expected = ["P0", "P1", "P2"]
    if steps == expected:
        return True, ""
    if len(steps) != len(expected):
        return False, f"Length mismatch. Expected {len(expected)} steps, got {len(steps)}."
    for i, (exp, got) in enumerate(zip(expected, steps)):
        if exp != got:
            return False, f"Step {i} mismatch. Expected '{exp}', got '{got}'."
    # Fallback for edge cases (should not be reached if lists are same length)
    return False, "Sequence does not match P0->P1->P2."

def parse_fleet_log_line(line: str) -> dict[str, Any]:
    """
    Parses a standardized fleet log line.
    Format: [TIMESTAMP] AGENT:CYCLE "MESSAGE"
    Example: [2026-04-20T00:16:42Z] Alchemist:119 "Submitted tile for review"

    Args:
        line: A single log line string.

    Returns:
        Dictionary with keys 'timestamp', 'agent', 'cycle', 'message'.
        Includes 'error' key if parsing fails.
    """
    pattern = r'^\[([^\]]+)\]\s+([^:]+):(\d+)\s+"([^"]*)"$'
    match = re.match(pattern, line.strip())
    if not match:
        return {"error": f"Failed to parse log line: {line}"}
    timestamp, agent, cycle_str, message = match.groups()
    try:
        cycle = int(cycle_str)
    except ValueError:
        return {"error": f"Invalid cycle number '{cycle_str}' in line: {line}"}
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "message": message
    }

def generate_training_tile(agent: str, task: str, output: str, metadata: dict) -> str:
    """
    Generates a training tile in standard format for PLATO room submission.

    Args:
        agent: Name of the agent generating the tile.
        task: Description of the task performed.
        output: The work output/content of the tile.
        metadata: Additional key-value pairs for the tile header.

    Returns:
        A string formatted as a markdown training tile.
    """
    # Construct frontmatter block
    frontmatter_lines = [f"agent: {agent}", f"task: {task}"]
    for key, value in metadata.items():
        frontmatter_lines.append(f"{key}: {value}")
    frontmatter = "\n".join(frontmatter_lines)

    tile_template = f"""---
{frontmatter}
---

# Training Tile

**Generated by:** {agent}
**Task:** {task}

## Output

{output}

---
*Tile generated for fleet training. Submit to active PLATO room.*
"""
    return tile_template

import re

def validate_deadband_protocol(steps):
    """
    Check if the steps list is exactly ["P0", "P1", "P2"].

    Parameters:
    steps (list): A list of strings.

    Returns:
    tuple: (bool, str) where bool is True if valid, False otherwise, and str is an error message.
    """
    if not isinstance(steps, list):
        return False, "Input must be a list"
    if steps == ["P0", "P1", "P2"]:
        return True, ""
    else:
        # Provide some error details
        if len(steps) != 3:
            return False, f"Expected 3 steps, got {len(steps)}"
        # Check each position
        expected = ["P0", "P1", "P2"]
        for i in range(3):
            if i >= len(steps):
                return False, f"Missing step {expected[i]}"
            if steps[i] != expected[i]:
                return False, f"At position {i}, expected '{expected[i]}' got '{steps[i]}'"
        # If we have more than 3 steps
        if len(steps) > 3:
            return False, "Extra steps after P2"
        return False, "Steps do not match P0, P1, P2 sequence"
