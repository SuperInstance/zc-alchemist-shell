def deadband_filter(values, lower, upper):
    """
    Apply a deadband filter to a sequence of values.
    
    Args:
        values: Iterable of numeric values
        lower: Lower deadband threshold
        upper: Upper deadband threshold
    Returns:
        List of filtered values
    """
    if lower > upper:
        raise ValueError("lower must be <= upper")
    
    filtered = []
    prev_output = 0.0
    
    for v in values:
        if lower <= v <= upper:
            filtered.append(prev_output)
        else:
            filtered.append(v)
            prev_output = v
    return filtered


def parse_fleet_log_line(line):
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format:
    [timestamp] [LEVEL] key1=value1 key2=value2 ...
    
    Args:
        line: String log line
    Returns:
        Dict with keys: timestamp, level, and all key=value pairs
    """
    import re
    
    # Extract timestamp and level
    ts_match = re.search(r'\[([^\]]+)\]', line)
    level_match = re.search(r'\[([A-Z]+)\]', line)
    
    if not ts_match or not level_match:
        raise ValueError("Invalid log line format")
    
    timestamp = ts_match.group(1)
    level = level_match.group(1)
    
    # Extract key=value pairs
    kv_pairs = re.findall(r'(\w+)=([^\s]+)', line)
    
    result = {'timestamp': timestamp, 'level': level}
    for k, v in kv_pairs:
        result[k] = v
    
    return result


def constraint_snap(point, line_start, line_end, tolerance):
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Args:
        point: (x, y) tuple
        line_start: (x1, y1) tuple
        line_end: (x2, y2) tuple
        tolerance: Maximum distance for snapping
    Returns:
        (x, y) tuple of snapped point or original point
    """
    import math
    
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector math for projection
    dx = x2 - x1
    dy = y2 - y1
    line_len_sq = dx*dx + dy*dy
    
    if line_len_sq == 0:
        # Line segment is a point
        dist = math.hypot(px - x1, py - y1)
        return (x1, y1) if dist <= tolerance else point
    
    # Projection parameter t
    t = ((px - x1) * dx + (py - y1) * dy) / line_len_sq
    t = max(0, min(1, t))  # Clamp to segment
    
    # Nearest point on segment
    nearest_x = x1 + t * dx
    nearest_y = y1 + t * dy
    
    # Check distance
    dist = math.hypot(px - nearest_x, py - nearest_y)
    
    return (nearest_x, nearest_y) if dist <= tolerance else point

def deadband_filter(values, lower, upper):
    """
    Deadband filter: if value is within [lower, upper], output previous output (or 0 if first).
    Otherwise output the value.
    """
    output = []
    prev = 0.0
    for v in values:
        if v >= lower and v <= upper:
            output.append(prev)
        else:
            output.append(v)
            prev = v
    return output


def parse_fleet_log_line(line):
    """
    Parse fleet log line like:
    [2026-04-20 03:01:42] [INFO] agent=Alchemist cycle=177 action=tile_submit status=accepted
    """
    import re
    pattern = r'\[([^\]]+)\] \[([A-Z]+)\] (.*)'
    match = re.match(pattern, line)
    if not match:
        return {}
    timestamp, level, rest = match.groups()
    result = {'timestamp': timestamp, 'level': level}
    # parse key=value pairs
    for part in rest.split():
        if '=' in part:
            k, v = part.split('=', 1)
            result[k] = v
    return result


def constraint_snap(point, line_start, line_end, tolerance):
    """
    Snap point to line segment if within tolerance.
    """
    import math
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # line vector
    dx = x2 - x1
    dy = y2 - y1
    
    # if line is a point
    if dx == 0 and dy == 0:
        d = math.sqrt((px - x1)**2 + (py - y1)**2)
        return (x1, y1) if d <= tolerance else point
    
    # projection
    t = ((px - x1) * dx + (py - y1) * dy) / (dx*dx + dy*dy)
    t = max(0, min(1, t))
    
    nearest_x = x1 + t * dx
    nearest_y = y1 + t * dy
    
    dist = math.sqrt((px - nearest_x)**2 + (py - nearest_y)**2)
    if dist <= tolerance:
        return (nearest_x, nearest_y)
    else:
        return point
