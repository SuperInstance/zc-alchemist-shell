def deadband_filter(values: list[float], deadband: float) -> list[float]:
    """
    Apply deadband protocol to a list of values.
    
    Args:
        values: List of numeric values
        deadband: Positive float threshold
    
    Returns:
        Filtered list where values within +/- deadband become 0
    
    Examples:
        >>> deadband_filter([1.5, 0.2, -0.3, 2.1], 0.5)
        [1.5, 0.0, 0.0, 2.1]
        >>> deadband_filter([0.0], 0.1)
        [0.0]
    """
    if not isinstance(deadband, (int, float)) or deadband < 0:
        raise ValueError("deadband must be a non-negative number")
    
    filtered = []
    for v in values:
        if abs(v) <= deadband:
            filtered.append(0.0)
        else:
            filtered.append(v)
    return filtered
```

- DeepSeek's error message more specific ("non-negative" vs "non-negative number")

from datetime import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse fleet log line format: "TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE"
    
    Example: "2026-04-20T05:06:00Z | Alchemist | 216 | ACTIVE | Testing models"
    
    Returns dict with keys: timestamp, agent, cycle, status, message.
    Cycle should be int, timestamp should be datetime object.
    
    Examples:
        >>> parse_fleet_log("2026-04-20T05:06:00Z | Alchemist | 216 | ACTIVE | Testing")
        {'timestamp': datetime.datetime(2026, 4, 20, 5, 6, 0), 'agent': 'Alchemist', 'cycle': 216, 'status': 'ACTIVE', 'message': 'Testing'}
        >>> parse_fleet_log("2026-04-19T12:00:00Z | Navigator | 100 | IDLE | Mapping")
        {'timestamp': datetime.datetime(2026, 4, 19, 12, 0, 0), 'agent': 'Navigator', 'cycle': 100, 'status': 'IDLE', 'message': 'Mapping'}
    """
    parts = log_line.split(" | ")
    if len(parts) != 5:
        raise ValueError(f"Invalid log format. Expected 5 parts, got {len(parts)}")
    
    timestamp_str, agent, cycle_str, status, message = parts
    
    try:
        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
    except ValueError:
        raise ValueError(f"Invalid timestamp format: {timestamp_str}")
    
    try:
        cycle = int(cycle_str)
    except ValueError:
        raise ValueError(f"Invalid cycle format: {cycle_str}")
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "status": status,
        "message": message
    }
```

import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse a fleet log line.
    
    Format: TIMESTAMP | AGENT | CYCLE | STATUS | MESSAGE
    
    Returns dict with timestamp as datetime, cycle as int.
    
    Examples:
        >>> log = "2026-04-20T05:06:00Z | Alchemist | 216 | ACTIVE | Testing models"
        >>> result = parse_fleet_log(log)
        >>> result['agent']
        'Alchemist'
        >>> result['cycle']
        216
    """
    components = log_line.strip().split(" | ")
    if len(components) != 5:
        raise ValueError("Log line must have exactly 5 components separated by ' | '")
    
    ts_str, agent, cycle_str, status, message = components
    
    # Parse timestamp (handle Zulu time)
    if ts_str.endswith('Z'):
        ts_str = ts_str[:-1] + '+00:00'
    try:
        timestamp = datetime.datetime.fromisoformat(ts_str)
    except ValueError as e:
        raise ValueError(f"Invalid timestamp: {ts_str}") from e
    
    # Parse cycle
    try:
        cycle = int(cycle_str)
    except ValueError as e:
        raise ValueError(f"Invalid cycle number: {cycle_str}") from e
    
    return {
        "timestamp": timestamp,
        "agent": agent,
        "cycle": cycle,
        "status": status
