# Function 1: validate_deadband_protocol
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of steps follows the Deadband Protocol strictly.
    Protocol: Must contain only "P0", "P1", "P2" and follow order P0 -> P1 -> P2.
    No skipping allowed. Returns (True, "") if valid, (False, error_msg) otherwise.
    """
    valid_steps = {"P0", "P1", "P2"}
    expected_order = ["P0", "P1", "P2"]
    
    # Check for invalid steps
    for step in steps:
        if step not in valid_steps:
            return False, f"Invalid step '{step}'. Must be one of {list(valid_steps)}"
    
    # Check order without skipping
    order_index = 0
    for step in steps:
        while order_index < len(expected_order) and expected_order[order_index] != step:
            order_index += 1
        if order_index >= len(expected_order):
            return False, f"Step '{step}' appears out of sequence or after protocol completion"
        # Move to next expected step for future iterations
        order_index += 1
    
    return True, ""

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet agent tile text into structured dictionary.
    Expected format lines like 'Key: value'. Returns dict with defaults for missing keys.
    """
    result = {
        "agent": "",
        "cycle": 0,
        "phase": 0,
        "task": "",
        "content": ""
    }
    
    lines = tile_text.strip().split('\n')
    content_lines = []
    
    for line in lines:
        if ':' in line:
            key, value = line.split(':', 1)
            key = key.strip().lower()
            value = value.strip()
            
            if key == "agent":
                result["agent"] = value
            elif key == "cycle":
                try:
                    result["cycle"] = int(value)
                except ValueError:
                    result["cycle"] = 0
            elif key == "phase":
                try:
                    result["phase"] = int(value)
                except ValueError:
                    result["phase"] = 0
            elif key == "task":
                result["task"] = value
        else:
            # Non-header lines go to content
            content_lines.append(line.strip())
    
    result["content"] = '\n'.join(content_lines).strip()
    return result

def simulate_channel_optimization(start: float, end: float, obstacles: list[tuple[float, float]]) -> list[float]:
    """
    Finds safe channels (P1) and optimal point (P2) given obstacles.
    Obstacles are (center, radius). Returns list of safe midpoints sorted, then optimal.
    """
    if start >= end:
        return []
    
    # Find all unsafe intervals
    unsafe_intervals = []
    for center, radius in obstacles:
        obs_start = center - radius
        obs_end = center + radius
        # Clip to channel bounds
        obs_start = max(obs_start, start)
        obs_end = min(obs_end, end)
        if obs_start < obs_end:
            unsafe_intervals.append((obs_start, obs_end))
    
    # Sort and merge overlapping unsafe intervals
    unsafe_intervals.sort()
    merged = []
    for interval in unsafe_intervals:
        if not merged or merged[-1][1] < interval[0]:
            merged.append(list(interval))
        else:
            merged[-1][1] = max(merged[-1][1], interval[1])
    
    # Find safe intervals (gaps between unsafe)
    safe_intervals = []
    current = start
    for unsafe_start, unsafe_end in merged:
        if current < unsafe_start:
            safe_intervals.append((current, unsafe_start))
        current = unsafe_end
    if current < end:
        safe_intervals.append((current, end))
    
    # Calculate midpoints of safe intervals (P1)
    safe_midpoints = [(a + b) / 2 for a, b in safe_intervals]
    safe_midpoints.sort()
    
    # Find largest safe interval for P2 optimization
    if not safe_intervals:
        return safe_midpoints  # No safe channel
    
    largest_interval = max(safe_intervals, key=lambda x: x[1] - x[0])
    optimal_point = (largest_interval[0] + largest_interval[1]) / 2
    
    return safe_midpoints + [optimal_point]

def validate_deadband_protocol(steps):
    """
    Validate Deadband Protocol steps.
    Returns (True, "") if valid, (False, error_message) otherwise.
    """
    allowed = {"P0", "P1", "P2"}
    order = ["P0", "P1", "P2"]
    
    seen = set()
    last_index = -1
    
    for i, step in enumerate(steps):
        if step not in allowed:
            return False, f"Invalid step '{step}' at position {i}. Allowed: {allowed}"
        
        current_index = order.index(step)
        if current_index < last_index:
            return False, f"Protocol violation at step {i}: '{step}' appears before its required predecessor"
        
        if step in seen and step != order[current_index]:
            return False, f"Duplicate step '{step}' at position {i} before completing protocol"
        
        seen.add(step)
        last_index = current_index
    
    # Check if all required steps are present (no skipping)
    if len(seen) < 3:
        missing = [s for s in order if s not in seen]
        return False, f"Missing required steps: {missing}"
    
    return True, ""
