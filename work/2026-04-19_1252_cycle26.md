## **2. Function Specifications**

### **Function 1: `deadband_filter`**
- **Input:** `values: List[float]`, `deadband: float = 0.1`
- **Output:** `List[float]`
- **Logic:** Implements P0/P1/P2 protocol:
  - P0: Remove values where `abs(value) < deadband` (negative space).
  - P1: Keep values where `deadband <= abs(value) <= 1.0` (safe channel).
  - P2: For remaining values, apply scaling: `value * 0.9` (optimize within channel).
- **Edge cases:** Empty list, negative deadband, values outside [-1, 1].

### **Function 2: `parse_fleet_log`**
- **Input:** `log_line: str`
- **Output:** `Dict[str, Union[str, int, float]]`
- **Format:** `"2026-04-19 12:38:23 | AGENT=Alchemist | CYCLE=26 | TILES=4 | STATUS=ACTIVE"`
- **Parse:** Split by `|`, trim, split key=value, convert types (int/float if numeric).
- **Edge cases:** Missing fields, malformed lines, extra spaces.

### **Function 3: `validate_training_tile`**
- **Input:** `tile: dict`, `schema: dict`
- **Output:** `Tuple[bool, List[str]]` (valid, errors)
- **Logic:** Validate `tile` against `schema` (required keys, types, value ranges).
- **Example schema:**
  ```json
  {
    "agent": {"type": "str", "required": true},
    "cycle": {"type": "int", "min": 1, "max": 100},
    "content": {"type": "str", "required": true}
  }
  ```
- **Edge cases:** Extra keys, missing required, type mismatches.

from typing import List, Dict, Union, Tuple

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol to a list of float values.
    
    P0: Remove values within deadband (abs(value) < deadband).
    P1: Keep values in safe channel (deadband <= abs(value) <= 1.0).
    P2: Scale remaining values by 0.9 (optimization).
    
    Args:
        values: List of float values.
        deadband: Deadband threshold (must be positive).
    
    Returns:
        Filtered and scaled list of values.
    
    Raises:
        ValueError: If deadband is negative.
    """
    if deadband < 0:
        raise ValueError("deadband must be non-negative")
    
    # P0: Map negative space (remove values inside deadband)
    filtered = [v for v in values if abs(v) >= deadband]
    
    # P1: Find safe channel (keep values within [-1, 1] after deadband)
    safe = [v for v in filtered if abs(v) <= 1.0]
    
    # P2: Optimize within channel (scale by 0.9)
    optimized = [v * 0.9 for v in safe]
    
    return optimized


def parse_fleet_log(log_line: str) -> Dict[str, Union[str, int, float]]:
    """
    Parse a structured fleet log line into a dictionary.
    
    Expected format: "TIMESTAMP | KEY1=VALUE1 | KEY2=VALUE2 | ..."
    
    Args:
        log_line: A single log line string.
    
    Returns:
        Dictionary with parsed key-value pairs.
        Numeric values are converted to int or float.
    
    Raises:
        ValueError: If log line is malformed.
    """
    if not log_line or "|" not in log_line:
        raise ValueError("Invalid log line format")
    
    result = {}
    parts = log_line.split("|")
    
    for part in parts:
        part = part.strip()
        if not part:
            continue
        
        if "=" not in part:
            # First part might be timestamp without key
            if " " in part and "-" in part and ":" in part:
                result["timestamp"] = part
            continue
        
        key, value = part.split("=", 1)
        key = key.strip()
        value = value.strip()
        
        # Try to convert numeric values
        try:
            if "." in value:
                value = float(value)
            else:
                value = int(value)
        except ValueError:
            # Keep as string if conversion fails
            pass
        
        result[key] = value
    
    return result


def validate_training_tile(tile: dict, schema: dict) -> Tuple[bool, List[str]]:
    """
    Validate a training tile against a schema.
    
    Args:
        tile: The training tile dictionary to validate.
        schema: Schema definition dict.
            Format: {field: {"type": str, "required": bool, "min": num, "max": num}}
    
    Returns:
        Tuple of (is_valid, list_of_error_messages).
    """
    errors = []
    
    # Check required fields
    for field, rules in schema.items():
        if rules.get("required", False) and field not in tile:
            errors.append(f"Missing required field: {field}")
    
    # Check each field in tile
    for field, value in tile.items():
        if field not in schema:
            errors.append(f"Unexpected field: {field}")
            continue
        
        rules = schema[field]
        expected_type = rules.get("type", "any")
        
        # Type checking
        if expected_type != "any":
            type_ok = False
            if expected_type == "str" and isinstance(value, str):
                type_ok = True
            elif expected_type == "int" and isinstance(value, int):
                type_ok = True
            elif expected_type == "float" and isinstance(value, (int, float)):
                type_ok = True
            elif expected_type == "bool" and isinstance(value, bool):
                type_ok = True
            elif expected_type == "list" and isinstance(value, list):
                type_ok = True
            elif expected_type == "dict" and isinstance(value, dict):
                type_ok = True
            
            if not type_ok:
                errors.append(f"Field '{field}' should be {expected_type}, got {type(value).__name__}")
        
        # Range checking for numbers
        if isinstance(value, (int, float)):
            if "min" in rules and value < rules["min"]:
                errors.append(f"Field '{field}' value {value} below minimum {rules['min']}")
            if "max" in rules and value > rules["max"]:
                errors.append(f"Field '{field}' value {value} above maximum {rules['max']}")
        
        # Length checking for strings
        if isinstance(value, str) and "min_length" in rules:
            if len(value) < rules["min_length"]:
                errors.append(f"Field '{field}' length {len(value)} below minimum {rules['min_length']}")
    
    return len(errors) == 0, errors
