```python
import re
from datetime import datetime
from typing import Dict, Optional

def parse_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured log line in format: [TIMESTAMP] LEVEL: Message
    
    Args:
        line: Log line string, e.g., "[2026-04-19 22:15:03] INFO: Agent started"
    
    Returns:
        Dictionary with keys 'timestamp', 'level', 'message', or None if parsing fails.
    """
    pattern = r'^\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+): (.+)$'
    match = re.match(pattern, line.strip())
    if not match:
        return None
    return {
        'timestamp': match.group(1),
        'level': match.group(2),
        'message': match.group(3)
    }
```

#### **Function 2: deadband_filter**
```python
from typing import List, Tuple

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply Deadband Protocol filtering to a list of float values.
    
    P0: Remove values that are NaN or infinite (negative space).
    P1: Keep values within deadband threshold from the median (safe channel).
    P2: Return filtered list sorted for optimal processing.
    
    Args:
        values: List of float values, may contain noise/outliers.
        deadband: Absolute threshold around median (default 0.1).
    
    Returns:
        Filtered list of values within deadband of median.
    """
    import math
    
    # P0: Map negative space
    valid = [v for v in values if not (math.isnan(v) or math.isinf(v))]
    if not valid:
        return []
    
    # P1: Find safe channel (median ± deadband)
    sorted_vals = sorted(valid)
    median = sorted_vals[len(sorted_vals) // 2]
    lower, upper = median - deadband, median + deadband
    
    # P2: Optimize within channel
    filtered = [v for v in sorted_vals if lower <= v <= upper]
    return filtered
```

```python
from typing import List

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Implements the Deadband Protocol for filtering a list of float values.
    
    Steps:
    1. Remove invalid values (NaN, inf) – P0: negative space.
    2. Compute median and define safe channel (median ± deadband) – P1.
    3. Return values within safe channel, sorted – P2 optimization.
    
    Args:
        values: Input list of floats.
        deadband: Allowed deviation from median.
    
    Returns:
        Filtered list of values within deadband of median.
    """
    import math
    import statistics
    
    # P0: Remove invalid values
    valid_vals = [v for v in values if not (math.isnan(v) or math.isinf(v))]
    if not valid_vals:
        return []
    
    # P1: Find safe channel
    median_val = statistics.median(valid_vals)
    lower_bound = median_val - deadband
    upper_bound = median_val + deadband
    
    # P2: Optimize within channel
    filtered = [v for v in valid_vals if lower_bound <= v <= upper_bound]
    filtered.sort()
    return filtered
```
