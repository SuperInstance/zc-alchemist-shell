```python
from typing import List, Tuple

def validate_deadband_protocol(actions: List[str]) -> Tuple[bool, str]:
    """
    Validates that a list of actions follows the Deadband Protocol priority order:
    P0 (map negative space) → P1 (find safe channels) → P2 (optimize within channels).
    
    Each action string must start with 'P0:', 'P1:', or 'P2:' prefix.
    The sequence must be non-decreasing: P0 can repeat, then P1, then P2.
    Skipping a priority (e.g., P0 then P2) is invalid.
    
    Args:
        actions: List of action strings with priority prefixes.
    
    Returns:
        Tuple of (is_valid: bool, error_message: str).
        If valid, error_message is empty string.
    
    Example:
        >>> validate_deadband_protocol(['P0: avoid rocks', 'P1: find channel', 'P2: optimize'])
        (True, '')
        >>> validate_deadband_protocol(['P1: find channel', 'P0: avoid rocks'])
        (False, 'Invalid priority jump from P1 to P0')
    """
    if not actions:
        return True, "Empty action list is trivially valid"
    
    valid_prefixes = {'P0', 'P1', 'P2'}
    last_priority = -1
    priority_map = {'P0': 0, 'P1': 1, 'P2': 2}
    
    for i, action in enumerate(actions):
        # Check prefix format
        if not isinstance(action, str):
            return False, f"Action at index {i} is not a string"
        
        parts = action.split(':', 1)
        if len(parts) < 2:
            return False, f"Action '{action}' missing ':' separator"
        
        prefix = parts[0]
        if prefix not in valid_prefixes:
            return False, f"Invalid priority prefix '{prefix}' in action '{action}'"
        
        current_priority = priority_map[prefix]
        
        # Check non-decreasing sequence
        if current_priority < last_priority:
            return False, f"Invalid priority jump from {list(priority_map.keys())[last_priority]} to {prefix}"
        
        last_priority = current_priority
    
    return True, ""
```

```python
    """
    Validate that actions follow Deadband Protocol priority sequence (P0→P1→P2).
    
    Args:
        actions: List of strings where each should start with 'P0:', 'P1:', or 'P2:'.
    
    Returns:
        (bool, str): True if valid with empty message, False with error description.
    
    Example:
        >>> validate_deadband_protocol(['P0: avoid', 'P1: explore', 'P2: optimize'])
        (True, '')
        >>> validate_deadband_protocol(['P1: explore', 'P0: avoid'])
        (False, 'Priority regression from P1 to P0 at index 1')
    """
    if not isinstance(actions, list):
        return False, "Input must be a list"
    
    priority_order = {'P0': 0, 'P1': 1, 'P2': 2}
    seen_priority = -1
    
    for idx, action in enumerate(actions):
        if not isinstance(action, str):
            return False, f"Item at index {idx} is not a string"
        
        # Extract prefix
        if ':' not in action:
            return False, f"Action '{action}' missing required ':'"
        
        prefix = action.split(':', 1)[0]
        if prefix not in priority_order:
            return False, f"Unknown priority prefix '{prefix}' in '{action}'"
        
        curr = priority_order[prefix]
        if curr < seen_priority:
            return False, f"Priority regression from {list(priority_order.keys())[seen_priority]} to {prefix} at index {idx}"
        
        seen_priority = curr
    
    return True, ""
```
