**DeepSeek-chat Output:**

import re
from datetime import datetime

def parse_log_line(line: str) -> dict:
    """
    Parse a common Apache/NGINX combined log format line.
    Example: '127.0.0.1 - - [10/Apr/2026:08:47:22 +0000] "GET /api/status HTTP/1.1" 200 1024'
    Returns dict with ip, timestamp, method, path, protocol, status_code, bytes_sent.
    """
    # Common combined log format regex
    pattern = r'^(\S+) \S+ \S+ \[([^\]]+)\] "(\S+) (\S+) (\S+)" (\d+) (\d+)$'
    match = re.match(pattern, line.strip())
    
    if not match:
        raise ValueError(f"Log line does not match expected format: {line}")
    
    ip, timestamp_str, method, path, protocol, status_code, bytes_sent = match.groups()
    
    # Parse timestamp (e.g., 10/Apr/2026:08:47:22 +0000)
    try:
        # Remove square brackets and parse
        timestamp = datetime.strptime(timestamp_str, "%d/%b/%Y:%H:%M:%S %z")
    except ValueError:
        timestamp = timestamp_str  # Fallback to string if parsing fails
    
    return {
        "ip": ip,
        "timestamp": timestamp,
        "method": method,
        "path": path,
        "protocol": protocol,
        "status_code": int(status_code),
        "bytes_sent": int(bytes_sent)
    }
```

from datetime import datetime

def parse_log_line(line):
    """
    Parses a log line in the combined log format.
    Args:
        line (str): A log line string.
    Returns:
        dict: A dictionary containing parsed components.
    """
    # Regex for combined log format
    log_pattern = r'^(\S+) (\S+) (\S+) \[([^\]]+)\] "(\S+) (\S+) (\S+)" (\d+) (\d+)$'
    match = re.match(log_pattern, line)
    
    if match:
        groups = match.groups()
        ip = groups[0]
        timestamp = groups[3]
        method = groups[4]
        url = groups[5]
        protocol = groups[6]
        status = int(groups[7])
        size = int(groups[8])
        
        # Convert timestamp to datetime object
        try:
            dt = datetime.strptime(timestamp, "%d/%b/%Y:%H:%M:%S %z")
        except:
            dt = timestamp  # keep as string if parsing fails
            
        return {
            "ip": ip,
            "timestamp": dt,
            "method": method,
            "path": url,
            "protocol": protocol,
            "status_code": status,
            "bytes_sent": size
        }
    else:
        raise ValueError("Invalid log line format")
```

*   **Correctness:** Both are functionally correct for the standard combined log format. DeepSeek's regex explicitly skips the `- -` user/auth fields with `\S+`, while Groq captures them as groups 1 & 2 (unused). DeepSeek's pattern is slightly more precise.
*   **Clarity:** DeepSeek includes a more detailed docstring with an example. Groq's docstring is adequate.
*   **Robustness:** Both handle parsing failures gracefully by falling back to the timestamp string.
*   **Style:** Both follow PEP 8. DeepSeek uses type hints (`-> dict`), Groq does not.
*   **Edge:** DeepSeek strips the input line (`line.strip()`), Groq does not—a minor robustness difference.

### **Function B: `find_safe_channels`**

**DeepSeek-chat Output:**
```python
from collections import deque
from typing import List, Tuple

def find_safe_channels(grid: List[List[int]], start: Tuple[int, int]) -> List[Tuple[int, int]]:
    """
    Perform BFS to find all reachable safe channels (0) from start position.
    Grid: 0 = safe channel, 1 = rock/blocked.
    Movement: up, down, left, right (no diagonals).
    Returns list of (row, col) coordinates of reachable safe cells.
    """
    if not grid or not grid[0]:
        return []
    
    rows, cols = len(grid), len(grid[0])
    sr, sc = start
    
    # Validate start
    if not (0 <= sr < rows and 0 <= sc < cols):
        return []
    if grid[sr][sc] == 1:
        return []  # Start is blocked
    
    visited = [[False] * cols for _ in range(rows)]
    queue = deque([(sr, sc)])
    visited[sr][sc] = True
    result = [(sr, sc)]
    
    # Directions: up, down, left, right
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    
    while queue:
        r, c = queue.popleft()
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            
            if 0 <= nr < rows and 0 <= nc < cols:
                if not visited[nr][nc] and grid[nr][nc] == 0:
                    visited[nr][nc] = True
                    queue.append((nr, nc))
                    result.append((nr, nc))
    
    return result
```
