from typing import List, Dict, Any, Tuple, Optional

def validate_deadband_protocol(sequence: List[str]) -> bool:
    """
    Validate if a sequence follows the Deadband Protocol.

    The protocol requires priorities in order: P0 before P1 before P2.
    Skipping a priority that appears later is invalid.
    Missing a later priority is allowed if it doesn't appear.

    Args:
        sequence: List of priority strings like ["P0", "P1", "P2"].

    Returns:
        True if sequence is valid, False otherwise.
    """
    allowed = {"P0", "P1", "P2"}
    seen = set()
    last_index = -1
    priority_order = {"P0": 0, "P1": 1, "P2": 2}

    for item in sequence:
        if item not in allowed:
            return False
        current_index = priority_order[item]
        if current_index < last_index:
            return False
        if current_index - last_index > 1:
            # Check if we skipped a required priority
            for skipped in range(last_index + 1, current_index):
                skipped_priority = ["P0", "P1", "P2"][skipped]
                if skipped_priority in sequence:
                    return False
        last_index = current_index
        seen.add(item)
    return True

def parse_fleet_tile(raw_tile: str) -> Dict[str, Any]:
    """
    Parse a raw fleet training tile string.

    Expected format: "AGENT=<id>;CYCLE=<number>;CONTENT=<text>;HASH=<md5>"

    Args:
        raw_tile: The raw tile string.

    Returns:
        Dictionary with parsed fields and validation flag.
    """
    result = {
        "agent_id": None,
        "cycle": None,
        "content": None,
        "content_hash": None,
        "is_valid_format": False
    }
    try:
        parts = raw_tile.split(';')
        if len(parts) != 4:
            return result
        data = {}
        for part in parts:
            if '=' not in part:
                return result
            key, value = part.split('=', 1)
            data[key.strip()] = value.strip()
        required = ["AGENT", "CYCLE", "CONTENT", "HASH"]
        if not all(req in data for req in required):
            return result
        result["agent_id"] = data["AGENT"]
        result["cycle"] = int(data["CYCLE"])
        result["content"] = data["CONTENT"]
        result["content_hash"] = data["HASH"]
        result["is_valid_format"] = True
    except (ValueError, AttributeError):
        pass
    return result

def simulate_channel_optimization(
    channels: List[float],
    strategy: str = "max",
    target: Optional[float] = None
) -> Tuple[int, float]:
    """
    Find optimal channel based on strategy.

    Args:
        channels: List of channel values.
        strategy: Optimization strategy: "max", "min", or "closest_to_target".
        target: Target value for "closest_to_target" strategy.

    Returns:
        Tuple of (optimal_index, optimal_value).

    Raises:
        ValueError: If strategy is invalid or target is missing.
    """
    if not channels:
        raise ValueError("Channels list cannot be empty")
    if strategy == "max":
        optimal_value = max(channels)
        optimal_index = channels.index(optimal_value)
    elif strategy == "min":
        optimal_value = min(channels)
        optimal_index = channels.index(optimal_value)
    elif strategy == "closest_to_target":
        if target is None:
            raise ValueError("Target must be provided for 'closest_to_target' strategy")
        optimal_index = min(
            range(len(channels)),
            key=lambda i: abs(channels[i] - target)
        )
        optimal_value = channels[optimal_index]
    else:
        raise ValueError(f"Invalid strategy: {strategy}")
    return optimal_index, optimal_value
