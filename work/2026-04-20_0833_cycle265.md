import random

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates if steps follow Deadband Protocol: P0 before P1 before P2.
    
    Args:
        steps: List of step strings, e.g., ["P0: map rocks", "P1: find channel"]
    
    Returns:
        Tuple of (is_valid, error_message)
    """
    # Extract priority prefixes
    priorities = []
    for i, step in enumerate(steps):
        step_lower = step.lower().strip()
        if step_lower.startswith("p0:"):
            priorities.append(0)
        elif step_lower.startswith("p1:"):
            priorities.append(1)
        elif step_lower.startswith("p2:"):
            priorities.append(2)
        else:
            return False, f"Step {i} '{step}' does not start with P0:, P1:, or P2:"
    
    # Check ordering
    for i in range(1, len(priorities)):
        if priorities[i] < priorities[i-1]:
            return False, f"Protocol violation at step {i}: P{priorities[i]} after P{priorities[i-1]}"
    
    return True, "Protocol valid"

def parse_fleet_tile(tile_text: str) -> dict:
    """
    Parses a fleet training tile text into structured data.
    
    Expected format (from our cycles):
    # Cycle X — Agent
    **Phase Y: ...**
    **Task:** ...
    ---
    Content...
    
    Returns dict with keys: agent, cycle, phase, task, content, timestamp
    """
    result = {
        "agent": "unknown",
        "cycle": 0,
        "phase": 0,
        "task": "",
        "content": "",
        "timestamp": None
    }
    
    lines = tile_text.strip().split('\n')
    
    # Parse header line
    if lines and lines[0].startswith('# Cycle'):
        header_match = re.match(r'# Cycle (\d+) — (\w+)', lines[0])
        if header_match:
            result["cycle"] = int(header_match.group(1))
            result["agent"] = header_match.group(2)
    
    # Parse phase and task
    for i, line in enumerate(lines):
        if line.startswith('**Phase'):
            phase_match = re.search(r'Phase (\d+)', line)
            if phase_match:
                result["phase"] = int(phase_match.group(1))
        
        if line.startswith('**Task:**'):
            result["task"] = line.replace('**Task:**', '').strip()
    
    # Find content separator
    sep_index = -1
    for i, line in enumerate(lines):
        if line.strip() == '---':
            sep_index = i
            break
    
    if sep_index >= 0 and sep_index + 1 < len(lines):
        result["content"] = '\n'.join(lines[sep_index + 1:]).strip()
    
    # Simple timestamp extraction (if present)
    for line in lines:
        if 'UTC' in line or '2026' in line:
            result["timestamp"] = line.strip()
            break
    
    return result

def simulate_channel_optimization(
    initial_position: float, 
    channel_bounds: Tuple[float, float], 
    noise_std: float = 0.1
) -> List[float]:
    """
    Simulates P2 optimization within a safe channel.
    
    Performs a random walk biased toward channel center.
    
    Args:
        initial_position: Starting point (must be within bounds)
        channel_bounds: (lower_bound, upper_bound)
        noise_std: Standard deviation of random noise
    
    Returns:
        List of positions over 100 simulation steps
    """
    lower, upper = channel_bounds
    if not (lower <= initial_position <= upper):
        raise ValueError(f"Initial position {initial_position} not in bounds [{lower}, {upper}]")
    
    center = (lower + upper) / 2.0
    positions = [initial_position]
    current = initial_position
    
    for _ in range(99):  # 99 more steps for total 100
        # Bias toward center (P2 optimization)
        bias = (center - current) * 0.3  # Attraction factor
        
        # Random noise
        noise = random.gauss(0, noise_std)
        
        # Update position
        new_position = current + bias + noise
        
        # Enforce channel bounds (P1 safe channel)
        new_position = max(lower, min(upper, new_position))
        
        positions.append(new_position)
        current = new_position
    
    return positions
