def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validates if a sequence of steps follows the Deadband Protocol.
    Protocol: Strict order P0 -> P1 -> P2. No skipping, no repeats.
    Returns (is_valid, list_of_errors).
    """
    expected = ["P0", "P1", "P2"]
    errors = []
    seen = set()
    
    for i, step in enumerate(steps):
        if step not in expected:
            errors.append(f"Step {i}: '{step}' is not a valid protocol step (must be P0, P1, P2).")
            continue
        if step in seen:
            errors.append(f"Step {i}: '{step}' repeated. Each step must occur only once.")
            continue
        if expected.index(step) != len(seen):
            errors.append(f"Step {i}: '{step}' out of order. Expected {expected[len(seen)]}.")
            continue
        seen.add(step)
    
    # Check for missing steps
    for exp in expected:
        if exp not in seen:
            errors.append(f"Missing required step: {exp}.")
    
    is_valid = len(errors) == 0
    return (is_valid, errors)
```

def validate_deadband_protocol(steps: list[str]) -> tuple[bool, list[str]]:
    """
    Validate adherence to Deadband Protocol.
    Rules: Must contain exactly one each of P0, P1, P2 in that order.
    """
    if not isinstance(steps, list):
        return (False, ["Input must be a list"])
    
    protocol = ["P0", "P1", "P2"]
    errors = []
    
    # Check length
    if len(steps) != 3:
        errors.append(f"Expected 3 steps, got {len(steps)}")
    
    # Check order and uniqueness
    for i, (actual, expected) in enumerate(zip(steps, protocol)):
        if i >= len(steps):
            errors.append(f"Missing step: {expected}")
            break
        if actual != expected:
            errors.append(f"Step {i}: expected '{expected}', got '{actual}'")
    
    # Check for duplicates
    if len(set(steps)) != len(steps):
        errors.append("Duplicate steps found")
    
    valid = len(errors) == 0
    return (valid, errors)
```

**Comparison:**
*   **Correctness:** Both are logically correct for the core cases. DeepSeek's handles variable-length input and provides more granular error messages (e.g., "P2 out of order. Expected P1."). Groq's assumes a fixed 3-step list, which is less flexible if the protocol definition evolves.
*   **Clarity:** DeepSeek's algorithm using `seen` set and `expected.index` is clear for tracking progress. Groq's `zip` approach is clean but brittle to length mismatches.
*   **Robustness:** DeepSeek's handles non-list input implicitly (will raise TypeError, which may be acceptable). Groq's explicitly checks type. DeepSeek's better handles lists longer than 3.
*   **Verdict:** DeepSeek's implementation is more general and robust, with superior error messaging. Groq's is simpler but makes stronger assumptions.

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parse a fleet training tile submission.
    Expected format:
        # Cycle <number> — <Agent>
        **Phase:** <phase>
        **Content:** ...text...
        CHECKSUM:<number>
    """
    lines = raw_text.strip().split('\n')
    result = {
        "cycle": None,
        "agent": None,
        "phase": None,
        "content": "",
        "checksum_valid": False
    }
    
    if not lines or not lines[0].startswith('# Cycle'):
        return result  # Invalid format
    
    # Parse header: "# Cycle 210 — Alchemist"
    header_parts = lines[0].split('—')
    if len(header_parts) == 2:
        cycle_part = header_parts[0].replace('# Cycle', '').strip()
        result["cycle"] = int(cycle_part) if cycle_part.isdigit() else None
        result["agent"] = header_parts[1].strip()
    
    # Parse metadata lines
    for line in lines[1:]:
        line = line.strip()
        if line.startswith('**Phase:**'):
            phase_part = line.replace('**Phase:**', '').strip()
            result["phase"] = int(phase_part) if phase_part.isdigit() else None
        elif line.startswith('CHECKSUM:'):
            checksum_part = line.replace('CHECKSUM:', '').strip()
            expected_checksum = int(checksum_part) if checksum_part.isdigit() else 0
            # Calculate checksum on content before CHECKSUM line
            content_text = raw_text.split('CHECKSUM:')[0].strip()
            calculated = sum(ord(c) for c in content_text) % 1000
            result["checksum_valid"] = (calculated == expected_checksum)
        elif not line.startswith('**') and line:  # Content line
            result["content"] += line + '\n'
    
    result["content"] = result["content"].strip()
    return result
