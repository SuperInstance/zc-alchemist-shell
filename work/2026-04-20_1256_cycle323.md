from typing import List

def deadband_filter(values: List[float], threshold: float) -> List[float]:
    """
    Apply a deadband filter to a list of float values.
    
    Implements the fleet's Deadband Protocol P0 principle: avoid unnecessary
    changes. If the absolute difference between the current input value and
    the last output value is less than threshold, the last output is repeated.
    
    Args:
        values: Input list of float values.
        threshold: Deadband threshold (non‑negative).
    
    Returns:
        Filtered list of floats.
    """
    if not values:
        return []
    
    # P0: Ensure threshold is non‑negative
    if threshold < 0:
        raise ValueError("Threshold must be non‑negative")
    
    filtered = [values[0]]
    for v in values[1:]:
        if abs(v - filtered[-1]) < threshold:
            # Within deadband, repeat previous output (P0)
            filtered.append(filtered[-1])
        else:
            # Outside deadband, pass through (P1 channel)
            filtered.append(v)
    return filtered
```

import re
from typing import Any, Dict

def parse_fleet_log(log_line: str) -> Dict[str, Any]:
    """
    Parse a fleet log line into a structured dictionary.
    
    Expected format: "YYYY‑MM‑DD HH:MM:SS [LEVEL] key1=value1 key2=value2 ..."
    Extracts agent, cycle, phase, tile_count fields; converts types.
    Missing fields become None.
    """
    # Regex to capture key=value pairs after the timestamp and level
    pattern = r'(\w+)=([^\s]+)'
    matches = re.findall(pattern, log_line)
    
    # Default dict with None for expected fields
    result = {"agent": None, "cycle": None, "phase": None, "tile_count": None}
    
    for key, val in matches:
        if key in result:
            if key in ("cycle", "phase", "tile_count"):
                try:
                    result[key] = int(val)
                except ValueError:
                    # If conversion fails, keep as string? Spec says int.
                    # Fleet practice: log as error, but here we fallback to None.
                    result[key] = None
            else:
                result[key] = val
    return result
```

**Function 3: constraint_snap**
```python
import math
from typing import List, Tuple

def constraint_snap(
    point: Tuple[float, float],
    candidates: List[Tuple[float, float]],
    tolerance: float
) -> Tuple[float, float]:
    """
    Snap a point to the nearest candidate within tolerance.
    
    Mimics geometric snapping from constraint‑theory‑core.
    If multiple candidates are within tolerance, choose the closest.
    If none, return the original point (P1: safe fallback).
    """
    if tolerance < 0:
        raise ValueError("Tolerance must be non‑negative")
    
    px, py = point
    best_dist = float('inf')
    best_candidate = point
    
    for cx, cy in candidates:
        dist = math.hypot(px - cx, py - cy)
        if dist <= tolerance and dist < best_dist:
            best_dist = dist
            best_candidate = (cx, cy)
    
    return best_candidate
```
