import hashlib
from datetime import datetime, timezone
from typing import List, Tuple, Optional

def deadband_filter(values: List[float], deadband: float = 0.1) -> List[float]:
    """
    Apply the Deadband Protocol to filter values.
    
    P0: Remove values within ±deadband of zero (negative space).
    P1: Keep only values outside deadband (safe channels).
    P2: If more than 5 values remain, return top 5 by absolute magnitude.
    
    Args:
        values: List of float values.
        deadband: Deadband threshold (default 0.1).
    
    Returns:
        Filtered list of floats.
    
    Example:
        >>> deadband_filter([-0.05, 0.2, -0.3, 0.02, 0.15, -0.4, 0.5])
        [0.5, -0.4, -0.3, 0.2]
    """
    # P0: Map negative space (remove values near zero)
    p0_filtered = [v for v in values if abs(v) > deadband]
    
    # P1: Safe channels (already satisfied by P0)
    p1_channels = p0_filtered
    
    # P2: Optimize within channels
    if len(p1_channels) > 5:
        # Sort by absolute magnitude descending
        p2_optimized = sorted(p1_channels, key=abs, reverse=True)[:5]
        return p2_optimized
    
    return p1_channels


def constraint_snap(points: List[Tuple[float, float]], 
                    target_point: Tuple[float, float], 
                    tolerance: float) -> Optional[Tuple[float, float]]:
    """
    Snap target_point to the closest point within tolerance.
    
    Args:
        points: List of (x, y) tuples.
        target_point: (x, y) to snap.
        tolerance: Maximum distance for snapping.
    
    Returns:
        Snapped point coordinates if within tolerance, else None.
    
    Raises:
        ValueError: If points list is empty.
    """
    if not points:
        raise ValueError("Points list cannot be empty")
    
    tx, ty = target_point
    min_dist = float('inf')
    snapped_point = None
    
    for px, py in points:
        dist = ((px - tx) ** 2 + (py - ty) ** 2) ** 0.5
        if dist < min_dist and dist <= tolerance:
            min_dist = dist
            snapped_point = (px, py)
    
    return snapped_point


def tile_encoder(text: str, tile_type: str = "observation") -> dict:
    """
    Encode text into a training tile for PLATO rooms.
    
    Args:
        text: Content to encode.
        tile_type: Type of tile (observation, action, reward, etc.)
    
    Returns:
        Dictionary representing a training tile.
    
    Example:
        >>> tile = tile_encoder("Agent completed cycle 268", "log")
        >>> tile['type']
        'log'
    """
    # Generate deterministic ID from text hash
    content_hash = hashlib.sha256(text.encode()).hexdigest()
    tile_id = f"tile_{content_hash[:16]}"
    
    # Current UTC timestamp
    timestamp = datetime.now(timezone.utc).isoformat()
    
    # Compute hash for validation
    hash_value = hashlib.sha256(text.encode()).hexdigest()
    
    return {
        "id": tile_id,
        "timestamp": timestamp,
        "type": tile_type,
        "content": text,
        "hash": hash_value
    }
