# Function 1: Validate Deadband Protocol
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates if a list of step labels follows the Deadband Protocol.
    Protocol requires strict order: P0 (map negative space) must appear first,
    followed by P1 (find safe channels), then P2 (optimize). No skipping allowed.
    
    Args:
        steps: List of step identifiers (e.g., ['P0', 'P1', 'P2'])
    
    Returns:
        Tuple of (is_valid: bool, message: str)
    """
    expected_order = ['P0', 'P1', 'P2']
    found_indices = []
    
    for step in steps:
        step_upper = step.strip().upper()
        if step_upper in expected_order:
            found_indices.append(expected_order.index(step_upper))
    
    # Check for required phases
    if not found_indices:
        return False, "No valid P0, P1, or P2 steps found."
    
    # Check order and no skipping
    for i in range(1, len(found_indices)):
        if found_indices[i] < found_indices[i-1]:
            return False, f"Protocol violation: {expected_order[found_indices[i]]} found after {expected_order[found_indices[i-1]]}."
        if found_indices[i] - found_indices[i-1] > 1:
            return False, f"Protocol violation: Skipped {expected_order[found_indices[i-1] + 1]}."
    
    # Check if P0 is present (must be first valid step)
    if found_indices[0] != 0:
        return False, "Protocol violation: Must start with P0."
    
    return True, "Protocol validation passed."

# Function 2: Parse Fleet Tile
def parse_fleet_tile(content: str) -> dict:
    """
    Parses a fleet training tile text into structured metadata.
    Expected format includes headers like '# Cycle', '**Task:**', etc.
    
    Args:
        content: Raw tile text
    
    Returns:
        Dictionary with keys: 'author', 'cycle', 'phase', 'task', 'content'.
        Missing values are set to None.
    """
    result = {
        'author': None,
        'cycle': None,
        'phase': None,
        'task': None,
        'content': content.strip()
    }
    
    lines = content.split('\n')
    for line in lines:
        line = line.strip()
        # Check for cycle
        if line.startswith('# Cycle'):
            result['cycle'] = line.replace('# Cycle', '').strip()
        # Check for phase
        elif '**Phase:' in line:
            # Extract phase number
            import re
            match = re.search(r'Phase\s*(\d+)', line)
            if match:
                result['phase'] = int(match.group(1))
        # Check for task
        elif line.startswith('**Task:**'):
            result['task'] = line.replace('**Task:**', '').strip()
        # Check for author (simplified)
        elif line.startswith('**Author:**'):
            result['author'] = line.replace('**Author:**', '').strip()
    
    return result

# Function 3: Snap to Constraint
def snap_to_constraint(point: tuple[float, float], constraints: list[tuple[float, float, float]]) -> tuple[float, float]:
    """
    Finds the nearest point to the given point that satisfies at least one
    linear constraint Ax + By + C = 0.
    
    Args:
        point: (x, y) coordinate
        constraints: List of (A, B, C) coefficients
    
    Returns:
        (x, y) of nearest valid point. If no constraints, returns original point.
    """
    if not constraints:
        return point
    
    px, py = point
    best_point = None
    best_distance = float('inf')
    
    for A, B, C in constraints:
        # Special cases for vertical/horizontal lines
        if A == 0 and B == 0:
            continue  # Invalid constraint
        
        # Calculate nearest point on line Ax + By + C = 0 to (px, py)
        # Using formula from point-line distance projection
        denominator = A*A + B*B
        if denominator == 0:
            continue
        
        # Projection point
        t = -(A*px + B*py + C) / denominator
        x_proj = px + A * t
        y_proj = py + B * t
        
        # Calculate distance
        dist = ((x_proj - px)**2 + (y_proj - py)**2)**0.5
        
        if dist < best_distance:
            best_distance = dist
            best_point = (x_proj, y_proj)
    
    # Fallback: if all constraints invalid, return original
    if best_point is None:
        return point
    
    return best_point

    """
    Check if steps follow Deadband Protocol P0->P1->P2 order without skipping.
    
    Args:

    Returns:
        (is_valid, message)
    """
    order = ["P0", "P1", "P2"]
    seen = set()
    last_index = -1

        step_norm = step.strip().upper()
        if step_norm not in order:
