```python
import re
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validate if steps follow Deadband Protocol P0->P1->P2 strictly.
    
    Args:
        steps: List of step strings like ["P0", "P1", "P2"]
    
    Returns:
        Tuple of (is_valid: bool, error_message: str)
    """
    valid_phases = {"P0", "P1", "P2"}
    expected_order = ["P0", "P1", "P2"]
    
    if not steps:
        return False, "Empty step list"
    
    # Check all steps are valid phases
    for i, step in enumerate(steps):
        if step not in valid_phases:
            return False, f"Invalid phase '{step}' at position {i}"
    
    # Check no skipping and correct order
    last_index = -1
    for step in steps:
        current_index = expected_order.index(step)
        if current_index < last_index:
            return False, f"Phase regression: '{step}' appears after higher priority phase"
        if current_index > last_index + 1:
            return False, f"Phase skip detected: '{step}' jumps from {expected_order[last_index] if last_index >=0 else 'start'}"
        last_index = current_index
    
    return True, ""
```

```python
from typing import List, Tuple

def validate_deadband_protocol(steps: List[str]) -> Tuple[bool, str]:
    """
    Validates that the steps list follows the Deadband Protocol:
    - Only contains P0, P1, P2
    - Appears in non-decreasing order P0 -> P1 -> P2
    - No phase is skipped
    """
    allowed = {"P0", "P1", "P2"}
    
    # Check all steps are allowed
    for idx, step in enumerate(steps):
        if step not in allowed:
            return False, f"Invalid step '{step}' at index {idx}"
    
    # Check order and skips
    phase_to_num = {"P0": 0, "P1": 1, "P2": 2}
    prev_num = -1
    for idx, step in enumerate(steps):
        curr_num = phase_to_num[step]
        if curr_num < prev_num:
            return False, f"Protocol regression at index {idx}: {step} after higher phase"
        if curr_num > prev_num + 1:
            return False, f"Protocol skip at index {idx}: jumped to {step} from phase {prev_num}"
        prev_num = curr_num
    
    return True, ""
```

### **2.2 Function 2: `parse_fleet_manifest`**

#### **DeepSeek-chat Output:**

```python
import re
from typing import Dict, List

def parse_fleet_manifest(manifest_text: str) -> Dict:
    """
    Parse fleet manifest text into structured data.
    
    Handles formats:
    - "Name: role [STATUS]"
    - "Name - role [STATUS]"
    
    Returns:
        Dictionary with "agents" list of agent dicts.
    """
    agents = []
    lines = manifest_text.strip().split('\n')
    
    for line_num, line in enumerate(lines, 1):
        line = line.strip()
        if not line:
            continue
            
        # Try "Name: role [STATUS]" format
        match = re.match(r'^([^:]+):\s*([^\[\]]+?)\s*\[([^\]]+)\]$', line)
        if not match:
            # Try "Name - role [STATUS]" format
            match = re.match(r'^([^-]+)-\s*([^\[\]]+?)\s*\[([^\]]+)\]$', line)
        
        if match:
            name = match.group(1).strip()
            role = match.group(2).strip()
            status = match.group(3).strip()
            agents.append({
                "name": name,
                "role": role,
                "status": status
            })
        else:
            # Skip unparseable lines with warning
            continue
    
    return {"agents": agents}
```

```python
import re
from typing import Dict, List

def parse_fleet_manifest(manifest_text: str) -> Dict:
    """
    Parses a fleet manifest text into a structured dictionary.
    Expected line formats:
      "AgentName: role [STATUS]"
      "AgentName - role [STATUS]"
    """
    agents = []
    
    for line in manifest_text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        
        # Match "Name: role [STATUS]" or "Name - role [STATUS]"
        pattern = r'^([^:\-]+)(?::|\-)\s*(.+?)\s*\[([^\]]+)\]$'
        match = re.match(pattern, line)
        
        if match:
            name = match.group(1).strip()
            role = match.group(2).strip()
            status = match.group(3).strip()
            agents.append({
                "name": name,
                "role": role,
                "status": status
            })
        # else: ignore malformed lines
    
    return {"agents": agents}
```
