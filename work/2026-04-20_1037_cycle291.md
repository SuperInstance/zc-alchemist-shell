def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validates a sequence of Deadband Protocol steps.
    Fleet Doctrine P0, P1, P2 must appear in strict order, no skips, no repeats.

    Args:
        steps: List of step labels (e.g., ["P0", "P1", "P2"]).

    Returns:
        Tuple of (is_valid, message). Message explains violation if invalid.
    """
    valid_sequence = ["P0", "P1", "P2"]
    seen = set()

    for i, step in enumerate(steps):
        # Check for invalid step label
        if step not in valid_sequence:
            return False, f"Invalid step '{step}'. Must be one of {valid_sequence}"

        # Check for repeat
        if step in seen:
            return False, f"Repeated step '{step}'. Steps must be unique."
        seen.add(step)

        # Check for skip or wrong order
        expected_step = valid_sequence[i] if i < len(valid_sequence) else None
        if step != expected_step:
            if expected_step is None:
                return False, "Too many steps. Sequence must be exactly P0, P1, P2."
            else:
                return False, f"Out of order. Expected '{expected_step}', got '{step}'."

    # Check if all required steps are present
    if len(seen) < len(valid_sequence):
        missing = [s for s in valid_sequence if s not in seen]
        return False, f"Missing steps: {missing}. Sequence incomplete."

    return True, "Protocol sequence is valid."
