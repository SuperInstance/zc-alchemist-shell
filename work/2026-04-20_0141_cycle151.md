# deadband_filter
def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol P0/P1/P2 logic.
    """
    if not values:
        return []
    
    # P0: Remove rocks (values outside ±2.0)
    safe_range = 2.0
    p0_filtered = [v for v in values if -safe_range <= v <= safe_range]
    
    if not p0_filtered:
        return []
    
    # P1: Find safe channels
    result = [p0_filtered[0]]
    for v in p0_filtered[1:]:
        if abs(v - result[-1]) <= deadband:
            result.append(v)
        # Else: not in same channel, skip (P1 boundary)
    
    # P2: Simple smoothing within channel (moving average of window 2)
    if len(result) >= 2:
        smoothed = [result[0]]
        for i in range(1, len(result)):
            smoothed.append((result[i-1] + result[i]) / 2)
        return smoothed
    
    return result

from datetime import datetime

def parse_fleet_log(log_line: str) -> dict:
    """
    Parse fleet log line.
    """
    import re
    
    # Extract timestamp
    ts_match = re.search(r'\[(.*?)\]', log_line)
    if not ts_match:
        raise ValueError("No timestamp found")
    
    timestamp = datetime.strptime(ts_match.group(1), "%Y-%m-%d %H:%M:%S")
    
    # Extract key=value pairs
    pairs = re.findall(r'(\w+)=([^\s]+)', log_line)
    
    result = {"timestamp": timestamp}
    for key, value in pairs:
        if key == "CYCLE" or key == "TILES":
            result[key.lower()] = int(value)
        else:
            result[key.lower()] = value
    
    return result

# deadband_filter
def deadband_filter(values, deadband=0.1):
    """
    Implements the Deadband Protocol for signal filtering.
    """
    # P0: eliminate values outside safe range
    SAFE_LIMIT = 2.0
    p0_vals = [v for v in values if abs(v) <= SAFE_LIMIT]
    
    if len(p0_vals) == 0:
        return []
    
    # P1: identify safe channels
    filtered = []
    last_kept = p0_vals[0]
    filtered.append(last_kept)
    
    for v in p0_vals[1:]:
        if abs(v - last_kept) <= deadband:
            filtered.append(v)
            last_kept = v
        # otherwise discard (not in same channel)
    
    # P2: optimize within channel via exponential smoothing
    alpha = 0.5
    smoothed = [filtered[0]]
    for i in range(1, len(filtered)):
        smoothed_val = alpha * filtered[i] + (1 - alpha) * smoothed[i-1]
        smoothed.append(smoothed_val)
    
    return smoothed

import re
from datetime import datetime

def parse_fleet_log(log_line):
    pattern = r'\[(?P<timestamp>\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\]\s*(?P<rest>.*)'
    match = re.match(pattern, log_line)
    if not match:
        return {}
    
    data = {'timestamp': datetime.strptime(match.group('timestamp'), '%Y-%m-%d %H:%M:%S')}
    
    # parse key=value pairs
    rest = match.group('rest')
    for pair in rest.split():
        if '=' in pair:
            key, val = pair.split('=', 1)
            key = key.lower()
            if key in ['cycle', 'tiles']:
                data[key] = int(val)
            else:
                data[key] = val
    
    return data

def constraint_snap(point, constraints):

    best_point = point
    best_distance = float('inf')
    
    for axis, value in constraints:
        if axis == 'x':
            dist = abs(x0 - value)
            candidate = (value, y0)
        elif axis == 'y':
            dist = abs(y0 - value)
            candidate = (x0, value)
        else:
            continue
        
        if dist < best_distance:
            best_distance = dist
            best_point = candidate
    
    return best_point
