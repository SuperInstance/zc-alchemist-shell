import re
from typing import List, Dict, Tuple, Any

def deadband_filter(values: List[float], center: float, deadband_width: float) -> List[float]:
    """
    Apply Deadband Protocol to a list of values.
    
    Args:
        values: List of float values to filter.
        center: The center point of the deadband.
        deadband_width: Half-width of the deadband region.
    
    Returns:
        List of filtered values where values within center ± deadband_width
        are replaced with center.
    
    Example:
        >>> deadband_filter([1.0, 2.0, 3.0, 4.0], 2.5, 0.6)
        [1.0, 2.5, 2.5, 4.0]
    """
    if not values:
        return []
    
    return [
        center if abs(v - center) <= deadband_width else v
        for v in values
    ]


def parse_fleet_log(log_line: str) -> Dict[str, str]:
    """
    Parse a standardized fleet log line.
    
    Args:
        log_line: String in format "[TIMESTAMP] SERVICE: MESSAGE"
    
    Returns:
        Dictionary with keys 'timestamp', 'service', 'message'.
        Returns {"timestamp": "ERROR", "service": "ERROR", "message": "ERROR"}
        for malformed lines.
    """
    pattern = r'^\[(.+?)\] (\w+): (.+)$'
    match = re.match(pattern, log_line)
    
    if match:
        return {
            "timestamp": match.group(1),
            "service": match.group(2),
            "message": match.group(3)
        }
    else:
        return {"timestamp": "ERROR", "service": "ERROR", "message": "ERROR"}


def validate_tile(tile_data: Dict) -> Tuple[bool, str]:
    """
    Validate a training tile structure for PLATO room ingestion.
    
    Args:
        tile_data: Dictionary containing tile data.
    
    Returns:
        Tuple of (is_valid, error_message).
    """
    required_keys = {"id", "agent", "content", "cycle"}
    
    # Check for all required keys
    missing_keys = required_keys - set(tile_data.keys())
    if missing_keys:
        return False, f"Missing required keys: {missing_keys}"
    
    # Validate types and constraints
    if not isinstance(tile_data.get("content"), str):
        return False, "Key 'content' must be a string"
    
    if not tile_data["content"].strip():
        return False, "Key 'content' must be non-empty"
    
    if not isinstance(tile_data.get("cycle"), int):
        return False, "Key 'cycle' must be an integer"
    
    # Agent must be non-empty string
    if not isinstance(tile_data.get("agent"), str) or not tile_data["agent"]:
        return False, "Key 'agent' must be a non-empty string"
    
    # ID can be int or str
    id_val = tile_data.get("id")
    if not isinstance(id_val, (int, str)):
        return False, "Key 'id' must be integer or string"
    
    return True, ""
