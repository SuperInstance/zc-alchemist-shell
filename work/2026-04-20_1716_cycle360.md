# Function A: Validate Deadband Protocol
def validate_deadband_protocol(steps: list[str]) -> tuple[bool, str]:
    """
    Validate that a list of steps follows the Deadband Protocol.
    Protocol: Strict P0 → P1 → P2 priority. No skipping phases.
    
    Args:
        steps: List of step labels, e.g., ["P0", "P1", "P2"]
    
    Returns:
        Tuple of (is_valid, error_message). Error message empty if valid.
    """
    expected_order = ["P0", "P1", "P2"]
    seen = set()
    
    for step in steps:
        if step not in expected_order:
            return False, f"Invalid step '{step}'. Must be one of {expected_order}"
        
        # Check for out-of-order
        for higher_priority in expected_order:
            if step == higher_priority:
                break
            if higher_priority not in seen:
                return False, f"Step '{step}' encountered before '{higher_priority}'. Cannot skip phases."
        
        seen.add(step)
    
    return True, ""

import hashlib
import re

def parse_fleet_tile(raw_text: str) -> dict:
    """
    Parse raw training tile text into structured metadata.
    
    Args:
        raw_text: Raw tile content as string.
    
    Returns:
        Dict with keys: valid, type, cycle, content_sha.
    """
    result = {
        "valid": False,
        "type": "unknown",
        "cycle": -1,
        "content_sha": ""
    }
    
    if not raw_text or not isinstance(raw_text, str):
        return result
    
    # Compute SHA-256
    sha = hashlib.sha256(raw_text.encode('utf-8')).hexdigest()
    result["content_sha"] = sha
    
    # Extract cycle number
    cycle_match = re.search(r'Cycle\s+(\d+)', raw_text, re.IGNORECASE)
    if cycle_match:
        result["cycle"] = int(cycle_match.group(1))
    
    # Detect tile type
    text_lower = raw_text.lower()
    if any(word in text_lower for word in ["function", "def ", "import ", "class "]):
        result["type"] = "code"
    elif any(word in text_lower for word in ["analysis", "compare", "benchmark"]):
        result["type"] = "analysis"
    elif any(word in text_lower for word in ["documentation", "readme", "guide"]):
        result["type"] = "documentation"
    elif "cycle" in text_lower and "phase" in text_lower:
        result["type"] = "log"
    else:
        result["type"] = "unknown"
    
    # Consider valid if we extracted at least a cycle or type
    if result["cycle"] != -1 or result["type"] != "unknown":
        result["valid"] = True
    
    return result

    """
    Checks if steps follow Deadband Protocol: P0, then P1, then P2.
    No skipping allowed.
    """
    order = ["P0", "P1", "P2"]
    for i, step in enumerate(steps):
        if step not in order:
            return False, f"Bad step {step}"
        if i > 0:
            prev_idx = order.index(steps[i-1])
            curr_idx = order.index(step)
            if curr_idx < prev_idx:
                return False, f"Out of order: {steps[i-1]} -> {step}"
            if curr_idx > prev_idx + 1:
                return False, f"Skipped phase: {steps[i-1]} -> {step}"
    return True, ""
