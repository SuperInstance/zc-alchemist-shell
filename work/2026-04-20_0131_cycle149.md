import re
from datetime import datetime
from typing import Dict, Optional

def parse_log_line(line: str) -> Optional[Dict[str, str]]:
    """
    Parse a structured log line format: 'YYYY-MM-DD HH:MM:SS LEVEL [MODULE] Message'
    Returns a dictionary with keys: timestamp, level, module, message.
    Returns None if line doesn't match expected format.
    """
    pattern = r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (\w+) \[([^\]]+)\] (.+)$'
    match = re.match(pattern, line)
    if not match:
        return None
    return {
        'timestamp': match.group(1),
        'level': match.group(2),
        'module': match.group(3),
        'message': match.group(4)
    }

from typing import List, Tuple

def deadband_filter(values: List[float], p0_thresh: float = -10.0, p1_thresh: float = 5.0) -> List[Tuple[int, float, str]]:
    """
    Apply deadband protocol to a list of values.
    P0: value <= p0_thresh → 'UNSAFE'
    P1: p0_thresh < value <= p1_thresh → 'SAFE_CHANNEL'
    P2: value > p1_thresh → 'OPTIMIZE'
    Returns list of tuples (index, value, status).
    """
    result = []
    for i, v in enumerate(values):
        if v <= p0_thresh:
            status = "UNSAFE"
        elif v <= p1_thresh:
            status = "SAFE_CHANNEL"
        else:
            status = "OPTIMIZE"
        result.append((i, v, status))
    return result

if __name__ == "__main__":
    # Example 1: Mixed values
    data1 = [-15.0, -5.0, 0.0, 3.0, 10.0, 20.0]
    print(deadband_filter(data1))
    
    # Example 2: All P0
    data2 = [-20.0, -12.0, -8.0]
    print(deadband_filter(data2, p0_thresh=-5.0))
    
    # Example 3: Edge cases
    data3 = [-10.0, 5.0, 5.1]
    print(deadband_filter(data3))

def deadband_filter(values: List[float], p0_limit: float = -10.0, p1_limit: float = 5.0) -> List[Tuple[int, float, str]]:
    """
    Implements fleet Deadband Protocol.
    P0 (negative space): value <= p0_limit → 'AVOID'
    P1 (safe channel): p0_limit < value <= p1_limit → 'SAFE'
    P2 (optimize): value > p1_limit → 'OPTIMIZE'
    """
    results = []
    for idx, val in enumerate(values):
        if val <= p0_limit:
            band = "AVOID"
        elif val <= p1_limit:
            band = "SAFE"
        else:
            band = "OPTIMIZE"
        results.append((idx, val, band))
    return results

if __name__ == "__main__":
