**Prompt:**
```
Write a Python function `deadband_filter` that implements the Cocapn fleet's Deadband Protocol.

Input: 
- `values`: list of floats
- `deadband`: float (default 0.1)

Protocol (P0/P1/P2):
- P0: Remove any values where abs(value) < deadband (negative space)
- P1: Keep only values where deadband <= abs(value) <= 1.0 (safe channel)
- P2: Return normalized list: each value divided by max(abs(values)) (optimize within channel)

Return filtered, normalized list. Include docstring and example.
```

**DeepSeek-chat output:**
```python
def deadband_filter(values, deadband=0.1):
    """
    Apply Deadband Protocol (P0/P1/P2) to a list of float values.
    
    P0: Remove values in negative space (|value| < deadband)
    P1: Keep values in safe channel (deadband <= |value| <= 1.0)
    P2: Normalize remaining values by max absolute value.
    
    Args:
        values: List of float values
        deadband: Threshold for P0 removal (default 0.1)
    
    Returns:
        List of filtered, normalized floats
    """
    if not values:
        return []
    
    # P0: Map negative space (remove)
    p1_candidates = [v for v in values if abs(v) >= deadband]
    
    # P1: Find safe channel
    safe = [v for v in p1_candidates if abs(v) <= 1.0]
    
    if not safe:
        return []
    
    # P2: Optimize within channel (normalize)
    max_abs = max(abs(v) for v in safe)
    normalized = [v / max_abs for v in safe]
    
    return normalized
```

```python
def deadband_filter(values, deadband=0.1):
    """
    Implements the Deadband Protocol for signal filtering.
    
    Steps:
    1. P0: Remove values where absolute value is less than deadband.
    2. P1: Keep values where absolute value is between deadband and 1.0 inclusive.
    3. P2: Normalize the remaining values by dividing by the maximum absolute value.
    
    Parameters:
    values (list of float): Input list of values.
    deadband (float): Deadband threshold (default 0.1).
    
    Returns:
    list of float: Filtered and normalized values.
    """
    # P0: Remove values in negative space
    filtered = [v for v in values if abs(v) >= deadband]
    
    # P1: Keep values in safe channel
    filtered = [v for v in filtered if abs(v) <= 1.0]
    
    if not filtered:
        return []
    
    # P2: Normalize
    max_val = max(abs(v) for v in filtered)
    normalized = [v / max_val for v in filtered]
    
    return normalized

# Example usage
sample = [0.05, 0.3, 0.9, 1.5, -0.2, -0.01]
print(deadband_filter(sample))
# Output: [0.3, 0.9, -0.2] -> normalized -> [0.333..., 1.0, -0.222...]
```

**DeepSeek-chat output:**
```python
import subprocess
import os

def git_repo_analyzer(repo_path):
    """
    Analyze a local git repository for basic stats.
    
    Args:
        repo_path: Path to git repository
        
    Returns:
        Dictionary with commit_count, branch_count, recent_authors, latest_commit_hash
        Returns None if repo_path is not a valid git repository
    """
    def run_git(cmd):
        try:
            result = subprocess.run(
                cmd,
                cwd=repo_path,
                capture_output=True,
                text=True,
                check=True
            )
            return result.stdout.strip()
        except (subprocess.CalledProcessError, FileNotFoundError):
            return None
    
    # Check if it's a git repo
    if run_git(["git", "rev-parse", "--git-dir"]) is None:
        return None
    
    stats = {}
    
    # Commit count
    commit_out = run_git(["git", "rev-list", "--count", "HEAD"])
    stats["commit_count"] = int(commit_out) if commit_out else 0
    
    # Branch count (local)
    branch_out = run_git(["git", "branch", "--list"])
    if branch_out:
        branches = [b.strip() for b in branch_out.split('\n') if b.strip()]
        stats["branch_count"] = len(branches)
    else:
        stats["branch_count"] = 0
    
    # Recent authors (last 20 commits, get unique, take last 5)
    author_out = run_git(["git", "log", "--format=%an", "-n", "20"])
    if author_out:
        authors = []
        seen = set()
        # Preserve order from most recent
        for author in author_out.split('\n'):
            if author not in seen:
                seen.add(author)
                authors.append(author)
        stats["recent_authors"] = authors[:5]
    else:
        stats["recent_authors"] = []
    
    # Latest commit hash
    hash_out = run_git(["git", "rev-parse", "HEAD"])
    stats["latest_commit_hash"] = hash_out if hash_out else ""
    
    return stats

# Example
if __name__ == "__main__":
    result = git_repo_analyzer(".")
    print(result)
```
