```python
def deadband_filter(data: float, lower: float, upper: float) -> float:
    """
    Apply a deadband filter to a single data point.
    
    If data is within [lower, upper] inclusive, return 0.
    Otherwise, return data unchanged.
    
    Args:
        data: Input value
        lower: Lower deadband threshold
        upper: Upper deadband threshold
    
    Returns:
        Filtered value
    
    Raises:
        ValueError: If lower > upper
    """
    if lower > upper:
        raise ValueError(f"Lower ({lower}) cannot be greater than upper ({upper})")
    
    if lower <= data <= upper:
        return 0.0
    return data
```

#### **Function 2: parse_fleet_log_line**

```python
from datetime import datetime

def parse_fleet_log_line(line: str) -> dict:
    """
    Parse a structured fleet log line.
    
    Expected format: "YYYY-MM-DD HH:MM | Agent | Event | Status"
    
    Args:
        line: Log line string
    
    Returns:
        Dictionary with keys: timestamp, agent, event, status
    
    Raises:
        ValueError: If line format is invalid
    """
    parts = line.split(" | ")
    if len(parts) != 4:
        raise ValueError(f"Expected 4 pipe-separated parts, got {len(parts)}")
    
    timestamp_str, agent, event, status = parts
    
    try:
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M")
    except ValueError:
        raise ValueError(f"Invalid timestamp format: {timestamp_str}")
    
    return {
        "timestamp": timestamp,
        "agent": agent.strip(),
        "event": event.strip(),
        "status": status.strip()
    }
```

#### **Function 3: constraint_snap**
```python
import math

def constraint_snap(
    point: tuple[float, float],
    line_start: tuple[float, float],
    line_end: tuple[float, float],
    tolerance: float
) -> tuple[float, float]:
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Uses vector projection to find closest point on infinite line,
    then clamps to segment endpoints.
    
    Args:
        point: (x, y) coordinate to snap
        line_start: (x, y) start of line segment
        line_end: (x, y) end of line segment
        tolerance: Maximum distance for snapping
    
    Returns:
        Original point or snapped point
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector from line_start to line_end
    dx = x2 - x1
    dy = y2 - y1
    
    # Vector from line_start to point
    vx = px - x1
    vy = py - y1
    
    # Dot product and squared length
    dot = vx * dx + vy * dy
    len_sq = dx * dx + dy * dy
    
    # Handle zero-length segment
    if len_sq == 0:
        # Segment is a point
        dist = math.hypot(px - x1, py - y1)
        return (x1, y1) if dist <= tolerance else (px, py)
    
    # Projection parameter t (0 to 1 for segment)
    t = max(0, min(1, dot / len_sq))
    
    # Closest point on segment
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    
    # Distance to closest point
    dist = math.hypot(px - closest_x, py - closest_y)
    
    return (closest_x, closest_y) if dist <= tolerance else (px, py)
```

```python
import numpy as np

def constraint_snap(point, line_start, line_end, tolerance):
    point = np.array(point)
    line_start = np.array(line_start)
    line_end = np.array(line_end)
    
    # Vector calculations
    line_vec = line_end - line_start
    point_vec = point - line_start
    
    line_len = np.linalg.norm(line_vec)
    if line_len == 0:
        return tuple(point)
    
    t = np.dot(point_vec, line_vec) / (line_len ** 2)
    t = max(0, min(1, t))
    
    closest = line_start + t * line_vec
    dist = np.linalg.norm(point - closest)
    
    if dist <= tolerance:
        return tuple(closest)
    else:
        return tuple(point)
```
