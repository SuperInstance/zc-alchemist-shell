```python
def deadband_filter(values, lower, upper):
    """
    Apply a deadband filter to input values.
    
    Args:
        values: Single numeric value or list/array of values
        lower: Lower bound of deadband
        upper: Upper bound of deadband
    
    Returns:
        Filtered value(s) where values within [lower, upper] return 0,
        otherwise value is shifted by the nearest bound.
    
    Example:
        >>> deadband_filter(5, 2, 8)
        0
        >>> deadband_filter(1, 2, 8)
        -1  # 1 - 2 = -1
        >>> deadband_filter(10, 2, 8)
        2   # 10 - 8 = 2
    """
    import numpy as np
    
    # Convert to numpy array for vectorized operations
    values_arr = np.asarray(values)
    result = np.zeros_like(values_arr)
    
    # Below deadband
    mask_below = values_arr < lower
    result[mask_below] = values_arr[mask_below] - lower
    
    # Above deadband  
    mask_above = values_arr > upper
    result[mask_above] = values_arr[mask_above] - upper
    
    # Within deadband already zeros
    
    # Return scalar if input was scalar
    if np.isscalar(values):
        return result.item()
    return result
```

```python
import re
from datetime import datetime

def parse_fleet_log_line(line):
    """
    Parse a structured fleet log line into dictionary.
    
    Expected format: "YYYY-MM-DD HH:MM [AGENT] LEVEL: Message"
    
    Args:
        line (str): Log line to parse
    
    Returns:
        dict with keys: timestamp, agent, level, message
        Returns None if line doesn't match expected format.
    
    Example:
        >>> parse_fleet_log_line("2026-04-19 23:10 [Oracle1] INFO: Tile accepted")
        {'timestamp': datetime(2026, 4, 19, 23, 10),
         'agent': 'Oracle1',
         'level': 'INFO',
         'message': 'Tile accepted'}
    """
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}) \[([^\]]+)\] (\w+): (.+)'
    match = re.match(pattern, line.strip())
    
    if not match:
        return None
    
    date_str, agent, level, message = match.groups()
    
    try:
        timestamp = datetime.strptime(date_str, "%Y-%m-%d %H:%M")
    except ValueError:
        timestamp = date_str  # Keep as string if parsing fails
    
    return {
        'timestamp': timestamp,
        'agent': agent,
        'level': level,
        'message': message.strip()
    }
```

```python
import math

def constraint_snap_point(point, line_start, line_end, tolerance=0.01):
    """
    Snap a point to the nearest point on a line segment if within tolerance.
    
    Args:
        point: (x, y) tuple
        line_start: (x1, y1) tuple
        line_end: (x2, y2) tuple
        tolerance: Maximum distance for snapping
    
    Returns:
        (x, y) tuple - snapped point if within tolerance, original point otherwise
    """
    px, py = point
    x1, y1 = line_start
    x2, y2 = line_end
    
    # Vector math for projection
    dx = x2 - x1
    dy = y2 - y1
    line_len_sq = dx*dx + dy*dy
    
    # Handle zero-length segment
    if line_len_sq == 0:
        dist = math.hypot(px - x1, py - y1)
        return line_start if dist <= tolerance else point
    
    # Project point onto line
    t = ((px - x1) * dx + (py - y1) * dy) / line_len_sq
    
    # Clamp to segment
    t = max(0, min(1, t))
    
    # Closest point on segment
    closest_x = x1 + t * dx
    closest_y = y1 + t * dy
    
    # Check distance
    dist = math.hypot(px - closest_x, py - closest_y)
    
    if dist <= tolerance:
        return (closest_x, closest_y)
    return point
```

```python
def deadband_filter(values, lower, upper):
    """
    Implements a deadband filter.
    
    Parameters:
    values (float or list): Input value(s)
    lower (float): Lower deadband threshold
    upper (float): Upper deadband threshold
    
    Returns:
    float or list: Filtered output where values in [lower, upper] map to 0,
    others are offset by nearest threshold.
    """
    import numpy as np
    
    if isinstance(values, (int, float)):
        # Single value
        if lower <= values <= upper:
            return 0.0
        elif values < lower:
            return values - lower
        else:
            return values - upper
    else:
        # Assume iterable
        result = []
        for v in values:
            if lower <= v <= upper:
                result.append(0.0)
            elif v < lower:
                result.append(v - lower)
            else:
                result.append(v - upper)
        return np.array(result) if isinstance(values, np.ndarray) else result
```

```python
    """
    Parse fleet log line format: "DATE TIME [AGENT] LEVEL: MESSAGE"

    pattern = r'(\d{4}-\d{2}-\d{2})\s+(\d{2}:\d{2})\s+\[([^\]]+)\]\s+(\w+):\s*(.+)'
    match = re.match(pattern, line)
    
    if match:
        date, time, agent, level, message = match.groups()
        return {
            'date': date,
            'time': time,
            'agent': agent,
            'level': level,
            'message': message
        }
    return None
```
