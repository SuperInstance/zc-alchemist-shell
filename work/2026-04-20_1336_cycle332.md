def deadband_filter(values: list[float], deadband: float = 0.1) -> list[float]:
    """
    Apply Deadband Protocol P0/P1/P2 to a signal.
    
    P0: Remove values in negative space (|value| < deadband → 0.0)
    P1: Find safe channels (quantize to nearest 0.5 multiple if |value| > deadband)
    P2: Return filtered list.
    """
    result = []
    for v in values:
        # P0: Map negative space
        if abs(v) < deadband:
            result.append(0.0)
        else:
            # P1: Find safe channel (quantize to nearest 0.5)
            quantized = round(v * 2) / 2
            result.append(quantized)
    return result
```

from datetime import datetime

def parse_fleet_state(log_line: str) -> dict:
    """
    Parse a fleet state log line into structured data.
    """
    try:
        parts = log_line.split(" | ")
        if len(parts) != 4:
            return {"error": "malformed line"}
        
        timestamp_str = parts[0].strip()
        agent_part = parts[1].strip()
        cycle_part = parts[2].strip()
        tiles_part = parts[3].strip()
        
        # Parse timestamp
        timestamp = datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M UTC")
        
        # Parse agent
        agent = agent_part.split("=")[1] if "=" in agent_part else "Unknown"
        
        # Parse cycle
        cycle = int(cycle_part.split("=")[1]) if "=" in cycle_part else 0
        
        # Parse phase and tiles
        phase = 0
        tiles_submitted = 0
        tiles_accepted = 0
        
        if "PHASE=" in log_line:
            phase_part = [p for p in parts if p.startswith("PHASE=")][0]
            phase = int(phase_part.split("=")[1])
        
        if "TILES=" in log_line:
            tiles_part = [p for p in parts if p.startswith("TILES=")][0]
            tiles_str = tiles_part.split("=")[1]
            if "/" in tiles_str:
                submitted, accepted = tiles_str.split("/")
                tiles_submitted = int(submitted)
                tiles_accepted = int(accepted)
        
        return {
            "timestamp": timestamp,
            "agent": agent,
            "cycle": cycle,
            "phase": phase,
            "tiles_submitted": tiles_submitted,
            "tiles_accepted": tiles_accepted
        }
    except Exception as e:
        return {"error": f"parse failed: {str(e)}"}
```

**Function 3: `validate_training_tile`**
```python
def validate_training_tile(tile: dict, schema: dict) -> tuple[bool, list[str]]:
    """
    Validate a training tile against a JSON schema.
    """
    errors = []
    
    # Check required fields
    required = schema.get("required", [])
    for field in required:
        if field not in tile:
            errors.append(f"Missing required field: {field}")
    
    # Type checking
    type_map = schema.get("properties", {})
    for field, expected_type in type_map.items():
        if field in tile:
            actual_type = type(tile[field]).__name__
            if expected_type == "string" and actual_type != "str":
                errors.append(f"Field {field} should be string, got {actual_type}")
            elif expected_type == "integer" and actual_type != "int":
                errors.append(f"Field {field} should be integer, got {actual_type}")
            elif expected_type == "number" and actual_type not in ["int", "float"]:
                errors.append(f"Field {field} should be number, got {actual_type}")
    
    # Special validation for tile content
    if "content" in tile and not isinstance(tile["content"], str):
        errors.append("Content must be a string")
    
    if "timestamp" in tile:
        try:
            datetime.fromisoformat(tile["timestamp"].replace("Z", "+00:00"))
        except:
            errors.append("Invalid timestamp format")
    
    return len(errors) == 0, errors
```
