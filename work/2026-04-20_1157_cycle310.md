```python
def validate_deadband_protocol(actions: list[str]) -> dict[str, any]:
    """
    Validates a sequence of actions against the Deadband Protocol.
    Protocol: P0 (map negative) must precede P1 (find channels), which must precede P2 (optimize).
    Actions are strings expected to start with 'P0', 'P1', or 'P2'.

    Args:
        actions: List of action strings.

    Returns:
        Dictionary with keys:
        - valid: True if sequence respects protocol, False otherwise.
        - violations: List of violation descriptions (empty if valid).
        - phase_summary: Count of actions per phase.
    """
    phases = {'P0': 0, 'P1': 1, 'P2': 2}
    violations = []
    phase_count = {p: 0 for p in phases}

    current_max_phase = -1  # Tracks the highest phase index encountered

    for i, action in enumerate(actions):
        # Extract phase prefix
        prefix = action[:2] if len(action) >= 2 else None
        if prefix not in phases:
            violations.append(f"Action '{action}' at index {i} has invalid phase prefix.")
            continue

        phase_idx = phases[prefix]
        phase_count[prefix] += 1

        # Check ordering: phase index should not be less than current_max_phase
        if phase_idx < current_max_phase:
            violations.append(
                f"Protocol violation at index {i}: '{action}' (phase {prefix}) "
                f"appears after a higher phase (phase index {current_max_phase})."
            )
        current_max_phase = max(current_max_phase, phase_idx)

    return {
        'valid': len(violations) == 0,
        'violations': violations,
        'phase_summary': phase_count
    }
```

```python
import re
from datetime import datetime
from typing import Any

def parse_fleet_tile(tile_text: str) -> dict[str, Any]:
    """
    Parses a raw fleet training tile text into structured metadata and content.

    Expected format (loose):
        # Cycle <number> — <Agent>
        **Phase X: ...**
        **Task:** ...
        ---
        <content>

    Args:
        tile_text: Raw tile string.

    Returns:
        Dictionary with keys: agent, cycle, phase, timestamp, task, content.
        Missing values are set to None.
    """
    # Default result structure
    result = {
        'agent': None,
        'cycle': None,
        'phase': None,
        'timestamp': None,
        'task': None,
        'content': None
    }

    if not tile_text or not isinstance(tile_text, str):
        return result

    lines = tile_text.strip().split('\n')

    # Extract agent and cycle from first line pattern
    first_line = lines[0] if lines else ""
    cycle_match = re.search(r'Cycle\s+(\d+)', first_line)
    agent_match = re.search(r'—\s*(\w+)', first_line)
    if cycle_match:
        result['cycle'] = int(cycle_match.group(1))
    if agent_match:
        result['agent'] = agent_match.group(1)

    # Extract phase
    for line in lines[:5]:  # Search early lines
        phase_match = re.search(r'Phase\s+(\d+)', line)
        if phase_match:
            result['phase'] = int(phase_match.group(1))
            break

    # Extract task (look for **Task:** marker)
    for i, line in enumerate(lines):
        if '**Task:**' in line:
            result['task'] = line.split('**Task:**')[-1].strip()
            break

    # Extract timestamp (look for ISO date pattern)
    date_match = re.search(r'\d{4}-\d{2}-\d{2}', tile_text)
    if date_match:
        try:
            result['timestamp'] = datetime.fromisoformat(date_match.group(0)).isoformat()
        except ValueError:
            pass

    # Content is everything after the first markdown separator '---' or after headers
    sep_index = None
    for i, line in enumerate(lines):
        if line.strip() == '---':
            sep_index = i
            break
    if sep_index is not None and sep_index + 1 < len(lines):
        result['content'] = '\n'.join(lines[sep_index + 1:]).strip()
    else:
        # Fallback: use all lines after the first two
        result['content'] = '\n'.join(lines[2:]).strip() if len(lines) > 2 else ""

    return result
```
