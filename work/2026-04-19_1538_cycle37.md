"""
Deadband Protocol Validator
P0: Map negative space (what NOT to do)
P1: Find safe channels (where you CAN be)
P2: Optimize within channels (best path)
Strict priority: Never skip to P2.
"""

def validate_deadband_protocol(action_list):
    """
    Validate a list of actions against the Deadband Protocol.
    
    Args:
        action_list (list of str): List of action descriptions.
        
    Returns:
        dict: {
            'p0_count': int,
            'p1_count': int, 
            'p2_count': int,
            'violations': list of dicts {
                'action': str,
                'expected_priority': str,
                'actual_priority': str,
                'rule': str
            },
            'has_skip_to_p2': bool
        }
    """
    if not isinstance(action_list, list):
        raise TypeError("action_list must be a list")
    
    # Priority classification rules (simplified for demo)
    p0_keywords = ['avoid', 'never', 'stop', 'don\'t', 'do not', 'prevent', 'block']
    p1_keywords = ['find', 'explore', 'map', 'discover', 'locate', 'identify']
    p2_keywords = ['optimize', 'improve', 'enhance', 'refine', 'tune', 'accelerate']
    
    results = {
        'p0_count': 0,
        'p1_count': 0,
        'p2_count': 0,
        'violations': [],
        'has_skip_to_p2': False
    }
    
    # Track sequence for P2 skipping violation
    action_sequence = []
    
    for i, action in enumerate(action_list):
        if not isinstance(action, str):
            continue
            
        action_lower = action.lower()
        
        # Classify priority
        priority = None
        if any(kw in action_lower for kw in p0_keywords):
            priority = 'P0'
            results['p0_count'] += 1
        elif any(kw in action_lower for kw in p1_keywords):
            priority = 'P1'
            results['p1_count'] += 1
        elif any(kw in action_lower for kw in p2_keywords):
            priority = 'P2'
            results['p2_count'] += 1
        else:
            # Default to P1 if no keywords match (safe assumption)
            priority = 'P1'
            results['p1_count'] += 1
        
        action_sequence.append(priority)
        
        # Check for priority violations (simplified rule: P2 shouldn't come before P0/P1)
        if priority == 'P2' and i > 0:
            if 'P0' not in action_sequence[:i] and 'P1' not in action_sequence[:i]:
                results['violations'].append({
                    'action': action,
                    'expected_priority': 'P0 or P1',
                    'actual_priority': 'P2',
                    'rule': 'P2 action appears before any P0/P1 action'
                })
                results['has_skip_to_p2'] = True
    
    return results

"""
Deadband Protocol Validation Function
"""

def validate_deadband_protocol(actions):
    """
    Classifies actions into P0, P1, P2 based on Deadband Protocol.
    
    Parameters:
    actions (list): A list of action strings.
    
    Returns:
    dict: A dictionary containing counts and violations.
    """
    # Input validation
    if not actions:
        return {"p0_count": 0, "p1_count": 0, "p2_count": 0, "violations": [], "has_skip_to_p2": False}
    
    # Define keyword mappings
    p0_triggers = ["avoid", "never", "stop", "don't", "do not", "prevent"]
    p1_triggers = ["find", "explore", "map", "discover", "locate"]
    p2_triggers = ["optimize", "improve", "enhance", "refine"]
    
    # Initialize results
    results = {
        "p0_count": 0,
        "p1_count": 0,
        "p2_count": 0,
        "violations": [],
        "has_skip_to_p2": False
    }
    
    # Track priority sequence
    sequence = []
    
    for idx, action in enumerate(actions):
        action_lower = action.lower()
        detected = False
        
        # Check P0
        for trigger in p0_triggers:
            if trigger in action_lower:
                results["p0_count"] += 1
                sequence.append("P0")
                detected = True
                break
        
        if not detected:
            # Check P1
            for trigger in p1_triggers:
                if trigger in action_lower:
                    results["p1_count"] += 1
                    sequence.append("P1")
                    detected = True
                    break
        
        if not detected:
            # Check P2
            for trigger in p2_triggers:
                if trigger in action_lower:
                    results["p2_count"] += 1
                    sequence.append("P2")
                    detected = True
                    break
        
        # If no triggers found, default to P1
        if not detected:
            results["p1_count"] += 1
            sequence.append("P1")
        
        # Check for P2 skipping violation
        if sequence[-1] == "P2" and idx > 0:
            prior_p0_p1 = any(p in ["P0", "P1"] for p in sequence[:-1])
            if not prior_p0_p1:
                results["violations"].append({
                    "action": action,
                    "expected_priority": "P0 or P1",
                    "actual_priority": "P2",
                    "rule": "P2 action occurred before P0/P1 in sequence"
                })
                results["has_skip_to_p2"] = True
    
    return results
