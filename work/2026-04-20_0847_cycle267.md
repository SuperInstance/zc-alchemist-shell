def deadband_filter(values, deadband=0.1, mode='p0'):
    """
    Apply Deadband Protocol filtering to a list of float values.
    
    Args:
        values: List of float values to filter
        deadband: Threshold for considering a value "safe" (default 0.1)
        mode: 'p0', 'p1', or 'p2' for protocol phase
    
    Returns:
        Filtered list based on protocol phase
    
    P0: Remove values that hit rocks (outside deadband)
    P1: Keep only values within safe channels (within deadband)
    P2: Optimize by normalizing values within safe range
    """
    if not values:
        return []
    
    # P0: Map negative space - identify unsafe values
    unsafe = [abs(v) > deadband for v in values]
    
    if mode == 'p0':
        # Return mask of unsafe values (what NOT to do)
        return unsafe
    elif mode == 'p1':
        # P1: Find safe channels
        safe_values = [v for v in values if abs(v) <= deadband]
        return safe_values
    elif mode == 'p2':
        # P2: Optimize within channels
        safe_values = [v for v in values if abs(v) <= deadband]
        if not safe_values:
            return []
        # Normalize to [-1, 1] range within deadband
        normalized = [v / deadband for v in safe_values]
        return normalized
    else:
        raise ValueError(f"Invalid mode: {mode}. Use 'p0', 'p1', or 'p2'")

def deadband_filter(values, deadband=0.1, mode='p0'):
    """
    Implements the Deadband Protocol for signal processing.
    
    The Deadband Protocol has three phases:
    P0: Identify dangerous values (outside deadband)
    P1: Extract safe values (within deadband)  
    P2: Optimize safe values
    
    Parameters:
    values (list): Input values
    deadband (float): Threshold for safety
    mode (str): 'p0', 'p1', or 'p2'
    
    Returns:
    list: Processed values according to mode
    """
    # Input validation
    if not isinstance(values, list):
        raise TypeError("values must be a list")
    
    if mode not in ['p0', 'p1', 'p2']:
        raise ValueError("mode must be 'p0', 'p1', or 'p2'")
    
    # P0: Identify what NOT to do
    if mode == 'p0':
        dangerous = []
        for val in values:
            if abs(val) > deadband:
                dangerous.append(True)
            else:
                dangerous.append(False)
        return dangerous
    
    # P1: Find safe channels
    elif mode == 'p1':
        safe = []
        for val in values:
            if abs(val) <= deadband:
                safe.append(val)
        return safe
    
    # P2: Optimize within channels
    elif mode == 'p2':
        safe = []
        for val in values:
            if abs(val) <= deadband:
                safe.append(val)
        
        if not safe:
            return []
        
        # Apply optimization: scale to maximum utilization
        max_val = max(abs(v) for v in safe) if safe else 1.0
        optimized = [v / max_val for v in safe]
        return optimized

    """
    Parse semi-structured fleet log entries into structured data.
    
    Expected format:
    [2026-04-20 08:42:15] AGENT=Alchemist CYCLE=267 STATUS=ACTIVE TILES=3
    
    Returns:
        List of dictionaries with parsed log data
    """
    entries = []
    
    # Regex pattern for fleet log format
    pattern = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\]\s+(.*)'
    
    for line in log_text.strip().split('\n'):
        match = re.match(pattern, line)
        if not match:
            continue
            
        timestamp_str, data_str = match.groups()
        
        # Parse timestamp
        try:
            timestamp = datetime.strptime(timestamp_str, '%Y-%m-%d %H:%M:%S')
        except ValueError:
            timestamp = None
        
        # Parse key-value pairs
        data = {}
        pairs = data_str.split()
        for pair in pairs:
            if '=' in pair:
                key, value = pair.split('=', 1)
                # Convert numeric values
                if value.isdigit():
                    value = int(value)
                elif value.replace('.', '', 1).isdigit() and value.count('.') == 1:
                    value = float(value)
                data[key.lower()] = value
        
        entries.append({
            'timestamp': timestamp,
            'raw': line,
            **data
        })
    
    return entries
```

import re

def parse_fleet_log(log_text):
    """
    Parses fleet log data from string format to structured dictionary.
    
    Log format example:
    [2026-04-20 08:42:15] AGENT=Alchemist CYCLE=267 STATUS=ACTIVE TILES=3
    
    Args:
        log_text (str): Multiline log text
    
    Returns:
        list: List of dictionaries, each representing a log entry
    """
    entries = []
    
    # Regular expression to match log entries
    log_regex = r'\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\]\s+(.+)'
    
    lines = log_text.strip().split('\n')
    
    for line in lines:
        match = re.search(log_regex, line)
        if match:
            timestamp = match.group(1)
            content = match.group(2)
            
            # Parse key-value pairs
            kv_pairs = content.split()
            entry_data = {"timestamp": timestamp, "raw_line": line}
            
            for pair in kv_pairs:
                if '=' in pair:
                    key, value = pair.split('=', 1)
                    
                    # Type conversion
                    if value.isdigit():
                        value = int(value)
                    elif value.replace('.', '', 1).isdigit() and '.' in value:
                        try:
                            value = float(value)
                        except:
                            pass
                    
                    entry_data[key.lower()] = value
            
            entries.append(entry_data)
    
    return entries
```
